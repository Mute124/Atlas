187int Atlas::BProject::run(int argc, char* argv[]) {
188 int code = 0;
189
190 bool shouldClose = false;
191
192
193
194 MinimizeWindow();
195 RequestWindowAttention(getAtlasEngine()->getWindow() );
196 while (!shouldClose) {
197 getAtlasEngine()->getInputRegistry().get()->checkAll("renderer");
198 shouldClose = getAtlasEngine()->getWindow().get()->shouldClose();
199
200 draw();
201 }
202
203 // This is here because it reduces CPU usage (by alot!).
204 std::this_thread::yield();
205 return code;
206}
