59RendererExtensions::Image RendererExtensions::CaptureScreen()
60{
61#if defined RMLUI_RENDERER_GL2 || defined RMLUI_RENDERER_GL3
62
63 int viewport[4] = {}; // x, y, width, height
64 glGetIntegerv(GL_VIEWPORT, viewport);
65
66 Image image;
67 image.num_components = 3;
68 image.width = viewport[2];
69 image.height = viewport[3];
70
71 if (image.width < 1 || image.height < 1)
72 return Image();
73
74 const int byte_size = image.width * image.height * image.num_components;
75 image.data = Rml::UniquePtr<Rml::byte[]>(new Rml::byte[byte_size]);
76
77 glReadPixels(0, 0, image.width, image.height, GL_RGB, GL_UNSIGNED_BYTE, image.data.get());
78
79 bool result = true;
80 GLenum err;
81 while ((err = glGetError()) != GL_NO_ERROR)
82 {
83 result = false;
84 Rml::Log::Message(Rml::Log::LT_ERROR, "Could not capture screenshot, got GL error: 0x%x", err);
85 }
86
87 if (!result)
88 return Image();
89
90 return image;
91
92#else
93
94 return Image();
95
96#endif
97}
