247bool Atlas::LoadShaderModule(const char* filePath, VkDevice device, VkShaderModule* outShaderModule)
248{
249 // open the file. With cursor at the end
250 std::ifstream file(filePath, std::ios::ate | std::ios::binary);
251
252 if (!file.is_open()) {
253 return false;
254 }
255
256 // find what the size of the file is by looking up the location of the cursor
257 // because the cursor is at the end, it gives the size directly in bytes
258 size_t fileSize = (size_t)file.tellg();
259
260 // spirv expects the buffer to be on uint32, so make sure to reserve a int
261 // vector big enough for the entire file
262 std::vector<uint32_t> buffer(fileSize / sizeof(uint32_t));
263
264 // put file cursor at beginning
265 file.seekg(0);
266
267 // load the entire file into the buffer
268 file.read((char*)buffer.data(), fileSize);
269
270 // now that the file is loaded into the buffer, we can close it
271 file.close();
272
273 // create a new shader module, using the buffer we loaded
274 VkShaderModuleCreateInfo createInfo = {};
275 createInfo.sType = VK_STRUCTURE_TYPE_SHADER_MODULE_CREATE_INFO;
276 createInfo.pNext = nullptr;
277
278 // codeSize has to be in bytes, so multply the ints in the buffer by size of
279 // int to know the real size of the buffer
280 createInfo.codeSize = buffer.size() * sizeof(uint32_t);
281 createInfo.pCode = buffer.data();
282
283 // check that the creation goes well.
284 VkShaderModule shaderModule;
285 if (vkCreateShaderModule(device, &createInfo, nullptr, &shaderModule) != VK_SUCCESS) {
286 return false;
287 }
288 *outShaderModule = shaderModule;
289 return true;
290}
