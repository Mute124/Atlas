68void Atlas::DescriptorAllocator::init_pool(VkDevice device, uint32_t maxSets, std::span<PoolSizeRatio> poolRatios)

70 std::vector<VkDescriptorPoolSize> poolSizes;
71

83 pool_info.pPoolSizes = poolSizes.data();
84
85 vkCreateDescriptorPool(device, &pool_info, nullptr, &pool);

152{
153 //create a descriptor pool that will hold 10 sets with 1 image each
154 std::vector<DescriptorAllocator::PoolSizeRatio> sizes =
155 {
156 { VK_DESCRIPTOR_TYPE_STORAGE_IMAGE, 1 }
157 };
158
159 mGlobalDescriptorAllocator.init_pool(mDevice, 10, sizes);
160
161 //make the descriptor set layout for our compute draw
162 {
163 DescriptorLayoutBuilder builder;
164 builder.addBinding(0, VK_DESCRIPTOR_TYPE_STORAGE_IMAGE);
165 mDrawImageDescriptorLayout = builder.build(mDevice, VK_SHADER_STAGE_COMPUTE_BIT);
166 }
167
168 // other code
169 //allocate a descriptor set for our draw image
170 mDrawImageDescriptors = mGlobalDescriptorAllocator.allocate(mDevice, mDrawImageDescriptorLayout);

173 imgInfo.imageLayout = VK_IMAGE_LAYOUT_GENERAL;
174 imgInfo.imageView = mDrawImage.imageView;
175
176 VkWriteDescriptorSet drawImageWrite = {};
177 drawImageWrite.sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;
178 drawImageWrite.pNext = nullptr;
179

181 drawImageWrite.dstSet = mDrawImageDescriptors;

188 //make sure both the descriptor allocator and the new layout get cleaned up properly

256void Atlas::VulkanRenderingBackend::initCommands()
257{
258 //create a command pool for commands submitted to the graphics queue.
259 //we also want the pool to allow for resetting of individual command buffers
260 VkCommandPoolCreateInfo commandPoolInfo = CreateCommandPoolCreateInfo(mGraphicsQueueFamily, VK_COMMAND_POOL_CREATE_RESET_COMMAND_BUFFER_BIT);
261
262 for (int i = 0; i < FRAME_OVERLAP; i++) {
263
264 vkCreateCommandPool(mDevice, &commandPoolInfo, nullptr, &mFrameDataArray.at(i).commandPool);
265
266 // allocate the default command buffer that we will use for rendering
267 VkCommandBufferAllocateInfo cmdAllocInfo = CreateCommandBufferAllocateInfo(mFrameDataArray.at(i).commandPool, 1);
268
269 vkAllocateCommandBuffers(mDevice, &cmdAllocInfo, &mFrameDataArray.at(i).mainCommandBuffer);
270 }
271
273 //VkCommandBufferAllocateInfo cmdAllocInfo = CreateCommandBufferAllocateInfo(mImmediateSubmitInfo.commandPool, 1);
274
275 //vkAllocateCommandBuffers(mDevice, &cmdAllocInfo, &mImmediateSubmitInfo.commandBuffer);
276
277 //mMainDeletionQueue.push([=]() {
278 // vkDestroyCommandPool(mDevice, mImmediateSubmitInfo.commandPool, nullptr);
279 //});
280
281 //create syncronization structures
282 //one fence to control when the gpu has finished rendering the frame,
283 //and 2 semaphores to syncronize rendering with swapchain
284 //we want the fence to start signalled so we can wait on it on the first frame
285 VkFenceCreateInfo fenceCreateInfo = CreateFenceCreateInfo(VK_FENCE_CREATE_SIGNALED_BIT);
286 VkSemaphoreCreateInfo semaphoreCreateInfo = CreateSemaphoreCreateInfo();
287
288 for (int i = 0; i < FRAME_OVERLAP; i++) {
289 vkCreateFence(mDevice, &fenceCreateInfo, nullptr, &mFrameDataArray.at(i).renderFence);
290
291 vkCreateSemaphore(mDevice, &semaphoreCreateInfo, nullptr, &mFrameDataArray.at(i).swapchainSemaphore);
292 vkCreateSemaphore(mDevice, &semaphoreCreateInfo, nullptr, &mFrameDataArray.at(i).renderSemaphore);
293 }
294
295 vkCreateFence(mDevice, &fenceCreateInfo, nullptr, &mImmediateSubmitInfo.fence);
296 mMainDeletionQueue.push([=]() { vkDestroyFence(mDevice, mImmediateSubmitInfo.fence, nullptr); });
297}

333void Atlas::VulkanRenderingBackend::ImmediateSubmit(std::function<void(VkCommandBuffer cmd)>&& function)
334{
335 VK_CHECK(vkResetFences(mDevice, 1, &mImmediateSubmitInfo.fence));
336 VK_CHECK(vkResetCommandBuffer(mImmediateSubmitInfo.commandBuffer, 0));
337
338 VkCommandBuffer cmd = mImmediateSubmitInfo.commandBuffer;
339
340 VkCommandBufferBeginInfo cmdBeginInfo = CreateCommandBufferBeginInfo(VK_COMMAND_BUFFER_USAGE_ONE_TIME_SUBMIT_BIT);
341
342 VK_CHECK(vkBeginCommandBuffer(cmd, &cmdBeginInfo));
343
344 function(cmd);
345
346 VK_CHECK(vkEndCommandBuffer(cmd));
347
348 VkCommandBufferSubmitInfo cmdinfo = SubmitCommandBufferInfo(cmd);
349 VkSubmitInfo2 submit = SubmitInfo(&cmdinfo, nullptr, nullptr);
350
351 // submit command buffer to the queue and execute it.
352 // _renderFence will now block until the graphic commands finish execution
353 VK_CHECK(vkQueueSubmit2(mGraphicsQueue, 1, &submit, mImmediateSubmitInfo.fence));
354
355 constexpr uint64_t cTimeout = 9999999999;
356
357 VK_CHECK(vkWaitForFences(mDevice, 1, &mImmediateSubmitInfo.fence, true, cTimeout));
358}

523void Atlas::VulkanRenderingBackend::draw()
524{
525 const static uint32_t cFenceCount = 1;
526
527 FrameData& currentFrame = getCurrentFrame();
528
529 resetFences(cFenceCount, currentFrame);
530
531 //request image from the swapchain
532 uint32_t swapchainImageIndex;
533 vkAcquireNextImageKHR(mDevice, mSwapchain, mNextImageTimeoutNS, currentFrame.swapchainSemaphore, nullptr, &swapchainImageIndex);
534
535 // reset command buffer
536 VkCommandBufferResetFlags cmdResetFlags = 0;
537
538 //naming it cmd for shorter writing
539 VkCommandBuffer cmd = currentFrame.mainCommandBuffer;
540
541 // now that we are sure that the commands finished executing, we can safely
542 // reset the command buffer to begin recording again.
543 vkResetCommandBuffer(cmd, cmdResetFlags);
544
545 //begin the command buffer recording. We will use this command buffer exactly once, so we want to let vulkan know that
546 VkCommandBufferBeginInfo cmdBeginInfo = CreateCommandBufferBeginInfo(VK_COMMAND_BUFFER_USAGE_ONE_TIME_SUBMIT_BIT);
547
548 mDrawExtent.width = mDrawImage.imageExtent.width;
549 mDrawExtent.height = mDrawImage.imageExtent.height;
550
551 vkBeginCommandBuffer(cmd, &cmdBeginInfo);
552
553 // transition our main draw image into general layout so we can write into it
554 // we will overwrite it all so we dont care about what was the older layout
555 TransitionImage(cmd, mDrawImage.image, VK_IMAGE_LAYOUT_UNDEFINED, VK_IMAGE_LAYOUT_GENERAL);
556
557 drawBackground(cmd);
558
559 //transition the draw image and the swapchain image into their correct transfer layouts
560 TransitionImage(cmd, mDrawImage.image, VK_IMAGE_LAYOUT_GENERAL, VK_IMAGE_LAYOUT_TRANSFER_SRC_OPTIMAL);
561 TransitionImage(cmd, mSwapchainImages[swapchainImageIndex], VK_IMAGE_LAYOUT_UNDEFINED, VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL);
562
563 // execute a copy from the draw image into the swapchain
564 CopyImageToImage(cmd, mDrawImage.image, mSwapchainImages[swapchainImageIndex], mDrawExtent, mSwapchainExtent);
565
566 // set swapchain image layout to Present so we can show it on the screen
567 TransitionImage(cmd, mSwapchainImages[swapchainImageIndex], VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL, VK_IMAGE_LAYOUT_PRESENT_SRC_KHR);
568
569 //finalize the command buffer (we can no longer add commands, but it can now be executed)
570 vkEndCommandBuffer(cmd);
571
572 //prepare the submission to the queue.
573 //we want to wait on the _presentSemaphore, as that semaphore is signaled when the swapchain is ready
574 //we will signal the _renderSemaphore, to signal that rendering has finished
575
576 VkCommandBufferSubmitInfo cmdinfo = SubmitCommandBufferInfo(cmd);
577
578 VkSemaphoreSubmitInfo waitInfo = SubmitSemaphoreInfo(VK_PIPELINE_STAGE_2_COLOR_ATTACHMENT_OUTPUT_BIT_KHR, currentFrame.swapchainSemaphore);
579 VkSemaphoreSubmitInfo signalInfo = SubmitSemaphoreInfo(VK_PIPELINE_STAGE_2_ALL_GRAPHICS_BIT, currentFrame.renderSemaphore);
580
581 VkSubmitInfo2 submit = SubmitInfo(&cmdinfo, &signalInfo, &waitInfo);
582
583 //submit command buffer to the queue and execute it.
584 // _renderFence will now block until the graphic commands finish execution
585 vkQueueSubmit2(mGraphicsQueue, 1, &submit, currentFrame.renderFence);
586
587 // prepare present
588 // this will put the image we just rendered to into the visible window.
589 // we want to wait on the _renderSemaphore for that,
590 // as its necessary that drawing commands have finished before the image is displayed to the user
591 VkPresentInfoKHR presentInfo = {};
592 presentInfo.sType = VK_STRUCTURE_TYPE_PRESENT_INFO_KHR;
593 presentInfo.pNext = nullptr;
594 presentInfo.pSwapchains = &mSwapchain;
595 presentInfo.swapchainCount = 1;
596
597 presentInfo.pWaitSemaphores = &currentFrame.renderSemaphore;
598 presentInfo.waitSemaphoreCount = 1;
599
600 presentInfo.pImageIndices = &swapchainImageIndex;
601
602 vkQueuePresentKHR(mGraphicsQueue, &presentInfo);
603
604 //increase the number of frames drawn
605 mCurrentFrameNumber++;
606}

608void Atlas::VulkanRenderingBackend::drawBackground(VkCommandBuffer cmd)
609{
610 //make a clear-color from frame number. This will flash with a 120 frame period.
611 VkClearColorValue clearValue;
612 float flash = std::abs(std::sin(mCurrentFrameNumber / 120.f));
613 clearValue = { { 0.0f, 0.0f, flash, 1.0f } };
614
615 VkImageSubresourceRange clearRange = CreateImageSubresourceRange(VK_IMAGE_ASPECT_COLOR_BIT);
616
617 //clear image
618 vkCmdClearColorImage(cmd, mDrawImage.image, VK_IMAGE_LAYOUT_GENERAL, &clearValue, 1, &clearRange);
619
620 // bind the gradient drawing compute pipeline
621 vkCmdBindPipeline(cmd, VK_PIPELINE_BIND_POINT_COMPUTE, mGradientPipeline);
622
623 // bind the descriptor set containing the draw image for the compute pipeline
624 vkCmdBindDescriptorSets(cmd, VK_PIPELINE_BIND_POINT_COMPUTE, mGradientPipelineLayout, 0, 1, &mDrawImageDescriptors, 0, nullptr);
625

916bool Atlas::LoadShaderModule(const char* filePath, VkDevice device, VkShaderModule* outShaderModule)
917{
918 // open the file. With cursor at the end
919 std::ifstream file(filePath, std::ios::ate | std::ios::binary);
920
921 if (!file.is_open()) {
922 return false;
923 }
924
925 // find what the size of the file is by looking up the location of the cursor
926 // because the cursor is at the end, it gives the size directly in bytes
927 size_t fileSize = (size_t)file.tellg();
928
929 // spirv expects the buffer to be on uint32, so make sure to reserve a int
930 // vector big enough for the entire file
931 std::vector<uint32_t> buffer(fileSize / sizeof(uint32_t));
932
933 // put file cursor at beginning
934 file.seekg(0);
935
936 // load the entire file into the buffer
937 file.read((char*)buffer.data(), fileSize);
938
939 // now that the file is loaded into the buffer, we can close it
940 file.close();
941
942 // create a new shader module, using the buffer we loaded
943 VkShaderModuleCreateInfo createInfo = {};
944 createInfo.sType = VK_STRUCTURE_TYPE_SHADER_MODULE_CREATE_INFO;
945 createInfo.pNext = nullptr;
946
947 // codeSize has to be in bytes, so multply the ints in the buffer by size of
948 // int to know the real size of the buffer
949 createInfo.codeSize = buffer.size() * sizeof(uint32_t);
950 createInfo.pCode = buffer.data();
951
952 // check that the creation goes well.
953 VkShaderModule shaderModule;
954 if (vkCreateShaderModule(device, &createInfo, nullptr, &shaderModule) != VK_SUCCESS) {
955 return false;
956 }
957 *outShaderModule = shaderModule;
958 return true;
959}
