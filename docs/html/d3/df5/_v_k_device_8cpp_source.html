339 ErrorLog("Error when building the triangle fragment shader module");

346 if (!LoadShaderModule("C:/Dev/Techstorm-v5/shaders/colored_triangle.vert.spv", mDevice, &triangleVertexShader)) {
347 ErrorLog("Error when building the triangle vertex shader module");
348 }
349 else {
350 TraceLog("Triangle vertex shader succesfully loaded");
351 }
352
353 //build the pipeline layout that controls the inputs/outputs of the shader

356 vkCreatePipelineLayout(mDevice, &pipeline_layout_info, nullptr, &_trianglePipelineLayout);
357
358 PipelineBuilder pipelineBuilder;
359
360 //use the triangle layout we created
361 pipelineBuilder.settings.pipelineLayout = _trianglePipelineLayout;
362 //connecting the vertex and pixel shaders to the pipeline
363 pipelineBuilder.setShaders(triangleVertexShader, triangleFragShader);
364 //it will draw triangles
365 pipelineBuilder.setInputTopography(VK_PRIMITIVE_TOPOLOGY_TRIANGLE_LIST);
366 //filled triangles
367 pipelineBuilder.setPolygonMode(VK_POLYGON_MODE_FILL);
368 //no backface culling
369 pipelineBuilder.setCullMode(VK_CULL_MODE_NONE, VK_FRONT_FACE_CLOCKWISE);
370 //no multisampling
371 pipelineBuilder.setMultisamplingNone();
372 //no blending
373 pipelineBuilder.disableBlending();
374 //no depth testing
375 pipelineBuilder.disableDepthTest();
376
377 //connect the image format we will draw into, from draw image
378 pipelineBuilder.setColorAttachmentFormat(mDrawImage.imageFormat);
379 pipelineBuilder.setDepthFormat(VK_FORMAT_UNDEFINED);
380
381 //finally build the pipeline
382 _trianglePipeline = pipelineBuilder.buildPipeline(mDevice);
383
384 //clean structures
385 vkDestroyShaderModule(mDevice, triangleFragShader, nullptr);
386 vkDestroyShaderModule(mDevice, triangleVertexShader, nullptr);
387
388 mMainDeletionQueue.push([&]() {
389 vkDestroyPipelineLayout(mDevice, _trianglePipelineLayout, nullptr);
390 vkDestroyPipeline(mDevice, _trianglePipeline, nullptr);

641 AllocatedBuffer staging = createBuffer(vertexBufferSize + indexBufferSize, VK_BUFFER_USAGE_TRANSFER_SRC_BIT, VMA_MEMORY_USAGE_CPU_ONLY);
642
643 void* data = staging.allocation->GetMappedData();
644
645 // copy vertex buffer
646 memcpy(data, vertices.data(), vertexBufferSize);
647
648 // copy index buffer
649 memcpy((char*)data + vertexBufferSize, indices.data(), indexBufferSize);
650
651 ImmediateSubmit([&](VkCommandBuffer cmd) {
652 VkBufferCopy vertexCopy{ 0 };
653 vertexCopy.dstOffset = 0;
654 vertexCopy.srcOffset = 0;
655 vertexCopy.size = vertexBufferSize;
656
657 vkCmdCopyBuffer(cmd, staging.buffer, newSurface.vertexBuffer.buffer, 1, &vertexCopy);
658
659 VkBufferCopy indexCopy{ 0 };
660 indexCopy.dstOffset = 0;
661 indexCopy.srcOffset = vertexBufferSize;
662 indexCopy.size = indexBufferSize;
663
664 vkCmdCopyBuffer(cmd, staging.buffer, newSurface.indexBuffer.buffer, 1, &indexCopy);
665 });
666
667 destroyBuffer(staging);
668
669 return newSurface;
670}

672void Atlas::VulkanRenderingBackend::ImmediateSubmit(std::function<void(VkCommandBuffer cmd)>&& function)
673{
674 vkResetFences(mDevice, 1, &mImmediateSubmitInfo.fence);
675
676 vkResetCommandBuffer(mImmediateSubmitInfo.commandBuffer, 0);
677
678 VkCommandBuffer cmd = mImmediateSubmitInfo.commandBuffer;
679
680 VkCommandBufferBeginInfo cmdBeginInfo = CreateCommandBufferBeginInfo(VK_COMMAND_BUFFER_USAGE_ONE_TIME_SUBMIT_BIT);
681
682 vkBeginCommandBuffer(cmd, &cmdBeginInfo);
683
684 function(cmd);
685
686 vkEndCommandBuffer(cmd);
687
688 VkCommandBufferSubmitInfo cmdinfo = SubmitCommandBufferInfo(cmd);
689 VkSubmitInfo2 submit = SubmitInfo(&cmdinfo, nullptr, nullptr);
690
691 // submit command buffer to the queue and execute it.
692 // _renderFence will now block until the graphic commands finish execution
693 vkQueueSubmit2(mGraphicsQueue, 1, &submit, mImmediateSubmitInfo.fence);
694
695 constexpr uint64_t cTimeout = 9999999999;
696
697 vkWaitForFences(mDevice, 1, &mImmediateSubmitInfo.fence, true, cTimeout);
698}

968void Atlas::VulkanRenderingBackend::endDrawingMode()
969{
970 FrameData& currentFrame = getCurrentFrame();
971
972 //finalize the command buffer (we can no longer add commands, but it can now be executed)
973 vkEndCommandBuffer(mCurrentDrawData.cmd);
974
975 //prepare the submission to the queue.
976 //we want to wait on the _presentSemaphore, as that semaphore is signaled when the swapchain is ready
977 //we will signal the _renderSemaphore, to signal that rendering has finished
978
979 VkCommandBufferSubmitInfo cmdinfo = SubmitCommandBufferInfo(mCurrentDrawData.cmd);
980
981 VkSemaphoreSubmitInfo waitInfo = SubmitSemaphoreInfo(VK_PIPELINE_STAGE_2_COLOR_ATTACHMENT_OUTPUT_BIT_KHR, currentFrame.swapchainSemaphore);
982 VkSemaphoreSubmitInfo signalInfo = SubmitSemaphoreInfo(VK_PIPELINE_STAGE_2_ALL_GRAPHICS_BIT, currentFrame.renderSemaphore);
983
984 VkSubmitInfo2 submit = SubmitInfo(&cmdinfo, &signalInfo, &waitInfo);
985
986 //submit command buffer to the queue and execute it.
987 // _renderFence will now block until the graphic commands finish execution
988 vkQueueSubmit2(mGraphicsQueue, 1, &submit, currentFrame.renderFence);
989
990 // prepare present
991 // this will put the image we just rendered to into the visible window.
992 // we want to wait on the _renderSemaphore for that,
993 // as its necessary that drawing commands have finished before the image is displayed to the user
994 VkPresentInfoKHR presentInfo = {};
995 presentInfo.sType = VK_STRUCTURE_TYPE_PRESENT_INFO_KHR;
996 presentInfo.pNext = nullptr;
997 presentInfo.pSwapchains = &mSwapchain;
998 presentInfo.swapchainCount = 1;
999
1000 presentInfo.pWaitSemaphores = &currentFrame.renderSemaphore;
1001 presentInfo.waitSemaphoreCount = 1;
1002
1003 presentInfo.pImageIndices = &mCurrentDrawData.swapchainImageIndex;
1004
1005 vkQueuePresentKHR(mGraphicsQueue, &presentInfo);
1006
1007 //increase the number of frames drawn
1008 mCurrentFrameNumber++;
1009
1010 // reset the current draw data to default values
1011 mCurrentDrawData = {};
1012}
