139void Atlas::SDLGameWindow::update()
140{
141 SDL_Event event;
142
143 while (SDL_PollEvent(&event) != 0) {
144 if (event.type == SDL_QUIT) {
145
146 std::cout << "Window closed" << std::endl;
147 mShouldClose = true;
148 }
149
150 if (event.type == SDL_WINDOWEVENT) {
151 if (event.window.event == SDL_WINDOWEVENT_MINIMIZED) {
152 //event= true;
153 std::cout << "Window minimized" << std::endl;
154 }
155 if (event.window.event == SDL_WINDOWEVENT_RESTORED) {
156 //stop_rendering = false;
157
158 std::cout << "Window restored" << std::endl;
159 }
160 }
161
162 //send SDL event to imgui for handling
