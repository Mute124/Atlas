71 Rml::TextureHandle LoadTexture(Rml::Vector2i& texture_dimensions, const Rml::String& source) override
72 {
73 Rml::FileInterface* file_interface = Rml::GetFileInterface();
74 Rml::FileHandle file_handle = file_interface->Open(source);
75 if (!file_handle)
76 return {};
77
78 file_interface->Seek(file_handle, 0, SEEK_END);
79 const size_t buffer_size = file_interface->Tell(file_handle);
80 file_interface->Seek(file_handle, 0, SEEK_SET);
81
82 using Rml::byte;
83 Rml::UniquePtr<byte[]> buffer(new byte[buffer_size]);
84 file_interface->Read(buffer.get(), buffer_size, file_handle);
85 file_interface->Close(file_handle);
86
87 const size_t i_ext = source.rfind('.');
88 Rml::String extension = (i_ext == Rml::String::npos ? Rml::String() : source.substr(i_ext + 1));
89
90 SDL_Surface* surface = IMG_LoadTyped_RW(SDL_RWFromMem(buffer.get(), int(buffer_size)), 1, extension.c_str());
91 if (!surface)
92 return {};
93
94 texture_dimensions = {surface->w, surface->h};
95
96 if (surface->format->format != SDL_PIXELFORMAT_RGBA32 && surface->format->format != SDL_PIXELFORMAT_BGRA32)
97 {
98 // Ensure correct format for premultiplied alpha conversion below. Additionally, fix rendering images with
99 // no alpha channel, see https://github.com/mikke89/RmlUi/issues/239
100 SDL_Surface* converted_surface = SDL_ConvertSurfaceFormat(surface, SDL_PixelFormatEnum::SDL_PIXELFORMAT_RGBA32, 0);
101 SDL_FreeSurface(surface);
102
103 if (!converted_surface)
104 return {};
105
106 surface = converted_surface;
107 }
108
109 // Convert colors to premultiplied alpha, which is necessary for correct alpha compositing.
110 byte* pixels = static_cast<byte*>(surface->pixels);
111 for (int i = 0; i < surface->w * surface->h * 4; i += 4)
112 {
113 const byte alpha = pixels[i + 3];
114 for (int j = 0; j < 3; ++j)
115 pixels[i + j] = byte(int(pixels[i + j]) * int(alpha) / 255);
116 }
117
118 SDL_Texture* texture = SDL_CreateTextureFromSurface(renderer, surface);
119 SDL_FreeSurface(surface);
120
121 return (Rml::TextureHandle)texture;
122 }

124 Rml::TextureHandle GenerateTexture(Rml::Span<const Rml::byte> source, Rml::Vector2i source_dimensions) override
125 {
126#if SDL_BYTEORDER == SDL_BIG_ENDIAN
127 Uint32 rmask = 0xff000000;
128 Uint32 gmask = 0x00ff0000;
129 Uint32 bmask = 0x0000ff00;
130 Uint32 amask = 0x000000ff;
131#else
132 Uint32 rmask = 0x000000ff;
133 Uint32 gmask = 0x0000ff00;
134 Uint32 bmask = 0x00ff0000;
135 Uint32 amask = 0xff000000;
136#endif
137
138 SDL_Surface* surface = SDL_CreateRGBSurfaceFrom((void*)source.data(), source_dimensions.x, source_dimensions.y, 32, source_dimensions.x * 4,
139 rmask, gmask, bmask, amask);
140 SDL_Texture* texture = SDL_CreateTextureFromSurface(renderer, surface);
141 SDL_SetTextureBlendMode(texture, SDL_BLENDMODE_BLEND);
142 SDL_FreeSurface(surface);
143 return (Rml::TextureHandle)texture;
144 }

168bool Backend::Initialize(const char* window_name, int width, int height, bool allow_resize)
169{
170 RMLUI_ASSERT(!data);
171
172 if (SDL_Init(SDL_INIT_VIDEO | SDL_INIT_EVENTS | SDL_INIT_TIMER) != 0)
173 return false;
174
175 // Submit click events when focusing the window.
176 SDL_SetHint(SDL_HINT_MOUSE_FOCUS_CLICKTHROUGH, "1");
177
178 // Request stencil buffer of at least 8-bit size to supporting clipping on transformed elements.
179 SDL_GL_SetAttribute(SDL_GL_STENCIL_SIZE, 8);
180 SDL_GL_SetAttribute(SDL_GL_DOUBLEBUFFER, 1);
181
182 // Enable linear filtering and MSAA for better-looking visuals, especially when transforms are applied.
183 SDL_SetHint(SDL_HINT_RENDER_SCALE_QUALITY, "linear");
184 SDL_GL_SetAttribute(SDL_GL_MULTISAMPLEBUFFERS, 1);
185 SDL_GL_SetAttribute(SDL_GL_MULTISAMPLESAMPLES, 2);
186
187 const Uint32 window_flags = (SDL_WINDOW_OPENGL | (allow_resize ? SDL_WINDOW_RESIZABLE : 0));
188
189 SDL_Window* window = SDL_CreateWindow(window_name, SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED, width, height, window_flags);
190 if (!window)
191 {
192 // Try again on low-quality settings.
193 SDL_SetHint(SDL_HINT_RENDER_SCALE_QUALITY, "nearest");
194 SDL_GL_SetAttribute(SDL_GL_MULTISAMPLEBUFFERS, 0);
195 SDL_GL_SetAttribute(SDL_GL_MULTISAMPLESAMPLES, 0);
196 window = SDL_CreateWindow(window_name, SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED, width, height, window_flags);
197 if (!window)
198 {
199 Rml::Log::Message(Rml::Log::LT_ERROR, "SDL error on create window: %s", SDL_GetError());
200 return false;
201 }
202 }
203
204 SDL_GLContext glcontext = SDL_GL_CreateContext(window);
205 int opengl_renderer_index = -1;
206 int num_render_drivers = SDL_GetNumRenderDrivers();
207 for (int i = 0; i < num_render_drivers; i++)
208 {
209 SDL_RendererInfo info;
210 if (SDL_GetRenderDriverInfo(i, &info) == 0)
211 {
212 if (strcmp(info.name, "opengl") == 0)
213 opengl_renderer_index = i;
214 }
215 }
216
217 SDL_Renderer* renderer = SDL_CreateRenderer(window, opengl_renderer_index, SDL_RENDERER_ACCELERATED | SDL_RENDERER_PRESENTVSYNC);
218 if (!renderer)
219 return false;
220
221 GLenum err = glewInit();
222 if (err != GLEW_OK)
223 {
224 Rml::Log::Message(Rml::Log::LT_ERROR, "GLEW error: %s", glewGetErrorString(err));
225 return false;
226 }
227
228 data = Rml::MakeUnique<BackendData>(renderer);
229
230 data->window = window;
231 data->glcontext = glcontext;
232 data->renderer = renderer;
233
234 data->system_interface.SetWindow(window);
235 data->render_interface.SetViewport(width, height);
236
237 return true;
238}

265bool Backend::ProcessEvents(Rml::Context* context, KeyDownCallback key_down_callback, bool power_save)
266{
267 RMLUI_ASSERT(data && context);
268
269 bool result = data->running;
270 data->running = true;
271
272 SDL_Event ev;
273 int has_event = 0;
274 if (power_save)
275 has_event = SDL_WaitEventTimeout(&ev, static_cast<int>(Rml::Math::Min(context->GetNextUpdateDelay(), 10.0) * 1000));
276 else
277 has_event = SDL_PollEvent(&ev);
278 while (has_event)
279 {
280 switch (ev.type)
281 {
282 case SDL_QUIT:
283 {
284 result = false;
285 }
286 break;
287 case SDL_KEYDOWN:
288 {
289 const Rml::Input::KeyIdentifier key = RmlSDL::ConvertKey(ev.key.keysym.sym);
290 const int key_modifier = RmlSDL::GetKeyModifierState();
291 const float native_dp_ratio = 1.f;
292
293 // See if we have any global shortcuts that take priority over the context.
294 if (key_down_callback && !key_down_callback(context, key, key_modifier, native_dp_ratio, true))
295 break;
296 // Otherwise, hand the event over to the context by calling the input handler as normal.
297 if (!RmlSDL::InputEventHandler(context, ev))
298 break;
299 // The key was not consumed by the context either, try keyboard shortcuts of lower priority.
300 if (key_down_callback && !key_down_callback(context, key, key_modifier, native_dp_ratio, false))
301 break;
302 }
303 break;
304 case SDL_WINDOWEVENT:
305 {
306 switch (ev.window.event)
307 {
308 case SDL_WINDOWEVENT_SIZE_CHANGED:
309 {
310 Rml::Vector2i dimensions(ev.window.data1, ev.window.data2);
311 data->render_interface.SetViewport(dimensions.x, dimensions.y);
312 }
313 break;
314 }
315 RmlSDL::InputEventHandler(context, ev);
316 }
317 break;
318 default:
319 {
320 RmlSDL::InputEventHandler(context, ev);
321 }
322 break;
323 }
324 has_event = SDL_PollEvent(&ev);
325 }
326
327 return result;
328}

337void Backend::BeginFrame()
338{
339 RMLUI_ASSERT(data);
340
341 SDL_SetRenderDrawColor(data->renderer, 0, 0, 0, 0);
342 SDL_RenderClear(data->renderer);
343
344 // SDL uses shaders that we need to disable here.
345 glUseProgramObjectARB(0);
346 glTexEnvf(GL_TEXTURE_ENV, GL_TEXTURE_ENV_MODE, GL_MODULATE);
347
348 data->render_interface.BeginFrame();
349}

351void Backend::PresentFrame()
352{
353 RMLUI_ASSERT(data);
354
355 data->render_interface.EndFrame();
356
357 // Draw a fake point just outside the screen to let SDL know that it needs to reset its state in case it wants to render a texture next frame.
358 SDL_SetRenderDrawBlendMode(data->renderer, SDL_BLENDMODE_NONE);
359 SDL_RenderDrawPoint(data->renderer, -1, -1);
360
361 SDL_RenderPresent(data->renderer);
362}
