59// Backend data stored in io.BackendRendererUserData to allow support for multiple Dear ImGui contexts

67bool ImGui_ImplSDLRenderer3_Init(SDL_Renderer* renderer)
68{
69 ImGuiIO& io = ImGui::GetIO();
70 IMGUI_CHECKVERSION();
71 IM_ASSERT(io.BackendRendererUserData == nullptr && "Already initialized a renderer backend!");
72 IM_ASSERT(renderer != nullptr && "SDL_Renderer not initialized!");
73
74 // Setup backend capabilities flags
75 ImGui_ImplSDLRenderer3_Data* bd = IM_NEW(ImGui_ImplSDLRenderer3_Data)();
76 io.BackendRendererUserData = (void*)bd;
77 io.BackendRendererName = "imgui_impl_sdlrenderer3";
78 io.BackendFlags |= ImGuiBackendFlags_RendererHasVtxOffset; // We can honor the ImDrawCmd::VtxOffset field, allowing for large meshes.
79
80 bd->Renderer = renderer;
81
82 return true;
83}

116// https://github.com/libsdl-org/SDL/issues/9009
117static int SDL_RenderGeometryRaw8BitColor(SDL_Renderer* renderer, ImVector<SDL_FColor>& colors_out, SDL_Texture* texture, const float* xy, int xy_stride, const SDL_Color* color, int color_stride, const float* uv, int uv_stride, int num_vertices, const void* indices, int num_indices, int size_indices)
118{
119 const Uint8* color2 = (const Uint8*)color;
120 colors_out.resize(num_vertices);
121 SDL_FColor* color3 = colors_out.Data;
122 for (int i = 0; i < num_vertices; i++)
123 {
124 color3[i].r = color->r / 255.0f;
125 color3[i].g = color->g / 255.0f;
126 color3[i].b = color->b / 255.0f;
127 color3[i].a = color->a / 255.0f;
128 color2 += color_stride;
129 color = (const SDL_Color*)color2;
130 }
131 return SDL_RenderGeometryRaw(renderer, texture, xy, xy_stride, color3, sizeof(*color3), uv, uv_stride, num_vertices, indices, num_indices, size_indices);
132}

134void ImGui_ImplSDLRenderer3_RenderDrawData(ImDrawData* draw_data, SDL_Renderer* renderer)
135{
136 ImGui_ImplSDLRenderer3_Data* bd = ImGui_ImplSDLRenderer3_GetBackendData();
137
138 // If there's a scale factor set by the user, use that instead
139 // If the user has specified a scale factor to SDL_Renderer already via SDL_RenderSetScale(), SDL will scale whatever we pass
140 // to SDL_RenderGeometryRaw() by that scale factor. In that case we don't want to be also scaling it ourselves here.
141 float rsx = 1.0f;
142 float rsy = 1.0f;
143 SDL_GetRenderScale(renderer, &rsx, &rsy);
144 ImVec2 render_scale;
145 render_scale.x = (rsx == 1.0f) ? draw_data->FramebufferScale.x : 1.0f;
146 render_scale.y = (rsy == 1.0f) ? draw_data->FramebufferScale.y : 1.0f;
147
148 // Avoid rendering when minimized, scale coordinates for retina displays (screen coordinates != framebuffer coordinates)
149 int fb_width = (int)(draw_data->DisplaySize.x * render_scale.x);
150 int fb_height = (int)(draw_data->DisplaySize.y * render_scale.y);
151 if (fb_width == 0 || fb_height == 0)
152 return;
153
154 // Backup SDL_Renderer state that will be modified to restore it afterwards
155 struct BackupSDLRendererState
156 {
157 SDL_Rect Viewport;
158 bool ViewportEnabled;
159 bool ClipEnabled;
160 SDL_Rect ClipRect;
161 };
162 BackupSDLRendererState old = {};
163 old.ViewportEnabled = SDL_RenderViewportSet(renderer);
164 old.ClipEnabled = SDL_RenderClipEnabled(renderer);
165 SDL_GetRenderViewport(renderer, &old.Viewport);
166 SDL_GetRenderClipRect(renderer, &old.ClipRect);
167
168 // Setup desired state
169 ImGui_ImplSDLRenderer3_SetupRenderState(renderer);
170
171 // Setup render state structure (for callbacks and custom texture bindings)
172 ImGuiPlatformIO& platform_io = ImGui::GetPlatformIO();
173 ImGui_ImplSDLRenderer3_RenderState render_state;
174 render_state.Renderer = renderer;
175 platform_io.Renderer_RenderState = &render_state;
176
177 // Will project scissor/clipping rectangles into framebuffer space
178 ImVec2 clip_off = draw_data->DisplayPos; // (0,0) unless using multi-viewports
179 ImVec2 clip_scale = render_scale;
180
181 // Render command lists
182 for (int n = 0; n < draw_data->CmdListsCount; n++)
183 {
184 const ImDrawList* draw_list = draw_data->CmdLists[n];
185 const ImDrawVert* vtx_buffer = draw_list->VtxBuffer.Data;
186 const ImDrawIdx* idx_buffer = draw_list->IdxBuffer.Data;
187
188 for (int cmd_i = 0; cmd_i < draw_list->CmdBuffer.Size; cmd_i++)
189 {
190 const ImDrawCmd* pcmd = &draw_list->CmdBuffer[cmd_i];
191 if (pcmd->UserCallback)
192 {
193 // User callback, registered via ImDrawList::AddCallback()
194 // (ImDrawCallback_ResetRenderState is a special callback value used by the user to request the renderer to reset render state.)
195 if (pcmd->UserCallback == ImDrawCallback_ResetRenderState)
196 ImGui_ImplSDLRenderer3_SetupRenderState(renderer);
197 else
198 pcmd->UserCallback(draw_list, pcmd);
199 }
200 else
201 {
202 // Project scissor/clipping rectangles into framebuffer space
203 ImVec2 clip_min((pcmd->ClipRect.x - clip_off.x) * clip_scale.x, (pcmd->ClipRect.y - clip_off.y) * clip_scale.y);
204 ImVec2 clip_max((pcmd->ClipRect.z - clip_off.x) * clip_scale.x, (pcmd->ClipRect.w - clip_off.y) * clip_scale.y);
205 if (clip_min.x < 0.0f) { clip_min.x = 0.0f; }
206 if (clip_min.y < 0.0f) { clip_min.y = 0.0f; }
207 if (clip_max.x > (float)fb_width) { clip_max.x = (float)fb_width; }
208 if (clip_max.y > (float)fb_height) { clip_max.y = (float)fb_height; }
209 if (clip_max.x <= clip_min.x || clip_max.y <= clip_min.y)
210 continue;
211
212 SDL_Rect r = { (int)(clip_min.x), (int)(clip_min.y), (int)(clip_max.x - clip_min.x), (int)(clip_max.y - clip_min.y) };
213 SDL_SetRenderClipRect(renderer, &r);
214
215 const float* xy = (const float*)(const void*)((const char*)(vtx_buffer + pcmd->VtxOffset) + offsetof(ImDrawVert, pos));
216 const float* uv = (const float*)(const void*)((const char*)(vtx_buffer + pcmd->VtxOffset) + offsetof(ImDrawVert, uv));
217 const SDL_Color* color = (const SDL_Color*)(const void*)((const char*)(vtx_buffer + pcmd->VtxOffset) + offsetof(ImDrawVert, col)); // SDL 2.0.19+
218
219 // Bind texture, Draw
220 SDL_Texture* tex = (SDL_Texture*)pcmd->GetTexID();
221 SDL_RenderGeometryRaw8BitColor(renderer, bd->ColorBuffer, tex,
222 xy, (int)sizeof(ImDrawVert),
223 color, (int)sizeof(ImDrawVert),
224 uv, (int)sizeof(ImDrawVert),
225 draw_list->VtxBuffer.Size - pcmd->VtxOffset,
226 idx_buffer + pcmd->IdxOffset, pcmd->ElemCount, sizeof(ImDrawIdx));
227 }
228 }
229 }
230 platform_io.Renderer_RenderState = NULL;
231
232 // Restore modified SDL_Renderer state
233 SDL_SetRenderViewport(renderer, old.ViewportEnabled ? &old.Viewport : nullptr);
234 SDL_SetRenderClipRect(renderer, old.ClipEnabled ? &old.ClipRect : nullptr);
235}

238bool ImGui_ImplSDLRenderer3_CreateFontsTexture()
239{
240 ImGuiIO& io = ImGui::GetIO();
241 ImGui_ImplSDLRenderer3_Data* bd = ImGui_ImplSDLRenderer3_GetBackendData();
242
243 // Build texture atlas
244 unsigned char* pixels;
245 int width, height;
246 io.Fonts->GetTexDataAsRGBA32(&pixels, &width, &height); // Load as RGBA 32-bit (75% of the memory is wasted, but default font is so small) because it is more likely to be compatible with user's existing shaders. If your ImTextureId represent a higher-level concept than just a GL texture id, consider calling GetTexDataAsAlpha8() instead to save on GPU memory.
247
248 // Upload texture to graphics system
249 // (Bilinear sampling is required by default. Set 'io.Fonts->Flags |= ImFontAtlasFlags_NoBakedLines' or 'style.AntiAliasedLinesUseTex = false' to allow point/nearest sampling)
250 bd->FontTexture = SDL_CreateTexture(bd->Renderer, SDL_PIXELFORMAT_ABGR8888, SDL_TEXTUREACCESS_STATIC, width, height);
251 if (bd->FontTexture == nullptr)
252 {
253 SDL_Log("error creating texture");
254 return false;
255 }
256 SDL_UpdateTexture(bd->FontTexture, nullptr, pixels, 4 * width);
257 SDL_SetTextureBlendMode(bd->FontTexture, SDL_BLENDMODE_BLEND);
258 SDL_SetTextureScaleMode(bd->FontTexture, SDL_SCALEMODE_LINEAR);
259
260 // Store our identifier
261 io.Fonts->SetTexID((ImTextureID)(intptr_t)bd->FontTexture);
262
263 return true;
264}
