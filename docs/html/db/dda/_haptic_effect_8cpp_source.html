35Atlas::HapticEffectUploadResult Atlas::MonoRumbleHapticEffect::upload(std::shared_ptr<HapticDevice> hapticDevice)
36{
37 HapticEffectUploadResult uploadResult{};
38
39 if (!SDL_HapticRumbleSupported(hapticDevice->getSDLHapticPtr().get())) {
40 const std::string cUnsupportedTraceMsg = std::format(
41 "The haptic device with the name of {} (id: {}) does not support rumble effects!",
42 SDL_HapticName(hapticDevice->getHapticDeviceID()),
43 hapticDevice->getHapticDeviceID()
44 );
45
46 WarnLog(cUnsupportedTraceMsg);
47 // The value of bSuccess is already false, so it doesn't need to be set.
48 return uploadResult;
49 }
50
51 // Tell the caller to ignore the bad effect ID because the rumble effect works differently.
52 // Additionally, this is set here rather than before the check to see if the effect is supported
53 // because the HapticDevice object will ignore the fact that it is not supported!
54 uploadResult.bIgnoreBadEffectID = true;
55
56 if (SDL_HapticRumbleInit(hapticDevice->getSDLHapticPtr().get()) == 0) {
57 uploadResult.bSuccess = true;
58
59 mbIsUploaded = true;
60 }
61
62 return uploadResult;
63}
