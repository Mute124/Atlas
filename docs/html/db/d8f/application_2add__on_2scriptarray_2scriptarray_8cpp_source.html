126// This optional callback is called when the template type is first used by the compiler.
127// It allows the application to validate if the template can be instantiated for the requested
128// subtype at compile time, instead of at runtime. The output argument dontGarbageCollect
129// allow the callback to tell the engine if the template instance type shouldn't be garbage collected,
130// i.e. no asOBJ_GC flag.
131static bool ScriptArrayTemplateCallback(asITypeInfo *ti, bool &dontGarbageCollect)
132{
133 // Make sure the subtype can be instantiated with a default factory/constructor,
134 // otherwise we won't be able to instantiate the elements.
135 int typeId = ti->GetSubTypeId();
136 if( typeId == asTYPEID_VOID )
137 return false;
138 if( (typeId & asTYPEID_MASK_OBJECT) && !(typeId & asTYPEID_OBJHANDLE) )
139 {
140 asITypeInfo *subtype = ti->GetEngine()->GetTypeInfoById(typeId);
141 asQWORD flags = subtype->GetFlags();
142 if( (flags & asOBJ_VALUE) && !(flags & asOBJ_POD) )
143 {
144 // Verify that there is a default constructor
145 bool found = false;
146 for( asUINT n = 0; n < subtype->GetBehaviourCount(); n++ )
147 {
148 asEBehaviours beh;
149 asIScriptFunction *func = subtype->GetBehaviourByIndex(n, &beh);
150 if( beh != asBEHAVE_CONSTRUCT ) continue;
151
152 if( func->GetParamCount() == 0 )
153 {
154 // Found the default constructor
155 found = true;
156 break;
157 }
158 }
159
160 if( !found )
161 {
162 // There is no default constructor
163 // TODO: Should format the message to give the name of the subtype for better understanding
164 ti->GetEngine()->WriteMessage("array", 0, 0, asMSGTYPE_ERROR, "The subtype has no default constructor");
165 return false;
166 }
167 }
168 else if( (flags & asOBJ_REF) )
169 {
170 bool found = false;
171
172 // If value assignment for ref type has been disabled then the array
173 // can be created if the type has a default factory function
174 if( !ti->GetEngine()->GetEngineProperty(asEP_DISALLOW_VALUE_ASSIGN_FOR_REF_TYPE) )
175 {
176 // Verify that there is a default factory
177 for( asUINT n = 0; n < subtype->GetFactoryCount(); n++ )
178 {
179 asIScriptFunction *func = subtype->GetFactoryByIndex(n);
180 if( func->GetParamCount() == 0 )
181 {
182 // Found the default factory
183 found = true;
184 break;
185 }
186 }
187 }
188
189 if( !found )
190 {
191 // No default factory
192 char buffer[1024];
193 snprintf(buffer, sizeof(buffer), "The subtype '%s' has no default factory", subtype ? subtype->GetEngine()->GetTypeDeclaration(subtype->GetTypeId()) : "UNKNOWN");
194 ti->GetEngine()->WriteMessage("array", 0, 0, asMSGTYPE_ERROR, buffer);
195 return false;
196 }
197 }
198
199 // If the object type is not garbage collected then the array also doesn't need to be
200 if( !(flags & asOBJ_GC) )
201 dontGarbageCollect = true;
202 }
203 else if( !(typeId & asTYPEID_OBJHANDLE) )
204 {
205 // Arrays with primitives cannot form circular references,
206 // thus there is no need to garbage collect them
207 dontGarbageCollect = true;
208 }
209 else
210 {
211 assert( typeId & asTYPEID_OBJHANDLE );
212
213 // It is not necessary to set the array as garbage collected for all handle types.
214 // If it is possible to determine that the handle cannot refer to an object type
215 // that can potentially form a circular reference with the array then it is not
216 // necessary to make the array garbage collected.
217 asITypeInfo *subtype = ti->GetEngine()->GetTypeInfoById(typeId);
218 asQWORD flags = subtype->GetFlags();
219 if( !(flags & asOBJ_GC) )
220 {
221 if( (flags & asOBJ_SCRIPT_OBJECT) )
222 {
223 // Even if a script class is by itself not garbage collected, it is possible
224 // that classes that derive from it may be, so it is not possible to know
225 // that no circular reference can occur.
226 if( (flags & asOBJ_NOINHERIT) )
227 {
228 // A script class declared as final cannot be inherited from, thus
229 // we can be certain that the object cannot be garbage collected.
230 dontGarbageCollect = true;
231 }
232 }
233 else
234 {
235 // For application registered classes we assume the application knows
236 // what it is doing and don't mark the array as garbage collected unless
237 // the type is also garbage collected.
238 dontGarbageCollect = true;
239 }
240 }
241 }
242
243 // The type is ok
244 return true;
245}

374CScriptArray::CScriptArray(asITypeInfo *ti, void *buf)
375{
376 // The object type should be the template instance of the array
377 assert( ti && string(ti->GetName()) == "array" );
378
379 refCount = 1;
380 gcFlag = false;
381 objType = ti;
382 objType->AddRef();
383 buffer = 0;
384
385 Precache();
386
387 asIScriptEngine *engine = ti->GetEngine();
388
389 // Determine element size
390 if( subTypeId & asTYPEID_MASK_OBJECT )
391 elementSize = sizeof(asPWORD);
392 else
393 elementSize = engine->GetSizeOfPrimitiveType(subTypeId);
394
395 // Determine the initial size from the buffer
396 asUINT length = *(asUINT*)buf;
397
398 // Make sure the array size isn't too large for us to handle
399 if( !CheckMaxSize(length) )
400 {
401 // Don't continue with the initialization
402 return;
403 }
404
405 // Copy the values of the array elements from the buffer
406 if( (ti->GetSubTypeId() & asTYPEID_MASK_OBJECT) == 0 )
407 {
408 CreateBuffer(&buffer, length);
409
410 // Copy the values of the primitive type into the internal buffer
411 if( length > 0 )
412 memcpy(At(0), (((asUINT*)buf)+1), length * elementSize);
413 }
414 else if( ti->GetSubTypeId() & asTYPEID_OBJHANDLE )
415 {
416 CreateBuffer(&buffer, length);
417
418 // Copy the handles into the internal buffer
419 if( length > 0 )
420 memcpy(At(0), (((asUINT*)buf)+1), length * elementSize);
421
422 // With object handles it is safe to clear the memory in the received buffer
423 // instead of increasing the ref count. It will save time both by avoiding the
424 // call the increase ref, and also relieve the engine from having to release
425 // its references too
426 memset((((asUINT*)buf)+1), 0, length * elementSize);
427 }
428 else if( ti->GetSubType()->GetFlags() & asOBJ_REF )
429 {
430 // Only allocate the buffer, but not the objects
431 subTypeId |= asTYPEID_OBJHANDLE;
432 CreateBuffer(&buffer, length);
433 subTypeId &= ~asTYPEID_OBJHANDLE;
434
435 // Copy the handles into the internal buffer
436 if( length > 0 )
437 memcpy(buffer->data, (((asUINT*)buf)+1), length * elementSize);
438
439 // For ref types we can do the same as for handles, as they are
440 // implicitly stored as handles.
441 memset((((asUINT*)buf)+1), 0, length * elementSize);
442 }
443 else
444 {
445 // TODO: Optimize by calling the copy constructor of the object instead of
446 // constructing with the default constructor and then assigning the value
447 // TODO: With C++11 ideally we should be calling the move constructor, instead
448 // of the copy constructor as the engine will just discard the objects in the
449 // buffer afterwards.
450 CreateBuffer(&buffer, length);
451
452 // For value types we need to call the opAssign for each individual object
453 for( asUINT n = 0; n < length; n++ )
454 {
455 void *obj = At(n);
456 asBYTE *srcObj = (asBYTE*)buf;
457 srcObj += 4 + n*ti->GetSubType()->GetSize();
458 engine->AssignScriptObject(obj, srcObj, ti->GetSubType());
459 }
460 }
461
462 // Notify the GC of the successful creation
463 if( objType->GetFlags() & asOBJ_GC )
464 objType->GetEngine()->NotifyGarbageCollectorOfNewObject(this, objType);
465}

559{
560 // At() will take care of the out-of-bounds checking, though
561 // if called from the application then nothing will be done

684void CScriptArray::RemoveRange(asUINT start, asUINT count)
685{
686 if (count == 0)
687 return;
688
689 if( buffer == 0 || start > buffer->numElements )
690 {
691 // If this is called from a script we raise a script exception
692 asIScriptContext *ctx = asGetActiveContext();
693 if (ctx)
694 ctx->SetException("Index out of bounds");
695 return;
696 }
697
698 // Cap count to the end of the array
699 if (start + count > buffer->numElements)
700 count = buffer->numElements - start;
701
702 // Destroy the elements that are being removed
703 Destruct(buffer, start, start + count);
704
705 // Compact the elements
706 // As objects in arrays of objects are not stored inline, it is safe to use memmove here
707 // since we're just copying the pointers to objects and not the actual objects.
708 memmove(buffer->data + start*elementSize, buffer->data + (start + count)*elementSize, (buffer->numElements - start - count)*elementSize);
709 buffer->numElements -= count;
710}

712// Internal
713void CScriptArray::Resize(int delta, asUINT at)
714{
715 if( delta < 0 )
716 {
717 if( -delta > (int)buffer->numElements )
718 delta = -(int)buffer->numElements;
719 if( at > buffer->numElements + delta )
720 at = buffer->numElements + delta;
721 }
722 else if( delta > 0 )
723 {
724 // Make sure the array size isn't too large for us to handle
725 if( delta > 0 && !CheckMaxSize(buffer->numElements + delta) )
726 return;
727
728 if( at > buffer->numElements )
729 at = buffer->numElements;
730 }
731
732 if( delta == 0 ) return;
733
734 if( buffer->maxElements < buffer->numElements + delta )
735 {
736 // Allocate memory for the buffer
737 SArrayBuffer *newBuffer = reinterpret_cast<SArrayBuffer*>(userAlloc(sizeof(SArrayBuffer)-1 + elementSize*(buffer->numElements + delta)));
738 if( newBuffer )
739 {
740 newBuffer->numElements = buffer->numElements + delta;
741 newBuffer->maxElements = newBuffer->numElements;
742 }
743 else
744 {
745 // Out of memory
746 asIScriptContext *ctx = asGetActiveContext();
747 if( ctx )
748 ctx->SetException("Out of memory");
749 return;
750 }
751
752 // As objects in arrays of objects are not stored inline, it is safe to use memcpy here
753 // since we're just copying the pointers to objects and not the actual objects.
754 memcpy(newBuffer->data, buffer->data, at*elementSize);
755 if( at < buffer->numElements )
756 memcpy(newBuffer->data + (at+delta)*elementSize, buffer->data + at*elementSize, (buffer->numElements-at)*elementSize);
757
758 // Initialize the new elements with default values
759 Construct(newBuffer, at, at+delta);
760
761 // Release the old buffer
762 userFree(buffer);
763
764 buffer = newBuffer;
765 }
766 else if( delta < 0 )
767 {
768 Destruct(buffer, at, at-delta);
769 // As objects in arrays of objects are not stored inline, it is safe to use memmove here
770 // since we're just copying the pointers to objects and not the actual objects.
771 memmove(buffer->data + at*elementSize, buffer->data + (at-delta)*elementSize, (buffer->numElements - (at-delta))*elementSize);
772 buffer->numElements += delta;
773 }
774 else
775 {
776 // As objects in arrays of objects are not stored inline, it is safe to use memmove here
777 // since we're just copying the pointers to objects and not the actual objects.
778 memmove(buffer->data + (at+delta)*elementSize, buffer->data + at*elementSize, (buffer->numElements - at)*elementSize);
779 Construct(buffer, at, at+delta);
780 buffer->numElements += delta;
781 }

840void CScriptArray::InsertAt(asUINT index, const CScriptArray &arr)
841{
842 if (index > buffer->numElements)
843 {
844 asIScriptContext *ctx = asGetActiveContext();
845 if (ctx)
846 ctx->SetException("Index out of bounds");
847 return;
848 }
849
850 if (objType != arr.objType)
851 {
852 // This shouldn't really be possible to happen when
853 // called from a script, but let's check for it anyway
854 asIScriptContext *ctx = asGetActiveContext();
855 if (ctx)
856 ctx->SetException("Mismatching array types");
857 return;
858 }
859
860 asUINT elements = arr.GetSize();
861 Resize(elements, index);
862 if (&arr != this)
863 {
864 for (asUINT n = 0; n < arr.GetSize(); n++)
865 {
866 // This const cast is allowed, since we know the
867 // value will only be used to make a copy of it
868 void *value = const_cast<void*>(arr.At(n));
869 SetValue(index + n, value);
870 }
871 }
872 else
873 {
874 // The array that is being inserted is the same as this one.
875 // So we should iterate over the elements before the index,
876 // and then the elements after
877 for (asUINT n = 0; n < index; n++)
878 {
879 // This const cast is allowed, since we know the
880 // value will only be used to make a copy of it
881 void *value = const_cast<void*>(arr.At(n));
882 SetValue(index + n, value);
883 }
884
885 for (asUINT n = index + elements, m = 0; n < arr.GetSize(); n++, m++)
886 {
887 // This const cast is allowed, since we know the
888 // value will only be used to make a copy of it
889 void *value = const_cast<void*>(arr.At(n));
890 SetValue(index + index + m, value);
891 }
892 }

978void CScriptArray::Construct(SArrayBuffer *buf, asUINT start, asUINT end)
979{
980 if( (subTypeId & asTYPEID_MASK_OBJECT) && !(subTypeId & asTYPEID_OBJHANDLE) )
981 {
982 // Create an object using the default constructor/factory for each element
983 void **max = (void**)(buf->data + end * sizeof(void*));
984 void **d = (void**)(buf->data + start * sizeof(void*));
985
986 asIScriptEngine *engine = objType->GetEngine();
987 asITypeInfo *subType = objType->GetSubType();
988
989 for( ; d < max; d++ )
990 {
991 *d = (void*)engine->CreateScriptObject(subType);
992 if( *d == 0 )
993 {
994 // Set the remaining entries to null so the destructor
995 // won't attempt to destroy invalid objects later
996 memset(d, 0, sizeof(void*)*(max-d));
997
998 // There is no need to set an exception on the context,
999 // as CreateScriptObject has already done that
1000 return;
1001 }
1002 }
1003 }
1004 else
1005 {
1006 // Set all elements to zero whether they are handles or primitives
1007 void *d = (void*)(buf->data + start * elementSize);
1008 memset(d, 0, (end-start)*elementSize);
1009 }

1085{
1086 if( objType != other.objType )
1087 return false;
1088
1089 if( GetSize() != other.GetSize() )
1090 return false;
1091
1092 asIScriptContext *cmpContext = 0;
1093 bool isNested = false;
1094
1095 if( subTypeId & ~asTYPEID_MASK_SEQNBR )
1096 {
1097 // Try to reuse the active context
1098 cmpContext = asGetActiveContext();

1100 {

1102 isNested = true;
1103 else
1104 cmpContext = 0;
1105 }
1106 if( cmpContext == 0 )
1107 {
1108 // TODO: Ideally this context would be retrieved from a pool, so we don't have to
1109 // create a new one everytime. We could keep a context with the array object
1110 // but that would consume a lot of resources as each context is quite heavy.
1111 cmpContext = objType->GetEngine()->CreateContext();
1112 }
1113 }
1114
1115 // Check if all elements are equal
1116 bool isEqual = true;
1117 SArrayCache *cache = reinterpret_cast<SArrayCache*>(objType->GetUserData(ARRAY_CACHE));
1118 for( asUINT n = 0; n < GetSize(); n++ )
1119 if( !Equals(At(n), other.At(n), cmpContext, cache) )
1120 {
1121 isEqual = false;
1122 break;
1123 }
1124
1125 if( cmpContext )
1126 {
1127 if( isNested )
1128 {
1129 asEContextState state = cmpContext->GetState();
1130 cmpContext->PopState();
1131 if( state == asEXECUTION_ABORTED )
1132 cmpContext->Abort();

1430// internal
1431void CScriptArray::Sort(asUINT startAt, asUINT count, bool asc)
1432{
1433 // Subtype isn't primitive and doesn't have opCmp
1434 SArrayCache *cache = reinterpret_cast<SArrayCache*>(objType->GetUserData(ARRAY_CACHE));
1435 if( subTypeId & ~asTYPEID_MASK_SEQNBR )
1436 {
1437 if( !cache || cache->cmpFunc == 0 )
1438 {
1439 asIScriptContext *ctx = asGetActiveContext();
1440 asITypeInfo* subType = objType->GetEngine()->GetTypeInfoById(subTypeId);
1441
1442 // Throw an exception
1443 if( ctx )
1444 {
1445 char tmp[512];
1446
1447 if( cache && cache->cmpFuncReturnCode == asMULTIPLE_FUNCTIONS )
1448#if defined(_MSC_VER) && _MSC_VER >= 1500 && !defined(__S3E__)
1449 sprintf_s(tmp, 512, "Type '%s' has multiple matching opCmp methods", subType->GetName());
1450#else
1451 snprintf(tmp, sizeof(tmp), "Type '%s' has multiple matching opCmp methods", subType->GetName());
1452#endif
1453 else
1454#if defined(_MSC_VER) && _MSC_VER >= 1500 && !defined(__S3E__)
1455 sprintf_s(tmp, 512, "Type '%s' does not have a matching opCmp method", subType->GetName());
1456#else
1457 snprintf(tmp, sizeof(tmp), "Type '%s' does not have a matching opCmp method", subType->GetName());
1458#endif
1459
1460 ctx->SetException(tmp);
1461 }
1462
1463 return;
1464 }
1465 }
1466
1467 // No need to sort
1468 if( count < 2 )
1469 {
1470 return;
1471 }
1472
1473 int start = startAt;
1474 int end = startAt + count;
1475
1476 // Check if we could access invalid item while sorting
1477 if( start >= (int)buffer->numElements || end > (int)buffer->numElements )
1478 {
1479 asIScriptContext *ctx = asGetActiveContext();
1480
1481 // Throw an exception
1482 if( ctx )
1483 {
1484 ctx->SetException("Index out of bounds");
1485 }
1486
1487 return;
1488 }
1489
1490 if( subTypeId & ~asTYPEID_MASK_SEQNBR )
1491 {
1492 asIScriptContext *cmpContext = 0;
1493 bool isNested = false;
1494
1495 // Try to reuse the active context
1496 cmpContext = asGetActiveContext();
1497 if( cmpContext )
1498 {
1499 if( cmpContext->GetEngine() == objType->GetEngine() && cmpContext->PushState() >= 0 )
1500 isNested = true;
1501 else
1502 cmpContext = 0;
1503 }
1504 if( cmpContext == 0 )
1505 cmpContext = objType->GetEngine()->RequestContext();
1506
1507 // Do the sorting
1508 struct {
1509 bool asc;
1510 asIScriptContext *cmpContext;
1511 asIScriptFunction *cmpFunc;
1512 bool operator()(void *a, void *b) const
1513 {
1514 if( !asc )
1515 {
1516 // Swap items
1517 void *TEMP = a;
1518 a = b;
1519 b = TEMP;
1520 }
1521
1522 int r = 0;
1523
1524 // Allow sort to work even if the array contains null handles
1525 if( a == 0 ) return true;
1526 if( b == 0 ) return false;
1527
1528 // Execute object opCmp
1529 if( cmpFunc )
1530 {
1531 // TODO: Add proper error handling
1532 r = cmpContext->Prepare(cmpFunc); assert(r >= 0);
1533 r = cmpContext->SetObject(a); assert(r >= 0);
1534 r = cmpContext->SetArgObject(0, b); assert(r >= 0);
1535 r = cmpContext->Execute();
1536
1537 if( r == asEXECUTION_FINISHED )
1538 {
1539 return (int)cmpContext->GetReturnDWord() < 0;
1540 }
1541 }
1542
1543 return false;
1544 }
1545 } customLess = {asc, cmpContext, cache ? cache->cmpFunc : 0};
1546 std::sort((void**)GetArrayItemPointer(start), (void**)GetArrayItemPointer(end), customLess);
1547
1548 // Clean up
1549 if( cmpContext )
1550 {
1551 if( isNested )
1552 {
1553 asEContextState state = cmpContext->GetState();
1554 cmpContext->PopState();
1555 if( state == asEXECUTION_ABORTED )
1556 cmpContext->Abort();
1557 }
1558 else
1559 objType->GetEngine()->ReturnContext(cmpContext);
1560 }
1561 }
1562 else
1563 {
1564 // TODO: Use std::sort for primitive types too
1565
1566 // Insertion sort
1567 asBYTE tmp[16];
1568 for( int i = start + 1; i < end; i++ )
1569 {
1570 Copy(tmp, GetArrayItemPointer(i));
1571
1572 int j = i - 1;
1573
1574 while( j >= start && Less(GetDataPointer(tmp), At(j), asc) )
1575 {
1576 Copy(GetArrayItemPointer(j + 1), GetArrayItemPointer(j));
1577 j--;
1578 }
1579
1580 Copy(GetArrayItemPointer(j + 1), tmp);
1581 }
1582 }
1583}

1585// Sort with script callback for comparing elements
1586void CScriptArray::Sort(asIScriptFunction *func, asUINT startAt, asUINT count)
1587{
1588 // No need to sort
1589 if (count < 2)
1590 return;
1591
1592 // Check if we could access invalid item while sorting
1593 asUINT start = startAt;
1594 asUINT end = asQWORD(startAt) + asQWORD(count) >= (asQWORD(1)<<32) ? 0xFFFFFFFF : startAt + count;
1595 if (end > buffer->numElements)
1596 end = buffer->numElements;
1597
1598 if (start >= buffer->numElements)
1599 {
1600 asIScriptContext *ctx = asGetActiveContext();
1601
1602 // Throw an exception
1603 if (ctx)
1604 ctx->SetException("Index out of bounds");
1605
1606 return;
1607 }
1608
1609 asIScriptContext *cmpContext = 0;
1610 bool isNested = false;
1611
1612 // Try to reuse the active context
1613 cmpContext = asGetActiveContext();
1614 if (cmpContext)
1615 {
1616 if (cmpContext->GetEngine() == objType->GetEngine() && cmpContext->PushState() >= 0)
1617 isNested = true;
1618 else
1619 cmpContext = 0;
1620 }
1621 if (cmpContext == 0)
1622 cmpContext = objType->GetEngine()->RequestContext();
1623
1624 // TODO: Security issue: If the array is accessed from the callback while the sort is going on the result may be unpredictable
1625 // For example, the callback resizes the array in the middle of the sort
1626 // Possible solution: set a lock flag on the array, and prohibit modifications while the lock flag is set
1627
1628 // Bubble sort
1629 // TODO: optimize: Use an efficient sort algorithm
1630 for (asUINT i = start; i+1 < end; i++)
1631 {
1632 asUINT best = i;
1633 for (asUINT j = i + 1; j < end; j++)
1634 {
1635 cmpContext->Prepare(func);
1636 cmpContext->SetArgAddress(0, At(j));
1637 cmpContext->SetArgAddress(1, At(best));
1638 int r = cmpContext->Execute();
1639 if (r != asEXECUTION_FINISHED)
1640 break;
1641 if (*(bool*)(cmpContext->GetAddressOfReturnValue()))
1642 best = j;
1643 }
1644
1645 // With Swap we guarantee that the array always sees all references
1646 // if the GC calls the EnumReferences in the middle of the sorting
1647 if( best != i )
1648 Swap(GetArrayItemPointer(i), GetArrayItemPointer(best));
1649 }
1650
1651 if (cmpContext)
1652 {
1653 if (isNested)
1654 {
1655 asEContextState state = cmpContext->GetState();
1656 cmpContext->PopState();
1657 if (state == asEXECUTION_ABORTED)
1658 cmpContext->Abort();
1659 }
1660 else
1661 objType->GetEngine()->ReturnContext(cmpContext);
1662 }
1663}

1665// internal
1666void CScriptArray::CopyBuffer(SArrayBuffer *dst, SArrayBuffer *src)
1667{
1668 asIScriptEngine *engine = objType->GetEngine();
1669 if( subTypeId & asTYPEID_OBJHANDLE )
1670 {
1671 // Copy the references and increase the reference counters
1672 if( dst->numElements > 0 && src->numElements > 0 )
1673 {
1674 int count = dst->numElements > src->numElements ? src->numElements : dst->numElements;
1675
1676 void **max = (void**)(dst->data + count * sizeof(void*));
1677 void **d = (void**)dst->data;
1678 void **s = (void**)src->data;
1679
1680 for( ; d < max; d++, s++ )
1681 {
1682 void *tmp = *d;
1683 *d = *s;
1684 if( *d )
1685 engine->AddRefScriptObject(*d, objType->GetSubType());
1686 // Release the old ref after incrementing the new to avoid problem incase it is the same ref
1687 if( tmp )
1688 engine->ReleaseScriptObject(tmp, objType->GetSubType());
1689 }
1690 }
1691 }
1692 else
1693 {
1694 if( dst->numElements > 0 && src->numElements > 0 )
1695 {
1696 int count = dst->numElements > src->numElements ? src->numElements : dst->numElements;
1697 if( subTypeId & asTYPEID_MASK_OBJECT )
1698 {
1699 // Call the assignment operator on all of the objects
1700 void **max = (void**)(dst->data + count * sizeof(void*));
1701 void **d = (void**)dst->data;
1702 void **s = (void**)src->data;
1703
1704 asITypeInfo *subType = objType->GetSubType();
1705 if (subType->GetFlags() & asOBJ_ASHANDLE)
1706 {
1707 // For objects that should work as handles we must use the opHndlAssign method
1708 // TODO: Must support alternative syntaxes as well
1709 // TODO: Move the lookup of the opHndlAssign method to Precache() so it is only done once
1710 string decl = string(subType->GetName()) + "& opHndlAssign(const " + string(subType->GetName()) + "&in)";
1711 asIScriptFunction *func = subType->GetMethodByDecl(decl.c_str());
1712 if (func)
1713 {
1714 // TODO: Reuse active context if existing
1715 asIScriptContext* ctx = engine->RequestContext();
1716 for (; d < max; d++, s++)
1717 {
1718 ctx->Prepare(func);
1719 ctx->SetObject(*d);
1720 ctx->SetArgAddress(0, *s);
1721 // TODO: Handle errors
1722 ctx->Execute();
1723 }
1724 engine->ReturnContext(ctx);
1725 }
1726 else
1727 {
1728 // opHndlAssign doesn't exist, so try ordinary value assign instead
1729 for (; d < max; d++, s++)
1730 engine->AssignScriptObject(*d, *s, subType);
1731 }
1732 }
1733 else
1734 for( ; d < max; d++, s++ )
1735 engine->AssignScriptObject(*d, *s, subType);
1736 }
1737 else
1738 {
1739 // Primitives are copied byte for byte
1740 memcpy(dst->data, src->data, count*elementSize);
1741 }
1742 }
1743 }

1746// internal
1747// Precache some info
1748void CScriptArray::Precache()
1749{
1750 subTypeId = objType->GetSubTypeId();
1751
1752 // Check if it is an array of objects. Only for these do we need to cache anything
1753 // Type ids for primitives and enums only has the sequence number part
1754 if( !(subTypeId & ~asTYPEID_MASK_SEQNBR) )
1755 return;
1756
1757 // The opCmp and opEquals methods are cached because the searching for the
1758 // methods is quite time consuming if a lot of array objects are created.
1759
1760 // First check if a cache already exists for this array type
1761 SArrayCache *cache = reinterpret_cast<SArrayCache*>(objType->GetUserData(ARRAY_CACHE));
1762 if( cache ) return;
1763
1764 // We need to make sure the cache is created only once, even
1765 // if multiple threads reach the same point at the same time
1766 asAcquireExclusiveLock();
1767
1768 // Now that we got the lock, we need to check again to make sure the
1769 // cache wasn't created while we were waiting for the lock
1770 cache = reinterpret_cast<SArrayCache*>(objType->GetUserData(ARRAY_CACHE));
1771 if( cache )
1772 {
1773 asReleaseExclusiveLock();
1774 return;
1775 }
1776
1777 // Create the cache
1778 cache = reinterpret_cast<SArrayCache*>(userAlloc(sizeof(SArrayCache)));
1779 if( !cache )
1780 {
1781 asIScriptContext *ctx = asGetActiveContext();
1782 if( ctx )
1783 ctx->SetException("Out of memory");
1784 asReleaseExclusiveLock();
1785 return;
1786 }
1787 memset(cache, 0, sizeof(SArrayCache));
1788
1789 // If the sub type is a handle to const, then the methods must be const too
1790 bool mustBeConst = (subTypeId & asTYPEID_HANDLETOCONST) ? true : false;
1791
1792 asITypeInfo *subType = objType->GetEngine()->GetTypeInfoById(subTypeId);
1793 if( subType )
1794 {
1795 for( asUINT i = 0; i < subType->GetMethodCount(); i++ )
1796 {
1797 asIScriptFunction *func = subType->GetMethodByIndex(i);
1798
1799 if( func->GetParamCount() == 1 && (!mustBeConst || func->IsReadOnly()) )
1800 {
1801 asDWORD flags = 0;
1802 int returnTypeId = func->GetReturnTypeId(&flags);
1803
1804 // The method must not return a reference
1805 if( flags != asTM_NONE )
1806 continue;
1807
1808 // opCmp returns an int and opEquals returns a bool
1809 bool isCmp = false, isEq = false;
1810 if( returnTypeId == asTYPEID_INT32 && strcmp(func->GetName(), "opCmp") == 0 )
1811 isCmp = true;
1812 if( returnTypeId == asTYPEID_BOOL && strcmp(func->GetName(), "opEquals") == 0 )
1813 isEq = true;
1814
1815 if( !isCmp && !isEq )
1816 continue;
1817
1818 // The parameter must either be a reference to the subtype or a handle to the subtype
1819 int paramTypeId;
1820 func->GetParam(0, &paramTypeId, &flags);
1821
1822 if( (paramTypeId & ~(asTYPEID_OBJHANDLE|asTYPEID_HANDLETOCONST)) != (subTypeId & ~(asTYPEID_OBJHANDLE|asTYPEID_HANDLETOCONST)) )
1823 continue;
1824
1825 if( (flags & asTM_INREF) )
1826 {
1827 if( (paramTypeId & asTYPEID_OBJHANDLE) || (mustBeConst && !(flags & asTM_CONST)) )
1828 continue;
1829 }
1830 else if( paramTypeId & asTYPEID_OBJHANDLE )
1831 {
1832 if( mustBeConst && !(paramTypeId & asTYPEID_HANDLETOCONST) )
1833 continue;
1834 }
1835 else
1836 continue;
1837
1838 if( isCmp )
1839 {
1840 if( cache->cmpFunc || cache->cmpFuncReturnCode )
1841 {
1842 cache->cmpFunc = 0;
1843 cache->cmpFuncReturnCode = asMULTIPLE_FUNCTIONS;
1844 }
1845 else
1846 cache->cmpFunc = func;
1847 }
1848 else if( isEq )
1849 {
1850 if( cache->eqFunc || cache->eqFuncReturnCode )
1851 {
1852 cache->eqFunc = 0;
1853 cache->eqFuncReturnCode = asMULTIPLE_FUNCTIONS;
1854 }
1855 else
1856 cache->eqFunc = func;
1857 }
1858 }
1859 }
1860 }
1861
1862 if( cache->eqFunc == 0 && cache->eqFuncReturnCode == 0 )
1863 cache->eqFuncReturnCode = asNO_FUNCTION;
1864 if( cache->cmpFunc == 0 && cache->cmpFuncReturnCode == 0 )
1865 cache->cmpFuncReturnCode = asNO_FUNCTION;
1866
1867 // Set the user data only at the end so others that retrieve it will know it is complete
1868 objType->SetUserData(cache, ARRAY_CACHE);
1869
1870 asReleaseExclusiveLock();
1871}

1873// GC behaviour
1874void CScriptArray::EnumReferences(asIScriptEngine *engine)
1875{
1876 // TODO: If garbage collection can be done from a separate thread, then this method must be
1877 // protected so that it doesn't get lost during the iteration if the array is modified
1878
1879 // If the array is holding handles, then we need to notify the GC of them
1880 if( subTypeId & asTYPEID_MASK_OBJECT )
1881 {
1882 void **d = (void**)buffer->data;
1883
1884 asITypeInfo *subType = engine->GetTypeInfoById(subTypeId);
1885 if ((subType->GetFlags() & asOBJ_REF))
1886 {
1887 // For reference types we need to notify the GC of each instance
1888 for (asUINT n = 0; n < buffer->numElements; n++)
1889 {
1890 if (d[n])
1891 engine->GCEnumCallback(d[n]);
1892 }
1893 }
1894 else if ((subType->GetFlags() & asOBJ_VALUE) && (subType->GetFlags() & asOBJ_GC))
1895 {
1896 // For value types we need to forward the enum callback
1897 // to the object so it can decide what to do
1898 for (asUINT n = 0; n < buffer->numElements; n++)
1899 {
1900 if (d[n])
1901 engine->ForwardGCEnumReferences(d[n], subType);
1902 }
1903 }
1904 }
