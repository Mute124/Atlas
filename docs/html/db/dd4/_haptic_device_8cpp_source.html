86void Atlas::HapticDevice::open()
87{
88 // Make sure that the haptic device is not already open because that can cause issues.
89 if (mbIsOpen == true) {
90 const std::string cAlreadyOpenMsg = std::format(
91 "Haptic device with the name of {} (id: {}) is already open. This haptic device will be closed and opened again.",
92 SDL_HapticName(getHapticDeviceID()),
93 getHapticDeviceID()
94 );
95
96 //"Trying to open a haptic device that is already open!"
97 WarnLog(cAlreadyOpenMsg);
98
99 close();
100 }
101
102 // TODO: This may cause an issue because this may or may not open a haptic device because of the comparator used in the if statement.
103 else if (HasReachedMaxHapticDevices()) {
104 const std::string cMaxDevicesReachedMsg = std::format(
105 "Max haptic devices reached. The haptic device with the name of {} (id: {}) will not be opened.",
106 SDL_HapticName(getHapticDeviceID()),
107 getHapticDeviceID()
108 );
109
110 //"Max haptic devices reached. This haptic device will not be opened."
111 WarnLog(cMaxDevicesReachedMsg);
112 //throw std::runtime_error("Max haptic devices reached.");
113 }
114
115
116 mSDLHapticPtr = std::shared_ptr<SDL_Haptic>(SDL_HapticOpen(getHapticDeviceID()));
117
118 if (!SDL_HapticOpened(getHapticDeviceID())) {
119
120 const std::string cSdlErrorMsg = SDL_GetError();
121
122 const std::string cFailureMsg = std::format(
123 "Failed to open haptic device with the name of {} (id: {}) because of the following error: {}.",
124 SDL_HapticName(getHapticDeviceID()),
125 getHapticDeviceID(),
126 cSdlErrorMsg
127 );
128
129 ErrorLog(cFailureMsg);
130 //throw std::runtime_error("Failed to open haptic device.");
131 }
132
133 /*if (mSDLHapticPtr == nullptr) {
134 throw std::runtime_error("Failed to open haptic device.");
135 }*/
136
137 const std::string cSuccessfulOpenMsg = std::format(
138 "Successfully opened haptic device with the name of {} (id: {}).",
139 SDL_HapticName(getHapticDeviceID()),
140 getHapticDeviceID()
141 );
142
143 TraceLog(cSuccessfulOpenMsg);
144
145 mbIsOpen = true;
146 sCurrentHapticDevicesCount++;
147
148
150}

152void Atlas::HapticDevice::close()
153{
154 if (mbIsOpen == true) {
155 const std::string cCloseAttemptTraceMsg = std::format(
156 "Closing haptic device with the name of {} (id: {}).",
157 SDL_HapticName(getHapticDeviceID()),
158 getHapticDeviceID()
159 );
160
161 TraceLog(cCloseAttemptTraceMsg);
162
163 SDL_HapticClose(mSDLHapticPtr.get());
164 mbIsOpen = false;
165 sCurrentHapticDevicesCount--;
166 }
