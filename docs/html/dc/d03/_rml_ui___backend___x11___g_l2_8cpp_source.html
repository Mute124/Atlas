53// Attach the OpenGL context to the window.
54static bool AttachToNative(GLXContext& out_gl_context, Display* display, Window window, XVisualInfo* visual_info)
55{
56 GLXContext gl_context = glXCreateContext(display, visual_info, nullptr, GL_TRUE);
57 if (!gl_context)
58 return false;
59
60 if (!glXMakeCurrent(display, window, gl_context))
61 return false;
62
63 if (!glXIsDirect(display, gl_context))
64 Rml::Log::Message(Rml::Log::LT_INFO, "OpenGL context does not support direct rendering; performance is likely to be poor.");
65
66 out_gl_context = gl_context;
67
68 Window root_window;
69 int x, y;
70 unsigned int width, height;
71 unsigned int border_width, depth;
72 XGetGeometry(display, window, &root_window, &x, &y, &width, &height, &border_width, &depth);
73
74 return true;
75}
76
77// Shutdown the OpenGL context.
78static void DetachFromNative(GLXContext gl_context, Display* display)
79{
80 glXMakeCurrent(display, 0L, nullptr);
81 glXDestroyContext(display, gl_context);
82}

103bool Backend::Initialize(const char* window_name, int width, int height, bool allow_resize)
104{
105 RMLUI_ASSERT(!data);
106
107 Display* display = XOpenDisplay(0);
108 if (!display)
109 return false;
110
111 int screen = XDefaultScreen(display);
112
113 // Fetch an appropriate 32-bit visual interface.
114 int attribute_list[] = {GLX_RGBA, GLX_DOUBLEBUFFER, GLX_RED_SIZE, 8, GLX_GREEN_SIZE, 8, GLX_BLUE_SIZE, 8, GLX_DEPTH_SIZE, 24, GLX_STENCIL_SIZE, 8,
115 0L};
116
117 XVisualInfo* visual_info = glXChooseVisual(display, screen, attribute_list);
118 if (!visual_info)
119 return false;
120
121 // Build up our window attributes.
122 XSetWindowAttributes window_attributes;
123 window_attributes.colormap = XCreateColormap(display, RootWindow(display, visual_info->screen), visual_info->visual, AllocNone);
124 window_attributes.border_pixel = 0;
125 window_attributes.event_mask = ExposureMask | KeyPressMask | ButtonPressMask | StructureNotifyMask;
126
127 // Create the window.
128 Window window = XCreateWindow(display, RootWindow(display, visual_info->screen), 0, 0, width, height, 0, visual_info->depth, InputOutput,
129 visual_info->visual, CWBorderPixel | CWColormap | CWEventMask, &window_attributes);
130
131 // Handle delete events in windowed mode.
132 Atom delete_atom = XInternAtom(display, "WM_DELETE_WINDOW", True);
133 XSetWMProtocols(display, window, &delete_atom, 1);
134
135 // Capture the events we're interested in.
136 XSelectInput(display, window,
137 KeyPressMask | KeyReleaseMask | ButtonPressMask | ButtonReleaseMask | LeaveWindowMask | PointerMotionMask | StructureNotifyMask);
138
139 if (!allow_resize)
140 {
141 // Force the window to remain at the fixed size by asking the window manager nicely, it may choose to ignore us
142 XSizeHints* win_size_hints = XAllocSizeHints(); // Allocate a size hint structure
143 if (!win_size_hints)
144 {
145 Rml::Log::Message(Rml::Log::LT_ERROR, "XAllocSizeHints - out of memory");
146 }
147 else
148 {
149 // Initialize the structure and specify which hints will be providing
150 win_size_hints->flags = PSize | PMinSize | PMaxSize;
151
152 // Set the sizes we want the window manager to use
153 win_size_hints->base_width = width;
154 win_size_hints->base_height = height;
155 win_size_hints->min_width = width;
156 win_size_hints->min_height = height;
157 win_size_hints->max_width = width;
158 win_size_hints->max_height = height;
159
160 // Pass the size hints to the window manager.
161 XSetWMNormalHints(display, window, win_size_hints);
162
163 // Free the size buffer
164 XFree(win_size_hints);
165 }
166 }
167
168 // Set the window title and show the window.
169 XSetStandardProperties(display, window, window_name, "", 0L, nullptr, 0, nullptr);
170 XMapRaised(display, window);
171
172 GLXContext gl_context = {};
173 if (!AttachToNative(gl_context, display, window, visual_info))
174 return false;
175
176 data = Rml::MakeUnique<BackendData>(display);
177
178 data->display = display;
179 data->window = window;
180 data->gl_context = gl_context;
181
182 data->system_interface.SetWindow(window);
183 data->render_interface.SetViewport(width, height);
184
185 return true;
186}

210bool Backend::ProcessEvents(Rml::Context* context, KeyDownCallback key_down_callback, bool power_save)
211{
212 RMLUI_ASSERT(data && context);
213
214 Display* display = data->display;
215 bool result = data->running;
216 data->running = true;
217
218 if (power_save && XPending(display) == 0)
219 {
220 int display_fd = ConnectionNumber(display);
221 fd_set fds{};
222 FD_ZERO(&fds);
223 FD_SET(display_fd, &fds);
224
225 double timeout = Rml::Math::Min(context->GetNextUpdateDelay(), 10.0);
226 struct timeval tv {};
227 double seconds;
228 tv.tv_usec = std::modf(timeout, &seconds) * 1000000.0;
229 tv.tv_sec = seconds;
230
231 int ready_fd_count;
232 do
233 {
234 ready_fd_count = select(display_fd + 1, &fds, NULL, NULL, &tv);
235 // We don't care about the return value as long as select didn't error out
236 RMLUI_ASSERT(ready_fd_count >= 0);
237 } while (XPending(display) == 0 && ready_fd_count != 0);
238 }
239
240 while (XPending(display) > 0)
241 {
242 XEvent ev;
243 XNextEvent(display, &ev);
244
245 switch (ev.type)
246 {
247 case ClientMessage:
248 {
249 // The only message we register for is WM_DELETE_WINDOW, so if we receive a client message then the window has been closed.
250 char* event_type = XGetAtomName(display, ev.xclient.message_type);
251 if (strcmp(event_type, "WM_PROTOCOLS") == 0)
252 data->running = false;
253 XFree(event_type);
254 event_type = nullptr;
255 }
256 break;
257 case ConfigureNotify:
258 {
259 int x = ev.xconfigure.width;
260 int y = ev.xconfigure.height;
261
262 context->SetDimensions({x, y});
263 data->render_interface.SetViewport(x, y);
264 }
265 break;
266 case KeyPress:
267 {
268 Rml::Input::KeyIdentifier key = RmlX11::ConvertKey(display, ev.xkey.keycode);
269 const int key_modifier = RmlX11::GetKeyModifierState(ev.xkey.state);
270 const float native_dp_ratio = 1.f;
271
272 // See if we have any global shortcuts that take priority over the context.
273 if (key_down_callback && !key_down_callback(context, key, key_modifier, native_dp_ratio, true))
274 break;
275 // Otherwise, hand the event over to the context by calling the input handler as normal.
276 if (!RmlX11::HandleInputEvent(context, display, ev))
277 break;
278 // The key was not consumed by the context either, try keyboard shortcuts of lower priority.
279 if (key_down_callback && !key_down_callback(context, key, key_modifier, native_dp_ratio, false))
280 break;
281 }
282 break;
283 case SelectionRequest:
284 {
285 data->system_interface.HandleSelectionRequest(ev);
286 }
287 break;
288 default:
289 {
290 // Pass unhandled events to the platform layer's input handler.
291 RmlX11::HandleInputEvent(context, display, ev);
292 }
293 break;
294 }
295 }
296
297 return result;
298}
