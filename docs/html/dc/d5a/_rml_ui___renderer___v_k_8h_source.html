175 template <typename Func>
176 void UploadToGPU(Func&& p_user_commands) noexcept
177 {
178 RMLUI_VK_ASSERTMSG(m_p_command_buffer, "you didn't initialize VkCommandBuffer");
179
180 VkCommandBufferBeginInfo info_command = {};
181
182 info_command.sType = VK_STRUCTURE_TYPE_COMMAND_BUFFER_BEGIN_INFO;
183 info_command.pNext = nullptr;
184 info_command.pInheritanceInfo = nullptr;
185 info_command.flags = VK_COMMAND_BUFFER_USAGE_ONE_TIME_SUBMIT_BIT;
186
187 VkResult status = vkBeginCommandBuffer(m_p_command_buffer, &info_command);
188
189 RMLUI_VK_ASSERTMSG(status == VkResult::VK_SUCCESS, "failed to vkBeginCommandBuffer");
190
191 p_user_commands(m_p_command_buffer);
192
193 status = vkEndCommandBuffer(m_p_command_buffer);
194
195 RMLUI_VK_ASSERTMSG(status == VkResult::VK_SUCCESS, "faield to vkEndCommandBuffer");
196
197 Submit();
198 Wait();
199 }
200
201 private:
202 void Create_Fence() noexcept
203 {
204 VkFenceCreateInfo info = {};
205 info.sType = VK_STRUCTURE_TYPE_FENCE_CREATE_INFO;
206 info.pNext = nullptr;
207 info.flags = 0;
208
209 VkResult status = vkCreateFence(m_p_device, &info, nullptr, &m_p_fence);
210
211 RMLUI_VK_ASSERTMSG(status == VkResult::VK_SUCCESS, "failed to vkCreateFence");
212 }
213
214 void Create_CommandBuffer() noexcept
215 {
216 RMLUI_VK_ASSERTMSG(m_p_command_pool, "you have to initialize VkCommandPool before calling this method!");
217
218 VkCommandBufferAllocateInfo info = {};
219 info.sType = VK_STRUCTURE_TYPE_COMMAND_BUFFER_ALLOCATE_INFO;
220 info.pNext = nullptr;
221 info.commandPool = m_p_command_pool;
222 info.commandBufferCount = 1;
223 info.level = VK_COMMAND_BUFFER_LEVEL_PRIMARY;
224
225 VkResult status = vkAllocateCommandBuffers(m_p_device, &info, &m_p_command_buffer);
226
227 RMLUI_VK_ASSERTMSG(status == VkResult::VK_SUCCESS, "failed to vkAllocateCommandBuffers");
228 }
229
230 void Create_CommandPool(uint32_t queue_family_index) noexcept
231 {
232 VkCommandPoolCreateInfo info = {};
233
234 info.sType = VK_STRUCTURE_TYPE_COMMAND_POOL_CREATE_INFO;
235 info.pNext = nullptr;
236 info.queueFamilyIndex = queue_family_index;
237 info.flags = 0;
238
239 VkResult status = vkCreateCommandPool(m_p_device, &info, nullptr, &m_p_command_pool);
240
241 RMLUI_VK_ASSERTMSG(status == VkResult::VK_SUCCESS, "failed to vkCreateCommandPool");
242 }
243
244 void Create_All(uint32_t queue_family_index) noexcept
245 {
246 Create_Fence();
247 Create_CommandPool(queue_family_index);
248 Create_CommandBuffer();
249 }
250
251 void Wait() noexcept
252 {
253 RMLUI_VK_ASSERTMSG(m_p_fence, "you must initialize your VkFence");
254
255 vkWaitForFences(m_p_device, 1, &m_p_fence, VK_TRUE, UINT64_MAX);
256 vkResetFences(m_p_device, 1, &m_p_fence);
257 vkResetCommandPool(m_p_device, m_p_command_pool, 0);
258 }
259
260 void Submit() noexcept
261 {
262 VkSubmitInfo info = {};
263
264 info.sType = VK_STRUCTURE_TYPE_SUBMIT_INFO;
265 info.pNext = nullptr;
266 info.waitSemaphoreCount = 0;
267 info.signalSemaphoreCount = 0;
268 info.pSignalSemaphores = nullptr;
269 info.pWaitSemaphores = nullptr;
270 info.pWaitDstStageMask = nullptr;
271 info.pCommandBuffers = &m_p_command_buffer;
272 info.commandBufferCount = 1;
273
274 auto status = vkQueueSubmit(m_p_graphics_queue, 1, &info, m_p_fence);
275
276 RMLUI_VK_ASSERTMSG(status == VkResult::VK_SUCCESS, "failed to vkQueueSubmit");
277 }

281 VkFence m_p_fence;
282 VkCommandBuffer m_p_command_buffer;
283 VkCommandPool m_p_command_pool;
284 VkQueue m_p_graphics_queue;

352 class DescriptorPoolManager {
353 public:
354 DescriptorPoolManager() : m_allocated_descriptor_count{}, m_p_descriptor_pool{} {}
355 ~DescriptorPoolManager()
356 {
357 RMLUI_VK_ASSERTMSG(m_allocated_descriptor_count <= 0, "something is wrong. You didn't free some VkDescriptorSet");
358 }
359
360 void Initialize(VkDevice p_device, uint32_t count_uniform_buffer, uint32_t count_image_sampler, uint32_t count_sampler,
361 uint32_t count_storage_buffer) noexcept
362 {
363 RMLUI_VK_ASSERTMSG(p_device, "you can't pass an invalid VkDevice here");
364
365 Rml::Array<VkDescriptorPoolSize, 5> sizes;
366 sizes[0] = {VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER_DYNAMIC, count_uniform_buffer};
367 sizes[1] = {VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER, count_uniform_buffer};
368 sizes[2] = {VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER, count_image_sampler};
369 sizes[3] = {VK_DESCRIPTOR_TYPE_SAMPLER, count_sampler};
370 sizes[4] = {VK_DESCRIPTOR_TYPE_STORAGE_BUFFER, count_storage_buffer};
371
372 VkDescriptorPoolCreateInfo info = {};
373 info.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_POOL_CREATE_INFO;
374 info.pNext = nullptr;
375 info.flags = VK_DESCRIPTOR_POOL_CREATE_FREE_DESCRIPTOR_SET_BIT;
376 info.maxSets = 1000;
377 info.poolSizeCount = static_cast<uint32_t>(sizes.size());
378 info.pPoolSizes = sizes.data();
379
380 auto status = vkCreateDescriptorPool(p_device, &info, nullptr, &m_p_descriptor_pool);
381 RMLUI_VK_ASSERTMSG(status == VkResult::VK_SUCCESS, "failed to vkCreateDescriptorPool");
382 }
383
384 void Shutdown(VkDevice p_device)
385 {
386 RMLUI_VK_ASSERTMSG(p_device, "you can't pass an invalid VkDevice here");
387
388 vkDestroyDescriptorPool(p_device, m_p_descriptor_pool, nullptr);
389 }
390
391 uint32_t Get_AllocatedDescriptorCount() const noexcept { return m_allocated_descriptor_count; }
392
393 bool Alloc_Descriptor(VkDevice p_device, VkDescriptorSetLayout* p_layouts, VkDescriptorSet* p_sets,
394 uint32_t descriptor_count_for_creation = 1) noexcept
395 {
396 RMLUI_VK_ASSERTMSG(p_layouts, "you have to pass a valid and initialized VkDescriptorSetLayout (probably you must create it)");
397 RMLUI_VK_ASSERTMSG(p_device, "you must pass a valid VkDevice here");
398
399 VkDescriptorSetAllocateInfo info = {};
400 info.sType = VK_STRUCTURE_TYPE_DESCRIPTOR_SET_ALLOCATE_INFO;
401 info.pNext = nullptr;
402 info.descriptorPool = m_p_descriptor_pool;
403 info.descriptorSetCount = descriptor_count_for_creation;
404 info.pSetLayouts = p_layouts;
405
406 auto status = vkAllocateDescriptorSets(p_device, &info, p_sets);
407 RMLUI_VK_ASSERTMSG(status == VkResult::VK_SUCCESS, "failed to vkAllocateDescriptorSets");
408
409 m_allocated_descriptor_count += descriptor_count_for_creation;
410
411 return status == VkResult::VK_SUCCESS;
412 }
413
414 void Free_Descriptors(VkDevice p_device, VkDescriptorSet* p_sets, uint32_t descriptor_count = 1) noexcept
415 {
416 RMLUI_VK_ASSERTMSG(p_device, "you must pass a valid VkDevice here");
417
418 if (p_sets)
419 {
420 m_allocated_descriptor_count -= descriptor_count;
421 vkFreeDescriptorSets(p_device, m_p_descriptor_pool, descriptor_count, p_sets);
422 }
423 }

427 VkDescriptorPool m_p_descriptor_pool;
