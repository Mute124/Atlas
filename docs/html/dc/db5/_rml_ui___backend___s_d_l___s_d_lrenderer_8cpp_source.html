55bool Backend::Initialize(const char* window_name, int width, int height, bool allow_resize)
56{
57 RMLUI_ASSERT(!data);
58
59 if (SDL_Init(SDL_INIT_VIDEO | SDL_INIT_EVENTS | SDL_INIT_TIMER) != 0)
60 return false;
61
62 // Submit click events when focusing the window.
63 SDL_SetHint(SDL_HINT_MOUSE_FOCUS_CLICKTHROUGH, "1");
64
65 const Uint32 window_flags = (allow_resize ? SDL_WINDOW_RESIZABLE : 0);
66 SDL_Window* window = SDL_CreateWindow(window_name, SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED, width, height, window_flags);
67 if (!window)
68 {
69 Rml::Log::Message(Rml::Log::LT_ERROR, "SDL error on create window: %s\n", SDL_GetError());
70 return false;
71 }
72
73 /*
74 * Force a specific back-end
75 SDL_SetHint(SDL_HINT_RENDER_BATCHING, "1");
76 SDL_SetHint(SDL_HINT_RENDER_DRIVER, "software");
77 SDL_SetHint(SDL_HINT_RENDER_DRIVER, "opengles2");
78 SDL_SetHint(SDL_HINT_RENDER_DRIVER, "direct3d");
79 SDL_SetHint(SDL_HINT_RENDER_DRIVER, "opengl");
80 */
81
82 SDL_Renderer* renderer = SDL_CreateRenderer(window, -1, SDL_RENDERER_ACCELERATED | SDL_RENDERER_PRESENTVSYNC);
83 if (!renderer)
84 return false;
85
86 data = Rml::MakeUnique<BackendData>(renderer);
87
88 data->window = window;
89 data->renderer = renderer;
90
91 data->system_interface.SetWindow(window);
92
93 SDL_RendererInfo renderer_info;
94 if (SDL_GetRendererInfo(renderer, &renderer_info) == 0)
95 {
96 data->system_interface.LogMessage(Rml::Log::LT_INFO, Rml::CreateString("Using SDL renderer: %s\n", renderer_info.name));
97 }
98
99 return true;
100}

126bool Backend::ProcessEvents(Rml::Context* context, KeyDownCallback key_down_callback, bool power_save)
127{
128 RMLUI_ASSERT(data && context);
129
130 bool result = data->running;
131 SDL_Event ev;
132
133 int has_event = 0;
134 if (power_save)
135 has_event = SDL_WaitEventTimeout(&ev, static_cast<int>(Rml::Math::Min(context->GetNextUpdateDelay(), 10.0) * 1000));
136 else
137 has_event = SDL_PollEvent(&ev);
138 while (has_event)
139 {
140 switch (ev.type)
141 {
142 case SDL_QUIT:
143 {
144 result = false;
145 }
146 break;
147 case SDL_KEYDOWN:
148 {
149 const Rml::Input::KeyIdentifier key = RmlSDL::ConvertKey(ev.key.keysym.sym);
150 const int key_modifier = RmlSDL::GetKeyModifierState();
151 const float native_dp_ratio = 1.f;
152
153 // See if we have any global shortcuts that take priority over the context.
154 if (key_down_callback && !key_down_callback(context, key, key_modifier, native_dp_ratio, true))
155 break;
156 // Otherwise, hand the event over to the context by calling the input handler as normal.
157 if (!RmlSDL::InputEventHandler(context, ev))
158 break;
159 // The key was not consumed by the context either, try keyboard shortcuts of lower priority.
160 if (key_down_callback && !key_down_callback(context, key, key_modifier, native_dp_ratio, false))
161 break;
162 }
163 break;
164 default:
165 {
166 RmlSDL::InputEventHandler(context, ev);
167 }
168 break;
169 }
170 has_event = SDL_PollEvent(&ev);
171 }
172
173 return result;
174}
