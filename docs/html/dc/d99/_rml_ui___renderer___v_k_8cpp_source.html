106{
107 RMLUI_ZoneScopedN("Vulkan - CompileGeometry");
108
109 VkDescriptorSet p_current_descriptor_set = nullptr;
110 p_current_descriptor_set = m_p_descriptor_set;
111
112 RMLUI_VK_ASSERTMSG(p_current_descriptor_set,
113 "you can't have here an invalid pointer of VkDescriptorSet. Two reason might be. 1. - you didn't allocate it "

115
116 auto* p_geometry_handle = new geometry_handle_t{};
117
118 uint32_t* pCopyDataToBuffer = nullptr;
119 const void* pData = reinterpret_cast<const void*>(vertices.data());
120

126

128 &p_geometry_handle->m_p_index, &p_geometry_handle->m_p_index_allocation);
129 RMLUI_VK_ASSERTMSG(status, "failed to AllocIndexBuffer");
130

134
135 return Rml::CompiledGeometryHandle(p_geometry_handle);

138void RenderInterface_VK::RenderGeometry(Rml::CompiledGeometryHandle geometry, Rml::Vector2f translation, Rml::TextureHandle texture)
139{
140 RMLUI_ZoneScopedN("Vulkan - RenderCompiledGeometry");
141
142 if (m_p_current_command_buffer == nullptr)
143 return;
144
145 RMLUI_VK_ASSERTMSG(m_p_current_command_buffer, "must be valid otherwise you can't render now!!! (can't be)");
146
147 texture_data_t* p_texture = reinterpret_cast<texture_data_t*>(texture);
148
149 VkDescriptorImageInfo info_descriptor_image = {};
150 if (p_texture && p_texture->m_p_vk_descriptor_set == nullptr)
151 {
152 VkDescriptorSet p_texture_set = nullptr;
153 m_manager_descriptors.Alloc_Descriptor(m_p_device, &m_p_descriptor_set_layout_texture, &p_texture_set);
154
155 info_descriptor_image.imageView = p_texture->m_p_vk_image_view;
156 info_descriptor_image.sampler = p_texture->m_p_vk_sampler;
157 info_descriptor_image.imageLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
158
159 VkWriteDescriptorSet info_write = {};
160
161 info_write.sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;
162 info_write.dstSet = p_texture_set;
163 info_write.descriptorType = VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER;
164 info_write.dstBinding = 2;
165 info_write.pImageInfo = &info_descriptor_image;
166 info_write.descriptorCount = 1;
167
168 vkUpdateDescriptorSets(m_p_device, 1, &info_write, 0, nullptr);
169 p_texture->m_p_vk_descriptor_set = p_texture_set;
170 }
171
172 geometry_handle_t* p_casted_compiled_geometry = reinterpret_cast<geometry_handle_t*>(geometry);
173
174 m_user_data_for_vertex_shader.m_translate = translation;
175
176 VkDescriptorSet p_current_descriptor_set = nullptr;
177 p_current_descriptor_set = m_p_descriptor_set;
178
179 RMLUI_VK_ASSERTMSG(p_current_descriptor_set,
180 "you can't have here an invalid pointer of VkDescriptorSet. Two reason might be. 1. - you didn't allocate it "
181 "at all or 2. - Somehing is wrong with allocation and somehow it was corrupted by something.");
182
183 shader_vertex_user_data_t* p_data = nullptr;
184
185 if (p_casted_compiled_geometry->m_p_shader_allocation == nullptr)
186 {
187 // it means it was freed in ReleaseCompiledGeometry method
188 bool status = m_memory_pool.Alloc_GeneralBuffer(sizeof(m_user_data_for_vertex_shader), reinterpret_cast<void**>(&p_data),
189 &p_casted_compiled_geometry->m_p_shader, &p_casted_compiled_geometry->m_p_shader_allocation);
190 RMLUI_VK_ASSERTMSG(status, "failed to allocate VkDescriptorBufferInfo for uniform data to shaders");
191 }
192 else
193 {
194 // it means our state is dirty and we need to update data, but it is not right in terms of architecture, for real better experience would
195 // be great to free all "compiled" geometries and "re-build" them in one general way, but here I got only three callings for
196 // font-face-layer textures (load_document example) and that shit. So better to think how to make it right, if it is fine okay, if it is
197 // not okay and like we really expect that ReleaseCompiledGeometry for all objects that needs to be rebuilt so better to implement that,
198 // but still it is a big architectural thing (or at least you need to do something big commits here to implement a such feature), so my
199 // implementation doesn't break anything what we had, but still it looks strange. If I get callings for releasing maybe I need to use it
200 // for all objects not separately????? Otherwise it is better to provide method for resizing (or some kind of "resizing" callback) for
201 // recalculating all geometry IDK, so it means you pass the existed geometry that wasn't pass to ReleaseCompiledGeometry, but from another
202 // hand you need to re-build compiled geometry again so we have two kinds of geometry one is compiled and never changes and one is dynamic
203 // and it goes through pipeline InitializationOfProgram...->Compile->Render->Release->Compile->Render->Release...
204
205 m_memory_pool.Free_GeometryHandle_ShaderDataOnly(p_casted_compiled_geometry);
206 bool status = m_memory_pool.Alloc_GeneralBuffer(sizeof(m_user_data_for_vertex_shader), reinterpret_cast<void**>(&p_data),
207 &p_casted_compiled_geometry->m_p_shader, &p_casted_compiled_geometry->m_p_shader_allocation);
208 RMLUI_VK_ASSERTMSG(status, "failed to allocate VkDescriptorBufferInfo for uniform data to shaders");
209 }
210
211 if (p_data)
212 {
213 p_data->m_transform = m_user_data_for_vertex_shader.m_transform;
214 p_data->m_translate = m_user_data_for_vertex_shader.m_translate;
215 }
216 else
217 {
218 RMLUI_VK_ASSERTMSG(p_data, "you can't reach this zone, it means something bad");
219 }
220
221 const uint32_t pDescriptorOffsets = static_cast<uint32_t>(p_casted_compiled_geometry->m_p_shader.offset);
222
223 VkDescriptorSet p_texture_descriptor_set = nullptr;
224
225 if (p_texture)
226 {
227 p_texture_descriptor_set = p_texture->m_p_vk_descriptor_set;
228 }
229
230 VkDescriptorSet p_sets[] = {p_current_descriptor_set, p_texture_descriptor_set};
231 int real_size_of_sets = 2;
232
233 if (p_texture == nullptr)
234 real_size_of_sets = 1;
235
236 vkCmdBindDescriptorSets(m_p_current_command_buffer, VK_PIPELINE_BIND_POINT_GRAPHICS, m_p_pipeline_layout, 0, real_size_of_sets, p_sets, 1,
237 &pDescriptorOffsets);
238
239 if (m_is_use_stencil_pipeline)
240 {
241 vkCmdBindPipeline(m_p_current_command_buffer, VK_PIPELINE_BIND_POINT_GRAPHICS, m_p_pipeline_stencil_for_region_where_geometry_will_be_drawn);
242 }
243 else
244 {
245 if (p_texture)
246 {
247 if (m_is_apply_to_regular_geometry_stencil)
248 {
249 vkCmdBindPipeline(m_p_current_command_buffer, VK_PIPELINE_BIND_POINT_GRAPHICS,
250 m_p_pipeline_stencil_for_regular_geometry_that_applied_to_region_with_textures);
251 }
252 else
253 {
254 vkCmdBindPipeline(m_p_current_command_buffer, VK_PIPELINE_BIND_POINT_GRAPHICS, m_p_pipeline_with_textures);
255 }
256 }
257 else
258 {
259 if (m_is_apply_to_regular_geometry_stencil)
260 {
261 vkCmdBindPipeline(m_p_current_command_buffer, VK_PIPELINE_BIND_POINT_GRAPHICS,
262 m_p_pipeline_stencil_for_regular_geometry_that_applied_to_region_without_textures);
263 }
264 else
265 {
266 vkCmdBindPipeline(m_p_current_command_buffer, VK_PIPELINE_BIND_POINT_GRAPHICS, m_p_pipeline_without_textures);
267 }
268 }
269 }
270
271 vkCmdBindVertexBuffers(m_p_current_command_buffer, 0, 1, &p_casted_compiled_geometry->m_p_vertex.buffer,
272 &p_casted_compiled_geometry->m_p_vertex.offset);
273
274 vkCmdBindIndexBuffer(m_p_current_command_buffer, p_casted_compiled_geometry->m_p_index.buffer, p_casted_compiled_geometry->m_p_index.offset,
275 VK_INDEX_TYPE_UINT32);
276
277 vkCmdDrawIndexed(m_p_current_command_buffer, p_casted_compiled_geometry->m_num_indices, 1, 0, 0, 0);
278}

407Rml::TextureHandle RenderInterface_VK::LoadTexture(Rml::Vector2i& texture_dimensions, const Rml::String& source)
408{
409 Rml::FileInterface* file_interface = Rml::GetFileInterface();
410 Rml::FileHandle file_handle = file_interface->Open(source);
411 if (!file_handle)
412 {
413 return false;
414 }
415
416 file_interface->Seek(file_handle, 0, SEEK_END);
417 size_t buffer_size = file_interface->Tell(file_handle);
418 file_interface->Seek(file_handle, 0, SEEK_SET);
419
420 if (buffer_size <= sizeof(TGAHeader))
421 {
422 Rml::Log::Message(Rml::Log::LT_ERROR, "Texture file size is smaller than TGAHeader, file is not a valid TGA image.");
423 file_interface->Close(file_handle);
424 return false;
425 }
426
427 using Rml::byte;
428 Rml::UniquePtr<byte[]> buffer(new byte[buffer_size]);
429 file_interface->Read(buffer.get(), buffer_size, file_handle);
430 file_interface->Close(file_handle);
431
432 TGAHeader header;
433 memcpy(&header, buffer.get(), sizeof(TGAHeader));
434
435 int color_mode = header.bitsPerPixel / 8;
436 const size_t image_size = header.width * header.height * 4; // We always make 32bit textures
437
438 if (header.dataType != 2)
439 {
440 Rml::Log::Message(Rml::Log::LT_ERROR, "Only 24/32bit uncompressed TGAs are supported.");
441 return false;
442 }
443
444 // Ensure we have at least 3 colors
445 if (color_mode < 3)
446 {
447 Rml::Log::Message(Rml::Log::LT_ERROR, "Only 24 and 32bit textures are supported.");
448 return false;
449 }
450
451 const byte* image_src = buffer.get() + sizeof(TGAHeader);
452 Rml::UniquePtr<byte[]> image_dest_buffer(new byte[image_size]);
453 byte* image_dest = image_dest_buffer.get();
454
455 // Targa is BGR, swap to RGB, flip Y axis, and convert to premultiplied alpha.
456 for (long y = 0; y < header.height; y++)
457 {
458 long read_index = y * header.width * color_mode;
459 long write_index = ((header.imageDescriptor & 32) != 0) ? read_index : (header.height - y - 1) * header.width * 4;
460 for (long x = 0; x < header.width; x++)
461 {
462 image_dest[write_index] = image_src[read_index + 2];
463 image_dest[write_index + 1] = image_src[read_index + 1];
464 image_dest[write_index + 2] = image_src[read_index];
465 if (color_mode == 4)
466 {
467 const byte alpha = image_src[read_index + 3];
468 for (size_t j = 0; j < 3; j++)
469 image_dest[write_index + j] = byte((image_dest[write_index + j] * alpha) / 255);
470 image_dest[write_index + 3] = alpha;
471 }
472 else
473 image_dest[write_index + 3] = 255;
474
475 write_index += 4;
476 read_index += color_mode;
477 }
478 }
479
480 texture_dimensions.x = header.width;
481 texture_dimensions.y = header.height;
482
483 return GenerateTexture({image_dest, image_size}, texture_dimensions);
484}

492/*
493 How vulkan works with textures efficiently?
494
495 You need to create buffer that has CPU memory accessibility it means it uses your RAM memory for storing data and it has only CPU visibility (RAM)
496 After you create buffer that has GPU memory accessibility it means it uses by your video hardware and it has only VRAM (Video RAM) visibility
497
498 So you copy data to CPU_buffer and after you copy that thing to GPU_buffer, but delete CPU_buffer
499
500 So it means you "uploaded" data to GPU
501
502 Again, you need to "write" data into CPU buffer after you need to copy that data from buffer to GPU buffer and after that buffer go to GPU.
503
504 RAW_POINTER_DATA_BYTES_LITERALLY->COPY_TO->CPU->COPY_TO->GPU->Releasing_CPU <= that's how works uploading textures in Vulkan if you want to have
505 efficient handling otherwise it is cpu_to_gpu visibility and it means you create only ONE buffer that is accessible for CPU and for GPU, but it
506 will cause the worst performance...
507*/
508Rml::TextureHandle RenderInterface_VK::CreateTexture(Rml::Span<const Rml::byte> source, Rml::Vector2i dimensions, const Rml::String& name)
509{
510 RMLUI_ZoneScopedN("Vulkan - GenerateTexture");
511
512 RMLUI_VK_ASSERTMSG(!source.empty(), "you pushed not valid data for copying to buffer");
513 RMLUI_VK_ASSERTMSG(m_p_allocator, "you have to initialize Vma Allocator for this method");
514 (void)name;
515
516 int width = dimensions.x;
517 int height = dimensions.y;
518
519 RMLUI_VK_ASSERTMSG(width, "invalid width");
520 RMLUI_VK_ASSERTMSG(height, "invalid height");
521
522 VkDeviceSize image_size = source.size();
523 VkFormat format = VkFormat::VK_FORMAT_R8G8B8A8_UNORM;
524
525 buffer_data_t cpu_buffer = CreateResource_StagingBuffer(image_size, VK_BUFFER_USAGE_TRANSFER_SRC_BIT);
526
527 void* data;
528 vmaMapMemory(m_p_allocator, cpu_buffer.m_p_vma_allocation, &data);
529 memcpy(data, source.data(), static_cast<size_t>(image_size));
530 vmaUnmapMemory(m_p_allocator, cpu_buffer.m_p_vma_allocation);
531
532 VkExtent3D extent_image = {};
533 extent_image.width = static_cast<uint32_t>(width);
534 extent_image.height = static_cast<uint32_t>(height);
535 extent_image.depth = 1;
536
537 auto* p_texture = new texture_data_t{};
538
539 VkImageCreateInfo info = {};
540 info.sType = VK_STRUCTURE_TYPE_IMAGE_CREATE_INFO;
541 info.pNext = nullptr;
542 info.imageType = VK_IMAGE_TYPE_2D;
543 info.format = format;
544 info.extent = extent_image;
545 info.mipLevels = 1;
546 info.arrayLayers = 1;

573 /*
574 * So Vulkan works only through VkCommandBuffer, it is for remembering API commands what you want to call from GPU
575 * So on CPU side you need to create a scope that consists of two things
576 * vkBeginCommandBuffer
577 * ... <= here your commands what you want to place into your command buffer and send it to GPU through vkQueueSubmit function
578 * vkEndCommandBuffer
579 *
580 * So commands start to work ONLY when you called the vkQueueSubmit otherwise you just "place" commands into your command buffer but you
581 * didn't issue any thing in order to start the work on GPU side. ALWAYS remember that just sumbit means execute async mode, so you have to wait
582 * operations before they exeecute fully otherwise you will get some errors or write/read concurrent state and all other stuff, vulkan validation
583 * will notify you :) (in most cases)
584 *
585 * BUT you need always sync what you have done when you called your vkQueueSubmit function, so it is wait method, but generally you can create
586 * another queue and isolate all stuff tbh
587 *
588 * So understing these principles you understand how to work with API and your GPU
589 *
590 * There's nothing hard, but it makes all stuff on programmer side if you remember OpenGL and how it was easy to load texture upload it and create
591 * buffers and it In OpenGL all stuff is handled by driver and other things, not a programmer definitely
592 *
593 * What we do here? We need to change the layout of our image. it means where we want to use it. So in our case we want to see that this image
594 * will be in shaders Because the initial state of create object is VK_IMAGE_LAYOUT_UNDEFINED means you can't just pass that VkImage handle to
595 * your functions and wait that it comes to shaders for exmaple No it doesn't work like that you have to have the explicit states of your resource
596 * and where it goes
597 *
598 * In our case we want to see in our pixel shader so we need to change transfer into this flag VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL, because we
599 * want to copy so it means some transfer thing, but after we say it goes to pixel after our copying operation
600 */
601 m_upload_manager.UploadToGPU([p_image, extent_image, cpu_buffer](VkCommandBuffer p_cmd) {
602 VkImageSubresourceRange range = {};

612 info_barrier.newLayout = VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL;

614 info_barrier.subresourceRange = range;
615 info_barrier.srcAccessMask = 0;
616 info_barrier.dstAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
617
618 vkCmdPipelineBarrier(p_cmd, VK_PIPELINE_STAGE_TOP_OF_PIPE_BIT, VK_PIPELINE_STAGE_TRANSFER_BIT, 0, 0, nullptr, 0, nullptr, 1, &info_barrier);
619
620 VkBufferImageCopy region = {};

622 region.bufferRowLength = 0;
623 region.bufferImageHeight = 0;
624
625 region.imageSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;

631 vkCmdCopyBufferToImage(p_cmd, cpu_buffer.m_p_vk_buffer, p_image, VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL, 1, &region);

645 });
646
647 DestroyResource_StagingBuffer(cpu_buffer);

659 info_image_view.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
660
661 VkImageView p_image_view = nullptr;
662 status = vkCreateImageView(m_p_device, &info_image_view, nullptr, &p_image_view);
663 RMLUI_VK_ASSERTMSG(status == VkResult::VK_SUCCESS, "failed to vkCreateImageView");

1026void RenderInterface_VK::Initialize_QueueIndecies() noexcept
1027{
1028 RMLUI_VK_ASSERTMSG(m_p_physical_device, "you must initialize your physical device");
1029 RMLUI_VK_ASSERTMSG(m_p_surface, "you must initialize VkSurfaceKHR before calling this method");
1030
1031 uint32_t queue_family_count = 0;
1032
1033 vkGetPhysicalDeviceQueueFamilyProperties(m_p_physical_device, &queue_family_count, nullptr);
1034
1035 RMLUI_VK_ASSERTMSG(queue_family_count >= 1, "failed to vkGetPhysicalDeviceQueueFamilyProperties (getting count)");
1036
1037 Rml::Vector<VkQueueFamilyProperties> queue_props;
1038 queue_props.resize(queue_family_count);
1039
1040 vkGetPhysicalDeviceQueueFamilyProperties(m_p_physical_device, &queue_family_count, queue_props.data());
1041
1042 RMLUI_VK_ASSERTMSG(queue_family_count >= 1, "failed to vkGetPhysicalDeviceQueueFamilyProperties (filling vector of VkQueueFamilyProperties)");
1043
1044 constexpr uint32_t kUint32Undefined = uint32_t(-1);
1045
1046 m_queue_index_compute = kUint32Undefined;
1047 m_queue_index_graphics = kUint32Undefined;
1048 m_queue_index_present = kUint32Undefined;
1049
1050 for (uint32_t i = 0; i < queue_family_count; ++i)
1051 {
1052 if ((queue_props[i].queueFlags & VK_QUEUE_GRAPHICS_BIT) != 0)
1053 {
1054 if (m_queue_index_graphics == kUint32Undefined)
1055 m_queue_index_graphics = i;
1056
1057 VkBool32 is_support_present;
1058
1059 vkGetPhysicalDeviceSurfaceSupportKHR(m_p_physical_device, i, m_p_surface, &is_support_present);
1060
1061 // User's videocard may have same index for two queues like graphics and present
1062
1063 if (is_support_present == VK_TRUE)
1064 {
1065 m_queue_index_graphics = i;
1066 m_queue_index_present = m_queue_index_graphics;
1067 break;
1068 }
1069 }
1070 }
1071
1072 if (m_queue_index_present == static_cast<uint32_t>(-1))
1073 {
1074 Rml::Log::Message(Rml::Log::LT_WARNING, "[Vulkan] User doesn't have one index for two queues, so we need to find for present queue index");
1075
1076 for (uint32_t i = 0; i < queue_family_count; ++i)
1077 {
1078 VkBool32 is_support_present;
