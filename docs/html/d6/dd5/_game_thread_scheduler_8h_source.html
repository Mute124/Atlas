228 void init(uint8_t numThreads) {
229 mIsRunning = true;
230 setStatus(EGameThreadSchedulerStatus::Initializing);
231
232 for(size_t i = 0; i < numThreads; ++i) {
233
234 mWorkersVector.emplace_back([this]() {
235
236 while(true) {
237 Task task;
238 {
239 std::unique_lock<std::mutex> lock(mQueueMutex);
240
241 mWorkAvailableConditional.wait(lock, [this]() {
242 return !isTaskQueueEmpty() || !mIsRunning;
243 });
244
245 if(!mIsRunning && isTaskQueueEmpty())
246 return;
247
248 task = std::move(mTaskQueue.front());
249 mTaskQueue.pop();
250 }
251
252 task(); // Run task
253 }
254 });
255 }
256
257 setStatus(EGameThreadSchedulerStatus::Running);

299 template<typename Func, typename... Args>
300 auto schedule(Func&& func, Args&&... args) -> std::future<std::invoke_result_t<Func, Args...>> {
301
302 // Determine the return type of the callable object (Func) when invoked with the given arguments (Args...)
303 using ReturnType = std::invoke_result_t<Func, Args...>;
304
305 // Create a shared pointer to a packaged task that wraps the callable object (Func) and its arguments (Args...)
306 // The packaged task is a callable object that can be executed later, and its return value can be retrieved using a future
307 auto taskPtr = std::make_shared<std::packaged_task<ReturnType()>>(
308 // Use std::bind to create a callable object that invokes the original callable object (Func) with the given arguments (Args...)
309 // The std::forward function is used to preserve the value category (lvalue or rvalue) of the original arguments
310 std::bind(std::forward<Func>(func), std::forward<Args>(args)...)
311 );
312
313 // Create a wrapper function that executes the packaged task when called
314 // This wrapper function is used to decouple the execution of the task from the scheduling mechanism
315 std::function<void()> wrapper = [taskPtr]() {
316 (*taskPtr)();
317 };
318
319 // Acquire a lock on the scheduler's queue mutex to ensure thread-safe access to the queue
320 {
321 std::lock_guard<std::mutex> lock(mQueueMutex);
322
323 // make sure that the scheduler is not shutting down
324 if (!mIsRunning) {
325 throw std::runtime_error("Scheduler is shutting down, cannot schedule new tasks.");
326 }
327
328 // Forward the callable object (Func) and its arguments (Args...) to the packaged task.
329 // In other words, finally put the callable object (Func) and its arguments (Args...) into the queue for execution
330 mTaskQueue.emplace(std::move(wrapper));
331
332 }
333
334 // let worker threads know that there is a new task
335 mWorkAvailableConditional.notify_one();
336
337 return taskPtr->get_future();
338 }
