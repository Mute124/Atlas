26import sys
27import os
28import subprocess
29
30# Compiles all .frag and .vert files in this directory to SPIR-V binary C character arrays. Requires 'glslc' installed and available system-wide.
31
32out_file = "ShadersCompiledSPV.h"
33
34current_dir = os.path.dirname(os.path.realpath(__file__));
35
36temp_spirv_path = os.path.join(current_dir, ".temp.spv")
37out_path = os.path.join(current_dir, out_file)
38
39with open(out_path,'w') as result_file:
40 result_file.write('// RmlUi SPIR-V Vulkan shaders compiled using command: \'python compile_shaders.py\'. Do not edit manually.\n\n#include <stdint.h>\n')
41
42 for file in os.listdir(current_dir):
43 if file.endswith(".vert") or file.endswith(".frag"):
44 shader_path = os.path.join(current_dir, file)
45 variable_name = os.path.splitext(file)[0]
46
47 print("Compiling '{}' to SPIR-V using glslc.".format(file))
48
49 subprocess.run(["glslc", shader_path, "-o", temp_spirv_path], check = True)
50
51 print("Success, writing output to variable '{}' in {}".format(variable_name, out_file))
52
53 i = 0
54 result_file.write('\nalignas(uint32_t) static const unsigned char {}[] = {{'.format(variable_name))
55 for b in open(temp_spirv_path, 'rb').read():
56 if i % 20 == 0:
57 result_file.write('\n\t')
58 result_file.write('0x%02X,' % b)
59 i += 1
60
61 result_file.write('\n};\n')
62
63 os.remove(temp_spirv_path)
