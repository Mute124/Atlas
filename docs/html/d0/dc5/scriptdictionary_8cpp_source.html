11// Object types are cached as user data to avoid costly runtime lookups
12
13// We just define a number here that we assume nobody else is using for
14// object type user data. The add-ons have reserved the numbers 1000
15// through 1999 for this purpose, so we should be fine.
16const asPWORD DICTIONARY_CACHE = 1003;
17
18// This cache holds the object type of the dictionary type and array type
19// so it isn't necessary to look this up each time the dictionary or array
20// is created.
21struct SDictionaryCache
22{
23 asITypeInfo *dictType;
24 asITypeInfo *arrayType;
25 asITypeInfo *keyType;
26
27 // This is called from RegisterScriptDictionary
28 static void Setup(asIScriptEngine *engine)
29 {
30 SDictionaryCache *cache = reinterpret_cast<SDictionaryCache*>(engine->GetUserData(DICTIONARY_CACHE));
31 if( cache == 0 )
32 {
33 cache = new SDictionaryCache;
34 engine->SetUserData(cache, DICTIONARY_CACHE);
35 engine->SetEngineUserDataCleanupCallback(SDictionaryCache::Cleanup, DICTIONARY_CACHE);
36
37 cache->dictType = engine->GetTypeInfoByName("dictionary");
38 cache->arrayType = engine->GetTypeInfoByDecl("array<string>");
39 cache->keyType = engine->GetTypeInfoByDecl("string");
40 }
41 }
42
43 // This is called from the engine when shutting down
44 static void Cleanup(asIScriptEngine *engine)
45 {
46 SDictionaryCache *cache = reinterpret_cast<SDictionaryCache*>(engine->GetUserData(DICTIONARY_CACHE));
47 if( cache )
48 delete cache;
49 }
50};

77{
78 // We start with one reference

80 gcFlag = false;

82 // Keep a reference to the engine for as long as we live

94CScriptDictionary::CScriptDictionary(asBYTE *buffer)
95{
96 // This constructor will always be called from a script
97 // so we can get the engine from the active context
98 asIScriptContext *ctx = asGetActiveContext();
99 Init(ctx->GetEngine());
100
101 // Determine if the dictionary key type is registered as reference type or value type
102 SDictionaryCache& cache = *reinterpret_cast<SDictionaryCache*>(engine->GetUserData(DICTIONARY_CACHE));
103 bool keyAsRef = cache.keyType->GetFlags() & asOBJ_REF ? true : false;
104
105 // Initialize the dictionary from the buffer
106 asUINT length = *(asUINT*)buffer;
107 buffer += 4;
108
109 while( length-- )
110 {
111 // Align the buffer pointer on a 4 byte boundary in
112 // case previous value was smaller than 4 bytes
113 if( asPWORD(buffer) & 0x3 )
114 buffer += 4 - (asPWORD(buffer) & 0x3);
115
116 // Get the name value pair from the buffer and insert it in the dictionary
117 dictKey_t name;
118 if (keyAsRef)
119 {
120 name = **(dictKey_t**)buffer;
121 buffer += sizeof(dictKey_t*);
122 }
123 else
124 {
125 name = *(dictKey_t*)buffer;
126 buffer += sizeof(dictKey_t);
127 }
128
129 // Get the type id of the value
130 int typeId = *(int*)buffer;
131 buffer += sizeof(int);
132
133 // Depending on the type id, the value will inline in the buffer or a pointer
134 void *ref = (void*)buffer;
135
136 if( typeId >= asTYPEID_INT8 && typeId <= asTYPEID_DOUBLE )
137 {
138 // Convert primitive values to either int64 or double, so we can use the overloaded Set methods
139 asINT64 i64;
140 double d;
141 switch( typeId )
142 {
143 case asTYPEID_INT8: i64 = *(char*) ref; break;
144 case asTYPEID_INT16: i64 = *(short*) ref; break;
145 case asTYPEID_INT32: i64 = *(int*) ref; break;
146 case asTYPEID_INT64: i64 = *(asINT64*) ref; break;
147 case asTYPEID_UINT8: i64 = *(unsigned char*) ref; break;
148 case asTYPEID_UINT16: i64 = *(unsigned short*)ref; break;
149 case asTYPEID_UINT32: i64 = *(unsigned int*) ref; break;
150 case asTYPEID_UINT64: i64 = *(asINT64*) ref; break;
151 case asTYPEID_FLOAT: d = *(float*) ref; break;
152 case asTYPEID_DOUBLE: d = *(double*) ref; break;
153 }
154
155 if( typeId >= asTYPEID_FLOAT )
156 Set(name, d);
157 else
158 Set(name, i64);
159 }
160 else
161 {
162 if( (typeId & asTYPEID_MASK_OBJECT) &&
163 !(typeId & asTYPEID_OBJHANDLE) &&
164 (engine->GetTypeInfoById(typeId)->GetFlags() & asOBJ_REF) )
165 {
166 // Dereference the pointer to get the reference to the actual object
167 ref = *(void**)ref;
168 }
169
170 Set(name, ref, typeId);
171 }
172
173 // Advance the buffer pointer with the size of the value
174 if( typeId & asTYPEID_MASK_OBJECT )
175 {
176 asITypeInfo *ti = engine->GetTypeInfoById(typeId);
177 if( ti->GetFlags() & asOBJ_VALUE )
178 buffer += ti->GetSize();
179 else
180 buffer += sizeof(void*);
181 }
182 else if( typeId == 0 )
183 {
184 // null pointer
185 buffer += sizeof(void*);
186 }
187 else
188 {
189 buffer += engine->GetSizeOfPrimitiveType(typeId);
190 }
191 }
192}

727bool CScriptDictValue::Get(asIScriptEngine *engine, void *value, int typeId) const
728{
729 // Return the value
730 if( typeId & asTYPEID_OBJHANDLE )
731 {
732 // A handle can be retrieved if the stored type is a handle of same or compatible type
733 // or if the stored type is an object that implements the interface that the handle refer to.
734 if( (m_typeId & asTYPEID_MASK_OBJECT) )
735 {
736 // Don't allow the get if the stored handle is to a const, but the desired handle is not
737 if( (m_typeId & asTYPEID_HANDLETOCONST) && !(typeId & asTYPEID_HANDLETOCONST) )
738 return false;
739
740 // RefCastObject will increment the refcount if successful
741 engine->RefCastObject(m_valueObj, engine->GetTypeInfoById(m_typeId), engine->GetTypeInfoById(typeId), reinterpret_cast<void**>(value));
742
743 return true;
744 }
745 }
746 else if( typeId & asTYPEID_MASK_OBJECT )
747 {
748 // Verify that the copy can be made
749 bool isCompatible = false;
750
751 // Allow a handle to be value assigned if the wanted type is not a handle
752 if( (m_typeId & ~(asTYPEID_OBJHANDLE | asTYPEID_HANDLETOCONST) ) == typeId && m_valueObj != 0 )
753 isCompatible = true;
754
755 // Copy the object into the given reference
756 if( isCompatible )
757 {
758 engine->AssignScriptObject(value, m_valueObj, engine->GetTypeInfoById(typeId));
759
760 return true;
761 }
762 }
763 else
764 {
765 if( m_typeId == typeId )
766 {
767 int size = engine->GetSizeOfPrimitiveType(typeId);
768 memcpy(value, &m_valueInt, size);
769 return true;
770 }
771
772 // We know all numbers are stored as either int64 or double, since we register overloaded functions for those
773 // Only bool and enums needs to be treated separately
774 if( typeId == asTYPEID_DOUBLE )
775 {
776 if( m_typeId == asTYPEID_INT64 )
777 *(double*)value = double(m_valueInt);
778 else if (m_typeId == asTYPEID_BOOL)
779 {
780 // Use memcpy instead of type cast to make sure the code is endianess agnostic
781 char localValue;
782 memcpy(&localValue, &m_valueInt, sizeof(char));
783 *(double*)value = localValue ? 1.0 : 0.0;
784 }
785 else if (m_typeId > asTYPEID_DOUBLE && (m_typeId & asTYPEID_MASK_OBJECT) == 0)
786 {
787 // Use memcpy instead of type cast to make sure the code is endianess agnostic
788 int localValue;
789 memcpy(&localValue, &m_valueInt, sizeof(int));
790 *(double*)value = double(localValue); // enums are 32bit
791 }
792 else
793 {
794 // The stored type is an object
795 // TODO: Check if the object has a conversion operator to a primitive value
796 *(double*)value = 0;
797 return false;
798 }
799 return true;
800 }
801 else if( typeId == asTYPEID_INT64 )
802 {
803 if( m_typeId == asTYPEID_DOUBLE )
804 *(asINT64*)value = asINT64(m_valueFlt);
805 else if (m_typeId == asTYPEID_BOOL)
806 {
807 // Use memcpy instead of type cast to make sure the code is endianess agnostic
808 char localValue;
809 memcpy(&localValue, &m_valueInt, sizeof(char));
810 *(asINT64*)value = localValue ? 1 : 0;
811 }
812 else if (m_typeId > asTYPEID_DOUBLE && (m_typeId & asTYPEID_MASK_OBJECT) == 0)
813 {
814 // Use memcpy instead of type cast to make sure the code is endianess agnostic
815 int localValue;
816 memcpy(&localValue, &m_valueInt, sizeof(int));
817 *(asINT64*)value = localValue; // enums are 32bit
818 }
819 else
820 {
821 // The stored type is an object
822 // TODO: Check if the object has a conversion operator to a primitive value
823 *(asINT64*)value = 0;
824 return false;
825 }
826 return true;
827 }
828 else if( typeId > asTYPEID_DOUBLE && (m_typeId & asTYPEID_MASK_OBJECT) == 0 )
829 {
830 // The desired type is an enum. These are always 32bit integers
831 if( m_typeId == asTYPEID_DOUBLE )
832 *(int*)value = int(m_valueFlt);
833 else if( m_typeId == asTYPEID_INT64 )
834 *(int*)value = int(m_valueInt);
835 else if (m_typeId == asTYPEID_BOOL)
836 {
837 // Use memcpy instead of type cast to make sure the code is endianess agnostic
838 char localValue;
839 memcpy(&localValue, &m_valueInt, sizeof(char));
840 *(int*)value = localValue ? 1 : 0;
841 }
842 else if (m_typeId > asTYPEID_DOUBLE && (m_typeId & asTYPEID_MASK_OBJECT) == 0)
843 {
844 // Use memcpy instead of type cast to make sure the code is endianess agnostic
845 int localValue;
846 memcpy(&localValue, &m_valueInt, sizeof(int));
847 *(int*)value = localValue; // enums are 32bit
848 }
849 else
850 {
851 // The stored type is an object
852 // TODO: Check if the object has a conversion operator to a primitive value
853 *(int*)value = 0;
854 return false;
855 }
856 return true;
857 }
858 else if( typeId == asTYPEID_BOOL )
859 {
860 if (m_typeId & asTYPEID_OBJHANDLE)
861 {
862 // TODO: Check if the object has a conversion operator to a primitive value
863 *(bool*)value = m_valueObj ? true : false;
864 }
865 else if( m_typeId & asTYPEID_MASK_OBJECT )
866 {
867 // TODO: Check if the object has a conversion operator to a primitive value
868 *(bool*)value = true;
869 }
870 else
871 {
872 // Compare only the bytes that were actually set
873 asQWORD zero = 0;
874 int size = engine->GetSizeOfPrimitiveType(m_typeId);
875 *(bool*)value = memcmp(&m_valueInt, &zero, size) == 0 ? false : true;
876 }
877 return true;
878 }
879 }
880
881 // It was not possible to retrieve the value using the desired typeId
882 return false;
883}
