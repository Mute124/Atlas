58bool Backend::Initialize(const char* window_name, int width, int height, bool allow_resize)
59{
60 RMLUI_ASSERT(!data);
61
62 if (SDL_Init(SDL_INIT_VIDEO | SDL_INIT_EVENTS | SDL_INIT_TIMER) != 0)
63 return false;
64
65 // Submit click events when focusing the window.
66 SDL_SetHint(SDL_HINT_MOUSE_FOCUS_CLICKTHROUGH, "1");
67
68 const Uint32 window_flags = (SDL_WINDOW_VULKAN | (allow_resize ? SDL_WINDOW_RESIZABLE : 0));
69
70 SDL_Window* window = SDL_CreateWindow(window_name, SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED, width, height, window_flags);
71 if (!window)
72 {
73 Rml::Log::Message(Rml::Log::LT_ERROR, "SDL error on create window: %s", SDL_GetError());
74 return false;
75 }
76
77 data = Rml::MakeUnique<BackendData>();
78 data->window = window;
79
80 Rml::Vector<const char*> extensions;
81 {
82 unsigned int count;
83 if (!SDL_Vulkan_GetInstanceExtensions(window, &count, nullptr))
84 {
85 data.reset();
86 Rml::Log::Message(Rml::Log::LT_ERROR, "Failed to get required vulkan extensions");
87 return false;
88 }
89 extensions.resize(count);
90 if (!SDL_Vulkan_GetInstanceExtensions(window, &count, extensions.data()))
91 {
92 data.reset();
93 Rml::Log::Message(Rml::Log::LT_ERROR, "Failed to get required vulkan extensions");
94 return false;
95 }
96 }
97
98 if (!data->render_interface.Initialize(std::move(extensions),
99 [](VkInstance instance, VkSurfaceKHR* out_surface) { return (bool)SDL_Vulkan_CreateSurface(data->window, instance, out_surface); }))
100 {
101 data.reset();
102 Rml::Log::Message(Rml::Log::LT_ERROR, "Failed to initialize Vulkan render interface");
103 return false;
104 }
105
106 data->system_interface.SetWindow(window);
107 data->render_interface.SetViewport(width, height);
108
109 return true;
110}

137static bool WaitForValidSwapchain()
138{
139 bool result = true;
140
141 // In some situations the swapchain may become invalid, such as when the window is minimized. In this state the renderer cannot accept any render
142 // calls. Since we don't have full control over the main loop here we may risk calls to Context::Render if we were to return. Instead, we keep the
143 // application inside this loop until we are able to recreate the swapchain and render again.
144 while (!data->render_interface.IsSwapchainValid())
145 {
146 SDL_Event ev;
147 while (SDL_PollEvent(&ev))
148 {
149 if (ev.type == SDL_QUIT)
150 {
151 // Restore the window so that we can recreate the swapchain, and then properly release all resource and shutdown cleanly.
152 SDL_RestoreWindow(data->window);
153 result = false;
154 }
155 }
156 SDL_Delay(10);
157 data->render_interface.RecreateSwapchain();
158 }
159
160 return result;
161}

163bool Backend::ProcessEvents(Rml::Context* context, KeyDownCallback key_down_callback, bool power_save)
164{
165 RMLUI_ASSERT(data && context);
166
167 bool result = data->running;
168 SDL_Event ev;
169
170 int has_event = 0;
171 if (power_save)
172 has_event = SDL_WaitEventTimeout(&ev, static_cast<int>(Rml::Math::Min(context->GetNextUpdateDelay(), 10.0) * 1000));
173 else
174 has_event = SDL_PollEvent(&ev);
175 while (has_event)
176 {
177 switch (ev.type)
178 {
179 case SDL_QUIT:
180 {
181 result = false;
182 }
183 break;
184 case SDL_KEYDOWN:
185 {
186 const Rml::Input::KeyIdentifier key = RmlSDL::ConvertKey(ev.key.keysym.sym);
187 const int key_modifier = RmlSDL::GetKeyModifierState();
188 const float native_dp_ratio = 1.f;
189
190 // See if we have any global shortcuts that take priority over the context.
191 if (key_down_callback && !key_down_callback(context, key, key_modifier, native_dp_ratio, true))
192 break;
193 // Otherwise, hand the event over to the context by calling the input handler as normal.
194 if (!RmlSDL::InputEventHandler(context, ev))
195 break;
196 // The key was not consumed by the context either, try keyboard shortcuts of lower priority.
197 if (key_down_callback && !key_down_callback(context, key, key_modifier, native_dp_ratio, false))
198 break;
199 }
200 break;
201 case SDL_WINDOWEVENT:
202 {
203 if (ev.window.event == SDL_WINDOWEVENT_SIZE_CHANGED)
204 {
205 Rml::Vector2i dimensions(ev.window.data1, ev.window.data2);
206 context->SetDimensions(dimensions);
207 data->render_interface.SetViewport(dimensions.x, dimensions.y);
208 break;
209 }
210 }
211 break;
212 default:
213 {
214 RmlSDL::InputEventHandler(context, ev);
215 }
216 break;
217 }
218 has_event = SDL_PollEvent(&ev);
219 }
220
221 if (!WaitForValidSwapchain())
222 result = false;
223
224 return result;
225}
