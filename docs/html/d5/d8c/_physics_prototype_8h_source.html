97// Layer that objects can be in, determines which other objects it can collide with
98// Typically you at least want to have 1 layer for moving bodies and 1 layer for static bodies, but you can have more
99// layers if you want. E.g. you could have a layer for high detail collision (which is not used by the physics simulation
100// but only if you do collision testing).
101namespace Layers
102{
103 static constexpr JPH::ObjectLayer NON_MOVING = 0;
104 static constexpr JPH::ObjectLayer MOVING = 1;
105 static constexpr JPH::ObjectLayer TERRIAN = 2;
106 static constexpr JPH::ObjectLayer NUM_LAYERS = 3;
107};

110class ObjectLayerPairFilterImpl : public JPH::ObjectLayerPairFilter
111{
112public:
113 virtual bool ShouldCollide(JPH::ObjectLayer inObject1, JPH::ObjectLayer inObject2) const override
114 {
115 using namespace JPH;
116 switch (inObject1)
117 {
118 case Layers::NON_MOVING:
119 return inObject2 == Layers::MOVING; // Non moving only collides with moving
120 case Layers::MOVING:
121 return true; // Moving collides with everything
122 case Layers::TERRIAN:
123 return inObject2 != Layers::TERRIAN;
124 default:
125 JPH_ASSERT(false);
126 return false;
127 }
128 }

131// Each broadphase layer results in a separate bounding volume tree in the broad phase. You at least want to have
132// a layer for non-moving and moving objects to avoid having to update a tree full of static objects every frame.
133// You can have a 1-on-1 mapping between object layers and broadphase layers (like in this case) but if you have
134// many object layers you'll be creating many broad phase trees, which is not efficient. If you want to fine tune
135// your broadphase layers define JPH_TRACK_BROADPHASE_STATS and look at the stats reported on the TTY.
136namespace BroadPhaseLayers
137{
138 static constexpr JPH::BroadPhaseLayer NON_MOVING(0);
139 static constexpr JPH::BroadPhaseLayer MOVING(1);
140 static constexpr JPH::uint NUM_LAYERS(2);
141};
