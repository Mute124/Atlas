118static const char* shader_frag_gradient = RMLUI_SHADER_HEADER R"(
119#define LINEAR 0
120#define RADIAL 1
121#define CONIC 2
122#define REPEATING_LINEAR 3
123#define REPEATING_RADIAL 4
124#define REPEATING_CONIC 5
125#define PI 3.14159265
126
127uniform int _func; // one of the above definitions
128uniform vec2 _p; // linear: starting point, radial: center, conic: center
129uniform vec2 _v; // linear: vector to ending point, radial: 2d curvature (inverse radius), conic: angled unit vector
130uniform vec4 _stop_colors[MAX_NUM_STOPS];
131uniform float _stop_positions[MAX_NUM_STOPS]; // normalized, 0 -> starting point, 1 -> ending point
132uniform int _num_stops;
133
134in vec2 fragTexCoord;
135in vec4 fragColor;
136out vec4 finalColor;
137
138vec4 mix_stop_colors(float t) {
139 vec4 color = _stop_colors[0];
140
141 for (int i = 1; i < _num_stops; i++)
142 color = mix(color, _stop_colors[i], smoothstep(_stop_positions[i-1], _stop_positions[i], t));
143
144 return color;
145}
146
147void main() {
148 float t = 0.0;
149
150 if (_func == LINEAR || _func == REPEATING_LINEAR)
151 {
152 float dist_square = dot(_v, _v);
153 vec2 V = fragTexCoord - _p;
154 t = dot(_v, V) / dist_square;
155 }
156 else if (_func == RADIAL || _func == REPEATING_RADIAL)
157 {
158 vec2 V = fragTexCoord - _p;
159 t = length(_v * V);
160 }
161 else if (_func == CONIC || _func == REPEATING_CONIC)
162 {
163 mat2 R = mat2(_v.x, -_v.y, _v.y, _v.x);
164 vec2 V = R * (fragTexCoord - _p);
165 t = 0.5 + atan(-V.x, V.y) / (2.0 * PI);
166 }
167
168 if (_func == REPEATING_LINEAR || _func == REPEATING_RADIAL || _func == REPEATING_CONIC)
169 {
170 float t0 = _stop_positions[0];
171 float t1 = _stop_positions[_num_stops - 1];
172 t = t0 + mod(t - t0, t1 - t0);
173 }
174
175 finalColor = fragColor * mix_stop_colors(t);
176}
177)";

233void main() {
234 // The general case uses a 4x5 color matrix for full rgba transformation, plus a constant term with the last column.

443class Uniforms {

485static void CheckGLError(const char* operation_name)
486{
487#ifdef RMLUI_DEBUG
488 GLenum error_code = glGetError();
489 if (error_code != GL_NO_ERROR)
490 {
491 static const Rml::Pair<GLenum, const char*> error_names[] = {{GL_INVALID_ENUM, "GL_INVALID_ENUM"}, {GL_INVALID_VALUE, "GL_INVALID_VALUE"},
492 {GL_INVALID_OPERATION, "GL_INVALID_OPERATION"}, {GL_OUT_OF_MEMORY, "GL_OUT_OF_MEMORY"}};
493 const char* error_str = "''";
494 for (auto& err : error_names)
495 {
496 if (err.first == error_code)
497 {
498 error_str = err.second;
499 break;
500 }
501 }
502 Rml::Log::Message(Rml::Log::LT_ERROR, "OpenGL error during %s. Error code 0x%x (%s).", operation_name, error_code, error_str);
503 }
504#endif
505 (void)operation_name;
506}

508// Create the shader, 'shader_type' is either GL_VERTEX_SHADER or GL_FRAGMENT_SHADER.
509static bool CreateShader(GLuint& out_shader_id, GLenum shader_type, const char* code_string)
510{
511 RMLUI_ASSERT(shader_type == GL_VERTEX_SHADER || shader_type == GL_FRAGMENT_SHADER);
512
513 GLuint id = glCreateShader(shader_type);
514 glShaderSource(id, 1, (const GLchar**)&code_string, NULL);
515 glCompileShader(id);
516
517 GLint status = 0;
518 glGetShaderiv(id, GL_COMPILE_STATUS, &status);
519 if (status == GL_FALSE)
520 {
521 GLint info_log_length = 0;
522 glGetShaderiv(id, GL_INFO_LOG_LENGTH, &info_log_length);
523 char* info_log_string = new char[info_log_length + 1];
524 glGetShaderInfoLog(id, info_log_length, NULL, info_log_string);
525
526 Rml::Log::Message(Rml::Log::LT_ERROR, "Compile failure in OpenGL shader: %s", info_log_string);
527 delete[] info_log_string;
528 glDeleteShader(id);
529 return false;
530 }
531
532 CheckGLError("CreateShader");
533
534 out_shader_id = id;
535 return true;
536}

538static bool CreateProgram(GLuint& out_program, Uniforms& inout_uniform_map, ProgramId program_id, GLuint vertex_shader, GLuint fragment_shader)
539{
540 GLuint id = glCreateProgram();
541 RMLUI_ASSERT(id);
542
543 for (GLuint i = 0; i < (GLuint)VertexAttribute::Count; i++)
544 glBindAttribLocation(id, i, vertex_attribute_names[i]);
545
546 CheckGLError("BindAttribLocations");
547
548 glAttachShader(id, vertex_shader);
549 glAttachShader(id, fragment_shader);
550
551 glLinkProgram(id);
552
553 glDetachShader(id, vertex_shader);
554 glDetachShader(id, fragment_shader);
555
556 GLint status = 0;
557 glGetProgramiv(id, GL_LINK_STATUS, &status);
558 if (status == GL_FALSE)
559 {
560 GLint info_log_length = 0;
561 glGetProgramiv(id, GL_INFO_LOG_LENGTH, &info_log_length);
562 char* info_log_string = new char[info_log_length + 1];
563 glGetProgramInfoLog(id, info_log_length, NULL, info_log_string);
564
565 Rml::Log::Message(Rml::Log::LT_ERROR, "OpenGL program linking failure: %s", info_log_string);
566 delete[] info_log_string;
567 glDeleteProgram(id);
568 return false;
569 }
570
571 out_program = id;
572
573 // Make a lookup table for the uniform locations.
574 GLint num_active_uniforms = 0;
575 glGetProgramiv(id, GL_ACTIVE_UNIFORMS, &num_active_uniforms);
576
577 constexpr size_t name_size = 64;
578 GLchar name_buf[name_size] = "";
579 for (int unif = 0; unif < num_active_uniforms; ++unif)
580 {
581 GLint array_size = 0;
582 GLenum type = 0;
583 GLsizei actual_length = 0;
584 glGetActiveUniform(id, unif, name_size, &actual_length, &array_size, &type, name_buf);
585 GLint location = glGetUniformLocation(id, name_buf);
586
587 // See if we have the name in our pre-defined name list.
588 UniformId program_uniform = UniformId::Count;
589 for (int i = 0; i < (int)UniformId::Count; i++)
590 {
591 const char* uniform_name = program_uniform_names[i];
592 if (strcmp(name_buf, uniform_name) == 0)
593 {
594 program_uniform = (UniformId)i;
595 break;
596 }
597 }
598
599 if ((size_t)program_uniform < (size_t)UniformId::Count)
600 {
601 inout_uniform_map.Insert(program_id, program_uniform, location);
602 }
603 else
604 {
605 Rml::Log::Message(Rml::Log::LT_ERROR, "OpenGL program uses unknown uniform '%s'.", name_buf);
606 return false;
607 }
608 }
609
610 CheckGLError("CreateProgram");
611
612 return true;
613}

615static bool CreateFramebuffer(FramebufferData& out_fb, int width, int height, int samples, FramebufferAttachment attachment,
616 GLuint shared_depth_stencil_buffer)
617{
618#ifdef RMLUI_PLATFORM_EMSCRIPTEN
619 constexpr GLint wrap_mode = GL_CLAMP_TO_EDGE;
620#else
621 constexpr GLint wrap_mode = GL_CLAMP_TO_BORDER; // GL_REPEAT GL_MIRRORED_REPEAT GL_CLAMP_TO_EDGE
622#endif
623
624 constexpr GLenum color_format = GL_RGBA8; // GL_RGBA8 GL_SRGB8_ALPHA8 GL_RGBA16F
625 constexpr GLint min_mag_filter = GL_LINEAR; // GL_NEAREST
626 const Rml::Colourf border_color(0.f, 0.f);
627
628 GLuint framebuffer = 0;
629 glGenFramebuffers(1, &framebuffer);
630 glBindFramebuffer(GL_FRAMEBUFFER, framebuffer);
631
632 GLuint color_tex_buffer = 0;
633 GLuint color_render_buffer = 0;
634 if (samples > 0)
635 {
636 glGenRenderbuffers(1, &color_render_buffer);
637 glBindRenderbuffer(GL_RENDERBUFFER, color_render_buffer);
638 glRenderbufferStorageMultisample(GL_RENDERBUFFER, samples, color_format, width, height);
639 glFramebufferRenderbuffer(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_RENDERBUFFER, color_render_buffer);
640 }
641 else
642 {
643 glGenTextures(1, &color_tex_buffer);
644 glBindTexture(GL_TEXTURE_2D, color_tex_buffer);
645 glTexImage2D(GL_TEXTURE_2D, 0, color_format, width, height, 0, GL_RGBA, GL_UNSIGNED_BYTE, NULL);
646
647 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, min_mag_filter);
648 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, min_mag_filter);
649 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, wrap_mode);
650 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, wrap_mode);
651#ifndef RMLUI_PLATFORM_EMSCRIPTEN
652 glTexParameterfv(GL_TEXTURE_2D, GL_TEXTURE_BORDER_COLOR, &border_color[0]);
653#endif
654
655 glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, color_tex_buffer, 0);
656 }
657
658 // Create depth/stencil buffer storage attachment.
659 GLuint depth_stencil_buffer = 0;
660 if (attachment != FramebufferAttachment::None)
661 {
662 if (shared_depth_stencil_buffer)
663 {
664 // Share depth/stencil buffer
665 depth_stencil_buffer = shared_depth_stencil_buffer;
666 }
667 else
668 {
669 // Create new depth/stencil buffer
670 glGenRenderbuffers(1, &depth_stencil_buffer);
671 glBindRenderbuffer(GL_RENDERBUFFER, depth_stencil_buffer);
672
673 glRenderbufferStorageMultisample(GL_RENDERBUFFER, samples, GL_DEPTH24_STENCIL8, width, height);
674 }
675
676 glFramebufferRenderbuffer(GL_FRAMEBUFFER, GL_DEPTH_STENCIL_ATTACHMENT, GL_RENDERBUFFER, depth_stencil_buffer);
677 }
678
679 const GLuint framebuffer_status = glCheckFramebufferStatus(GL_FRAMEBUFFER);
680 if (framebuffer_status != GL_FRAMEBUFFER_COMPLETE)
681 {
682 Rml::Log::Message(Rml::Log::LT_ERROR, "OpenGL framebuffer could not be generated. Error code %x.", framebuffer_status);
683 return false;
684 }
685
686 glBindFramebuffer(GL_FRAMEBUFFER, 0);
687 glBindTexture(GL_TEXTURE_2D, 0);
688 glBindRenderbuffer(GL_RENDERBUFFER, 0);
689
690 CheckGLError("CreateFramebuffer");
691
692 out_fb = {};
693 out_fb.width = width;
694 out_fb.height = height;
695 out_fb.framebuffer = framebuffer;
696 out_fb.color_tex_buffer = color_tex_buffer;
697 out_fb.color_render_buffer = color_render_buffer;
698 out_fb.depth_stencil_buffer = depth_stencil_buffer;
699 out_fb.owns_depth_stencil_buffer = !shared_depth_stencil_buffer;
700
701 return true;
702}

728static bool CreateShaders(ProgramData& data)
729{
730 RMLUI_ASSERT(std::all_of(data.vert_shaders.begin(), data.vert_shaders.end(), [](auto&& value) { return value == 0; }));
731 RMLUI_ASSERT(std::all_of(data.frag_shaders.begin(), data.frag_shaders.end(), [](auto&& value) { return value == 0; }));
732 RMLUI_ASSERT(std::all_of(data.programs.begin(), data.programs.end(), [](auto&& value) { return value == 0; }));
733 auto ReportError = [](const char* type, const char* name) {
734 Rml::Log::Message(Rml::Log::LT_ERROR, "Could not create OpenGL %s: '%s'.", type, name);
735 return false;
736 };
737
738 for (const VertShaderDefinition& def : vert_shader_definitions)
739 {
740 if (!CreateShader(data.vert_shaders[def.id], GL_VERTEX_SHADER, def.code_str))
741 return ReportError("vertex shader", def.name_str);
742 }
743
744 for (const FragShaderDefinition& def : frag_shader_definitions)
745 {
746 if (!CreateShader(data.frag_shaders[def.id], GL_FRAGMENT_SHADER, def.code_str))
747 return ReportError("fragment shader", def.name_str);
748 }
749
750 for (const ProgramDefinition& def : program_definitions)
751 {
752 if (!CreateProgram(data.programs[def.id], data.uniforms, def.id, data.vert_shaders[def.vert_shader], data.frag_shaders[def.frag_shader]))
753 return ReportError("program", def.name_str);
754 }
755
756 glUseProgram(data.programs[ProgramId::BlendMask]);
757 glUniform1i(data.uniforms.Get(ProgramId::BlendMask, UniformId::TexMask), 1);
758
759 glUseProgram(0);
760
761 return true;
762}

832 glGetIntegerv(GL_BLEND_EQUATION_RGB, &glstate_backup.blend_equation_rgb);
833 glGetIntegerv(GL_BLEND_EQUATION_ALPHA, &glstate_backup.blend_equation_alpha);

880

882
883 SetTransform(nullptr);
884

886 glBindFramebuffer(GL_FRAMEBUFFER, render_layers.GetTopLayer().framebuffer);
887 glClear(GL_COLOR_BUFFER_BIT);
888
889 UseProgram(ProgramId::None);

896void RenderInterface_GL3::EndFrame()
897{
898 const Gfx::FramebufferData& fb_active = render_layers.GetTopLayer();
899 const Gfx::FramebufferData& fb_postprocess = render_layers.GetPostprocessPrimary();
900
901 // Resolve MSAA to postprocess framebuffer.
902 glBindFramebuffer(GL_READ_FRAMEBUFFER, fb_active.framebuffer);
903 glBindFramebuffer(GL_DRAW_FRAMEBUFFER, fb_postprocess.framebuffer);
904
905 glBlitFramebuffer(0, 0, fb_active.width, fb_active.height, 0, 0, fb_postprocess.width, fb_postprocess.height, GL_COLOR_BUFFER_BIT, GL_NEAREST);
906
907 // Draw to backbuffer
908 glBindFramebuffer(GL_FRAMEBUFFER, 0);
909
910 // Assuming we have an opaque background, we can just write to it with the premultiplied alpha blend mode and we'll get the correct result.
911 // Instead, if we had a transparent destination that didn't use premultiplied alpha, we would need to perform a manual un-premultiplication step.
912 glActiveTexture(GL_TEXTURE0);
913 Gfx::BindTexture(fb_postprocess);
914 UseProgram(ProgramId::Passthrough);
915 DrawFullscreenQuad();
916
917 render_layers.EndFrame();
918
919 // Restore GL state.
920 if (glstate_backup.enable_cull_face)
921 glEnable(GL_CULL_FACE);
922 else
923 glDisable(GL_CULL_FACE);
924
925 if (glstate_backup.enable_blend)
926 glEnable(GL_BLEND);
927 else
928 glDisable(GL_BLEND);
929
930 if (glstate_backup.enable_stencil_test)
931 glEnable(GL_STENCIL_TEST);
932 else
933 glDisable(GL_STENCIL_TEST);
934
935 if (glstate_backup.enable_scissor_test)
936 glEnable(GL_SCISSOR_TEST);
937 else
938 glDisable(GL_SCISSOR_TEST);
939
940 if (glstate_backup.enable_depth_test)
941 glEnable(GL_DEPTH_TEST);
942 else
943 glDisable(GL_DEPTH_TEST);
944
945 glViewport(glstate_backup.viewport[0], glstate_backup.viewport[1], glstate_backup.viewport[2], glstate_backup.viewport[3]);
946 glScissor(glstate_backup.scissor[0], glstate_backup.scissor[1], glstate_backup.scissor[2], glstate_backup.scissor[3]);
947
948 glActiveTexture(glstate_backup.active_texture);
949
950 glClearStencil(glstate_backup.stencil_clear_value);
951 glClearColor(glstate_backup.color_clear_value[0], glstate_backup.color_clear_value[1], glstate_backup.color_clear_value[2],
952 glstate_backup.color_clear_value[3]);
953 glColorMask(glstate_backup.color_writemask[0], glstate_backup.color_writemask[1], glstate_backup.color_writemask[2],
954 glstate_backup.color_writemask[3]);
955
956 glBlendEquationSeparate(glstate_backup.blend_equation_rgb, glstate_backup.blend_equation_alpha);
957 glBlendFuncSeparate(glstate_backup.blend_src_rgb, glstate_backup.blend_dst_rgb, glstate_backup.blend_src_alpha, glstate_backup.blend_dst_alpha);
958
959 glStencilFuncSeparate(GL_FRONT, glstate_backup.stencil_front.func, glstate_backup.stencil_front.ref, glstate_backup.stencil_front.value_mask);
960 glStencilMaskSeparate(GL_FRONT, glstate_backup.stencil_front.writemask);
961 glStencilOpSeparate(GL_FRONT, glstate_backup.stencil_front.fail, glstate_backup.stencil_front.pass_depth_fail,
962 glstate_backup.stencil_front.pass_depth_pass);
963
964 glStencilFuncSeparate(GL_BACK, glstate_backup.stencil_back.func, glstate_backup.stencil_back.ref, glstate_backup.stencil_back.value_mask);
965 glStencilMaskSeparate(GL_BACK, glstate_backup.stencil_back.writemask);
966 glStencilOpSeparate(GL_BACK, glstate_backup.stencil_back.fail, glstate_backup.stencil_back.pass_depth_fail,
967 glstate_backup.stencil_back.pass_depth_pass);
968
969 Gfx::CheckGLError("EndFrame");
970}

1123void RenderInterface_GL3::RenderToClipMask(Rml::ClipMaskOperation operation, Rml::CompiledGeometryHandle geometry, Rml::Vector2f translation)
1124{
1125 RMLUI_ASSERT(glIsEnabled(GL_STENCIL_TEST));
1126 using Rml::ClipMaskOperation;
1127
1128 const bool clear_stencil = (operation == ClipMaskOperation::Set || operation == ClipMaskOperation::SetInverse);
1129 if (clear_stencil)
1130 {
1131 // @performance Increment the reference value instead of clearing each time.
1132 glClear(GL_STENCIL_BUFFER_BIT);
1133 }
1134
1135 GLint stencil_test_value = 0;
1136 glGetIntegerv(GL_STENCIL_REF, &stencil_test_value);
1137
1138 glColorMask(GL_FALSE, GL_FALSE, GL_FALSE, GL_FALSE);
1139 glStencilFunc(GL_ALWAYS, GLint(1), GLuint(-1));
1140
1141 switch (operation)
1142 {
1143 case ClipMaskOperation::Set:
1144 {
1145 glStencilOp(GL_KEEP, GL_KEEP, GL_REPLACE);
1146 stencil_test_value = 1;
1147 }
1148 break;
1149 case ClipMaskOperation::SetInverse:
1150 {
1151 glStencilOp(GL_KEEP, GL_KEEP, GL_REPLACE);
1152 stencil_test_value = 0;
1153 }
1154 break;
1155 case ClipMaskOperation::Intersect:
1156 {
1157 glStencilOp(GL_KEEP, GL_KEEP, GL_INCR);
1158 stencil_test_value += 1;
1159 }
1160 break;
1161 }
1162
1163 RenderGeometry(geometry, translation, {});
1164
1165 // Restore state
1166 // @performance Cache state so we don't toggle it unnecessarily.
1167 glColorMask(GL_TRUE, GL_TRUE, GL_TRUE, GL_TRUE);
1168 glStencilOp(GL_KEEP, GL_KEEP, GL_KEEP);
1169 glStencilFunc(GL_EQUAL, stencil_test_value, GLuint(-1));
1170}

1191Rml::TextureHandle RenderInterface_GL3::LoadTexture(Rml::Vector2i& texture_dimensions, const Rml::String& source)
1192{
1193 Rml::FileInterface* file_interface = Rml::GetFileInterface();
1194 Rml::FileHandle file_handle = file_interface->Open(source);
1195 if (!file_handle)
1196 {
1197 return false;
1198 }
1199
1200 file_interface->Seek(file_handle, 0, SEEK_END);
1201 size_t buffer_size = file_interface->Tell(file_handle);
1202 file_interface->Seek(file_handle, 0, SEEK_SET);
1203
1204 if (buffer_size <= sizeof(TGAHeader))
1205 {
1206 Rml::Log::Message(Rml::Log::LT_ERROR, "Texture file size is smaller than TGAHeader, file is not a valid TGA image.");
1207 file_interface->Close(file_handle);
1208 return false;
1209 }
1210
1211 using Rml::byte;
1212 Rml::UniquePtr<byte[]> buffer(new byte[buffer_size]);
1213 file_interface->Read(buffer.get(), buffer_size, file_handle);
1214 file_interface->Close(file_handle);
1215
1216 TGAHeader header;
1217 memcpy(&header, buffer.get(), sizeof(TGAHeader));
1218
1219 int color_mode = header.bitsPerPixel / 8;
1220 const size_t image_size = header.width * header.height * 4; // We always make 32bit textures
1221
1222 if (header.dataType != 2)
1223 {
1224 Rml::Log::Message(Rml::Log::LT_ERROR, "Only 24/32bit uncompressed TGAs are supported.");
1225 return false;
1226 }
1227
1228 // Ensure we have at least 3 colors
1229 if (color_mode < 3)
1230 {
1231 Rml::Log::Message(Rml::Log::LT_ERROR, "Only 24 and 32bit textures are supported.");
1232 return false;
1233 }
1234
1235 const byte* image_src = buffer.get() + sizeof(TGAHeader);
1236 Rml::UniquePtr<byte[]> image_dest_buffer(new byte[image_size]);
1237 byte* image_dest = image_dest_buffer.get();
1238
1239 // Targa is BGR, swap to RGB, flip Y axis, and convert to premultiplied alpha.
1240 for (long y = 0; y < header.height; y++)
1241 {
1242 long read_index = y * header.width * color_mode;
1243 long write_index = ((header.imageDescriptor & 32) != 0) ? read_index : (header.height - y - 1) * header.width * 4;
1244 for (long x = 0; x < header.width; x++)
1245 {
1246 image_dest[write_index] = image_src[read_index + 2];
1247 image_dest[write_index + 1] = image_src[read_index + 1];
1248 image_dest[write_index + 2] = image_src[read_index];
1249 if (color_mode == 4)
1250 {
1251 const byte alpha = image_src[read_index + 3];
1252 for (size_t j = 0; j < 3; j++)
1253 image_dest[write_index + j] = byte((image_dest[write_index + j] * alpha) / 255);
1254 image_dest[write_index + 3] = alpha;
1255 }
1256 else
1257 image_dest[write_index + 3] = 255;
1258
1259 write_index += 4;
1260 read_index += color_mode;
1261 }
1262 }
1263
1264 texture_dimensions.x = header.width;
1265 texture_dimensions.y = header.height;
1266
1267 return GenerateTexture({image_dest, image_size}, texture_dimensions);
1268}

1321static void SigmaToParameters(const float desired_sigma, int& out_pass_level, float& out_sigma)
1322{
1323 constexpr int max_num_passes = 10;
1324 static_assert(max_num_passes < 31, "");
1325 constexpr float max_single_pass_sigma = 3.0f;
1326 out_pass_level = Rml::Math::Clamp(Rml::Math::Log2(int(desired_sigma * (2.f / max_single_pass_sigma))), 0, max_num_passes);
1327 out_sigma = Rml::Math::Clamp(desired_sigma / float(1 << out_pass_level), 0.0f, max_single_pass_sigma);
1328}

1330static void SetTexCoordLimits(GLint tex_coord_min_location, GLint tex_coord_max_location, Rml::Rectanglei rectangle_flipped,
1331 Rml::Vector2i framebuffer_size)
1332{
1333 // Offset by half-texel values so that texture lookups are clamped to fragment centers, thereby avoiding color
1334 // bleeding from neighboring texels due to bilinear interpolation.
1335 const Rml::Vector2f min = (Rml::Vector2f(rectangle_flipped.p0) + Rml::Vector2f(0.5f)) / Rml::Vector2f(framebuffer_size);
1336 const Rml::Vector2f max = (Rml::Vector2f(rectangle_flipped.p1) - Rml::Vector2f(0.5f)) / Rml::Vector2f(framebuffer_size);
1337
1338 glUniform2f(tex_coord_min_location, min.x, min.y);
1339 glUniform2f(tex_coord_max_location, max.x, max.y);
1340}

1342static void SetBlurWeights(GLint weights_location, float sigma)
1343{
1344 constexpr int num_weights = BLUR_NUM_WEIGHTS;
1345 float weights[num_weights];
1346 float normalization = 0.0f;
1347 for (int i = 0; i < num_weights; i++)
1348 {
1349 if (Rml::Math::Absolute(sigma) < 0.1f)
1350 weights[i] = float(i == 0);
1351 else
1352 weights[i] = Rml::Math::Exp(-float(i * i) / (2.0f * sigma * sigma)) / (Rml::Math::SquareRoot(2.f * Rml::Math::RMLUI_PI) * sigma);
1353
1354 normalization += (i == 0 ? 1.f : 2.0f) * weights[i];
1355 }
1356 for (int i = 0; i < num_weights; i++)
1357 weights[i] /= normalization;
1358
1359 glUniform1fv(weights_location, (GLsizei)num_weights, &weights[0]);
1360}

1362void RenderInterface_GL3::RenderBlur(float sigma, const Gfx::FramebufferData& source_destination, const Gfx::FramebufferData& temp,
1363 const Rml::Rectanglei window_flipped)
1364{
1365 RMLUI_ASSERT(&source_destination != &temp && source_destination.width == temp.width && source_destination.height == temp.height);
1366 RMLUI_ASSERT(window_flipped.Valid());
1367
1368 int pass_level = 0;
1369 SigmaToParameters(sigma, pass_level, sigma);
1370
1371 const Rml::Rectanglei original_scissor = scissor_state;
1372
1373 // Begin by downscaling so that the blur pass can be done at a reduced resolution for large sigma.
1374 Rml::Rectanglei scissor = window_flipped;
1375
1376 UseProgram(ProgramId::Passthrough);
1377 SetScissor(scissor, true);
1378
1379 // Downscale by iterative half-scaling with bilinear filtering, to reduce aliasing.
1380 glViewport(0, 0, source_destination.width / 2, source_destination.height / 2);
1381
1382 // Scale UVs if we have even dimensions, such that texture fetches align perfectly between texels, thereby producing a 50% blend of
1383 // neighboring texels.
1384 const Rml::Vector2f uv_scaling = {(source_destination.width % 2 == 1) ? (1.f - 1.f / float(source_destination.width)) : 1.f,
1385 (source_destination.height % 2 == 1) ? (1.f - 1.f / float(source_destination.height)) : 1.f};
1386
1387 for (int i = 0; i < pass_level; i++)
1388 {
1389 scissor.p0 = (scissor.p0 + Rml::Vector2i(1)) / 2;
1390 scissor.p1 = Rml::Math::Max(scissor.p1 / 2, scissor.p0);
1391 const bool from_source = (i % 2 == 0);
1392 Gfx::BindTexture(from_source ? source_destination : temp);
1393 glBindFramebuffer(GL_FRAMEBUFFER, (from_source ? temp : source_destination).framebuffer);
1394 SetScissor(scissor, true);
1395
1396 DrawFullscreenQuad({}, uv_scaling);
1397 }
1398
1399 glViewport(0, 0, source_destination.width, source_destination.height);
1400
1401 // Ensure texture data end up in the temp buffer. Depending on the last downscaling, we might need to move it from the source_destination buffer.
1402 const bool transfer_to_temp_buffer = (pass_level % 2 == 0);
1403 if (transfer_to_temp_buffer)
1404 {
1405 Gfx::BindTexture(source_destination);
1406 glBindFramebuffer(GL_FRAMEBUFFER, temp.framebuffer);
1407 DrawFullscreenQuad();
1408 }
1409
1410 // Set up uniforms.
1411 UseProgram(ProgramId::Blur);
1412 SetBlurWeights(GetUniformLocation(UniformId::Weights), sigma);
1413 SetTexCoordLimits(GetUniformLocation(UniformId::TexCoordMin), GetUniformLocation(UniformId::TexCoordMax), scissor,
1414 {source_destination.width, source_destination.height});
1415
1416 const GLint texel_offset_location = GetUniformLocation(UniformId::TexelOffset);
1417 auto SetTexelOffset = [texel_offset_location](Rml::Vector2f blur_direction, int texture_dimension) {
1418 const Rml::Vector2f texel_offset = blur_direction * (1.0f / float(texture_dimension));
1419 glUniform2f(texel_offset_location, texel_offset.x, texel_offset.y);
1420 };
1421
1422 // Blur render pass - vertical.
1423 Gfx::BindTexture(temp);
1424 glBindFramebuffer(GL_FRAMEBUFFER, source_destination.framebuffer);
1425
1426 SetTexelOffset({0.f, 1.f}, temp.height);
1427 DrawFullscreenQuad();
1428
1429 // Blur render pass - horizontal.
1430 Gfx::BindTexture(source_destination);
1431 glBindFramebuffer(GL_FRAMEBUFFER, temp.framebuffer);
1432
1433 // Add a 1px transparent border around the blur region by first clearing with a padded scissor. This helps prevent
1434 // artifacts when upscaling the blur result in the later step. On Intel and AMD, we have observed that during
1435 // blitting with linear filtering, pixels outside the 'src' region can be blended into the output. On the other
1436 // hand, it looks like Nvidia clamps the pixels to the source edge, which is what we really want. Regardless, we
1437 // work around the issue with this extra step.
1438 SetScissor(scissor.Extend(1), true);
1439 glClear(GL_COLOR_BUFFER_BIT);
1440 SetScissor(scissor, true);
1441
1442 SetTexelOffset({1.f, 0.f}, source_destination.width);
1443 DrawFullscreenQuad();
1444
1445 // Blit the blurred image to the scissor region with upscaling.
1446 SetScissor(window_flipped, true);
1447 glBindFramebuffer(GL_READ_FRAMEBUFFER, temp.framebuffer);
1448 glBindFramebuffer(GL_DRAW_FRAMEBUFFER, source_destination.framebuffer);
1449
1450 const Rml::Vector2i src_min = scissor.p0;
1451 const Rml::Vector2i src_max = scissor.p1;
1452 const Rml::Vector2i dst_min = window_flipped.p0;
1453 const Rml::Vector2i dst_max = window_flipped.p1;
1454 glBlitFramebuffer(src_min.x, src_min.y, src_max.x, src_max.y, dst_min.x, dst_min.y, dst_max.x, dst_max.y, GL_COLOR_BUFFER_BIT, GL_LINEAR);
1455
1456 // The above upscale blit might be jittery at low resolutions (large pass levels). This is especially noticeable when moving an element with
1457 // backdrop blur around or when trying to click/hover an element within a blurred region since it may be rendered at an offset. For more stable
1458 // and accurate rendering we next upscale the blur image by an exact power-of-two. However, this may not fill the edges completely so we need to
1459 // do the above first. Note that this strategy may sometimes result in visible seams. Alternatively, we could try to enlarge the window to the
1460 // next power-of-two size and then downsample and blur that.
1461 const Rml::Vector2i target_min = src_min * (1 << pass_level);
1462 const Rml::Vector2i target_max = src_max * (1 << pass_level);
1463 if (target_min != dst_min || target_max != dst_max)
1464 {
1465 glBlitFramebuffer(src_min.x, src_min.y, src_max.x, src_max.y, target_min.x, target_min.y, target_max.x, target_max.y, GL_COLOR_BUFFER_BIT,
1466 GL_LINEAR);
1467 }
1468
1469 // Restore render state.
1470 SetScissor(original_scissor);
1471
1472 Gfx::CheckGLError("Blur");
1473}

1702void RenderInterface_GL3::RenderShader(Rml::CompiledShaderHandle shader_handle, Rml::CompiledGeometryHandle geometry_handle,
1703 Rml::Vector2f translation, Rml::TextureHandle /*texture*/)
1704{
1705 RMLUI_ASSERT(shader_handle && geometry_handle);
1706 const CompiledShader& shader = *reinterpret_cast<CompiledShader*>(shader_handle);
1707 const CompiledShaderType type = shader.type;
1708 const Gfx::CompiledGeometryData& geometry = *reinterpret_cast<Gfx::CompiledGeometryData*>(geometry_handle);
1709
1710 switch (type)
1711 {
1712 case CompiledShaderType::Gradient:
1713 {
1714 RMLUI_ASSERT(shader.stop_positions.size() == shader.stop_colors.size());
1715 const int num_stops = (int)shader.stop_positions.size();
1716
1717 UseProgram(ProgramId::Gradient);
1718 glUniform1i(GetUniformLocation(UniformId::Func), static_cast<int>(shader.gradient_function));
1719 glUniform2f(GetUniformLocation(UniformId::P), shader.p.x, shader.p.y);
1720 glUniform2f(GetUniformLocation(UniformId::V), shader.v.x, shader.v.y);
1721 glUniform1i(GetUniformLocation(UniformId::NumStops), num_stops);
1722 glUniform1fv(GetUniformLocation(UniformId::StopPositions), num_stops, shader.stop_positions.data());

1780 case FilterType::Passthrough:
1781 {

1907void RenderInterface_GL3::CompositeLayers(Rml::LayerHandle source_handle, Rml::LayerHandle destination_handle, Rml::BlendMode blend_mode,
1908 Rml::Span<const Rml::CompiledFilterHandle> filters)
1909{
1910 using Rml::BlendMode;
1911
1912 // Blit source layer to postprocessing buffer. Do this regardless of whether we actually have any filters to be
1913 // applied, because we need to resolve the multi-sampled framebuffer in any case.
1914 // @performance If we have BlendMode::Replace and no filters or mask then we can just blit directly to the destination.
1915 BlitLayerToPostprocessPrimary(source_handle);

1918 RenderFilters(filters);
1919
1920 // Render to the destination layer.
1921 glBindFramebuffer(GL_FRAMEBUFFER, render_layers.GetLayer(destination_handle).framebuffer);
1922 Gfx::BindTexture(render_layers.GetPostprocessPrimary());
1923
1924 UseProgram(ProgramId::Passthrough);
1925
1926 if (blend_mode == BlendMode::Replace)
1927 glDisable(GL_BLEND);
1928
1929 DrawFullscreenQuad();
1930
1931 if (blend_mode == BlendMode::Replace)
1932 glEnable(GL_BLEND);
1933
1934 if (destination_handle != render_layers.GetTopLayerHandle())
1935 glBindFramebuffer(GL_FRAMEBUFFER, render_layers.GetTopLayer().framebuffer);
1936
1937 Gfx::CheckGLError("CompositeLayers");

1946Rml::TextureHandle RenderInterface_GL3::SaveLayerAsTexture()
1947{
1948 RMLUI_ASSERT(scissor_state.Valid());
1949 const Rml::Rectanglei bounds = scissor_state;
1950
1951 Rml::TextureHandle render_texture = GenerateTexture({}, bounds.Size());
1952 if (!render_texture)
1953 return {};
1954
1955 BlitLayerToPostprocessPrimary(render_layers.GetTopLayerHandle());
1956
1957 EnableScissorRegion(false);
1958
1959 const Gfx::FramebufferData& source = render_layers.GetPostprocessPrimary();
1960 const Gfx::FramebufferData& destination = render_layers.GetPostprocessSecondary();
1961 glBindFramebuffer(GL_READ_FRAMEBUFFER, source.framebuffer);
1962 glBindFramebuffer(GL_DRAW_FRAMEBUFFER, destination.framebuffer);
1963
1964 // Flip the image vertically, as that convention is used for textures, and move to origin.
1965 glBlitFramebuffer( //
1966 bounds.Left(), source.height - bounds.Bottom(), // src0
1967 bounds.Right(), source.height - bounds.Top(), // src1
1968 0, bounds.Height(), // dst0
1969 bounds.Width(), 0, // dst1
1970 GL_COLOR_BUFFER_BIT, GL_NEAREST //
1971 );
1972
1973 glBindTexture(GL_TEXTURE_2D, (GLuint)render_texture);
1974
1975 const Gfx::FramebufferData& texture_source = destination;
1976 glBindFramebuffer(GL_READ_FRAMEBUFFER, texture_source.framebuffer);
1977 glCopyTexSubImage2D(GL_TEXTURE_2D, 0, 0, 0, 0, 0, bounds.Width(), bounds.Height());
1978
1979 SetScissor(bounds);
1980 glBindFramebuffer(GL_FRAMEBUFFER, render_layers.GetTopLayer().framebuffer);
1981 Gfx::CheckGLError("SaveLayerAsTexture");
1982
