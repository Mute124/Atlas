57 Rml::TextureHandle LoadTexture(Rml::Vector2i& texture_dimensions, const Rml::String& source) override
58 {
59 Rml::FileInterface* file_interface = Rml::GetFileInterface();
60 Rml::FileHandle file_handle = file_interface->Open(source);
61 if (!file_handle)
62 return {};
63
64 file_interface->Seek(file_handle, 0, SEEK_END);
65 const size_t buffer_size = file_interface->Tell(file_handle);
66 file_interface->Seek(file_handle, 0, SEEK_SET);
67
68 using Rml::byte;
69 Rml::UniquePtr<byte[]> buffer(new byte[buffer_size]);
70 file_interface->Read(buffer.get(), buffer_size, file_handle);
71 file_interface->Close(file_handle);
72
73 const size_t i_ext = source.rfind('.');
74 Rml::String extension = (i_ext == Rml::String::npos ? Rml::String() : source.substr(i_ext + 1));
75
76 SDL_Surface* surface = IMG_LoadTyped_RW(SDL_RWFromMem(buffer.get(), int(buffer_size)), 1, extension.c_str());
77 if (!surface)
78 return {};
79
80 texture_dimensions = {surface->w, surface->h};
81
82 if (surface->format->format != SDL_PIXELFORMAT_RGBA32)
83 {
84 // Ensure correct format for premultiplied alpha conversion and GenerateTexture below.
85 SDL_Surface* converted_surface = SDL_ConvertSurfaceFormat(surface, SDL_PixelFormatEnum::SDL_PIXELFORMAT_RGBA32, 0);
86 SDL_FreeSurface(surface);
87
88 if (!converted_surface)
89 return {};
90
91 surface = converted_surface;
92 }
93
94 // Convert colors to premultiplied alpha, which is necessary for correct alpha compositing.
95 const size_t pixels_byte_size = surface->w * surface->h * 4;
96 byte* pixels = static_cast<byte*>(surface->pixels);
97 for (size_t i = 0; i < pixels_byte_size; i += 4)
98 {
99 const byte alpha = pixels[i + 3];
100 for (size_t j = 0; j < 3; ++j)
101 pixels[i + j] = byte(int(pixels[i + j]) * int(alpha) / 255);
102 }
103
104 Rml::TextureHandle texture_handle = RenderInterface_GL3::GenerateTexture({pixels, pixels_byte_size}, texture_dimensions);
105
106 SDL_FreeSurface(surface);
107
108 return texture_handle;
109 }
110};

128bool Backend::Initialize(const char* window_name, int width, int height, bool allow_resize)
129{
130 RMLUI_ASSERT(!data);
131
132 if (SDL_Init(SDL_INIT_VIDEO | SDL_INIT_EVENTS | SDL_INIT_TIMER) != 0)
133 return false;
134
135 // Submit click events when focusing the window.
136 SDL_SetHint(SDL_HINT_MOUSE_FOCUS_CLICKTHROUGH, "1");
137
138#if defined RMLUI_PLATFORM_EMSCRIPTEN
139 // GLES 3.0 (WebGL 2.0)
140 SDL_GL_SetAttribute(SDL_GL_CONTEXT_FLAGS, 0);
141 SDL_GL_SetAttribute(SDL_GL_CONTEXT_PROFILE_MASK, SDL_GL_CONTEXT_PROFILE_ES);
142 SDL_GL_SetAttribute(SDL_GL_CONTEXT_MAJOR_VERSION, 3);
143 SDL_GL_SetAttribute(SDL_GL_CONTEXT_MINOR_VERSION, 0);
144#else
145 // GL 3.3 Core
146 SDL_GL_SetAttribute(SDL_GL_CONTEXT_FLAGS, 0);
147 SDL_GL_SetAttribute(SDL_GL_CONTEXT_PROFILE_MASK, SDL_GL_CONTEXT_PROFILE_CORE);
148 SDL_GL_SetAttribute(SDL_GL_CONTEXT_MAJOR_VERSION, 3);
149 SDL_GL_SetAttribute(SDL_GL_CONTEXT_MINOR_VERSION, 3);
150#endif
151
152 SDL_GL_SetAttribute(SDL_GL_DOUBLEBUFFER, 1);
153
154 const Uint32 window_flags = (SDL_WINDOW_OPENGL | (allow_resize ? SDL_WINDOW_RESIZABLE : 0));
155
156 SDL_Window* window = SDL_CreateWindow(window_name, SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED, width, height, window_flags);
157 if (!window)
158 {
159 Rml::Log::Message(Rml::Log::LT_ERROR, "SDL error on create window: %s", SDL_GetError());
160 return false;
161 }
162
163 SDL_GLContext glcontext = SDL_GL_CreateContext(window);
164 SDL_GL_MakeCurrent(window, glcontext);
165 SDL_GL_SetSwapInterval(1);
166
167 if (!RmlGL3::Initialize())
168 {
169 Rml::Log::Message(Rml::Log::LT_ERROR, "Failed to initialize OpenGL renderer");
170 return false;
171 }
172
173 data = Rml::MakeUnique<BackendData>();
174
175 if (!data->render_interface)
176 {
177 data.reset();
178 Rml::Log::Message(Rml::Log::LT_ERROR, "Failed to initialize OpenGL3 render interface");
179 return false;
180 }
181
182 data->window = window;
183 data->glcontext = glcontext;
184
185 data->system_interface.SetWindow(window);
186 data->render_interface.SetViewport(width, height);
187
188 return true;
189}

215bool Backend::ProcessEvents(Rml::Context* context, KeyDownCallback key_down_callback, bool power_save)
216{
217 RMLUI_ASSERT(data && context);
218
219#if defined RMLUI_PLATFORM_EMSCRIPTEN
220
221 // Ideally we would hand over control of the main loop to emscripten:
222 //
223 // // Hand over control of the main loop to the WebAssembly runtime.
224 // emscripten_set_main_loop_arg(EventLoopIteration, (void*)user_data_handle, 0, true);
225 //
226 // The above is the recommended approach. However, as we don't control the main loop here we have to make due with another approach. Instead, use
227 // Asyncify to yield by sleeping.
228 // Important: Must be linked with option -sASYNCIFY
229 emscripten_sleep(1);
230
231#endif
232
233 bool result = data->running;
234 data->running = true;
235
236 SDL_Event ev;
237 int has_event = 0;
238 if (power_save)
239 has_event = SDL_WaitEventTimeout(&ev, static_cast<int>(Rml::Math::Min(context->GetNextUpdateDelay(), 10.0) * 1000));
240 else
241 has_event = SDL_PollEvent(&ev);
242 while (has_event)
243 {
244 switch (ev.type)
245 {
246 case SDL_QUIT:
247 {
248 result = false;
249 }
250 break;
251 case SDL_KEYDOWN:
252 {
253 const Rml::Input::KeyIdentifier key = RmlSDL::ConvertKey(ev.key.keysym.sym);
254 const int key_modifier = RmlSDL::GetKeyModifierState();
255 const float native_dp_ratio = 1.f;
256
257 // See if we have any global shortcuts that take priority over the context.
258 if (key_down_callback && !key_down_callback(context, key, key_modifier, native_dp_ratio, true))
259 break;
260 // Otherwise, hand the event over to the context by calling the input handler as normal.
261 if (!RmlSDL::InputEventHandler(context, ev))
262 break;
263 // The key was not consumed by the context either, try keyboard shortcuts of lower priority.
264 if (key_down_callback && !key_down_callback(context, key, key_modifier, native_dp_ratio, false))
265 break;
266 }
267 break;
268 case SDL_WINDOWEVENT:
269 {
270 switch (ev.window.event)
271 {
272 case SDL_WINDOWEVENT_SIZE_CHANGED:
273 {
274 Rml::Vector2i dimensions(ev.window.data1, ev.window.data2);
275 data->render_interface.SetViewport(dimensions.x, dimensions.y);
276 }
277 break;
278 }
279 RmlSDL::InputEventHandler(context, ev);
280 }
281 break;
282 default:
283 {
284 RmlSDL::InputEventHandler(context, ev);
285 }
286 break;
287 }
288 has_event = SDL_PollEvent(&ev);
289 }
290
291 return result;
292}
