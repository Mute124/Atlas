33VkPipeline Atlas::PipelineBuilder::buildPipeline(VkDevice device)
34{
35 // make viewport state from our stored viewport and scissor.
36 // at the moment we wont support multiple viewports or scissors
37 VkPipelineViewportStateCreateInfo viewportState = {};
38 viewportState.sType = VK_STRUCTURE_TYPE_PIPELINE_VIEWPORT_STATE_CREATE_INFO;
39 viewportState.pNext = nullptr;
40
41 viewportState.viewportCount = 1;
42 viewportState.scissorCount = 1;
43
44 // setup dummy color blending. We arent using transparent objects yet
45 // the blending is just "no blend", but we do write to the color attachment
46 VkPipelineColorBlendStateCreateInfo colorBlending = {};
47 colorBlending.sType = VK_STRUCTURE_TYPE_PIPELINE_COLOR_BLEND_STATE_CREATE_INFO;
48 colorBlending.pNext = nullptr;
49
50 colorBlending.logicOpEnable = VK_FALSE;
51 colorBlending.logicOp = VK_LOGIC_OP_COPY;
52 colorBlending.attachmentCount = 1;
53 colorBlending.pAttachments = &settings.colorBlendAttachment;
54
55 // completely clear VertexInputStateCreateInfo, as we have no need for it
56 VkPipelineVertexInputStateCreateInfo vertexInputInfo = { .sType = VK_STRUCTURE_TYPE_PIPELINE_VERTEX_INPUT_STATE_CREATE_INFO };
57
58 // build the actual pipeline
59 // we now use all of the info structs we have been writing into into this one
60 // to create the pipeline
61 VkGraphicsPipelineCreateInfo pipelineInfo = { .sType = VK_STRUCTURE_TYPE_GRAPHICS_PIPELINE_CREATE_INFO };
62 // connect the renderInfo to the pNext extension mechanism
63 pipelineInfo.pNext = &settings.renderInfo;
64
65 pipelineInfo.stageCount = (uint32_t)shaderStages.size();
66 pipelineInfo.pStages = shaderStages.data();
67 pipelineInfo.pVertexInputState = &vertexInputInfo;
68 pipelineInfo.pInputAssemblyState = &settings.inputAssembly;
69 pipelineInfo.pViewportState = &viewportState;
70 pipelineInfo.pRasterizationState = &settings.rasterizer;
71 pipelineInfo.pMultisampleState = &settings.multisampling;
72 pipelineInfo.pColorBlendState = &colorBlending;
73 pipelineInfo.pDepthStencilState = &settings.depthStencil;
74 pipelineInfo.layout = settings.pipelineLayout;
75
76 VkDynamicState state[] = { VK_DYNAMIC_STATE_VIEWPORT, VK_DYNAMIC_STATE_SCISSOR };
77
78 VkPipelineDynamicStateCreateInfo dynamicInfo = { .sType = VK_STRUCTURE_TYPE_PIPELINE_DYNAMIC_STATE_CREATE_INFO };
79 dynamicInfo.pDynamicStates = &state[0];
80 dynamicInfo.dynamicStateCount = 2;
81
82 pipelineInfo.pDynamicState = &dynamicInfo;
83
84 // its easy to error out on create graphics pipeline, so we handle it a bit
85 // better than the common VK_CHECK case
86 VkPipeline newPipeline;
87 if (vkCreateGraphicsPipelines(device, VK_NULL_HANDLE, 1, &pipelineInfo,
88 nullptr, &newPipeline)
89 != VK_SUCCESS) {
90 //fmt::println("failed to create pipeline");
91 std::cout << "failed to create pipeline" << std::endl;
92 return VK_NULL_HANDLE; // failed to create graphics pipeline
93 }
94 else {
95 return newPipeline;
96 }
97}
