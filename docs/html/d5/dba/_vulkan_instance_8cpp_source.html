78uint16_t Atlas::VulkanInstanceWrapper::init() {
79 uint16_t initResult = 0;
80
81 vkb::InstanceBuilder instanceBuilder;
82
83 VulkanInstanceWrapper::InitConfiguration& initConfiguration = mInitConfiguration;
84
85 instanceBuilder.set_app_name(initConfiguration.applicationName.c_str());
86 instanceBuilder.set_engine_name(initConfiguration.engineName.c_str());
87
88 instanceBuilder.set_engine_version(initConfiguration.atlasVersion.major, initConfiguration.atlasVersion.minor, initConfiguration.atlasVersion.patch);
89 instanceBuilder.require_api_version(initConfiguration.vulkanVersion.major, initConfiguration.vulkanVersion.minor, initConfiguration.vulkanVersion.patch);
90
91 instanceBuilder.request_validation_layers(initConfiguration.bEnableValidationLayers);
92
93 if (initConfiguration.optionalDebugMessengerCallback.has_value()) {
94 instanceBuilder.set_debug_callback(initConfiguration.optionalDebugMessengerCallback.value());
95 }
96 else {
97#ifdef ATLAS_USE_VULKAN_DEFAULT_DEBUG_MESSENGER
98 instanceBuilder.use_default_debug_messenger();
99#else
100 instanceBuilder.set_debug_callback(DefaultVulkanDebugCallback);
101#endif
102 }
103
104 auto instanceReturn = instanceBuilder.build();
105
106 ATLAS_ASSERT(instanceReturn.has_value(), "Failed to create Vulkan instance.");
107
108 mVulkanBootstrapInstance = instanceReturn.value();
109 mVulkanInstance = instanceReturn.value().instance;
110 mDebugMessenger = instanceReturn.value().debug_messenger;
111
112 return initResult;
113}

137Atlas::ELogLevel Atlas::TranslateVulkanLogLevel(VkDebugUtilsMessageSeverityFlagBitsEXT messageSeverity)
138{
139 ELogLevel logLevel;
140
141 switch (messageSeverity)
142 {
143 case VK_DEBUG_UTILS_MESSAGE_SEVERITY_VERBOSE_BIT_EXT:
144 // Use ELogLevel::trace here because the trace log level is meant for verbose logging (usually this is logged in a log file anyways).
145 logLevel = ELogLevel::trace;
146 break;
147 case VK_DEBUG_UTILS_MESSAGE_SEVERITY_INFO_BIT_EXT:
148 logLevel = ELogLevel::info;
149 break;
150 case VK_DEBUG_UTILS_MESSAGE_SEVERITY_WARNING_BIT_EXT:
151 logLevel = ELogLevel::warn;
152 break;
153 case VK_DEBUG_UTILS_MESSAGE_SEVERITY_ERROR_BIT_EXT:
154 logLevel = ELogLevel::error;
155 break;
156 case VK_DEBUG_UTILS_MESSAGE_SEVERITY_FLAG_BITS_MAX_ENUM_EXT:
157 logLevel = ELogLevel::critical;
158 break;
159 default:
160 logLevel = ELogLevel::info;
161 break;
162 }
163
164 return logLevel;
165}
