130Atlas::ELogLevel Atlas::TranslateVulkanLogLevel(VkDebugUtilsMessageSeverityFlagBitsEXT messageSeverity)
131{
132 ELogLevel logLevel;
133
134 switch (messageSeverity)
135 {
136 case VK_DEBUG_UTILS_MESSAGE_SEVERITY_VERBOSE_BIT_EXT:
137 // Use ELogLevel::trace here because the trace log level is meant for verbose logging (usually this is logged in a log file anyways).
138 logLevel = ELogLevel::trace;
139 break;
140 case VK_DEBUG_UTILS_MESSAGE_SEVERITY_INFO_BIT_EXT:
141 logLevel = ELogLevel::info;
142 break;
143 case VK_DEBUG_UTILS_MESSAGE_SEVERITY_WARNING_BIT_EXT:
144 logLevel = ELogLevel::warn;
145 break;
146 case VK_DEBUG_UTILS_MESSAGE_SEVERITY_ERROR_BIT_EXT:
147 logLevel = ELogLevel::error;
148 break;
149 case VK_DEBUG_UTILS_MESSAGE_SEVERITY_FLAG_BITS_MAX_ENUM_EXT:
150 logLevel = ELogLevel::critical;
151 break;
152 default:
153 logLevel = ELogLevel::info;
154 break;
155 }
156
157 return logLevel;
158}
