129static void CheckGLError(const char* operation_name)
130{
131#ifdef RMLUI_DEBUG
132 GLenum error_code = glGetError();
133 if (error_code != GL_NO_ERROR)
134 {
135 static const Rml::Pair<GLenum, const char*> error_names[] = {{GL_INVALID_ENUM, "GL_INVALID_ENUM"}, {GL_INVALID_VALUE, "GL_INVALID_VALUE"},
136 {GL_INVALID_OPERATION, "GL_INVALID_OPERATION"}, {GL_OUT_OF_MEMORY, "GL_OUT_OF_MEMORY"}};
137 const char* error_str = "''";
138 for (auto& err : error_names)
139 {
140 if (err.first == error_code)
141 {
142 error_str = err.second;
143 break;
144 }
145 }
146 Rml::Log::Message(Rml::Log::LT_ERROR, "OpenGL error during %s. Error code 0x%x (%s).", operation_name, error_code, error_str);
147 }
148#endif
149 (void)operation_name;
150}

152// Create the shader, 'shader_type' is either GL_VERTEX_SHADER or GL_FRAGMENT_SHADER.
153static GLuint CreateShader(GLenum shader_type, const char* code_string)
154{
155 GLuint id = glCreateShader(shader_type);
156
157 glShaderSource(id, 1, (const GLchar**)&code_string, NULL);
158 glCompileShader(id);
159
160 GLint status = 0;
161 glGetShaderiv(id, GL_COMPILE_STATUS, &status);
162 if (status == GL_FALSE)
163 {
164 GLint info_log_length = 0;
165 glGetShaderiv(id, GL_INFO_LOG_LENGTH, &info_log_length);
166 char* info_log_string = new char[info_log_length + 1];
167 glGetShaderInfoLog(id, info_log_length, NULL, info_log_string);
168
169 Rml::Log::Message(Rml::Log::LT_ERROR, "Compile failure in OpenGL shader: %s", info_log_string);
170 delete[] info_log_string;
171 glDeleteShader(id);
172 return 0;
173 }
174
175 CheckGLError("CreateShader");
176
177 return id;
178}

189static bool CreateProgram(GLuint vertex_shader, GLuint fragment_shader, ProgramData& out_program)
190{
191 GLuint id = glCreateProgram();
192 RMLUI_ASSERT(id);
193
194 BindAttribLocations(id);
195
196 glAttachShader(id, vertex_shader);
197 glAttachShader(id, fragment_shader);
198
199 glLinkProgram(id);
200
201 glDetachShader(id, vertex_shader);
202 glDetachShader(id, fragment_shader);
203
204 GLint status = 0;
205 glGetProgramiv(id, GL_LINK_STATUS, &status);
206 if (status == GL_FALSE)
207 {
208 GLint info_log_length = 0;
209 glGetProgramiv(id, GL_INFO_LOG_LENGTH, &info_log_length);
210 char* info_log_string = new char[info_log_length + 1];
211 glGetProgramInfoLog(id, info_log_length, NULL, info_log_string);
212
213 Rml::Log::Message(Rml::Log::LT_ERROR, "OpenGL program linking failure: %s", info_log_string);
214 delete[] info_log_string;
215 glDeleteProgram(id);
216 return false;
217 }
218
219 out_program = {};
220 out_program.id = id;
221
222 // Make a lookup table for the uniform locations.
223 GLint num_active_uniforms = 0;
224 glGetProgramiv(id, GL_ACTIVE_UNIFORMS, &num_active_uniforms);
225
226 constexpr size_t name_size = 64;
227 GLchar name_buf[name_size] = "";
228 for (int unif = 0; unif < num_active_uniforms; ++unif)
229 {
230 GLint array_size = 0;
231 GLenum type = 0;
232 GLsizei actual_length = 0;
233 glGetActiveUniform(id, unif, name_size, &actual_length, &array_size, &type, name_buf);
234 GLint location = glGetUniformLocation(id, name_buf);
235
236 // See if we have the name in our pre-defined name list.
237 ProgramUniform program_uniform = ProgramUniform::Count;
238 for (int i = 0; i < (int)ProgramUniform::Count; i++)
239 {
240 const char* uniform_name = program_uniform_names[i];
241 if (strcmp(name_buf, uniform_name) == 0)
242 {
243 program_uniform = (ProgramUniform)i;
244 break;
245 }
246 }
247
248 if ((size_t)program_uniform < (size_t)ProgramUniform::Count)
249 {
250 out_program.uniform_locations[(size_t)program_uniform] = location;
251 }
252 else
253 {
254 Rml::Log::Message(Rml::Log::LT_ERROR, "OpenGL program uses unknown uniform '%s'.", name_buf);
255 return false;
256 }
257 }
258
259 CheckGLError("CreateProgram");
260
261 return true;
262}

264static bool CreateShaders(ShadersData& out_shaders)
265{
266 out_shaders = {};
267 GLuint& main_vertex = out_shaders.shader_main_vertex;
268 GLuint& main_fragment_color = out_shaders.shader_main_fragment_color;
269 GLuint& main_fragment_texture = out_shaders.shader_main_fragment_texture;
270
271 main_vertex = CreateShader(GL_VERTEX_SHADER, shader_main_vertex);
272 if (!main_vertex)
273 {
274 Rml::Log::Message(Rml::Log::LT_ERROR, "Could not create OpenGL shader: 'shader_main_vertex'.");
275 return false;
276 }
277 main_fragment_color = CreateShader(GL_FRAGMENT_SHADER, shader_main_fragment_color);
278 if (!main_fragment_color)
279 {
280 Rml::Log::Message(Rml::Log::LT_ERROR, "Could not create OpenGL shader: 'shader_main_fragment_color'.");
281 return false;
282 }
283 main_fragment_texture = CreateShader(GL_FRAGMENT_SHADER, shader_main_fragment_texture);
284 if (!main_fragment_texture)
285 {
286 Rml::Log::Message(Rml::Log::LT_ERROR, "Could not create OpenGL shader: 'shader_main_fragment_texture'.");
287 return false;
288 }
289
290 if (!CreateProgram(main_vertex, main_fragment_color, out_shaders.program_color))
291 {
292 Rml::Log::Message(Rml::Log::LT_ERROR, "Could not create OpenGL program: 'program_color'.");
293 return false;
294 }
295 if (!CreateProgram(main_vertex, main_fragment_texture, out_shaders.program_texture))
296 {
297 Rml::Log::Message(Rml::Log::LT_ERROR, "Could not create OpenGL program: 'program_texture'.");
298 return false;
299 }
300
301 return true;
302}

338void RenderInterface_BackwardCompatible_GL3::BeginFrame()
339{
340 RMLUI_ASSERT(viewport_width >= 0 && viewport_height >= 0);
341
342 // Backup GL state.
343 glstate_backup.enable_cull_face = glIsEnabled(GL_CULL_FACE);
344 glstate_backup.enable_blend = glIsEnabled(GL_BLEND);
345 glstate_backup.enable_stencil_test = glIsEnabled(GL_STENCIL_TEST);
346 glstate_backup.enable_scissor_test = glIsEnabled(GL_SCISSOR_TEST);
347
348 glGetIntegerv(GL_VIEWPORT, glstate_backup.viewport);
349 glGetIntegerv(GL_SCISSOR_BOX, glstate_backup.scissor);
350
351 glGetIntegerv(GL_STENCIL_CLEAR_VALUE, &glstate_backup.stencil_clear_value);
352 glGetFloatv(GL_COLOR_CLEAR_VALUE, glstate_backup.color_clear_value);
353
354 glGetIntegerv(GL_BLEND_EQUATION_RGB, &glstate_backup.blend_equation_rgb);
355 glGetIntegerv(GL_BLEND_EQUATION_ALPHA, &glstate_backup.blend_equation_alpha);
356 glGetIntegerv(GL_BLEND_SRC_RGB, &glstate_backup.blend_src_rgb);
357 glGetIntegerv(GL_BLEND_DST_RGB, &glstate_backup.blend_dst_rgb);
358 glGetIntegerv(GL_BLEND_SRC_ALPHA, &glstate_backup.blend_src_alpha);
359 glGetIntegerv(GL_BLEND_DST_ALPHA, &glstate_backup.blend_dst_alpha);
360
361 glGetIntegerv(GL_STENCIL_FUNC, &glstate_backup.stencil_front.func);
362 glGetIntegerv(GL_STENCIL_REF, &glstate_backup.stencil_front.ref);
363 glGetIntegerv(GL_STENCIL_VALUE_MASK, &glstate_backup.stencil_front.value_mask);
364 glGetIntegerv(GL_STENCIL_WRITEMASK, &glstate_backup.stencil_front.writemask);
365 glGetIntegerv(GL_STENCIL_FAIL, &glstate_backup.stencil_front.fail);
366 glGetIntegerv(GL_STENCIL_PASS_DEPTH_FAIL, &glstate_backup.stencil_front.pass_depth_fail);
367 glGetIntegerv(GL_STENCIL_PASS_DEPTH_PASS, &glstate_backup.stencil_front.pass_depth_pass);
368
369 glGetIntegerv(GL_STENCIL_BACK_FUNC, &glstate_backup.stencil_back.func);
370 glGetIntegerv(GL_STENCIL_BACK_REF, &glstate_backup.stencil_back.ref);
371 glGetIntegerv(GL_STENCIL_BACK_VALUE_MASK, &glstate_backup.stencil_back.value_mask);
372 glGetIntegerv(GL_STENCIL_BACK_WRITEMASK, &glstate_backup.stencil_back.writemask);
373 glGetIntegerv(GL_STENCIL_BACK_FAIL, &glstate_backup.stencil_back.fail);
374 glGetIntegerv(GL_STENCIL_BACK_PASS_DEPTH_FAIL, &glstate_backup.stencil_back.pass_depth_fail);
375 glGetIntegerv(GL_STENCIL_BACK_PASS_DEPTH_PASS, &glstate_backup.stencil_back.pass_depth_pass);
376
377 // Setup expected GL state.
378 glViewport(0, 0, viewport_width, viewport_height);
379
380 glClearStencil(0);
381 glClearColor(0, 0, 0, 1);
382
383 glDisable(GL_CULL_FACE);
384
385 glEnable(GL_BLEND);
386 glBlendEquation(GL_FUNC_ADD);
387 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
388
389 glEnable(GL_STENCIL_TEST);
390 glStencilFunc(GL_ALWAYS, 1, GLuint(-1));
391 glStencilMask(GLuint(-1));
392 glStencilOp(GL_KEEP, GL_KEEP, GL_KEEP);
393
394 projection = Rml::Matrix4f::ProjectOrtho(0, (float)viewport_width, (float)viewport_height, 0, -10000, 10000);
395 SetTransform(nullptr);
396}

398void RenderInterface_BackwardCompatible_GL3::EndFrame()
399{
400 // Restore GL state.
401 if (glstate_backup.enable_cull_face)
402 glEnable(GL_CULL_FACE);
403 else
404 glDisable(GL_CULL_FACE);
405
406 if (glstate_backup.enable_blend)
407 glEnable(GL_BLEND);
408 else
409 glDisable(GL_BLEND);
410
411 if (glstate_backup.enable_stencil_test)
412 glEnable(GL_STENCIL_TEST);
413 else
414 glDisable(GL_STENCIL_TEST);
415
416 if (glstate_backup.enable_scissor_test)
417 glEnable(GL_SCISSOR_TEST);
418 else
419 glDisable(GL_SCISSOR_TEST);
420
421 glViewport(glstate_backup.viewport[0], glstate_backup.viewport[1], glstate_backup.viewport[2], glstate_backup.viewport[3]);
422 glScissor(glstate_backup.scissor[0], glstate_backup.scissor[1], glstate_backup.scissor[2], glstate_backup.scissor[3]);
423
424 glClearStencil(glstate_backup.stencil_clear_value);
425 glClearColor(glstate_backup.color_clear_value[0], glstate_backup.color_clear_value[1], glstate_backup.color_clear_value[2],
426 glstate_backup.color_clear_value[3]);
427
428 glBlendEquationSeparate(glstate_backup.blend_equation_rgb, glstate_backup.blend_equation_alpha);
429 glBlendFuncSeparate(glstate_backup.blend_src_rgb, glstate_backup.blend_dst_rgb, glstate_backup.blend_src_alpha, glstate_backup.blend_dst_alpha);
430
431 glStencilFuncSeparate(GL_FRONT, glstate_backup.stencil_front.func, glstate_backup.stencil_front.ref, glstate_backup.stencil_front.value_mask);
432 glStencilMaskSeparate(GL_FRONT, glstate_backup.stencil_front.writemask);
433 glStencilOpSeparate(GL_FRONT, glstate_backup.stencil_front.fail, glstate_backup.stencil_front.pass_depth_fail,
434 glstate_backup.stencil_front.pass_depth_pass);
435
436 glStencilFuncSeparate(GL_BACK, glstate_backup.stencil_back.func, glstate_backup.stencil_back.ref, glstate_backup.stencil_back.value_mask);
437 glStencilMaskSeparate(GL_BACK, glstate_backup.stencil_back.writemask);
438 glStencilOpSeparate(GL_BACK, glstate_backup.stencil_back.fail, glstate_backup.stencil_back.pass_depth_fail,
439 glstate_backup.stencil_back.pass_depth_pass);
440}

461Rml::CompiledGeometryHandle RenderInterface_BackwardCompatible_GL3::CompileGeometry(Rml::Vertex* vertices, int num_vertices, int* indices,
462 int num_indices, Rml::TextureHandle texture)
463{
464 constexpr GLenum draw_usage = GL_STATIC_DRAW;
465
466 GLuint vao = 0;
467 GLuint vbo = 0;
468 GLuint ibo = 0;
469
470 glGenVertexArrays(1, &vao);
471 glGenBuffers(1, &vbo);
472 glGenBuffers(1, &ibo);
473 glBindVertexArray(vao);
474
475 glBindBuffer(GL_ARRAY_BUFFER, vbo);
476 glBufferData(GL_ARRAY_BUFFER, sizeof(Rml::Vertex) * num_vertices, (const void*)vertices, draw_usage);
477
478 glEnableVertexAttribArray((GLuint)Gfx::VertexAttribute::Position);
479 glVertexAttribPointer((GLuint)Gfx::VertexAttribute::Position, 2, GL_FLOAT, GL_FALSE, sizeof(Rml::Vertex),
480 (const GLvoid*)(offsetof(Rml::Vertex, position)));
481
482 glEnableVertexAttribArray((GLuint)Gfx::VertexAttribute::Color0);
483 glVertexAttribPointer((GLuint)Gfx::VertexAttribute::Color0, 4, GL_UNSIGNED_BYTE, GL_TRUE, sizeof(Rml::Vertex),
484 (const GLvoid*)(offsetof(Rml::Vertex, colour)));
485
486 glEnableVertexAttribArray((GLuint)Gfx::VertexAttribute::TexCoord0);
487 glVertexAttribPointer((GLuint)Gfx::VertexAttribute::TexCoord0, 2, GL_FLOAT, GL_FALSE, sizeof(Rml::Vertex),
488 (const GLvoid*)(offsetof(Rml::Vertex, tex_coord)));
489
490 glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, ibo);
491 glBufferData(GL_ELEMENT_ARRAY_BUFFER, sizeof(int) * num_indices, (const void*)indices, draw_usage);
492
493 glBindVertexArray(0);
494 glBindBuffer(GL_ARRAY_BUFFER, 0);
495
496 Gfx::CheckGLError("CompileGeometry");
497
498 Gfx::CompiledGeometryData* geometry = new Gfx::CompiledGeometryData;
499 geometry->texture = texture;
500 geometry->vao = vao;
501 geometry->vbo = vbo;
502 geometry->ibo = ibo;
503 geometry->draw_count = num_indices;
504
505 return (Rml::CompiledGeometryHandle)geometry;
506}

508void RenderInterface_BackwardCompatible_GL3::RenderCompiledGeometry(Rml::CompiledGeometryHandle handle, const Rml::Vector2f& translation)
509{
510 Gfx::CompiledGeometryData* geometry = (Gfx::CompiledGeometryData*)handle;
511
512 if (geometry->texture)
513 {
514 glUseProgram(shaders->program_texture.id);
515 if (geometry->texture != TextureEnableWithoutBinding)
516 glBindTexture(GL_TEXTURE_2D, (GLuint)geometry->texture);
517 SubmitTransformUniform(ProgramId::Texture, shaders->program_texture.uniform_locations[(size_t)Gfx::ProgramUniform::Transform]);
518 glUniform2fv(shaders->program_texture.uniform_locations[(size_t)Gfx::ProgramUniform::Translate], 1, &translation.x);
519 }
520 else
521 {
522 glUseProgram(shaders->program_color.id);
523 glBindTexture(GL_TEXTURE_2D, 0);
524 SubmitTransformUniform(ProgramId::Color, shaders->program_color.uniform_locations[(size_t)Gfx::ProgramUniform::Transform]);
525 glUniform2fv(shaders->program_color.uniform_locations[(size_t)Gfx::ProgramUniform::Translate], 1, &translation.x);
526 }
527
528 glBindVertexArray(geometry->vao);
529 glDrawElements(GL_TRIANGLES, geometry->draw_count, GL_UNSIGNED_INT, (const GLvoid*)0);
530
531 glBindVertexArray(0);
532 glUseProgram(0);
533 glBindTexture(GL_TEXTURE_2D, 0);
534
535 Gfx::CheckGLError("RenderCompiledGeometry");
536}

549void RenderInterface_BackwardCompatible_GL3::EnableScissorRegion(bool enable)
550{
551 ScissoringState new_state = ScissoringState::Disable;
552
553 if (enable)
554 new_state = (transform_active ? ScissoringState::Stencil : ScissoringState::Scissor);
555
556 if (new_state != scissoring_state)
557 {
558 // Disable old
559 if (scissoring_state == ScissoringState::Scissor)
560 glDisable(GL_SCISSOR_TEST);
561 else if (scissoring_state == ScissoringState::Stencil)
562 glStencilFunc(GL_ALWAYS, 1, GLuint(-1));
563
564 // Enable new
565 if (new_state == ScissoringState::Scissor)
566 glEnable(GL_SCISSOR_TEST);
567 else if (new_state == ScissoringState::Stencil)
568 glStencilFunc(GL_EQUAL, 1, GLuint(-1));
569
570 scissoring_state = new_state;
571 }
572}

574void RenderInterface_BackwardCompatible_GL3::SetScissorRegion(int x, int y, int width, int height)
575{
576 if (transform_active)
577 {
578 const float left = float(x);
579 const float right = float(x + width);
580 const float top = float(y);
581 const float bottom = float(y + height);
582
583 Rml::Vertex vertices[4];
584 vertices[0].position = {left, top};
585 vertices[1].position = {right, top};
586 vertices[2].position = {right, bottom};
587 vertices[3].position = {left, bottom};
588
589 int indices[6] = {0, 2, 1, 0, 3, 2};
590
591 glClear(GL_STENCIL_BUFFER_BIT);
592 glColorMask(GL_FALSE, GL_FALSE, GL_FALSE, GL_FALSE);
593 glStencilFunc(GL_ALWAYS, 1, GLuint(-1));
594 glStencilOp(GL_KEEP, GL_KEEP, GL_REPLACE);
595
596 RenderGeometry(vertices, 4, indices, 6, 0, Rml::Vector2f(0, 0));
597
598 glStencilOp(GL_KEEP, GL_KEEP, GL_KEEP);
599 glStencilFunc(GL_EQUAL, 1, GLuint(-1));
600 glColorMask(GL_TRUE, GL_TRUE, GL_TRUE, GL_TRUE);
601 }
602 else
603 {
604 glScissor(x, viewport_height - (y + height), width, height);

627bool RenderInterface_BackwardCompatible_GL3::LoadTexture(Rml::TextureHandle& texture_handle, Rml::Vector2i& texture_dimensions,
628 const Rml::String& source)
629{
630 Rml::FileInterface* file_interface = Rml::GetFileInterface();
631 Rml::FileHandle file_handle = file_interface->Open(source);
632 if (!file_handle)
633 {
634 return false;
635 }
636
637 file_interface->Seek(file_handle, 0, SEEK_END);
638 size_t buffer_size = file_interface->Tell(file_handle);
639 file_interface->Seek(file_handle, 0, SEEK_SET);
640
641 if (buffer_size <= sizeof(TGAHeader))
642 {
643 Rml::Log::Message(Rml::Log::LT_ERROR, "Texture file size is smaller than TGAHeader, file is not a valid TGA image.");
644 file_interface->Close(file_handle);
645 return false;
646 }
647
648 using Rml::byte;
649 byte* buffer = new byte[buffer_size];
650 file_interface->Read(buffer, buffer_size, file_handle);
651 file_interface->Close(file_handle);
652
653 TGAHeader header;
654 memcpy(&header, buffer, sizeof(TGAHeader));
655
656 int color_mode = header.bitsPerPixel / 8;
657 int image_size = header.width * header.height * 4; // We always make 32bit textures
658
659 if (header.dataType != 2)
660 {
661 Rml::Log::Message(Rml::Log::LT_ERROR, "Only 24/32bit uncompressed TGAs are supported.");
662 delete[] buffer;
663 return false;
664 }
665
666 // Ensure we have at least 3 colors
667 if (color_mode < 3)
668 {
669 Rml::Log::Message(Rml::Log::LT_ERROR, "Only 24 and 32bit textures are supported.");
670 delete[] buffer;
671 return false;
672 }
673
674 const byte* image_src = buffer + sizeof(TGAHeader);
675 byte* image_dest = new byte[image_size];
676
677 // Targa is BGR, swap to RGB and flip Y axis
678 for (long y = 0; y < header.height; y++)
679 {
680 long read_index = y * header.width * color_mode;
681 long write_index = ((header.imageDescriptor & 32) != 0) ? read_index : (header.height - y - 1) * header.width * 4;
682 for (long x = 0; x < header.width; x++)
683 {
684 image_dest[write_index] = image_src[read_index + 2];
685 image_dest[write_index + 1] = image_src[read_index + 1];
686 image_dest[write_index + 2] = image_src[read_index];
687 if (color_mode == 4)
688 {
689 const int alpha = image_src[read_index + 3];
690#ifdef RMLUI_SRGB_PREMULTIPLIED_ALPHA
691 image_dest[write_index + 0] = (image_dest[write_index + 0] * alpha) / 255;
692 image_dest[write_index + 1] = (image_dest[write_index + 1] * alpha) / 255;
693 image_dest[write_index + 2] = (image_dest[write_index + 2] * alpha) / 255;
694#endif
695 image_dest[write_index + 3] = (byte)alpha;
696 }
697 else
698 {
699 image_dest[write_index + 3] = 255;
700 }
701
702 write_index += 4;
703 read_index += color_mode;
704 }
705 }
706
707 texture_dimensions.x = header.width;
708 texture_dimensions.y = header.height;
709
710 bool success = GenerateTexture(texture_handle, image_dest, texture_dimensions);
711
712 delete[] image_dest;
713 delete[] buffer;
714
715 return success;
716}

718bool RenderInterface_BackwardCompatible_GL3::GenerateTexture(Rml::TextureHandle& texture_handle, const Rml::byte* source,
719 const Rml::Vector2i& source_dimensions)
720{
721 GLuint texture_id = 0;
722 glGenTextures(1, &texture_id);
723 if (texture_id == 0)
724 {
725 Rml::Log::Message(Rml::Log::LT_ERROR, "Failed to generate texture.");
726 return false;
727 }
728
729 glBindTexture(GL_TEXTURE_2D, texture_id);
730
731 GLint internal_format = GL_RGBA8;
732 glTexImage2D(GL_TEXTURE_2D, 0, internal_format, source_dimensions.x, source_dimensions.y, 0, GL_RGBA, GL_UNSIGNED_BYTE, source);
733 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
734 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
735
736 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
737 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
738
739 texture_handle = (Rml::TextureHandle)texture_id;
740
741 glBindTexture(GL_TEXTURE_2D, 0);
742
743 return true;
744}
