63void RenderInterface_BackwardCompatible_GL2::BeginFrame()
64{
65 RMLUI_ASSERT(viewport_width >= 0 && viewport_height >= 0);
66 glViewport(0, 0, viewport_width, viewport_height);
67
68 glEnableClientState(GL_VERTEX_ARRAY);
69 glEnableClientState(GL_COLOR_ARRAY);
70 glDisableClientState(GL_TEXTURE_COORD_ARRAY);
71
72 glEnable(GL_BLEND);
73 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
74
75 Rml::Matrix4f projection = Rml::Matrix4f::ProjectOrtho(0, (float)viewport_width, (float)viewport_height, 0, -10000, 10000);
76 glMatrixMode(GL_PROJECTION);
77 glLoadMatrixf(projection.data());
78 glMatrixMode(GL_TEXTURE);
79 glLoadIdentity();
80 glMatrixMode(GL_MODELVIEW);
81 glLoadIdentity();
82
83 transform_enabled = false;
84}

95void RenderInterface_BackwardCompatible_GL2::RenderGeometry(Rml::Vertex* vertices, int /*num_vertices*/, int* indices, int num_indices,
96 const Rml::TextureHandle texture, const Rml::Vector2f& translation)
97{
98 glPushMatrix();
99 glTranslatef(translation.x, translation.y, 0);
100
101 glVertexPointer(2, GL_FLOAT, sizeof(Rml::Vertex), &vertices[0].position);
102 glColorPointer(4, GL_UNSIGNED_BYTE, sizeof(Rml::Vertex), &vertices[0].colour);
103
104 if (!texture)
105 {
106 glDisable(GL_TEXTURE_2D);
107 glDisableClientState(GL_TEXTURE_COORD_ARRAY);
108 }
109 else
110 {
111 glEnable(GL_TEXTURE_2D);
112
113 if (texture != TextureEnableWithoutBinding)
114 glBindTexture(GL_TEXTURE_2D, (GLuint)texture);
115
116 glEnableClientState(GL_TEXTURE_COORD_ARRAY);
117 glTexCoordPointer(2, GL_FLOAT, sizeof(Rml::Vertex), &vertices[0].tex_coord);
118 }
119
120 glDrawElements(GL_TRIANGLES, num_indices, GL_UNSIGNED_INT, indices);
121
122 glPopMatrix();
123}

147void RenderInterface_BackwardCompatible_GL2::SetScissorRegion(int x, int y, int width, int height)
148{
149 if (!transform_enabled)
150 {
151 glScissor(x, viewport_height - (y + height), width, height);
152 }
153 else
154 {
155 // clear the stencil buffer
156 glStencilMask(GLuint(-1));
157 glClear(GL_STENCIL_BUFFER_BIT);
158
159 // fill the stencil buffer
160 glColorMask(GL_FALSE, GL_FALSE, GL_FALSE, GL_FALSE);
161 glDepthMask(GL_FALSE);
162 glStencilFunc(GL_NEVER, 1, GLuint(-1));
163 glStencilOp(GL_REPLACE, GL_KEEP, GL_KEEP);
164
165 float fx = (float)x;
166 float fy = (float)y;
167 float fwidth = (float)width;
168 float fheight = (float)height;
169
170 // draw transformed quad
171 GLfloat vertices[] = {fx, fy, 0, fx, fy + fheight, 0, fx + fwidth, fy + fheight, 0, fx + fwidth, fy, 0};
172 glDisableClientState(GL_COLOR_ARRAY);
173 glVertexPointer(3, GL_FLOAT, 0, vertices);
174 GLushort indices[] = {1, 2, 0, 3};
175 glDrawElements(GL_TRIANGLE_STRIP, 4, GL_UNSIGNED_SHORT, indices);
176 glEnableClientState(GL_COLOR_ARRAY);
177
178 // prepare for drawing the real thing
179 glColorMask(GL_TRUE, GL_TRUE, GL_TRUE, GL_TRUE);
180 glDepthMask(GL_TRUE);
181 glStencilMask(0);
182 glStencilFunc(GL_EQUAL, 1, GLuint(-1));
183 }
184}

205bool RenderInterface_BackwardCompatible_GL2::LoadTexture(Rml::TextureHandle& texture_handle, Rml::Vector2i& texture_dimensions,
206 const Rml::String& source)
207{
208 Rml::FileInterface* file_interface = Rml::GetFileInterface();
209 Rml::FileHandle file_handle = file_interface->Open(source);
210 if (!file_handle)
211 {
212 return false;
213 }
214
215 file_interface->Seek(file_handle, 0, SEEK_END);
216 size_t buffer_size = file_interface->Tell(file_handle);
217 file_interface->Seek(file_handle, 0, SEEK_SET);
218
219 if (buffer_size <= sizeof(TGAHeader))
220 {
221 Rml::Log::Message(Rml::Log::LT_ERROR, "Texture file size is smaller than TGAHeader, file is not a valid TGA image.");
222 file_interface->Close(file_handle);
223 return false;
224 }
225
226 char* buffer = new char[buffer_size];
227 file_interface->Read(buffer, buffer_size, file_handle);
228 file_interface->Close(file_handle);
229
230 TGAHeader header;
231 memcpy(&header, buffer, sizeof(TGAHeader));
232
233 int color_mode = header.bitsPerPixel / 8;
234 int image_size = header.width * header.height * 4; // We always make 32bit textures
235
236 if (header.dataType != 2)
237 {
238 Rml::Log::Message(Rml::Log::LT_ERROR, "Only 24/32bit uncompressed TGAs are supported.");
239 delete[] buffer;
240 return false;
241 }
242
243 // Ensure we have at least 3 colors
244 if (color_mode < 3)
245 {
246 Rml::Log::Message(Rml::Log::LT_ERROR, "Only 24 and 32bit textures are supported.");
247 delete[] buffer;
248 return false;
249 }
250
251 const char* image_src = buffer + sizeof(TGAHeader);
252 unsigned char* image_dest = new unsigned char[image_size];
253
254 // Targa is BGR, swap to RGB and flip Y axis
255 for (long y = 0; y < header.height; y++)
256 {
257 long read_index = y * header.width * color_mode;
258 long write_index = ((header.imageDescriptor & 32) != 0) ? read_index : (header.height - y - 1) * header.width * color_mode;
259 for (long x = 0; x < header.width; x++)
260 {
261 image_dest[write_index] = image_src[read_index + 2];
262 image_dest[write_index + 1] = image_src[read_index + 1];
263 image_dest[write_index + 2] = image_src[read_index];
264 if (color_mode == 4)
265 image_dest[write_index + 3] = image_src[read_index + 3];
266 else
267 image_dest[write_index + 3] = 255;
268
269 write_index += 4;
270 read_index += color_mode;
271 }
272 }
273
274 texture_dimensions.x = header.width;
275 texture_dimensions.y = header.height;
276
277 bool success = GenerateTexture(texture_handle, image_dest, texture_dimensions);
278
279 delete[] image_dest;
280 delete[] buffer;
281
282 return success;
283}

285bool RenderInterface_BackwardCompatible_GL2::GenerateTexture(Rml::TextureHandle& texture_handle, const Rml::byte* source,
286 const Rml::Vector2i& source_dimensions)
287{
288 GLuint texture_id = 0;
289 glGenTextures(1, &texture_id);
290 if (texture_id == 0)
291 {
292 Rml::Log::Message(Rml::Log::LT_ERROR, "Failed to generate texture.");
293 return false;
294 }
295
296 glBindTexture(GL_TEXTURE_2D, texture_id);
297
298 glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA8, source_dimensions.x, source_dimensions.y, 0, GL_RGBA, GL_UNSIGNED_BYTE, source);
299 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
300 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
301
302 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
303 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
304
305 texture_handle = (Rml::TextureHandle)texture_id;
306
307 return true;
308}
