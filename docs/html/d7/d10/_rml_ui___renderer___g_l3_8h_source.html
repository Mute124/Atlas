134 /*
135 Manages render targets, including the layer stack and postprocessing framebuffers.
136
137 Layers can be pushed and popped, creating new framebuffers as needed. Typically, geometry is rendered to the top
138 layer. The layer framebuffers may have MSAA enabled.
139
140 Postprocessing framebuffers are separate from the layers, and are commonly used to apply texture-wide effects
141 such as filters. They are used both as input and output during rendering, and do not use MSAA.
142 */
143 class RenderLayerStack {
144 public:
145 RenderLayerStack();
146 ~RenderLayerStack();
147
148 // Push a new layer. All references to previously retrieved layers are invalidated.
149 Rml::LayerHandle PushLayer();
150
151 // Pop the top layer. All references to previously retrieved layers are invalidated.
152 void PopLayer();
153
154 const Gfx::FramebufferData& GetLayer(Rml::LayerHandle layer) const;
155 const Gfx::FramebufferData& GetTopLayer() const;
156 Rml::LayerHandle GetTopLayerHandle() const;
157
158 const Gfx::FramebufferData& GetPostprocessPrimary() { return EnsureFramebufferPostprocess(0); }
159 const Gfx::FramebufferData& GetPostprocessSecondary() { return EnsureFramebufferPostprocess(1); }
160 const Gfx::FramebufferData& GetPostprocessTertiary() { return EnsureFramebufferPostprocess(2); }
161 const Gfx::FramebufferData& GetBlendMask() { return EnsureFramebufferPostprocess(3); }
162
163 void SwapPostprocessPrimarySecondary();
164
165 void BeginFrame(int new_width, int new_height);
166 void EndFrame();
167
168 private:
169 void DestroyFramebuffers();
170 const Gfx::FramebufferData& EnsureFramebufferPostprocess(int index);
171
172 int width = 0, height = 0;
173
174 // The number of active layers is manually tracked since we re-use the framebuffers stored in the fb_layers stack.
175 int layers_size = 0;
176
177 Rml::Vector<Gfx::FramebufferData> fb_layers;
178 Rml::Vector<Gfx::FramebufferData> fb_postprocess;
179 };
