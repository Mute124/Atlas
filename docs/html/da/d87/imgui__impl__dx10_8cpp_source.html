84static void ImGui_ImplDX10_SetupRenderState(ImDrawData* draw_data, ID3D10Device* device)
85{
86 ImGui_ImplDX10_Data* bd = ImGui_ImplDX10_GetBackendData();
87
88 // Setup viewport
89 D3D10_VIEWPORT vp;
90 memset(&vp, 0, sizeof(D3D10_VIEWPORT));
91 vp.Width = (UINT)draw_data->DisplaySize.x;
92 vp.Height = (UINT)draw_data->DisplaySize.y;
93 vp.MinDepth = 0.0f;
94 vp.MaxDepth = 1.0f;
95 vp.TopLeftX = vp.TopLeftY = 0;
96 device->RSSetViewports(1, &vp);
97
98 // Bind shader and vertex buffers
99 unsigned int stride = sizeof(ImDrawVert);
100 unsigned int offset = 0;
101 device->IASetInputLayout(bd->pInputLayout);
102 device->IASetVertexBuffers(0, 1, &bd->pVB, &stride, &offset);
103 device->IASetIndexBuffer(bd->pIB, sizeof(ImDrawIdx) == 2 ? DXGI_FORMAT_R16_UINT : DXGI_FORMAT_R32_UINT, 0);
104 device->IASetPrimitiveTopology(D3D10_PRIMITIVE_TOPOLOGY_TRIANGLELIST);
105 device->VSSetShader(bd->pVertexShader);
106 device->VSSetConstantBuffers(0, 1, &bd->pVertexConstantBuffer);
107 device->PSSetShader(bd->pPixelShader);
108 device->PSSetSamplers(0, 1, &bd->pFontSampler);
109 device->GSSetShader(nullptr);
110
111 // Setup render state
112 const float blend_factor[4] = { 0.f, 0.f, 0.f, 0.f };
113 device->OMSetBlendState(bd->pBlendState, blend_factor, 0xffffffff);
114 device->OMSetDepthStencilState(bd->pDepthStencilState, 0);
115 device->RSSetState(bd->pRasterizerState);
116}

118// Render function
119void ImGui_ImplDX10_RenderDrawData(ImDrawData* draw_data)
120{
121 // Avoid rendering when minimized
122 if (draw_data->DisplaySize.x <= 0.0f || draw_data->DisplaySize.y <= 0.0f)
123 return;
124
125 ImGui_ImplDX10_Data* bd = ImGui_ImplDX10_GetBackendData();
126 ID3D10Device* device = bd->pd3dDevice;
127
128 // Create and grow vertex/index buffers if needed
129 if (!bd->pVB || bd->VertexBufferSize < draw_data->TotalVtxCount)
130 {
131 if (bd->pVB) { bd->pVB->Release(); bd->pVB = nullptr; }
132 bd->VertexBufferSize = draw_data->TotalVtxCount + 5000;
133 D3D10_BUFFER_DESC desc;
134 memset(&desc, 0, sizeof(D3D10_BUFFER_DESC));
135 desc.Usage = D3D10_USAGE_DYNAMIC;
136 desc.ByteWidth = bd->VertexBufferSize * sizeof(ImDrawVert);
137 desc.BindFlags = D3D10_BIND_VERTEX_BUFFER;
138 desc.CPUAccessFlags = D3D10_CPU_ACCESS_WRITE;
139 desc.MiscFlags = 0;
140 if (device->CreateBuffer(&desc, nullptr, &bd->pVB) < 0)
141 return;
142 }
143
144 if (!bd->pIB || bd->IndexBufferSize < draw_data->TotalIdxCount)
145 {
146 if (bd->pIB) { bd->pIB->Release(); bd->pIB = nullptr; }
147 bd->IndexBufferSize = draw_data->TotalIdxCount + 10000;
148 D3D10_BUFFER_DESC desc;
149 memset(&desc, 0, sizeof(D3D10_BUFFER_DESC));
150 desc.Usage = D3D10_USAGE_DYNAMIC;
151 desc.ByteWidth = bd->IndexBufferSize * sizeof(ImDrawIdx);
152 desc.BindFlags = D3D10_BIND_INDEX_BUFFER;
153 desc.CPUAccessFlags = D3D10_CPU_ACCESS_WRITE;
154 if (device->CreateBuffer(&desc, nullptr, &bd->pIB) < 0)
155 return;
156 }
157
158 // Copy and convert all vertices into a single contiguous buffer
159 ImDrawVert* vtx_dst = nullptr;
160 ImDrawIdx* idx_dst = nullptr;
161 bd->pVB->Map(D3D10_MAP_WRITE_DISCARD, 0, (void**)&vtx_dst);
162 bd->pIB->Map(D3D10_MAP_WRITE_DISCARD, 0, (void**)&idx_dst);
163 for (int n = 0; n < draw_data->CmdListsCount; n++)
164 {
165 const ImDrawList* draw_list = draw_data->CmdLists[n];
166 memcpy(vtx_dst, draw_list->VtxBuffer.Data, draw_list->VtxBuffer.Size * sizeof(ImDrawVert));
167 memcpy(idx_dst, draw_list->IdxBuffer.Data, draw_list->IdxBuffer.Size * sizeof(ImDrawIdx));
168 vtx_dst += draw_list->VtxBuffer.Size;
169 idx_dst += draw_list->IdxBuffer.Size;
170 }
171 bd->pVB->Unmap();
172 bd->pIB->Unmap();
173
174 // Setup orthographic projection matrix into our constant buffer
175 // Our visible imgui space lies from draw_data->DisplayPos (top left) to draw_data->DisplayPos+data_data->DisplaySize (bottom right). DisplayPos is (0,0) for single viewport apps.
176 {
177 void* mapped_resource;
178 if (bd->pVertexConstantBuffer->Map(D3D10_MAP_WRITE_DISCARD, 0, &mapped_resource) != S_OK)
179 return;
180 VERTEX_CONSTANT_BUFFER_DX10* constant_buffer = (VERTEX_CONSTANT_BUFFER_DX10*)mapped_resource;
181 float L = draw_data->DisplayPos.x;
182 float R = draw_data->DisplayPos.x + draw_data->DisplaySize.x;
183 float T = draw_data->DisplayPos.y;
184 float B = draw_data->DisplayPos.y + draw_data->DisplaySize.y;
185 float mvp[4][4] =
186 {
187 { 2.0f/(R-L), 0.0f, 0.0f, 0.0f },
188 { 0.0f, 2.0f/(T-B), 0.0f, 0.0f },
189 { 0.0f, 0.0f, 0.5f, 0.0f },
190 { (R+L)/(L-R), (T+B)/(B-T), 0.5f, 1.0f },
191 };
192 memcpy(&constant_buffer->mvp, mvp, sizeof(mvp));
193 bd->pVertexConstantBuffer->Unmap();
194 }

196 // Backup DX state that will be modified to restore it afterwards (unfortunately this is very ugly looking and verbose. Close your eyes!)
197 struct BACKUP_DX10_STATE
198 {
199 UINT ScissorRectsCount, ViewportsCount;
200 D3D10_RECT ScissorRects[D3D10_VIEWPORT_AND_SCISSORRECT_OBJECT_COUNT_PER_PIPELINE];
201 D3D10_VIEWPORT Viewports[D3D10_VIEWPORT_AND_SCISSORRECT_OBJECT_COUNT_PER_PIPELINE];
202 ID3D10RasterizerState* RS;
203 ID3D10BlendState* BlendState;
204 FLOAT BlendFactor[4];
205 UINT SampleMask;
206 UINT StencilRef;
207 ID3D10DepthStencilState* DepthStencilState;
208 ID3D10ShaderResourceView* PSShaderResource;
209 ID3D10SamplerState* PSSampler;
210 ID3D10PixelShader* PS;
211 ID3D10VertexShader* VS;
212 ID3D10GeometryShader* GS;
213 D3D10_PRIMITIVE_TOPOLOGY PrimitiveTopology;
214 ID3D10Buffer* IndexBuffer, *VertexBuffer, *VSConstantBuffer;
215 UINT IndexBufferOffset, VertexBufferStride, VertexBufferOffset;
216 DXGI_FORMAT IndexBufferFormat;
217 ID3D10InputLayout* InputLayout;
218 };
219 BACKUP_DX10_STATE old = {};
220 old.ScissorRectsCount = old.ViewportsCount = D3D10_VIEWPORT_AND_SCISSORRECT_OBJECT_COUNT_PER_PIPELINE;
221 device->RSGetScissorRects(&old.ScissorRectsCount, old.ScissorRects);
222 device->RSGetViewports(&old.ViewportsCount, old.Viewports);
223 device->RSGetState(&old.RS);
224 device->OMGetBlendState(&old.BlendState, old.BlendFactor, &old.SampleMask);
225 device->OMGetDepthStencilState(&old.DepthStencilState, &old.StencilRef);
226 device->PSGetShaderResources(0, 1, &old.PSShaderResource);
227 device->PSGetSamplers(0, 1, &old.PSSampler);
228 device->PSGetShader(&old.PS);
229 device->VSGetShader(&old.VS);
230 device->VSGetConstantBuffers(0, 1, &old.VSConstantBuffer);
231 device->GSGetShader(&old.GS);
232 device->IAGetPrimitiveTopology(&old.PrimitiveTopology);
233 device->IAGetIndexBuffer(&old.IndexBuffer, &old.IndexBufferFormat, &old.IndexBufferOffset);
234 device->IAGetVertexBuffers(0, 1, &old.VertexBuffer, &old.VertexBufferStride, &old.VertexBufferOffset);
235 device->IAGetInputLayout(&old.InputLayout);
236
237 // Setup desired DX state
238 ImGui_ImplDX10_SetupRenderState(draw_data, device);
239
240 // Render command lists
241 // (Because we merged all buffers into a single one, we maintain our own offset into them)
242 int global_vtx_offset = 0;
243 int global_idx_offset = 0;
244 ImVec2 clip_off = draw_data->DisplayPos;
245 for (int n = 0; n < draw_data->CmdListsCount; n++)
246 {
247 const ImDrawList* draw_list = draw_data->CmdLists[n];
248 for (int cmd_i = 0; cmd_i < draw_list->CmdBuffer.Size; cmd_i++)
249 {
250 const ImDrawCmd* pcmd = &draw_list->CmdBuffer[cmd_i];
251 if (pcmd->UserCallback)
252 {
253 // User callback, registered via ImDrawList::AddCallback()
254 // (ImDrawCallback_ResetRenderState is a special callback value used by the user to request the renderer to reset render state.)
255 if (pcmd->UserCallback == ImDrawCallback_ResetRenderState)
256 ImGui_ImplDX10_SetupRenderState(draw_data, device);
257 else
258 pcmd->UserCallback(draw_list, pcmd);
259 }
260 else
261 {
262 // Project scissor/clipping rectangles into framebuffer space
263 ImVec2 clip_min(pcmd->ClipRect.x - clip_off.x, pcmd->ClipRect.y - clip_off.y);
264 ImVec2 clip_max(pcmd->ClipRect.z - clip_off.x, pcmd->ClipRect.w - clip_off.y);
265 if (clip_max.x <= clip_min.x || clip_max.y <= clip_min.y)
266 continue;
267
268 // Apply scissor/clipping rectangle
269 const D3D10_RECT r = { (LONG)clip_min.x, (LONG)clip_min.y, (LONG)clip_max.x, (LONG)clip_max.y };
270 device->RSSetScissorRects(1, &r);
271
272 // Bind texture, Draw
273 ID3D10ShaderResourceView* texture_srv = (ID3D10ShaderResourceView*)pcmd->GetTexID();
274 device->PSSetShaderResources(0, 1, &texture_srv);
275 device->DrawIndexed(pcmd->ElemCount, pcmd->IdxOffset + global_idx_offset, pcmd->VtxOffset + global_vtx_offset);
276 }
277 }
278 global_idx_offset += draw_list->IdxBuffer.Size;
279 global_vtx_offset += draw_list->VtxBuffer.Size;
280 }
281
282 // Restore modified DX state
283 device->RSSetScissorRects(old.ScissorRectsCount, old.ScissorRects);
284 device->RSSetViewports(old.ViewportsCount, old.Viewports);
285 device->RSSetState(old.RS); if (old.RS) old.RS->Release();
286 device->OMSetBlendState(old.BlendState, old.BlendFactor, old.SampleMask); if (old.BlendState) old.BlendState->Release();
287 device->OMSetDepthStencilState(old.DepthStencilState, old.StencilRef); if (old.DepthStencilState) old.DepthStencilState->Release();
288 device->PSSetShaderResources(0, 1, &old.PSShaderResource); if (old.PSShaderResource) old.PSShaderResource->Release();
289 device->PSSetSamplers(0, 1, &old.PSSampler); if (old.PSSampler) old.PSSampler->Release();
290 device->PSSetShader(old.PS); if (old.PS) old.PS->Release();
291 device->VSSetShader(old.VS); if (old.VS) old.VS->Release();
292 device->GSSetShader(old.GS); if (old.GS) old.GS->Release();
293 device->VSSetConstantBuffers(0, 1, &old.VSConstantBuffer); if (old.VSConstantBuffer) old.VSConstantBuffer->Release();
294 device->IASetPrimitiveTopology(old.PrimitiveTopology);
295 device->IASetIndexBuffer(old.IndexBuffer, old.IndexBufferFormat, old.IndexBufferOffset); if (old.IndexBuffer) old.IndexBuffer->Release();
296 device->IASetVertexBuffers(0, 1, &old.VertexBuffer, &old.VertexBufferStride, &old.VertexBufferOffset); if (old.VertexBuffer) old.VertexBuffer->Release();
297 device->IASetInputLayout(old.InputLayout); if (old.InputLayout) old.InputLayout->Release();
298}

300static void ImGui_ImplDX10_CreateFontsTexture()
301{
302 // Build texture atlas
303 ImGui_ImplDX10_Data* bd = ImGui_ImplDX10_GetBackendData();
304 ImGuiIO& io = ImGui::GetIO();
305 unsigned char* pixels;
306 int width, height;
307 io.Fonts->GetTexDataAsRGBA32(&pixels, &width, &height);
308
309 // Upload texture to graphics system
310 {
311 D3D10_TEXTURE2D_DESC desc;
312 ZeroMemory(&desc, sizeof(desc));
313 desc.Width = width;
314 desc.Height = height;
315 desc.MipLevels = 1;
316 desc.ArraySize = 1;
317 desc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
318 desc.SampleDesc.Count = 1;
319 desc.Usage = D3D10_USAGE_DEFAULT;
320 desc.BindFlags = D3D10_BIND_SHADER_RESOURCE;
321 desc.CPUAccessFlags = 0;
322
323 ID3D10Texture2D* pTexture = nullptr;
324 D3D10_SUBRESOURCE_DATA subResource;
325 subResource.pSysMem = pixels;
326 subResource.SysMemPitch = desc.Width * 4;
327 subResource.SysMemSlicePitch = 0;
328 bd->pd3dDevice->CreateTexture2D(&desc, &subResource, &pTexture);
329 IM_ASSERT(pTexture != nullptr);
330
331 // Create texture view
332 D3D10_SHADER_RESOURCE_VIEW_DESC srv_desc;
333 ZeroMemory(&srv_desc, sizeof(srv_desc));
334 srv_desc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
335 srv_desc.ViewDimension = D3D10_SRV_DIMENSION_TEXTURE2D;
336 srv_desc.Texture2D.MipLevels = desc.MipLevels;
337 srv_desc.Texture2D.MostDetailedMip = 0;
338 bd->pd3dDevice->CreateShaderResourceView(pTexture, &srv_desc, &bd->pFontTextureView);
339 pTexture->Release();
340 }
341
342 // Store our identifier
343 io.Fonts->SetTexID((ImTextureID)bd->pFontTextureView);
344
345 // Create texture sampler
346 // (Bilinear sampling is required by default. Set 'io.Fonts->Flags |= ImFontAtlasFlags_NoBakedLines' or 'style.AntiAliasedLinesUseTex = false' to allow point/nearest sampling)
347 {
348 D3D10_SAMPLER_DESC desc;
349 ZeroMemory(&desc, sizeof(desc));
350 desc.Filter = D3D10_FILTER_MIN_MAG_MIP_LINEAR;
351 desc.AddressU = D3D10_TEXTURE_ADDRESS_CLAMP;
352 desc.AddressV = D3D10_TEXTURE_ADDRESS_CLAMP;
353 desc.AddressW = D3D10_TEXTURE_ADDRESS_CLAMP;
354 desc.MipLODBias = 0.f;
355 desc.ComparisonFunc = D3D10_COMPARISON_ALWAYS;
356 desc.MinLOD = 0.f;
357 desc.MaxLOD = 0.f;
358 bd->pd3dDevice->CreateSamplerState(&desc, &bd->pFontSampler);
359 }
360}

362bool ImGui_ImplDX10_CreateDeviceObjects()
363{
364 ImGui_ImplDX10_Data* bd = ImGui_ImplDX10_GetBackendData();
365 if (!bd->pd3dDevice)
366 return false;
367 if (bd->pFontSampler)
368 ImGui_ImplDX10_InvalidateDeviceObjects();
369
370 // By using D3DCompile() from <d3dcompiler.h> / d3dcompiler.lib, we introduce a dependency to a given version of d3dcompiler_XX.dll (see D3DCOMPILER_DLL_A)
371 // If you would like to use this DX10 sample code but remove this dependency you can:
372 // 1) compile once, save the compiled shader blobs into a file or source code and pass them to CreateVertexShader()/CreatePixelShader() [preferred solution]
373 // 2) use code to detect any version of the DLL and grab a pointer to D3DCompile from the DLL.
374 // See https://github.com/ocornut/imgui/pull/638 for sources and details.
375
376 // Create the vertex shader
377 {
378 static const char* vertexShader =
379 "cbuffer vertexBuffer : register(b0) \
380 {\
381 float4x4 ProjectionMatrix; \
382 };\
383 struct VS_INPUT\
384 {\
385 float2 pos : POSITION;\
386 float4 col : COLOR0;\
387 float2 uv : TEXCOORD0;\
388 };\
389 \
390 struct PS_INPUT\
391 {\
392 float4 pos : SV_POSITION;\
393 float4 col : COLOR0;\
394 float2 uv : TEXCOORD0;\
395 };\
396 \
397 PS_INPUT main(VS_INPUT input)\
398 {\
399 PS_INPUT output;\
400 output.pos = mul( ProjectionMatrix, float4(input.pos.xy, 0.f, 1.f));\
401 output.col = input.col;\
402 output.uv = input.uv;\
403 return output;\
404 }";
405
406 ID3DBlob* vertexShaderBlob;
407 if (FAILED(D3DCompile(vertexShader, strlen(vertexShader), nullptr, nullptr, nullptr, "main", "vs_4_0", 0, 0, &vertexShaderBlob, nullptr)))
408 return false; // NB: Pass ID3DBlob* pErrorBlob to D3DCompile() to get error showing in (const char*)pErrorBlob->GetBufferPointer(). Make sure to Release() the blob!
409 if (bd->pd3dDevice->CreateVertexShader(vertexShaderBlob->GetBufferPointer(), vertexShaderBlob->GetBufferSize(), &bd->pVertexShader) != S_OK)
410 {
411 vertexShaderBlob->Release();
412 return false;
413 }
414
415 // Create the input layout
416 D3D10_INPUT_ELEMENT_DESC local_layout[] =
417 {
418 { "POSITION", 0, DXGI_FORMAT_R32G32_FLOAT, 0, (UINT)offsetof(ImDrawVert, pos), D3D10_INPUT_PER_VERTEX_DATA, 0 },
419 { "TEXCOORD", 0, DXGI_FORMAT_R32G32_FLOAT, 0, (UINT)offsetof(ImDrawVert, uv), D3D10_INPUT_PER_VERTEX_DATA, 0 },
420 { "COLOR", 0, DXGI_FORMAT_R8G8B8A8_UNORM, 0, (UINT)offsetof(ImDrawVert, col), D3D10_INPUT_PER_VERTEX_DATA, 0 },
421 };
422 if (bd->pd3dDevice->CreateInputLayout(local_layout, 3, vertexShaderBlob->GetBufferPointer(), vertexShaderBlob->GetBufferSize(), &bd->pInputLayout) != S_OK)
423 {
424 vertexShaderBlob->Release();
425 return false;
426 }
427 vertexShaderBlob->Release();
428
429 // Create the constant buffer
430 {
431 D3D10_BUFFER_DESC desc;
432 desc.ByteWidth = sizeof(VERTEX_CONSTANT_BUFFER_DX10);
433 desc.Usage = D3D10_USAGE_DYNAMIC;
434 desc.BindFlags = D3D10_BIND_CONSTANT_BUFFER;
435 desc.CPUAccessFlags = D3D10_CPU_ACCESS_WRITE;
436 desc.MiscFlags = 0;
437 bd->pd3dDevice->CreateBuffer(&desc, nullptr, &bd->pVertexConstantBuffer);
438 }
439 }
440
441 // Create the pixel shader
442 {
443 static const char* pixelShader =
444 "struct PS_INPUT\
445 {\
446 float4 pos : SV_POSITION;\
447 float4 col : COLOR0;\
448 float2 uv : TEXCOORD0;\
449 };\
450 sampler sampler0;\
451 Texture2D texture0;\
452 \
453 float4 main(PS_INPUT input) : SV_Target\
454 {\
455 float4 out_col = input.col * texture0.Sample(sampler0, input.uv); \
456 return out_col; \
457 }";
458
459 ID3DBlob* pixelShaderBlob;
460 if (FAILED(D3DCompile(pixelShader, strlen(pixelShader), nullptr, nullptr, nullptr, "main", "ps_4_0", 0, 0, &pixelShaderBlob, nullptr)))
461 return false; // NB: Pass ID3DBlob* pErrorBlob to D3DCompile() to get error showing in (const char*)pErrorBlob->GetBufferPointer(). Make sure to Release() the blob!
462 if (bd->pd3dDevice->CreatePixelShader(pixelShaderBlob->GetBufferPointer(), pixelShaderBlob->GetBufferSize(), &bd->pPixelShader) != S_OK)
463 {
464 pixelShaderBlob->Release();
465 return false;
466 }
467 pixelShaderBlob->Release();
468 }
469
470 // Create the blending setup
471 {
472 D3D10_BLEND_DESC desc;
473 ZeroMemory(&desc, sizeof(desc));
474 desc.AlphaToCoverageEnable = false;
475 desc.BlendEnable[0] = true;
476 desc.SrcBlend = D3D10_BLEND_SRC_ALPHA;
477 desc.DestBlend = D3D10_BLEND_INV_SRC_ALPHA;
478 desc.BlendOp = D3D10_BLEND_OP_ADD;
479 desc.SrcBlendAlpha = D3D10_BLEND_ONE;
480 desc.DestBlendAlpha = D3D10_BLEND_INV_SRC_ALPHA;
481 desc.BlendOpAlpha = D3D10_BLEND_OP_ADD;
482 desc.RenderTargetWriteMask[0] = D3D10_COLOR_WRITE_ENABLE_ALL;
483 bd->pd3dDevice->CreateBlendState(&desc, &bd->pBlendState);
484 }
485
486 // Create the rasterizer state
487 {
488 D3D10_RASTERIZER_DESC desc;
489 ZeroMemory(&desc, sizeof(desc));
490 desc.FillMode = D3D10_FILL_SOLID;

535bool ImGui_ImplDX10_Init(ID3D10Device* device)
536{
537 ImGuiIO& io = ImGui::GetIO();
538 IMGUI_CHECKVERSION();
539 IM_ASSERT(io.BackendRendererUserData == nullptr && "Already initialized a renderer backend!");
540
541 // Setup backend capabilities flags
542 ImGui_ImplDX10_Data* bd = IM_NEW(ImGui_ImplDX10_Data)();
543 io.BackendRendererUserData = (void*)bd;
544 io.BackendRendererName = "imgui_impl_dx10";
545 io.BackendFlags |= ImGuiBackendFlags_RendererHasVtxOffset; // We can honor the ImDrawCmd::VtxOffset field, allowing for large meshes.
546
547 // Get factory from device
548 IDXGIDevice* pDXGIDevice = nullptr;
549 IDXGIAdapter* pDXGIAdapter = nullptr;
550 IDXGIFactory* pFactory = nullptr;
551 if (device->QueryInterface(IID_PPV_ARGS(&pDXGIDevice)) == S_OK)
552 if (pDXGIDevice->GetParent(IID_PPV_ARGS(&pDXGIAdapter)) == S_OK)
553 if (pDXGIAdapter->GetParent(IID_PPV_ARGS(&pFactory)) == S_OK)
554 {
555 bd->pd3dDevice = device;
556 bd->pFactory = pFactory;
557 }
558 if (pDXGIDevice) pDXGIDevice->Release();
559 if (pDXGIAdapter) pDXGIAdapter->Release();
560 bd->pd3dDevice->AddRef();
561
562 return true;
563}
