1// dear imgui: Renderer Backend for DirectX12
2// This needs to be used along with a Platform Backend (e.g. Win32)
3
4// Implemented features:
5// [X] Renderer: User texture binding. Use 'D3D12_GPU_DESCRIPTOR_HANDLE' as ImTextureID. Read the FAQ about ImTextureID!
6// [X] Renderer: Large meshes support (64k+ vertices) with 16-bit indices.
7// [X] Renderer: Expose selected render state for draw callbacks to use. Access in '(ImGui_ImplXXXX_RenderState*)GetPlatformIO().Renderer_RenderState'.
8
9// You can use unmodified imgui_impl_* files in your project. See examples/ folder for examples of using this.
10// Prefer including the entire imgui/ repository into your project (either as a copy or as a submodule), and only build the backends you need.

12// - FAQ https://dearimgui.com/faq
13// - Getting Started https://dearimgui.com/getting-started
14// - Documentation https://dearimgui.com/docs (same as your local docs/ folder).
15// - Introduction, links and more at the top of imgui.cpp

95static void ImGui_ImplDX12_SetupRenderState(ImDrawData* draw_data, ID3D12GraphicsCommandList* command_list, ImGui_ImplDX12_RenderBuffers* fr)
96{
97 ImGui_ImplDX12_Data* bd = ImGui_ImplDX12_GetBackendData();
98
99 // Setup orthographic projection matrix into our constant buffer
100 // Our visible imgui space lies from draw_data->DisplayPos (top left) to draw_data->DisplayPos+data_data->DisplaySize (bottom right).
101 VERTEX_CONSTANT_BUFFER_DX12 vertex_constant_buffer;
102 {
103 float L = draw_data->DisplayPos.x;
104 float R = draw_data->DisplayPos.x + draw_data->DisplaySize.x;
105 float T = draw_data->DisplayPos.y;
106 float B = draw_data->DisplayPos.y + draw_data->DisplaySize.y;
107 float mvp[4][4] =
108 {
109 { 2.0f/(R-L), 0.0f, 0.0f, 0.0f },
110 { 0.0f, 2.0f/(T-B), 0.0f, 0.0f },
111 { 0.0f, 0.0f, 0.5f, 0.0f },
112 { (R+L)/(L-R), (T+B)/(B-T), 0.5f, 1.0f },
113 };
114 memcpy(&vertex_constant_buffer.mvp, mvp, sizeof(mvp));
115 }
116
117 // Setup viewport
118 D3D12_VIEWPORT vp;
119 memset(&vp, 0, sizeof(D3D12_VIEWPORT));
120 vp.Width = draw_data->DisplaySize.x;
121 vp.Height = draw_data->DisplaySize.y;
122 vp.MinDepth = 0.0f;
123 vp.MaxDepth = 1.0f;
124 vp.TopLeftX = vp.TopLeftY = 0.0f;
125 command_list->RSSetViewports(1, &vp);
126
127 // Bind shader and vertex buffers
128 unsigned int stride = sizeof(ImDrawVert);
129 unsigned int offset = 0;
130 D3D12_VERTEX_BUFFER_VIEW vbv;
131 memset(&vbv, 0, sizeof(D3D12_VERTEX_BUFFER_VIEW));
132 vbv.BufferLocation = fr->VertexBuffer->GetGPUVirtualAddress() + offset;
133 vbv.SizeInBytes = fr->VertexBufferSize * stride;
134 vbv.StrideInBytes = stride;
135 command_list->IASetVertexBuffers(0, 1, &vbv);
136 D3D12_INDEX_BUFFER_VIEW ibv;
137 memset(&ibv, 0, sizeof(D3D12_INDEX_BUFFER_VIEW));
138 ibv.BufferLocation = fr->IndexBuffer->GetGPUVirtualAddress();
139 ibv.SizeInBytes = fr->IndexBufferSize * sizeof(ImDrawIdx);
140 ibv.Format = sizeof(ImDrawIdx) == 2 ? DXGI_FORMAT_R16_UINT : DXGI_FORMAT_R32_UINT;
141 command_list->IASetIndexBuffer(&ibv);
142 command_list->IASetPrimitiveTopology(D3D_PRIMITIVE_TOPOLOGY_TRIANGLELIST);
143 command_list->SetPipelineState(bd->pPipelineState);
144 command_list->SetGraphicsRootSignature(bd->pRootSignature);
145 command_list->SetGraphicsRoot32BitConstants(0, 16, &vertex_constant_buffer, 0);
146
147 // Setup blend factor
148 const float blend_factor[4] = { 0.f, 0.f, 0.f, 0.f };
149 command_list->OMSetBlendFactor(blend_factor);
150}

160// Render function
161void ImGui_ImplDX12_RenderDrawData(ImDrawData* draw_data, ID3D12GraphicsCommandList* command_list)
162{
163 // Avoid rendering when minimized
164 if (draw_data->DisplaySize.x <= 0.0f || draw_data->DisplaySize.y <= 0.0f)
165 return;
166
167 // FIXME: I'm assuming that this only gets called once per frame!
168 // If not, we can't just re-allocate the IB or VB, we'll have to do a proper allocator.
169 ImGui_ImplDX12_Data* bd = ImGui_ImplDX12_GetBackendData();
170 bd->frameIndex = bd->frameIndex + 1;
171 ImGui_ImplDX12_RenderBuffers* fr = &bd->pFrameResources[bd->frameIndex % bd->numFramesInFlight];
172
173 // Create and grow vertex/index buffers if needed
174 if (fr->VertexBuffer == nullptr || fr->VertexBufferSize < draw_data->TotalVtxCount)
175 {
176 SafeRelease(fr->VertexBuffer);
177 fr->VertexBufferSize = draw_data->TotalVtxCount + 5000;
178 D3D12_HEAP_PROPERTIES props;
179 memset(&props, 0, sizeof(D3D12_HEAP_PROPERTIES));
180 props.Type = D3D12_HEAP_TYPE_UPLOAD;
181 props.CPUPageProperty = D3D12_CPU_PAGE_PROPERTY_UNKNOWN;
182 props.MemoryPoolPreference = D3D12_MEMORY_POOL_UNKNOWN;
183 D3D12_RESOURCE_DESC desc;
184 memset(&desc, 0, sizeof(D3D12_RESOURCE_DESC));
185 desc.Dimension = D3D12_RESOURCE_DIMENSION_BUFFER;
186 desc.Width = fr->VertexBufferSize * sizeof(ImDrawVert);
187 desc.Height = 1;
188 desc.DepthOrArraySize = 1;
189 desc.MipLevels = 1;
190 desc.Format = DXGI_FORMAT_UNKNOWN;
191 desc.SampleDesc.Count = 1;
192 desc.Layout = D3D12_TEXTURE_LAYOUT_ROW_MAJOR;
193 desc.Flags = D3D12_RESOURCE_FLAG_NONE;
194 if (bd->pd3dDevice->CreateCommittedResource(&props, D3D12_HEAP_FLAG_NONE, &desc, D3D12_RESOURCE_STATE_GENERIC_READ, nullptr, IID_PPV_ARGS(&fr->VertexBuffer)) < 0)
195 return;
196 }
197 if (fr->IndexBuffer == nullptr || fr->IndexBufferSize < draw_data->TotalIdxCount)
198 {
199 SafeRelease(fr->IndexBuffer);
200 fr->IndexBufferSize = draw_data->TotalIdxCount + 10000;
201 D3D12_HEAP_PROPERTIES props;
202 memset(&props, 0, sizeof(D3D12_HEAP_PROPERTIES));
203 props.Type = D3D12_HEAP_TYPE_UPLOAD;
204 props.CPUPageProperty = D3D12_CPU_PAGE_PROPERTY_UNKNOWN;
205 props.MemoryPoolPreference = D3D12_MEMORY_POOL_UNKNOWN;
206 D3D12_RESOURCE_DESC desc;
207 memset(&desc, 0, sizeof(D3D12_RESOURCE_DESC));
208 desc.Dimension = D3D12_RESOURCE_DIMENSION_BUFFER;
209 desc.Width = fr->IndexBufferSize * sizeof(ImDrawIdx);
210 desc.Height = 1;
211 desc.DepthOrArraySize = 1;
212 desc.MipLevels = 1;
213 desc.Format = DXGI_FORMAT_UNKNOWN;
214 desc.SampleDesc.Count = 1;
215 desc.Layout = D3D12_TEXTURE_LAYOUT_ROW_MAJOR;
216 desc.Flags = D3D12_RESOURCE_FLAG_NONE;
217 if (bd->pd3dDevice->CreateCommittedResource(&props, D3D12_HEAP_FLAG_NONE, &desc, D3D12_RESOURCE_STATE_GENERIC_READ, nullptr, IID_PPV_ARGS(&fr->IndexBuffer)) < 0)
218 return;
219 }
220
221 // Upload vertex/index data into a single contiguous GPU buffer
222 // During Map() we specify a null read range (as per DX12 API, this is informational and for tooling only)
223 void* vtx_resource, *idx_resource;
224 D3D12_RANGE range = { 0, 0 };
225 if (fr->VertexBuffer->Map(0, &range, &vtx_resource) != S_OK)
226 return;
227 if (fr->IndexBuffer->Map(0, &range, &idx_resource) != S_OK)
228 return;
229 ImDrawVert* vtx_dst = (ImDrawVert*)vtx_resource;
230 ImDrawIdx* idx_dst = (ImDrawIdx*)idx_resource;
231 for (int n = 0; n < draw_data->CmdListsCount; n++)
232 {
233 const ImDrawList* draw_list = draw_data->CmdLists[n];
234 memcpy(vtx_dst, draw_list->VtxBuffer.Data, draw_list->VtxBuffer.Size * sizeof(ImDrawVert));
235 memcpy(idx_dst, draw_list->IdxBuffer.Data, draw_list->IdxBuffer.Size * sizeof(ImDrawIdx));
236 vtx_dst += draw_list->VtxBuffer.Size;
237 idx_dst += draw_list->IdxBuffer.Size;
238 }
239
240 // During Unmap() we specify the written range (as per DX12 API, this is informational and for tooling only)
241 range.End = (SIZE_T)((intptr_t)vtx_dst - (intptr_t)vtx_resource);
242 IM_ASSERT(range.End == draw_data->TotalVtxCount * sizeof(ImDrawVert));
243 fr->VertexBuffer->Unmap(0, &range);
244 range.End = (SIZE_T)((intptr_t)idx_dst - (intptr_t)idx_resource);
245 IM_ASSERT(range.End == draw_data->TotalIdxCount * sizeof(ImDrawIdx));
246 fr->IndexBuffer->Unmap(0, &range);
247
248 // Setup desired DX state
249 ImGui_ImplDX12_SetupRenderState(draw_data, command_list, fr);
250
251 // Setup render state structure (for callbacks and custom texture bindings)
252 ImGuiPlatformIO& platform_io = ImGui::GetPlatformIO();
253 ImGui_ImplDX12_RenderState render_state;
254 render_state.Device = bd->pd3dDevice;
255 render_state.CommandList = command_list;
256 platform_io.Renderer_RenderState = &render_state;
257
258 // Render command lists
259 // (Because we merged all buffers into a single one, we maintain our own offset into them)
260 int global_vtx_offset = 0;
261 int global_idx_offset = 0;
262 ImVec2 clip_off = draw_data->DisplayPos;
263 for (int n = 0; n < draw_data->CmdListsCount; n++)
264 {
265 const ImDrawList* draw_list = draw_data->CmdLists[n];
266 for (int cmd_i = 0; cmd_i < draw_list->CmdBuffer.Size; cmd_i++)
267 {
268 const ImDrawCmd* pcmd = &draw_list->CmdBuffer[cmd_i];
269 if (pcmd->UserCallback != nullptr)
270 {
271 // User callback, registered via ImDrawList::AddCallback()
272 // (ImDrawCallback_ResetRenderState is a special callback value used by the user to request the renderer to reset render state.)
273 if (pcmd->UserCallback == ImDrawCallback_ResetRenderState)
274 ImGui_ImplDX12_SetupRenderState(draw_data, command_list, fr);
275 else
276 pcmd->UserCallback(draw_list, pcmd);
277 }
278 else
279 {
280 // Project scissor/clipping rectangles into framebuffer space
281 ImVec2 clip_min(pcmd->ClipRect.x - clip_off.x, pcmd->ClipRect.y - clip_off.y);
282 ImVec2 clip_max(pcmd->ClipRect.z - clip_off.x, pcmd->ClipRect.w - clip_off.y);
283 if (clip_max.x <= clip_min.x || clip_max.y <= clip_min.y)
284 continue;
285
286 // Apply scissor/clipping rectangle
287 const D3D12_RECT r = { (LONG)clip_min.x, (LONG)clip_min.y, (LONG)clip_max.x, (LONG)clip_max.y };
288 command_list->RSSetScissorRects(1, &r);
289
290 // Bind texture, Draw
291 D3D12_GPU_DESCRIPTOR_HANDLE texture_handle = {};
292 texture_handle.ptr = (UINT64)pcmd->GetTexID();
293 command_list->SetGraphicsRootDescriptorTable(1, texture_handle);
294 command_list->DrawIndexedInstanced(pcmd->ElemCount, 1, pcmd->IdxOffset + global_idx_offset, pcmd->VtxOffset + global_vtx_offset, 0);
295 }
296 }
297 global_idx_offset += draw_list->IdxBuffer.Size;
298 global_vtx_offset += draw_list->VtxBuffer.Size;
299 }
300 platform_io.Renderer_RenderState = NULL;
301}

303static void ImGui_ImplDX12_CreateFontsTexture()
304{
305 // Build texture atlas
306 ImGuiIO& io = ImGui::GetIO();
307 ImGui_ImplDX12_Data* bd = ImGui_ImplDX12_GetBackendData();
308 unsigned char* pixels;
309 int width, height;
310 io.Fonts->GetTexDataAsRGBA32(&pixels, &width, &height);
311
312 // Upload texture to graphics system
313 {
314 D3D12_HEAP_PROPERTIES props;
315 memset(&props, 0, sizeof(D3D12_HEAP_PROPERTIES));
316 props.Type = D3D12_HEAP_TYPE_DEFAULT;
317 props.CPUPageProperty = D3D12_CPU_PAGE_PROPERTY_UNKNOWN;
318 props.MemoryPoolPreference = D3D12_MEMORY_POOL_UNKNOWN;
319
320 D3D12_RESOURCE_DESC desc;
321 ZeroMemory(&desc, sizeof(desc));
322 desc.Dimension = D3D12_RESOURCE_DIMENSION_TEXTURE2D;
323 desc.Alignment = 0;
324 desc.Width = width;
325 desc.Height = height;
326 desc.DepthOrArraySize = 1;
327 desc.MipLevels = 1;
328 desc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
329 desc.SampleDesc.Count = 1;
330 desc.SampleDesc.Quality = 0;
331 desc.Layout = D3D12_TEXTURE_LAYOUT_UNKNOWN;
332 desc.Flags = D3D12_RESOURCE_FLAG_NONE;
333
334 ID3D12Resource* pTexture = nullptr;
335 bd->pd3dDevice->CreateCommittedResource(&props, D3D12_HEAP_FLAG_NONE, &desc,
336 D3D12_RESOURCE_STATE_COPY_DEST, nullptr, IID_PPV_ARGS(&pTexture));
337
338 UINT uploadPitch = (width * 4 + D3D12_TEXTURE_DATA_PITCH_ALIGNMENT - 1u) & ~(D3D12_TEXTURE_DATA_PITCH_ALIGNMENT - 1u);
339 UINT uploadSize = height * uploadPitch;
340 desc.Dimension = D3D12_RESOURCE_DIMENSION_BUFFER;
341 desc.Alignment = 0;
342 desc.Width = uploadSize;
343 desc.Height = 1;
344 desc.DepthOrArraySize = 1;
345 desc.MipLevels = 1;
346 desc.Format = DXGI_FORMAT_UNKNOWN;
347 desc.SampleDesc.Count = 1;
348 desc.SampleDesc.Quality = 0;
349 desc.Layout = D3D12_TEXTURE_LAYOUT_ROW_MAJOR;
350 desc.Flags = D3D12_RESOURCE_FLAG_NONE;
351
352 props.Type = D3D12_HEAP_TYPE_UPLOAD;
353 props.CPUPageProperty = D3D12_CPU_PAGE_PROPERTY_UNKNOWN;
354 props.MemoryPoolPreference = D3D12_MEMORY_POOL_UNKNOWN;
355
356 ID3D12Resource* uploadBuffer = nullptr;
357 HRESULT hr = bd->pd3dDevice->CreateCommittedResource(&props, D3D12_HEAP_FLAG_NONE, &desc,
358 D3D12_RESOURCE_STATE_GENERIC_READ, nullptr, IID_PPV_ARGS(&uploadBuffer));
359 IM_ASSERT(SUCCEEDED(hr));
360
361 void* mapped = nullptr;
362 D3D12_RANGE range = { 0, uploadSize };
363 hr = uploadBuffer->Map(0, &range, &mapped);
364 IM_ASSERT(SUCCEEDED(hr));
365 for (int y = 0; y < height; y++)
366 memcpy((void*) ((uintptr_t) mapped + y * uploadPitch), pixels + y * width * 4, width * 4);
367 uploadBuffer->Unmap(0, &range);
368
369 D3D12_TEXTURE_COPY_LOCATION srcLocation = {};
370 srcLocation.pResource = uploadBuffer;
371 srcLocation.Type = D3D12_TEXTURE_COPY_TYPE_PLACED_FOOTPRINT;
372 srcLocation.PlacedFootprint.Footprint.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
373 srcLocation.PlacedFootprint.Footprint.Width = width;
374 srcLocation.PlacedFootprint.Footprint.Height = height;
375 srcLocation.PlacedFootprint.Footprint.Depth = 1;
376 srcLocation.PlacedFootprint.Footprint.RowPitch = uploadPitch;
377
378 D3D12_TEXTURE_COPY_LOCATION dstLocation = {};
379 dstLocation.pResource = pTexture;
380 dstLocation.Type = D3D12_TEXTURE_COPY_TYPE_SUBRESOURCE_INDEX;
381 dstLocation.SubresourceIndex = 0;
382
383 D3D12_RESOURCE_BARRIER barrier = {};
384 barrier.Type = D3D12_RESOURCE_BARRIER_TYPE_TRANSITION;
385 barrier.Flags = D3D12_RESOURCE_BARRIER_FLAG_NONE;
386 barrier.Transition.pResource = pTexture;
387 barrier.Transition.Subresource = D3D12_RESOURCE_BARRIER_ALL_SUBRESOURCES;
388 barrier.Transition.StateBefore = D3D12_RESOURCE_STATE_COPY_DEST;
389 barrier.Transition.StateAfter = D3D12_RESOURCE_STATE_PIXEL_SHADER_RESOURCE;
390
391 ID3D12Fence* fence = nullptr;
392 hr = bd->pd3dDevice->CreateFence(0, D3D12_FENCE_FLAG_NONE, IID_PPV_ARGS(&fence));
393 IM_ASSERT(SUCCEEDED(hr));
394
395 HANDLE event = CreateEvent(0, 0, 0, 0);
396 IM_ASSERT(event != nullptr);
397
398 D3D12_COMMAND_QUEUE_DESC queueDesc = {};
399 queueDesc.Type = D3D12_COMMAND_LIST_TYPE_DIRECT;
400 queueDesc.Flags = D3D12_COMMAND_QUEUE_FLAG_NONE;
401 queueDesc.NodeMask = 1;
402
403 ID3D12CommandQueue* cmdQueue = nullptr;
404 hr = bd->pd3dDevice->CreateCommandQueue(&queueDesc, IID_PPV_ARGS(&cmdQueue));
405 IM_ASSERT(SUCCEEDED(hr));
406
407 ID3D12CommandAllocator* cmdAlloc = nullptr;
408 hr = bd->pd3dDevice->CreateCommandAllocator(D3D12_COMMAND_LIST_TYPE_DIRECT, IID_PPV_ARGS(&cmdAlloc));
409 IM_ASSERT(SUCCEEDED(hr));
410
411 ID3D12GraphicsCommandList* cmdList = nullptr;
412 hr = bd->pd3dDevice->CreateCommandList(0, D3D12_COMMAND_LIST_TYPE_DIRECT, cmdAlloc, nullptr, IID_PPV_ARGS(&cmdList));
413 IM_ASSERT(SUCCEEDED(hr));
414
415 cmdList->CopyTextureRegion(&dstLocation, 0, 0, 0, &srcLocation, nullptr);
416 cmdList->ResourceBarrier(1, &barrier);
417
418 hr = cmdList->Close();
419 IM_ASSERT(SUCCEEDED(hr));
420
421 cmdQueue->ExecuteCommandLists(1, (ID3D12CommandList* const*)&cmdList);
422 hr = cmdQueue->Signal(fence, 1);
423 IM_ASSERT(SUCCEEDED(hr));
424
425 fence->SetEventOnCompletion(1, event);
426 WaitForSingleObject(event, INFINITE);
427
428 cmdList->Release();
429 cmdAlloc->Release();
430 cmdQueue->Release();
431 CloseHandle(event);
432 fence->Release();
433 uploadBuffer->Release();
434
435 // Create texture view
436 D3D12_SHADER_RESOURCE_VIEW_DESC srvDesc;
437 ZeroMemory(&srvDesc, sizeof(srvDesc));
438 srvDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
439 srvDesc.ViewDimension = D3D12_SRV_DIMENSION_TEXTURE2D;
440 srvDesc.Texture2D.MipLevels = desc.MipLevels;
441 srvDesc.Texture2D.MostDetailedMip = 0;
442 srvDesc.Shader4ComponentMapping = D3D12_DEFAULT_SHADER_4_COMPONENT_MAPPING;
443 bd->pd3dDevice->CreateShaderResourceView(pTexture, &srvDesc, bd->hFontSrvCpuDescHandle);
444 SafeRelease(bd->pFontTextureResource);
445 bd->pFontTextureResource = pTexture;
446 }
447
448 // Store our identifier
449 io.Fonts->SetTexID((ImTextureID)bd->hFontSrvGpuDescHandle.ptr);
450}

452bool ImGui_ImplDX12_CreateDeviceObjects()
453{
454 ImGui_ImplDX12_Data* bd = ImGui_ImplDX12_GetBackendData();
455 if (!bd || !bd->pd3dDevice)
456 return false;
457 if (bd->pPipelineState)
458 ImGui_ImplDX12_InvalidateDeviceObjects();
459
460 // Create the root signature
461 {
462 D3D12_DESCRIPTOR_RANGE descRange = {};
463 descRange.RangeType = D3D12_DESCRIPTOR_RANGE_TYPE_SRV;
464 descRange.NumDescriptors = 1;
465 descRange.BaseShaderRegister = 0;
466 descRange.RegisterSpace = 0;
467 descRange.OffsetInDescriptorsFromTableStart = 0;
468
469 D3D12_ROOT_PARAMETER param[2] = {};
470
471 param[0].ParameterType = D3D12_ROOT_PARAMETER_TYPE_32BIT_CONSTANTS;
472 param[0].Constants.ShaderRegister = 0;
473 param[0].Constants.RegisterSpace = 0;
474 param[0].Constants.Num32BitValues = 16;
475 param[0].ShaderVisibility = D3D12_SHADER_VISIBILITY_VERTEX;
476
477 param[1].ParameterType = D3D12_ROOT_PARAMETER_TYPE_DESCRIPTOR_TABLE;
478 param[1].DescriptorTable.NumDescriptorRanges = 1;
479 param[1].DescriptorTable.pDescriptorRanges = &descRange;
480 param[1].ShaderVisibility = D3D12_SHADER_VISIBILITY_PIXEL;
481
482 // Bilinear sampling is required by default. Set 'io.Fonts->Flags |= ImFontAtlasFlags_NoBakedLines' or 'style.AntiAliasedLinesUseTex = false' to allow point/nearest sampling.
483 D3D12_STATIC_SAMPLER_DESC staticSampler = {};

503 desc.Flags =

515 // (1) the current OS is Windows 7, and
516 // (2) there exists a version of d3d12.dll for Windows 7 (D3D12On7) in one of the following directories.

528 return false;

532 if (D3D12SerializeRootSignatureFn == nullptr)

539 bd->pd3dDevice->CreateRootSignature(0, blob->GetBufferPointer(), blob->GetBufferSize(), IID_PPV_ARGS(&bd->pRootSignature));
540 blob->Release();

543 // By using D3DCompile() from <d3dcompiler.h> / d3dcompiler.lib, we introduce a dependency to a given version of d3dcompiler_XX.dll (see D3DCOMPILER_DLL_A)
544 // If you would like to use this DX12 sample code but remove this dependency you can:
545 // 1) compile once, save the compiled shader blobs into a file or source code and assign them to psoDesc.VS/PS [preferred solution]
546 // 2) use code to detect any version of the DLL and grab a pointer to D3DCompile from the DLL.
547 // See https://github.com/ocornut/imgui/pull/638 for sources and details.

549 D3D12_GRAPHICS_PIPELINE_STATE_DESC psoDesc;
550 memset(&psoDesc, 0, sizeof(D3D12_GRAPHICS_PIPELINE_STATE_DESC));
551 psoDesc.NodeMask = 1;
552 psoDesc.PrimitiveTopologyType = D3D12_PRIMITIVE_TOPOLOGY_TYPE_TRIANGLE;
553 psoDesc.pRootSignature = bd->pRootSignature;
554 psoDesc.SampleMask = UINT_MAX;
555 psoDesc.NumRenderTargets = 1;

560 ID3DBlob* vertexShaderBlob;

563 // Create the vertex shader
564 {
565 static const char* vertexShader =

569 };\

575 };\

582 };\
583 \
584 PS_INPUT main(VS_INPUT input)\
585 {\
586 PS_INPUT output;\
587 output.pos = mul( ProjectionMatrix, float4(input.pos.xy, 0.f, 1.f));\
588 output.col = input.col;\
589 output.uv = input.uv;\

595 psoDesc.VS = { vertexShaderBlob->GetBufferPointer(), vertexShaderBlob->GetBufferSize() };

599 {
600 { "POSITION", 0, DXGI_FORMAT_R32G32_FLOAT, 0, (UINT)offsetof(ImDrawVert, pos), D3D12_INPUT_CLASSIFICATION_PER_VERTEX_DATA, 0 },
601 { "TEXCOORD", 0, DXGI_FORMAT_R32G32_FLOAT, 0, (UINT)offsetof(ImDrawVert, uv), D3D12_INPUT_CLASSIFICATION_PER_VERTEX_DATA, 0 },
602 { "COLOR", 0, DXGI_FORMAT_R8G8B8A8_UNORM, 0, (UINT)offsetof(ImDrawVert, col), D3D12_INPUT_CLASSIFICATION_PER_VERTEX_DATA, 0 },
603 };
604 psoDesc.InputLayout = { local_layout, 3 };
605 }

607 // Create the pixel shader
608 {
609 static const char* pixelShader =
610 "struct PS_INPUT\
611 {\
612 float4 pos : SV_POSITION;\
613 float4 col : COLOR0;\
614 float2 uv : TEXCOORD0;\
615 };\
616 SamplerState sampler0 : register(s0);\
617 Texture2D texture0 : register(t0);\
618 \
619 float4 main(PS_INPUT input) : SV_Target\
620 {\
621 float4 out_col = input.col * texture0.Sample(sampler0, input.uv); \
622 return out_col; \
623 }";
624
625 if (FAILED(D3DCompile(pixelShader, strlen(pixelShader), nullptr, nullptr, nullptr, "main", "ps_5_0", 0, 0, &pixelShaderBlob, nullptr)))
626 {
627 vertexShaderBlob->Release();
628 return false; // NB: Pass ID3DBlob* pErrorBlob to D3DCompile() to get error showing in (const char*)pErrorBlob->GetBufferPointer(). Make sure to Release() the blob!
629 }
630 psoDesc.PS = { pixelShaderBlob->GetBufferPointer(), pixelShaderBlob->GetBufferSize() };
631 }

633 // Create the blending setup
634 {
635 D3D12_BLEND_DESC& desc = psoDesc.BlendState;
636 desc.AlphaToCoverageEnable = false;
637 desc.RenderTarget[0].BlendEnable = true;
638 desc.RenderTarget[0].SrcBlend = D3D12_BLEND_SRC_ALPHA;

645 }

647 // Create the rasterizer state
648 {
649 D3D12_RASTERIZER_DESC& desc = psoDesc.RasterizerState;
650 desc.FillMode = D3D12_FILL_MODE_SOLID;
651 desc.CullMode = D3D12_CULL_MODE_NONE;
652 desc.FrontCounterClockwise = FALSE;
653 desc.DepthBias = D3D12_DEFAULT_DEPTH_BIAS;

659 desc.ForcedSampleCount = 0;

661 }

664 {
665 D3D12_DEPTH_STENCIL_DESC& desc = psoDesc.DepthStencilState;
666 desc.DepthEnable = false;
667 desc.DepthWriteMask = D3D12_DEPTH_WRITE_MASK_ALL;

669 desc.StencilEnable = false;
670 desc.FrontFace.StencilFailOp = desc.FrontFace.StencilDepthFailOp = desc.FrontFace.StencilPassOp = D3D12_STENCIL_OP_KEEP;
671 desc.FrontFace.StencilFunc = D3D12_COMPARISON_FUNC_ALWAYS;
672 desc.BackFace = desc.FrontFace;
673 }
674
675 HRESULT result_pipeline_state = bd->pd3dDevice->CreateGraphicsPipelineState(&psoDesc, IID_PPV_ARGS(&bd->pPipelineState));
676 vertexShaderBlob->Release();
677 pixelShaderBlob->Release();
678 if (result_pipeline_state != S_OK)
679 return false;

681 ImGui_ImplDX12_CreateFontsTexture();

686void ImGui_ImplDX12_InvalidateDeviceObjects()
687{
688 ImGui_ImplDX12_Data* bd = ImGui_ImplDX12_GetBackendData();
689 if (!bd || !bd->pd3dDevice)
690 return;

701 SafeRelease(fr->IndexBuffer);

706bool ImGui_ImplDX12_Init(ID3D12Device* device, int num_frames_in_flight, DXGI_FORMAT rtv_format, ID3D12DescriptorHeap* cbv_srv_heap,
707 D3D12_CPU_DESCRIPTOR_HANDLE font_srv_cpu_desc_handle, D3D12_GPU_DESCRIPTOR_HANDLE font_srv_gpu_desc_handle)

712

715 io.BackendRendererUserData = (void*)bd;
716 io.BackendRendererName = "imgui_impl_dx12";
717 io.BackendFlags |= ImGuiBackendFlags_RendererHasVtxOffset; // We can honor the ImDrawCmd::VtxOffset field, allowing for large meshes.
718
719 bd->pd3dDevice = device;
720 bd->RTVFormat = rtv_format;
721 bd->hFontSrvCpuDescHandle = font_srv_cpu_desc_handle;
722 bd->hFontSrvGpuDescHandle = font_srv_gpu_desc_handle;
723 bd->pFrameResources = new ImGui_ImplDX12_RenderBuffers[num_frames_in_flight];
724 bd->numFramesInFlight = num_frames_in_flight;
725 bd->pd3dSrvDescHeap = cbv_srv_heap;
726 bd->frameIndex = UINT_MAX;

729 for (int i = 0; i < num_frames_in_flight; i++)
