112vkb::PhysicalDeviceSelector Atlas::PhysicalDevice::selectDevice(VulkanInstanceWrapper& cVulkanInstanceRef)
113{
114 vkb::PhysicalDeviceSelector selector{cVulkanInstanceRef.getVulkanBootstrapInstance()};
115
116 if(mSelectionConstraints.preferredDeviceName.has_value()) {
117
118 selector.set_name(mSelectionConstraints.preferredDeviceName.value());
119
120 } else {
121
122 selector.set_minimum_version(mSelectionConstraints.minimumAPIVersion.major, mSelectionConstraints.minimumAPIVersion.minor);
123 selector.set_required_features_13(mSelectionConstraints.physicalDeviceFeatures.vulkan13Features);
124 selector.set_required_features_12(mSelectionConstraints.physicalDeviceFeatures.vulkan12Features);
125 selector.allow_any_gpu_device_type(mSelectionConstraints.bAllowAnyDeviceType);
126
127 if (mSelectionConstraints.bDeferSurfaceInit) {
128 selector.defer_surface_initialization();
129 }
130
131 if (mSelectionConstraints.bDisablePortabilitySubset) {
132 selector.disable_portability_subset();
133 }
134
135 if (mSelectionConstraints.bRequireDedicatedComputeQueue) {
136 selector.require_dedicated_compute_queue();
137 }
138
139 if (mSelectionConstraints.bRequireDedicatedTransferQueue) {
140 selector.require_dedicated_transfer_queue();
141 }
142
143 if (mSelectionConstraints.bRequireSeparateComputeQueue) {
144 selector.require_separate_compute_queue();
145 }
146
147 if (mSelectionConstraints.bRequireSeparateTransferQueue) {
148 selector.require_separate_transfer_queue();
149 }
150
151 selector.prefer_gpu_device_type(ToVkbPreferredDeviceType(mSelectionConstraints.preferredDeviceType));
152 selector.required_device_memory_size(mSelectionConstraints.requiredDeviceMemorySize);
153
154 selector.require_present(mSelectionConstraints.bRequirePresent);
155 selector.select_first_device_unconditionally(mSelectionConstraints.bAlwaysSelectFirstDevice);
156 }
157
158 selector.add_required_extensions(mSelectionConstraints.requiredDeviceExtensions);
159 selector.set_surface(mSelectionConstraints.surface);
160
161 return selector;
162}
