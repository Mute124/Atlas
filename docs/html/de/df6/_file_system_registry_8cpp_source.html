64{
65 std::string filename = path.substr(path.find_last_of("\\") + 1);
66 std::string extension = filename.substr(filename.find_last_of(".") + 1);
67
68 // create file meta

71 uint16_t index = GetNextFileIndex();

73 // add to configLookup table

75 this->mLookupTable[meta.filename] = index;
76 this->mLookupTable[meta.extension] = index;
77 this->mLookupTable[meta.sandboxPath] = index;
78
79 // Register file
80 this->mFileMetaRegistry[index] = std::make_shared<FileMeta>(meta);
81

84 if (this->mLoadOnRegisterExtensionsSet.contains(extension)) {
85 loadFile(this->mRegisteredFiles[index]);
86 }

88 std::string output = std::format("File with index[{}], with the meta of: {} was registered. ", index, meta.format());

119void Atlas::FileSystemRegistry::loadFile(std::shared_ptr<RegisteredFile> file) {
120 if (!sIsReady) {
121 std::string registryNotReadyMsg = std::format("Registry is not ready; cannot load file: {}!", file->getFileMeta()->format());
122
123 Log(registryNotReadyMsg, ELogLevel::WARNING);
124
125 throw ImpatientUsageException(registryNotReadyMsg, std::source_location::current());
126 return;
127 }
128
129 std::scoped_lock lock(file->getLoadMutex());
130 std::shared_ptr<FileMeta> fileMeta = file->getFileMeta();
131
132 if (fileMeta == nullptr) {
133 Log("FileMeta is null!");
134 return;
135 }
136
137 if (!mLoadingFunctions.contains(fileMeta->extension)) {
138 Log("No load function found for extension: " + fileMeta->extension);
139
140 throw MissingFileLoadFunctionException(fileMeta->extension, std::source_location::current());
141 return;
142 }
143
144 Log("Loading file: " + fileMeta->filename + " (" + fileMeta->extension + ")", ELogLevel::TRACE);
145 FileLoadFuncType loadFunc = mLoadingFunctions.at(fileMeta->extension);
146
147 if (loadFunc == nullptr) {
148 Log("No load function found for extension: " + fileMeta->extension);
149 return;
150 }
151
152 try {
153 file->mFileData = loadFunc(fileMeta);
154
155 if (!file->mFileData.has_value()) {
156 std::string msg = std::format("Failed to load file: {} because the load function returned null or empty data.", fileMeta->format());
157 throw AccessViolationException(msg, std::source_location::current());
158 }
159 } catch(std::bad_any_cast& e) {
160
161 std::string msg = std::format(
162 "Failed to load file: {} because it was not of the expected type. Exception: {}",
163 fileMeta->format(),
164 e.what()
165 );
166
167 Log(msg, ELogLevel::ERROR);
168 file->mFileData = nullptr;
169
170 return;
171 }
172
173 file->mIsLoaded = true;
174}
