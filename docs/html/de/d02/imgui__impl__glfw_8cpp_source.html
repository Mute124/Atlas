1// dear imgui: Platform Backend for GLFW
2// This needs to be used along with a Renderer (e.g. OpenGL3, Vulkan, WebGPU..)
3// (Info: GLFW is a cross-platform general purpose library for handling windows, inputs, OpenGL/Vulkan graphics context creation, etc.)
4// (Requires: GLFW 3.1+. Prefer GLFW 3.3+ or GLFW 3.4+ for full feature support.)
5
6// Implemented features:
7// [X] Platform: Clipboard support.
8// [X] Platform: Mouse support. Can discriminate Mouse/TouchScreen/Pen (Windows only).
9// [X] Platform: Keyboard support. Since 1.87 we are using the io.AddKeyEvent() function. Pass ImGuiKey values to all key functions e.g. ImGui::IsKeyPressed(ImGuiKey_Space). [Legacy GLFW_KEY_* values are obsolete since 1.87 and not supported since 1.91.5]
10// [X] Platform: Gamepad support. Enable with 'io.ConfigFlags |= ImGuiConfigFlags_NavEnableGamepad'.
11// [X] Platform: Mouse cursor shape and visibility. Disable with 'io.ConfigFlags |= ImGuiConfigFlags_NoMouseCursorChange' (note: the resizing cursors requires GLFW 3.4+).

175// Backend data stored in io.BackendPlatformUserData to allow support for multiple Dear ImGui contexts
176// It is STRONGLY preferred that you use docking branch with multi-viewports (== single Dear ImGui context + multiple windows) instead of multiple Dear ImGui contexts.
177// FIXME: multi-context support is not well tested and probably dysfunctional in this backend.
178// - Because glfwPollEvents() process all windows and some events may be called outside of it, you will need to register your own callbacks
179// (passing install_callbacks=false in ImGui_ImplGlfw_InitXXX functions), set the current dear imgui context and then call our callbacks.
180// - Otherwise we may need to store a GLFWWindow* -> ImGuiContext* map and handle this in the backend, adding a little bit of extra complexity to it.
181// FIXME: some shared resources (mouse cursor shape, gamepad) are mishandled when using multi-context.
182static ImGui_ImplGlfw_Data* ImGui_ImplGlfw_GetBackendData()
183{
184 return ImGui::GetCurrentContext() ? (ImGui_ImplGlfw_Data*)ImGui::GetIO().BackendPlatformUserData : nullptr;
185}

317// X11 does not include current pressed/released modifier key in 'mods' flags submitted by GLFW
318// See https://github.com/ocornut/imgui/issues/6034 and https://github.com/glfw/glfw/issues/1630
319static void ImGui_ImplGlfw_UpdateKeyModifiers(GLFWwindow* window)
320{
321 ImGuiIO& io = ImGui::GetIO();
322 io.AddKeyEvent(ImGuiMod_Ctrl, (glfwGetKey(window, GLFW_KEY_LEFT_CONTROL) == GLFW_PRESS) || (glfwGetKey(window, GLFW_KEY_RIGHT_CONTROL) == GLFW_PRESS));
323 io.AddKeyEvent(ImGuiMod_Shift, (glfwGetKey(window, GLFW_KEY_LEFT_SHIFT) == GLFW_PRESS) || (glfwGetKey(window, GLFW_KEY_RIGHT_SHIFT) == GLFW_PRESS));
324 io.AddKeyEvent(ImGuiMod_Alt, (glfwGetKey(window, GLFW_KEY_LEFT_ALT) == GLFW_PRESS) || (glfwGetKey(window, GLFW_KEY_RIGHT_ALT) == GLFW_PRESS));
325 io.AddKeyEvent(ImGuiMod_Super, (glfwGetKey(window, GLFW_KEY_LEFT_SUPER) == GLFW_PRESS) || (glfwGetKey(window, GLFW_KEY_RIGHT_SUPER) == GLFW_PRESS));
326}

334void ImGui_ImplGlfw_MouseButtonCallback(GLFWwindow* window, int button, int action, int mods)
335{
336 ImGui_ImplGlfw_Data* bd = ImGui_ImplGlfw_GetBackendData();
337 if (bd->PrevUserCallbackMousebutton != nullptr && ImGui_ImplGlfw_ShouldChainCallback(window))
338 bd->PrevUserCallbackMousebutton(window, button, action, mods);
339
340 ImGui_ImplGlfw_UpdateKeyModifiers(window);
341
342 ImGuiIO& io = ImGui::GetIO();
343 if (button >= 0 && button < ImGuiMouseButton_COUNT)
344 io.AddMouseButtonEvent(button, action == GLFW_PRESS);
345}

363static int ImGui_ImplGlfw_TranslateUntranslatedKey(int key, int scancode)
364{
365#if GLFW_HAS_GETKEYNAME && !defined(EMSCRIPTEN_USE_EMBEDDED_GLFW3)
366 // GLFW 3.1+ attempts to "untranslate" keys, which goes the opposite of what every other framework does, making using lettered shortcuts difficult.
367 // (It had reasons to do so: namely GLFW is/was more likely to be used for WASD-type game controls rather than lettered shortcuts, but IHMO the 3.1 change could have been done differently)
368 // See https://github.com/glfw/glfw/issues/1502 for details.
369 // Adding a workaround to undo this (so our keys are translated->untranslated->translated, likely a lossy process).
370 // This won't cover edge cases but this is at least going to cover common cases.
371 if (key >= GLFW_KEY_KP_0 && key <= GLFW_KEY_KP_EQUAL)
372 return key;
373 GLFWerrorfun prev_error_callback = glfwSetErrorCallback(nullptr);
374 const char* key_name = glfwGetKeyName(key, scancode);
375 glfwSetErrorCallback(prev_error_callback);
376#if GLFW_HAS_GETERROR && !defined(EMSCRIPTEN_USE_EMBEDDED_GLFW3) // Eat errors (see #5908)
377 (void)glfwGetError(nullptr);
378#endif
379 if (key_name && key_name[0] != 0 && key_name[1] == 0)
380 {
381 const char char_names[] = "`-=[]\\,;\'./";
382 const int char_keys[] = { GLFW_KEY_GRAVE_ACCENT, GLFW_KEY_MINUS, GLFW_KEY_EQUAL, GLFW_KEY_LEFT_BRACKET, GLFW_KEY_RIGHT_BRACKET, GLFW_KEY_BACKSLASH, GLFW_KEY_COMMA, GLFW_KEY_SEMICOLON, GLFW_KEY_APOSTROPHE, GLFW_KEY_PERIOD, GLFW_KEY_SLASH, 0 };
383 IM_ASSERT(IM_ARRAYSIZE(char_names) == IM_ARRAYSIZE(char_keys));
384 if (key_name[0] >= '0' && key_name[0] <= '9') { key = GLFW_KEY_0 + (key_name[0] - '0'); }
385 else if (key_name[0] >= 'A' && key_name[0] <= 'Z') { key = GLFW_KEY_A + (key_name[0] - 'A'); }
386 else if (key_name[0] >= 'a' && key_name[0] <= 'z') { key = GLFW_KEY_A + (key_name[0] - 'a'); }
387 else if (const char* p = strchr(char_names, key_name[0])) { key = char_keys[p - char_names]; }
388 }
389 // if (action == GLFW_PRESS) printf("key %d scancode %d name '%s'\n", key, scancode, key_name);
390#else
391 IM_UNUSED(scancode);
392#endif
393 return key;
394}

580static bool ImGui_ImplGlfw_Init(GLFWwindow* window, bool install_callbacks, GlfwClientApi client_api)
581{
582 ImGuiIO& io = ImGui::GetIO();
583 IMGUI_CHECKVERSION();
584 IM_ASSERT(io.BackendPlatformUserData == nullptr && "Already initialized a platform backend!");
585 //printf("GLFW_VERSION: %d.%d.%d (%d)", GLFW_VERSION_MAJOR, GLFW_VERSION_MINOR, GLFW_VERSION_REVISION, GLFW_VERSION_COMBINED);
586
587 // Setup backend capabilities flags
588 ImGui_ImplGlfw_Data* bd = IM_NEW(ImGui_ImplGlfw_Data)();
589 io.BackendPlatformUserData = (void*)bd;
590 io.BackendPlatformName = "imgui_impl_glfw";
591 io.BackendFlags |= ImGuiBackendFlags_HasMouseCursors; // We can honor GetMouseCursor() values (optional)
592 io.BackendFlags |= ImGuiBackendFlags_HasSetMousePos; // We can honor io.WantSetMousePos requests (optional, rarely used)
593
594 bd->Window = window;
595 bd->Time = 0.0;
596
597 ImGuiPlatformIO& platform_io = ImGui::GetPlatformIO();
598 platform_io.Platform_SetClipboardTextFn = [](ImGuiContext*, const char* text) { glfwSetClipboardString(NULL, text); };
599 platform_io.Platform_GetClipboardTextFn = [](ImGuiContext*) { return glfwGetClipboardString(NULL); };
600#ifdef __EMSCRIPTEN__
601 platform_io.Platform_OpenInShellFn = [](ImGuiContext*, const char* url) { ImGui_ImplGlfw_EmscriptenOpenURL(url); return true; };
602#endif
603
604 // Create mouse cursors
605 // (By design, on X11 cursors are user configurable and some cursors may be missing. When a cursor doesn't exist,
606 // GLFW will emit an error which will often be printed by the app, so we temporarily disable error reporting.
607 // Missing cursors will return nullptr and our _UpdateMouseCursor() function will use the Arrow cursor instead.)
608 GLFWerrorfun prev_error_callback = glfwSetErrorCallback(nullptr);
609 bd->MouseCursors[ImGuiMouseCursor_Arrow] = glfwCreateStandardCursor(GLFW_ARROW_CURSOR);
610 bd->MouseCursors[ImGuiMouseCursor_TextInput] = glfwCreateStandardCursor(GLFW_IBEAM_CURSOR);
611 bd->MouseCursors[ImGuiMouseCursor_ResizeNS] = glfwCreateStandardCursor(GLFW_VRESIZE_CURSOR);
612 bd->MouseCursors[ImGuiMouseCursor_ResizeEW] = glfwCreateStandardCursor(GLFW_HRESIZE_CURSOR);
613 bd->MouseCursors[ImGuiMouseCursor_Hand] = glfwCreateStandardCursor(GLFW_HAND_CURSOR);
614#if GLFW_HAS_NEW_CURSORS
615 bd->MouseCursors[ImGuiMouseCursor_ResizeAll] = glfwCreateStandardCursor(GLFW_RESIZE_ALL_CURSOR);
616 bd->MouseCursors[ImGuiMouseCursor_ResizeNESW] = glfwCreateStandardCursor(GLFW_RESIZE_NESW_CURSOR);
617 bd->MouseCursors[ImGuiMouseCursor_ResizeNWSE] = glfwCreateStandardCursor(GLFW_RESIZE_NWSE_CURSOR);
618 bd->MouseCursors[ImGuiMouseCursor_NotAllowed] = glfwCreateStandardCursor(GLFW_NOT_ALLOWED_CURSOR);
619#else
620 bd->MouseCursors[ImGuiMouseCursor_ResizeAll] = glfwCreateStandardCursor(GLFW_ARROW_CURSOR);
621 bd->MouseCursors[ImGuiMouseCursor_ResizeNESW] = glfwCreateStandardCursor(GLFW_ARROW_CURSOR);
622 bd->MouseCursors[ImGuiMouseCursor_ResizeNWSE] = glfwCreateStandardCursor(GLFW_ARROW_CURSOR);
623 bd->MouseCursors[ImGuiMouseCursor_NotAllowed] = glfwCreateStandardCursor(GLFW_ARROW_CURSOR);
624#endif
625 glfwSetErrorCallback(prev_error_callback);
626#if GLFW_HAS_GETERROR && !defined(__EMSCRIPTEN__) // Eat errors (see #5908)
627 (void)glfwGetError(nullptr);
628#endif
629
630 // Chain GLFW callbacks: our callbacks will call the user's previously installed callbacks, if any.
631 if (install_callbacks)
632 ImGui_ImplGlfw_InstallCallbacks(window);
633
634 // Set platform dependent data in viewport
635 ImGuiViewport* main_viewport = ImGui::GetMainViewport();
636 main_viewport->PlatformHandle = (void*)bd->Window;
637#ifdef _WIN32
638 main_viewport->PlatformHandleRaw = glfwGetWin32Window(bd->Window);
639#elif defined(__APPLE__)
640 main_viewport->PlatformHandleRaw = (void*)glfwGetCocoaWindow(bd->Window);
641#else
642 IM_UNUSED(main_viewport);
643#endif
644
645 // Windows: register a WndProc hook so we can intercept some messages.
646#ifdef _WIN32
647 bd->PrevWndProc = (WNDPROC)::GetWindowLongPtrW((HWND)main_viewport->PlatformHandleRaw, GWLP_WNDPROC);
648 IM_ASSERT(bd->PrevWndProc != nullptr);
649 ::SetWindowLongPtrW((HWND)main_viewport->PlatformHandleRaw, GWLP_WNDPROC, (LONG_PTR)ImGui_ImplGlfw_WndProc);
650#endif
651
652 // Emscripten: the same application can run on various platforms, so we detect the Apple platform at runtime
653 // to override io.ConfigMacOSXBehaviors from its default (which is always false in Emscripten).
654#ifdef __EMSCRIPTEN__
655#if EMSCRIPTEN_USE_PORT_CONTRIB_GLFW3 >= 34020240817
656 if (emscripten::glfw3::IsRuntimePlatformApple())
657 {
658 ImGui::GetIO().ConfigMacOSXBehaviors = true;
659
660 // Due to how the browser (poorly) handles the Meta Key, this line essentially disables repeats when used.
661 // This means that Meta + V only registers a single key-press, even if the keys are held.
662 // This is a compromise for dealing with this issue in ImGui since ImGui implements key repeat itself.
663 // See https://github.com/pongasoft/emscripten-glfw/blob/v3.4.0.20240817/docs/Usage.md#the-problem-of-the-super-key
664 emscripten::glfw3::SetSuperPlusKeyTimeouts(10, 10);
665 }
666#endif
667#endif
668
669 bd->ClientApi = client_api;
670 return true;
671}

775static inline float Saturate(float v) { return v < 0.0f ? 0.0f : v > 1.0f ? 1.0f : v; }
776static void ImGui_ImplGlfw_UpdateGamepads()
777{
778 ImGuiIO& io = ImGui::GetIO();
779 if ((io.ConfigFlags & ImGuiConfigFlags_NavEnableGamepad) == 0) // FIXME: Technically feeding gamepad shouldn't depend on this now that they are regular inputs.
780 return;
781
782 io.BackendFlags &= ~ImGuiBackendFlags_HasGamepad;
783#if GLFW_HAS_GAMEPAD_API && !defined(EMSCRIPTEN_USE_EMBEDDED_GLFW3)
784 GLFWgamepadstate gamepad;
785 if (!glfwGetGamepadState(GLFW_JOYSTICK_1, &gamepad))
786 return;
787 #define MAP_BUTTON(KEY_NO, BUTTON_NO, _UNUSED) do { io.AddKeyEvent(KEY_NO, gamepad.buttons[BUTTON_NO] != 0); } while (0)
788 #define MAP_ANALOG(KEY_NO, AXIS_NO, _UNUSED, V0, V1) do { float v = gamepad.axes[AXIS_NO]; v = (v - V0) / (V1 - V0); io.AddKeyAnalogEvent(KEY_NO, v > 0.10f, Saturate(v)); } while (0)
789#else
790 int axes_count = 0, buttons_count = 0;
791 const float* axes = glfwGetJoystickAxes(GLFW_JOYSTICK_1, &axes_count);
792 const unsigned char* buttons = glfwGetJoystickButtons(GLFW_JOYSTICK_1, &buttons_count);
793 if (axes_count == 0 || buttons_count == 0)
794 return;
795 #define MAP_BUTTON(KEY_NO, _UNUSED, BUTTON_NO) do { io.AddKeyEvent(KEY_NO, (buttons_count > BUTTON_NO && buttons[BUTTON_NO] == GLFW_PRESS)); } while (0)
796 #define MAP_ANALOG(KEY_NO, _UNUSED, AXIS_NO, V0, V1) do { float v = (axes_count > AXIS_NO) ? axes[AXIS_NO] : V0; v = (v - V0) / (V1 - V0); io.AddKeyAnalogEvent(KEY_NO, v > 0.10f, Saturate(v)); } while (0)
797#endif
798 io.BackendFlags |= ImGuiBackendFlags_HasGamepad;
799 MAP_BUTTON(ImGuiKey_GamepadStart, GLFW_GAMEPAD_BUTTON_START, 7);
800 MAP_BUTTON(ImGuiKey_GamepadBack, GLFW_GAMEPAD_BUTTON_BACK, 6);
801 MAP_BUTTON(ImGuiKey_GamepadFaceLeft, GLFW_GAMEPAD_BUTTON_X, 2); // Xbox X, PS Square
802 MAP_BUTTON(ImGuiKey_GamepadFaceRight, GLFW_GAMEPAD_BUTTON_B, 1); // Xbox B, PS Circle
803 MAP_BUTTON(ImGuiKey_GamepadFaceUp, GLFW_GAMEPAD_BUTTON_Y, 3); // Xbox Y, PS Triangle
804 MAP_BUTTON(ImGuiKey_GamepadFaceDown, GLFW_GAMEPAD_BUTTON_A, 0); // Xbox A, PS Cross
805 MAP_BUTTON(ImGuiKey_GamepadDpadLeft, GLFW_GAMEPAD_BUTTON_DPAD_LEFT, 13);
806 MAP_BUTTON(ImGuiKey_GamepadDpadRight, GLFW_GAMEPAD_BUTTON_DPAD_RIGHT, 11);
807 MAP_BUTTON(ImGuiKey_GamepadDpadUp, GLFW_GAMEPAD_BUTTON_DPAD_UP, 10);

827void ImGui_ImplGlfw_NewFrame()
828{
829 ImGuiIO& io = ImGui::GetIO();
830 ImGui_ImplGlfw_Data* bd = ImGui_ImplGlfw_GetBackendData();
831 IM_ASSERT(bd != nullptr && "Context or backend not initialized! Did you call ImGui_ImplGlfw_InitForXXX()?");
832
833 // Setup display size (every frame to accommodate for window resizing)
834 int w, h;
835 int display_w, display_h;
836 glfwGetWindowSize(bd->Window, &w, &h);
837 glfwGetFramebufferSize(bd->Window, &display_w, &display_h);
838 io.DisplaySize = ImVec2((float)w, (float)h);
839 if (w > 0 && h > 0)
840 io.DisplayFramebufferScale = ImVec2((float)display_w / (float)w, (float)display_h / (float)h);
841
842 // Setup time step
843 // (Accept glfwGetTime() not returning a monotonically increasing value. Seems to happens on disconnecting peripherals and probably on VMs and Emscripten, see #6491, #6189, #6114, #3644)
844 double current_time = glfwGetTime();
845 if (current_time <= bd->Time)
846 current_time = bd->Time + 0.00001f;
847 io.DeltaTime = bd->Time > 0.0 ? (float)(current_time - bd->Time) : (float)(1.0f / 60.0f);
848 bd->Time = current_time;
849
850 ImGui_ImplGlfw_UpdateMouseData();
851 ImGui_ImplGlfw_UpdateMouseCursor();
852
853 // Update game controllers (if enabled and available)
854 ImGui_ImplGlfw_UpdateGamepads();
855}
