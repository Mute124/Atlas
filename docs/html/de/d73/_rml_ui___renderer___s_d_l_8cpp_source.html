36RenderInterface_SDL::RenderInterface_SDL(SDL_Renderer* renderer) : renderer(renderer)
37{
38 // RmlUi serves vertex colors and textures with premultiplied alpha, set the blend mode accordingly.
39 // Equivalent to glBlendFunc(GL_ONE, GL_ONE_MINUS_SRC_ALPHA).
40 blend_mode = SDL_ComposeCustomBlendMode(SDL_BLENDFACTOR_ONE, SDL_BLENDFACTOR_ONE_MINUS_SRC_ALPHA, SDL_BLENDOPERATION_ADD, SDL_BLENDFACTOR_ONE,
41 SDL_BLENDFACTOR_ONE_MINUS_SRC_ALPHA, SDL_BLENDOPERATION_ADD);
42}

65void RenderInterface_SDL::RenderGeometry(Rml::CompiledGeometryHandle handle, Rml::Vector2f translation, Rml::TextureHandle texture)
66{
67 const GeometryView* geometry = reinterpret_cast<GeometryView*>(handle);
68 const Rml::Vertex* vertices = geometry->vertices.data();
69 const size_t num_vertices = geometry->vertices.size();
70 const int* indices = geometry->indices.data();
71 const size_t num_indices = geometry->indices.size();
72
73 SDL_FPoint* positions = new SDL_FPoint[num_vertices];
74
75 for (size_t i = 0; i < num_vertices; i++)
76 {
77 positions[i].x = vertices[i].position.x + translation.x;
78 positions[i].y = vertices[i].position.y + translation.y;
79 }
80
81 SDL_Texture* sdl_texture = (SDL_Texture*)texture;
82
83 SDL_RenderGeometryRaw(renderer, sdl_texture, &positions[0].x, sizeof(SDL_FPoint), (const SDL_Color*)&vertices->colour, sizeof(Rml::Vertex),
84 &vertices->tex_coord.x, sizeof(Rml::Vertex), (int)num_vertices, indices, (int)num_indices, 4);
85
86 delete[] positions;
87}

110Rml::TextureHandle RenderInterface_SDL::LoadTexture(Rml::Vector2i& texture_dimensions, const Rml::String& source)
111{
112 Rml::FileInterface* file_interface = Rml::GetFileInterface();
113 Rml::FileHandle file_handle = file_interface->Open(source);
114 if (!file_handle)
115 return {};
116
117 file_interface->Seek(file_handle, 0, SEEK_END);
118 size_t buffer_size = file_interface->Tell(file_handle);
119 file_interface->Seek(file_handle, 0, SEEK_SET);
120
121 using Rml::byte;
122 Rml::UniquePtr<byte[]> buffer(new byte[buffer_size]);
123 file_interface->Read(buffer.get(), buffer_size, file_handle);
124 file_interface->Close(file_handle);
125
126 const size_t i_ext = source.rfind('.');
127 Rml::String extension = (i_ext == Rml::String::npos ? Rml::String() : source.substr(i_ext + 1));
128
129 SDL_Surface* surface = IMG_LoadTyped_RW(SDL_RWFromMem(buffer.get(), int(buffer_size)), 1, extension.c_str());
130 if (!surface)
131 return {};
132
133 if (surface->format->format != SDL_PIXELFORMAT_RGBA32 && surface->format->format != SDL_PIXELFORMAT_BGRA32)
134 {
135 SDL_Surface* converted_surface = SDL_ConvertSurfaceFormat(surface, SDL_PIXELFORMAT_RGBA32, 0);
136 SDL_FreeSurface(surface);
137
138 if (!converted_surface)
139 return {};
140
141 surface = converted_surface;
142 }
143
144 // Convert colors to premultiplied alpha, which is necessary for correct alpha compositing.
145 byte* pixels = static_cast<byte*>(surface->pixels);
146 for (int i = 0; i < surface->w * surface->h * 4; i += 4)
147 {
148 const byte alpha = pixels[i + 3];
149 for (int j = 0; j < 3; ++j)
150 pixels[i + j] = byte(int(pixels[i + j]) * int(alpha) / 255);
151 }
152
153 SDL_Texture* texture = SDL_CreateTextureFromSurface(renderer, surface);
154
155 texture_dimensions = Rml::Vector2i(surface->w, surface->h);
156 SDL_FreeSurface(surface);
157
158 if (texture)
159 SDL_SetTextureBlendMode(texture, blend_mode);
160
161 return (Rml::TextureHandle)texture;
162}

164Rml::TextureHandle RenderInterface_SDL::GenerateTexture(Rml::Span<const Rml::byte> source, Rml::Vector2i source_dimensions)
165{
166 SDL_Surface* surface = SDL_CreateRGBSurfaceWithFormatFrom((void*)source.data(), source_dimensions.x, source_dimensions.y, 32,
167 source_dimensions.x * 4, SDL_PIXELFORMAT_RGBA32);
168
169 SDL_Texture* texture = SDL_CreateTextureFromSurface(renderer, surface);
170 SDL_SetTextureBlendMode(texture, blend_mode);
171
172 SDL_FreeSurface(surface);
173 return (Rml::TextureHandle)texture;
174}
