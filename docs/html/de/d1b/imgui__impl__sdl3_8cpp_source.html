141{
142 ImGui_ImplSDL3_Data* bd = ImGui_ImplSDL3_GetBackendData();
143 SDL_WindowID window_id = (SDL_WindowID)(intptr_t)viewport->PlatformHandle;
144 SDL_Window* window = SDL_GetWindowFromID(window_id);
145 if ((data->WantVisible == false || bd->ImeWindow != window) && bd->ImeWindow != NULL)
146 {
147 SDL_StopTextInput(bd->ImeWindow);
148 bd->ImeWindow = nullptr;
149 }
150 if (data->WantVisible)
151 {
152 SDL_Rect r;
153 r.x = (int)data->InputPos.x;
154 r.y = (int)data->InputPos.y;
155 r.w = 1;
156 r.h = (int)data->InputLineHeight;
157 SDL_SetTextInputArea(window, &r, 0);
158 SDL_StartTextInput(window);
159 bd->ImeWindow = window;
160 }
161}

314// You can read the io.WantCaptureMouse, io.WantCaptureKeyboard flags to tell if dear imgui wants to use your inputs.
315// - When io.WantCaptureMouse is true, do not dispatch mouse input data to your main application, or clear/overwrite your copy of the mouse data.
316// - When io.WantCaptureKeyboard is true, do not dispatch keyboard input data to your main application, or clear/overwrite your copy of the keyboard data.
317// Generally you may always pass all inputs to dear imgui, and hide them from your application based on those two flags.
318// If you have multiple SDL events and some of them are not meant to be used by dear imgui, you may need to filter events based on their windowID field.
319bool ImGui_ImplSDL3_ProcessEvent(const SDL_Event* event)
320{
321 ImGui_ImplSDL3_Data* bd = ImGui_ImplSDL3_GetBackendData();
322 IM_ASSERT(bd != nullptr && "Context or backend not initialized! Did you call ImGui_ImplSDL3_Init()?");
323 ImGuiIO& io = ImGui::GetIO();
324
325 switch (event->type)
326 {
327 case SDL_EVENT_MOUSE_MOTION:
328 {
329 if (ImGui_ImplSDL3_GetViewportForWindowID(event->motion.windowID) == NULL)
330 return false;
331 ImVec2 mouse_pos((float)event->motion.x, (float)event->motion.y);
332 io.AddMouseSourceEvent(event->motion.which == SDL_TOUCH_MOUSEID ? ImGuiMouseSource_TouchScreen : ImGuiMouseSource_Mouse);
333 io.AddMousePosEvent(mouse_pos.x, mouse_pos.y);
334 return true;
335 }
336 case SDL_EVENT_MOUSE_WHEEL:
337 {
338 if (ImGui_ImplSDL3_GetViewportForWindowID(event->wheel.windowID) == NULL)
339 return false;
340 //IMGUI_DEBUG_LOG("wheel %.2f %.2f, precise %.2f %.2f\n", (float)event->wheel.x, (float)event->wheel.y, event->wheel.preciseX, event->wheel.preciseY);
341 float wheel_x = -event->wheel.x;
342 float wheel_y = event->wheel.y;
343 io.AddMouseSourceEvent(event->wheel.which == SDL_TOUCH_MOUSEID ? ImGuiMouseSource_TouchScreen : ImGuiMouseSource_Mouse);
344 io.AddMouseWheelEvent(wheel_x, wheel_y);
345 return true;
346 }
347 case SDL_EVENT_MOUSE_BUTTON_DOWN:
348 case SDL_EVENT_MOUSE_BUTTON_UP:
349 {
350 if (ImGui_ImplSDL3_GetViewportForWindowID(event->button.windowID) == NULL)
351 return false;
352 int mouse_button = -1;
353 if (event->button.button == SDL_BUTTON_LEFT) { mouse_button = 0; }
354 if (event->button.button == SDL_BUTTON_RIGHT) { mouse_button = 1; }
355 if (event->button.button == SDL_BUTTON_MIDDLE) { mouse_button = 2; }
356 if (event->button.button == SDL_BUTTON_X1) { mouse_button = 3; }
357 if (event->button.button == SDL_BUTTON_X2) { mouse_button = 4; }
358 if (mouse_button == -1)
359 break;
360 io.AddMouseSourceEvent(event->button.which == SDL_TOUCH_MOUSEID ? ImGuiMouseSource_TouchScreen : ImGuiMouseSource_Mouse);
361 io.AddMouseButtonEvent(mouse_button, (event->type == SDL_EVENT_MOUSE_BUTTON_DOWN));
362 bd->MouseButtonsDown = (event->type == SDL_EVENT_MOUSE_BUTTON_DOWN) ? (bd->MouseButtonsDown | (1 << mouse_button)) : (bd->MouseButtonsDown & ~(1 << mouse_button));
363 return true;
364 }
365 case SDL_EVENT_TEXT_INPUT:
366 {
367 if (ImGui_ImplSDL3_GetViewportForWindowID(event->text.windowID) == NULL)
368 return false;
369 io.AddInputCharactersUTF8(event->text.text);
370 return true;
371 }
372 case SDL_EVENT_KEY_DOWN:
373 case SDL_EVENT_KEY_UP:
374 {
375 if (ImGui_ImplSDL3_GetViewportForWindowID(event->key.windowID) == NULL)
376 return false;
377 //IMGUI_DEBUG_LOG("SDL_EVENT_KEY_%d: key=%d, scancode=%d, mod=%X\n", (event->type == SDL_EVENT_KEY_DOWN) ? "DOWN" : "UP", event->key.key, event->key.scancode, event->key.mod);
378 ImGui_ImplSDL3_UpdateKeyModifiers((SDL_Keymod)event->key.mod);
379 ImGuiKey key = ImGui_ImplSDL3_KeyEventToImGuiKey(event->key.key, event->key.scancode);
380 io.AddKeyEvent(key, (event->type == SDL_EVENT_KEY_DOWN));
381 io.SetKeyEventNativeData(key, event->key.key, event->key.scancode, event->key.scancode); // To support legacy indexing (<1.87 user code). Legacy backend uses SDLK_*** as indices to IsKeyXXX() functions.
382 return true;
383 }
384 case SDL_EVENT_WINDOW_MOUSE_ENTER:
385 {
386 if (ImGui_ImplSDL3_GetViewportForWindowID(event->window.windowID) == NULL)
387 return false;
388 bd->MouseWindowID = event->window.windowID;
389 bd->MousePendingLeaveFrame = 0;
390 return true;
391 }
392 // - In some cases, when detaching a window from main viewport SDL may send SDL_WINDOWEVENT_ENTER one frame too late,
393 // causing SDL_WINDOWEVENT_LEAVE on previous frame to interrupt drag operation by clear mouse position. This is why
394 // we delay process the SDL_WINDOWEVENT_LEAVE events by one frame. See issue #5012 for details.
395 // FIXME: Unconfirmed whether this is still needed with SDL3.
396 case SDL_EVENT_WINDOW_MOUSE_LEAVE:
397 {
398 if (ImGui_ImplSDL3_GetViewportForWindowID(event->window.windowID) == NULL)
399 return false;
400 bd->MousePendingLeaveFrame = ImGui::GetFrameCount() + 1;
401 return true;
402 }
403 case SDL_EVENT_WINDOW_FOCUS_GAINED:
404 case SDL_EVENT_WINDOW_FOCUS_LOST:
405 {
406 if (ImGui_ImplSDL3_GetViewportForWindowID(event->window.windowID) == NULL)
407 return false;
408 io.AddFocusEvent(event->type == SDL_EVENT_WINDOW_FOCUS_GAINED);
409 return true;
410 }
411 case SDL_EVENT_GAMEPAD_ADDED:
412 case SDL_EVENT_GAMEPAD_REMOVED:
413 {
414 bd->WantUpdateGamepadsList = true;
415 return true;
416 }
417 }
418 return false;
419}

432static bool ImGui_ImplSDL3_Init(SDL_Window* window, SDL_Renderer* renderer, void* sdl_gl_context)
433{
434 ImGuiIO& io = ImGui::GetIO();
435 IMGUI_CHECKVERSION();
436 IM_ASSERT(io.BackendPlatformUserData == nullptr && "Already initialized a platform backend!");
437 IM_UNUSED(sdl_gl_context); // Unused in this branch
438
439 // Check and store if we are on a SDL backend that supports global mouse position
440 // ("wayland" and "rpi" don't support it, but we chose to use a white-list instead of a black-list)
441 bool mouse_can_use_global_state = false;
442#if SDL_HAS_CAPTURE_AND_GLOBAL_MOUSE
443 const char* sdl_backend = SDL_GetCurrentVideoDriver();
444 const char* global_mouse_whitelist[] = { "windows", "cocoa", "x11", "DIVE", "VMAN" };
445 for (int n = 0; n < IM_ARRAYSIZE(global_mouse_whitelist); n++)
446 if (strncmp(sdl_backend, global_mouse_whitelist[n], strlen(global_mouse_whitelist[n])) == 0)
447 mouse_can_use_global_state = true;
448#endif
449
450 // Setup backend capabilities flags
451 ImGui_ImplSDL3_Data* bd = IM_NEW(ImGui_ImplSDL3_Data)();
452 io.BackendPlatformUserData = (void*)bd;
453 io.BackendPlatformName = "imgui_impl_sdl3";
454 io.BackendFlags |= ImGuiBackendFlags_HasMouseCursors; // We can honor GetMouseCursor() values (optional)
455 io.BackendFlags |= ImGuiBackendFlags_HasSetMousePos; // We can honor io.WantSetMousePos requests (optional, rarely used)
456
457 bd->Window = window;
458 bd->WindowID = SDL_GetWindowID(window);

558static void ImGui_ImplSDL3_UpdateMouseData()
559{
560 ImGui_ImplSDL3_Data* bd = ImGui_ImplSDL3_GetBackendData();
561 ImGuiIO& io = ImGui::GetIO();
562
563 // We forward mouse input when hovered or captured (via SDL_EVENT_MOUSE_MOTION) or when focused (below)
564#if SDL_HAS_CAPTURE_AND_GLOBAL_MOUSE
565 // SDL_CaptureMouse() let the OS know e.g. that our imgui drag outside the SDL window boundaries shouldn't e.g. trigger other operations outside
566 SDL_CaptureMouse(bd->MouseButtonsDown != 0);
567 SDL_Window* focused_window = SDL_GetKeyboardFocus();
568 const bool is_app_focused = (bd->Window == focused_window);
569#else
570 SDL_Window* focused_window = bd->Window;
571 const bool is_app_focused = (SDL_GetWindowFlags(bd->Window) & SDL_WINDOW_INPUT_FOCUS) != 0; // SDL 2.0.3 and non-windowed systems: single-viewport only
572#endif
573 if (is_app_focused)
574 {
575 // (Optional) Set OS mouse position from Dear ImGui if requested (rarely used, only when io.ConfigNavMoveSetMousePos is enabled by user)
576 if (io.WantSetMousePos)
577 SDL_WarpMouseInWindow(bd->Window, io.MousePos.x, io.MousePos.y);
578
579 // (Optional) Fallback to provide mouse position when focused (SDL_EVENT_MOUSE_MOTION already provides this when hovered or captured)
580 if (bd->MouseCanUseGlobalState && bd->MouseButtonsDown == 0)
581 {
582 // Single-viewport mode: mouse position in client window coordinates (io.MousePos is (0,0) when the mouse is on the upper-left corner of the app window)
583 float mouse_x_global, mouse_y_global;
584 int window_x, window_y;
585 SDL_GetGlobalMouseState(&mouse_x_global, &mouse_y_global);
586 SDL_GetWindowPosition(focused_window, &window_x, &window_y);
587 io.AddMousePosEvent(mouse_x_global - window_x, mouse_y_global - window_y);
588 }
589 }
590}

653static inline float Saturate(float v) { return v < 0.0f ? 0.0f : v > 1.0f ? 1.0f : v; }
654static void ImGui_ImplSDL3_UpdateGamepadAnalog(ImGui_ImplSDL3_Data* bd, ImGuiIO& io, ImGuiKey key, SDL_GamepadAxis axis_no, float v0, float v1)

724void ImGui_ImplSDL3_NewFrame()
725{
726 ImGui_ImplSDL3_Data* bd = ImGui_ImplSDL3_GetBackendData();
727 IM_ASSERT(bd != nullptr && "Context or backend not initialized! Did you call ImGui_ImplSDL3_Init()?");
728 ImGuiIO& io = ImGui::GetIO();
729
730 // Setup display size (every frame to accommodate for window resizing)
731 int w, h;
732 int display_w, display_h;
733 SDL_GetWindowSize(bd->Window, &w, &h);
734 if (SDL_GetWindowFlags(bd->Window) & SDL_WINDOW_MINIMIZED)
735 w = h = 0;
736 SDL_GetWindowSizeInPixels(bd->Window, &display_w, &display_h);
737 io.DisplaySize = ImVec2((float)w, (float)h);
738 if (w > 0 && h > 0)
739 io.DisplayFramebufferScale = ImVec2((float)display_w / w, (float)display_h / h);
740
741 // Setup time step (we don't use SDL_GetTicks() because it is using millisecond resolution)
742 // (Accept SDL_GetPerformanceCounter() not returning a monotonically increasing value. Happens in VMs and Emscripten, see #6189, #6114, #3644)
743 static Uint64 frequency = SDL_GetPerformanceFrequency();
744 Uint64 current_time = SDL_GetPerformanceCounter();
745 if (current_time <= bd->Time)
746 current_time = bd->Time + 1;
747 io.DeltaTime = bd->Time > 0 ? (float)((double)(current_time - bd->Time) / frequency) : (float)(1.0f / 60.0f);
748 bd->Time = current_time;
749
750 if (bd->MousePendingLeaveFrame && bd->MousePendingLeaveFrame >= ImGui::GetFrameCount() && bd->MouseButtonsDown == 0)
751 {
752 bd->MouseWindowID = 0;
753 bd->MousePendingLeaveFrame = 0;
754 io.AddMousePosEvent(-FLT_MAX, -FLT_MAX);
755 }
756
757 ImGui_ImplSDL3_UpdateMouseData();
758 ImGui_ImplSDL3_UpdateMouseCursor();
759
760 // Update game controllers (if enabled and available)
761 ImGui_ImplSDL3_UpdateGamepads();
762}
