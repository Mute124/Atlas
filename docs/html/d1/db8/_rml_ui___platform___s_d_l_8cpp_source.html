70void SystemInterface_SDL::SetMouseCursor(const Rml::String& cursor_name)
71{
72 SDL_Cursor* cursor = nullptr;
73
74 if (cursor_name.empty() || cursor_name == "arrow")
75 cursor = cursor_default;
76 else if (cursor_name == "move")
77 cursor = cursor_move;
78 else if (cursor_name == "pointer")
79 cursor = cursor_pointer;
80 else if (cursor_name == "resize")
81 cursor = cursor_resize;
82 else if (cursor_name == "cross")
83 cursor = cursor_cross;
84 else if (cursor_name == "text")
85 cursor = cursor_text;
86 else if (cursor_name == "unavailable")
87 cursor = cursor_unavailable;
88 else if (Rml::StringUtilities::StartsWith(cursor_name, "rmlui-scroll"))
89 cursor = cursor_move;
90
91 if (cursor)
92 SDL_SetCursor(cursor);
93}

107bool RmlSDL::InputEventHandler(Rml::Context* context, SDL_Event& ev)
108{
109 bool result = true;
110
111 switch (ev.type)
112 {
113 case SDL_MOUSEMOTION: result = context->ProcessMouseMove(ev.motion.x, ev.motion.y, GetKeyModifierState()); break;
114 case SDL_MOUSEBUTTONDOWN:
115 result = context->ProcessMouseButtonDown(ConvertMouseButton(ev.button.button), GetKeyModifierState());
116 SDL_CaptureMouse(SDL_TRUE);
117 break;
118 case SDL_MOUSEBUTTONUP:
119 SDL_CaptureMouse(SDL_FALSE);
120 result = context->ProcessMouseButtonUp(ConvertMouseButton(ev.button.button), GetKeyModifierState());
121 break;
122 case SDL_MOUSEWHEEL: result = context->ProcessMouseWheel(float(-ev.wheel.y), GetKeyModifierState()); break;
123 case SDL_KEYDOWN:
124 result = context->ProcessKeyDown(ConvertKey(ev.key.keysym.sym), GetKeyModifierState());
125 if (ev.key.keysym.sym == SDLK_RETURN || ev.key.keysym.sym == SDLK_KP_ENTER)
126 result &= context->ProcessTextInput('\n');
127 break;
128 case SDL_KEYUP: result = context->ProcessKeyUp(ConvertKey(ev.key.keysym.sym), GetKeyModifierState()); break;
129 case SDL_TEXTINPUT: result = context->ProcessTextInput(Rml::String(&ev.text.text[0])); break;
130 case SDL_WINDOWEVENT:
131 {
132 switch (ev.window.event)
133 {
134 case SDL_WINDOWEVENT_SIZE_CHANGED:
135 {
136 Rml::Vector2i dimensions(ev.window.data1, ev.window.data2);
137 context->SetDimensions(dimensions);
138 }
139 break;
140 case SDL_WINDOWEVENT_LEAVE: context->ProcessMouseLeave(); break;
141 }
142 }
143 break;
144 default: break;
145 }
146
147 return result;
148}

286int RmlSDL::GetKeyModifierState()
287{
288 SDL_Keymod sdl_mods = SDL_GetModState();
289
290 int retval = 0;
291
292 if (sdl_mods & KMOD_CTRL)
293 retval |= Rml::Input::KM_CTRL;
294
295 if (sdl_mods & KMOD_SHIFT)
296 retval |= Rml::Input::KM_SHIFT;
297
298 if (sdl_mods & KMOD_ALT)
299 retval |= Rml::Input::KM_ALT;
300
301 if (sdl_mods & KMOD_NUM)
302 retval |= Rml::Input::KM_NUMLOCK;
303
304 if (sdl_mods & KMOD_CAPS)
305 retval |= Rml::Input::KM_CAPSLOCK;
306
307 return retval;
308}
