34void CSerializer::Clear()
35{
36 // Extra objects need to be released, since they are not stored in
37 // the module and we cannot rely on the application releasing them
38 for (size_t i = 0; i < m_extraObjects.size(); i++)
39 {
40 SExtraObject& o = m_extraObjects[i];
41 for (size_t i2 = 0; i2 < m_root.m_children.size(); i2++)
42 {
43 if (m_root.m_children[i2]->m_originalPtr == o.originalObject && m_root.m_children[i2]->m_restorePtr)
44 reinterpret_cast<asIScriptObject*>(m_root.m_children[i2]->m_restorePtr)->Release();
45 }
46 }
47 m_extraObjects.resize(0);
48
49 // Clean the serialized values before we remove the user types
50 m_root.Uninit();
51
52 if (m_engine)
53 m_engine->Release();
54 m_engine = 0;
55}

62int CSerializer::Store(asIScriptModule *mod)
63{
64 m_mod = mod;
65
66 // The engine must not be destroyed before we're completed, so we'll hold on to a reference
67 mod->GetEngine()->AddRef();
68 if( m_engine ) m_engine->Release();
69 m_engine = mod->GetEngine();
70
71 m_root.m_serializer = this;
72
73 // First store global variables
74 asUINT i;
75 for( i = 0; i < mod->GetGlobalVarCount(); i++ )
76 {
77 const char *name, *nameSpace;
78 int typeId;
79 mod->GetGlobalVar(i, &name, &nameSpace, &typeId);
80 m_root.m_children.push_back(new CSerializedValue(&m_root, name, nameSpace, mod->GetAddressOfGlobalVar(i), typeId));
81 }
82
83 // Second store extra objects
84 for( i = 0; i < m_extraObjects.size(); i++ )
85 m_root.m_children.push_back(new CSerializedValue(&m_root, "", "", m_extraObjects[i].originalObject, m_extraObjects[i].originalTypeId));
86
87 // For the handles that were stored, we need to substitute the stored pointer
88 // that is still pointing to the original object to an internal reference so
89 // it can be restored later on.
90 m_root.ReplaceHandles();
91
92 return 0;
93}

95// Retrieve all global variables after reload script.
96int CSerializer::Restore(asIScriptModule *mod)
97{
98 m_mod = mod;
99
100 // The engine must not be destroyed before we're completed, so we'll hold on to a reference
101 mod->GetEngine()->AddRef();
102 if( m_engine ) m_engine->Release();
103 m_engine = mod->GetEngine();
104
105 // First restore extra objects, i.e. the ones that are not directly seen from the module's global variables
106 asUINT i;
107 for( i = 0; i < m_extraObjects.size(); i++ )
108 {
109 SExtraObject &o = m_extraObjects[i];
110 asITypeInfo *type = m_mod->GetTypeInfoByName( o.originalClassName.c_str() );
111 if( type )
112 {
113 for( size_t i2 = 0; i2 < m_root.m_children.size(); i2++ )
114 {
115 if( m_root.m_children[i2]->m_originalPtr == o.originalObject )
116 {
117 // Create a new script object, but don't call its constructor as we will initialize the members.
118 // Calling the constructor may have unwanted side effects if for example the constructor changes
119 // any outside entities, such as setting global variables to point to new objects, etc.
120 void *newPtr = m_engine->CreateUninitializedScriptObject( type );
121 m_root.m_children[i2]->Restore( newPtr, type->GetTypeId() );
122 }
123 }
124 }
125 }
126
127 // Second restore the global variables
128 asUINT varCount = mod->GetGlobalVarCount();
129 for( i = 0; i < varCount; i++ )
130 {
131 const char *name, *nameSpace;
132 int typeId;
133 mod->GetGlobalVar(i, &name, &nameSpace, &typeId);
134
135 CSerializedValue *v = m_root.FindByName(name, nameSpace);
136 if( v )
137 v->Restore(mod->GetAddressOfGlobalVar(i), typeId);
138 }
139
140 // The handles that were restored needs to be
141 // updated to point to their final objects.
142 m_root.RestoreHandles();
143
144 return 0;
145}

214void CSerializedValue::ClearChildren()
215{
216 // If this value is for an object handle that created an object during the restore
217 // then it is necessary to release the handle here, so we won't get a memory leak
218 if( (m_typeId & asTYPEID_OBJHANDLE) && m_children.size() == 1 && m_children[0]->m_restorePtr )
219 {
220 m_serializer->m_engine->ReleaseScriptObject(m_children[0]->m_restorePtr, m_serializer->m_engine->GetTypeInfoById(m_children[0]->m_typeId));
221 }
222
223 for( size_t n = 0; n < m_children.size(); n++ )
224 delete m_children[n];
225 m_children.clear();
226}

304void CSerializedValue::Store(void *ref, int typeId)
305{
306 m_isInit = true;
307 SetType(typeId);
308 m_originalPtr = ref;
309
310 if( m_typeId & asTYPEID_OBJHANDLE )
311 {
312 m_handlePtr = *(void**)ref;
313 }
314 else if( m_typeId & asTYPEID_SCRIPTOBJECT )
315 {
316 asIScriptObject *obj = (asIScriptObject *)ref;
317 asITypeInfo *type = obj->GetObjectType();
318 SetType(type->GetTypeId());
319
320 // Store children
321 for( asUINT i = 0; i < type->GetPropertyCount(); i++ )
322 {
323 int childId;
324 const char *childName;
325 type->GetProperty(i, &childName, &childId);
326
327 m_children.push_back(new CSerializedValue(this, childName, "", obj->GetAddressOfProperty(i), childId));
328 }
329 }
330 else
331 {
332 int size = m_serializer->m_engine->GetSizeOfPrimitiveType(m_typeId);
333
334 if( size == 0 )
335 {
336 // if it is user type( string, array, etc ... )
337 if( m_serializer->m_userTypes[m_typeName] )
338 m_serializer->m_userTypes[m_typeName]->Store(this, m_originalPtr);
339 else
340 {
341 // POD-types can be stored without need for user type
342 asITypeInfo *type = GetType();
343 if( type && (type->GetFlags() & asOBJ_POD) )
344 size = GetType()->GetSize();
345
346 // It is not necessary to report an error here if it is not a POD-type as that will be done when restoring
347 }
348 }
349
350 if( size )
351 {
352 m_mem.resize(size);
353 memcpy(&m_mem[0], ref, size);
354 }
355 }

358void CSerializedValue::Restore(void *ref, int typeId)
359{
360 if( !m_isInit || !ref )
361 return;
362
363 // Verify that the stored type matched the new type of the value being restored
364 if( typeId <= asTYPEID_DOUBLE && typeId != m_typeId ) return; // TODO: We may try to do a type conversion for primitives
365 if( (typeId & ~asTYPEID_MASK_SEQNBR) ^ (m_typeId & ~asTYPEID_MASK_SEQNBR) ) return;
366 asITypeInfo *type = m_serializer->m_engine->GetTypeInfoById(typeId);
367 if( type && m_typeName != type->GetName() ) return;
368
369 // Set the new pointer and type
370 m_restorePtr = ref;
371 SetType(typeId);
372
373 // Restore the value
374 if( m_typeId & asTYPEID_OBJHANDLE )
375 {
376 // if need create objects
377 if( m_children.size() == 1 )
378 {
379 asITypeInfo *ctype = m_children[0]->GetType();
380
381 if( ctype->GetFactoryCount() == 0 )
382 {
383 // There are no factories, so assume the same pointer is going to be used
384 m_children[0]->m_restorePtr = m_handlePtr;
385
386 // Increase the refCount for the object as it will be released upon clean-up
387 m_serializer->m_engine->AddRefScriptObject(m_handlePtr, ctype);
388 }
389 else
390 {
391 // Create a new script object, but don't call its constructor as we will initialize the members.
392 // Calling the constructor may have unwanted side effects if for example the constructor changes
393 // any outside entities, such as setting global variables to point to new objects, etc.
394 void* newObject = 0;
395 if( type->GetFlags() & asOBJ_SCRIPT_OBJECT )
396 newObject = m_serializer->m_engine->CreateUninitializedScriptObject(ctype);
397 // Allocate an unitialized object for the custom type
398 else if (m_serializer->m_userTypes[m_typeName])
399 {
400 newObject = m_serializer->m_userTypes[m_typeName]->AllocateUnitializedMemory(this);
401 }
402 else
403 {
404 std::string str = "Cannot allocate memory for type '";
405 str += type->GetName();
406 str += "'";
407 m_serializer->m_engine->WriteMessage("", 0, 0, asMSGTYPE_ERROR, str.c_str());
408 }
409
410 m_children[0]->Restore(newObject, ctype->GetTypeId());
411 }
412 }
413 }
414 else if( m_typeId & asTYPEID_SCRIPTOBJECT )
415 {
416 asIScriptObject *obj = (asIScriptObject *)ref;
417
418 // Retrieve children
419 for( asUINT i = 0; i < type->GetPropertyCount() ; i++ )
420 {
421 const char *nameProperty;
422 int ptypeId;
423 type->GetProperty(i, &nameProperty, &ptypeId);
424
425 CSerializedValue *var = FindByName(nameProperty, "");
426 if( var )
427 var->Restore(obj->GetAddressOfProperty(i), ptypeId);
428 }
429 }
430 else
431 {
432 if( m_mem.size() )
433 {
434 // POD values can be restored with direct copy
435 memcpy(ref, &m_mem[0], m_mem.size());
436 }
437 else if( m_serializer->m_userTypes[m_typeName] )
438 {
439 // user type restore
440 m_serializer->m_userTypes[m_typeName]->Restore(this, m_restorePtr);
441 }
442 else
443 {
444 std::string str = "Cannot restore type '";
445 str += type->GetName();
446 str += "'";
447 m_serializer->m_engine->WriteMessage("", 0, 0, asMSGTYPE_ERROR, str.c_str());
448 }
449 }

452void CSerializedValue::CancelDuplicates(CSerializedValue *from)
453{
454 std::vector<void*> ptrs;
455 from->GetAllPointersOfChildren(&ptrs);
456
457 for( size_t i = 0; i < ptrs.size(); ++i )
458 {
459 CSerializedValue *find = m_serializer->m_root.FindByPtrInHandles(ptrs[i]);
460
461 while( find )
462 {
463 // cancel create object
464 find->ClearChildren();
465
466 // Find next link to this ptr
467 find = m_serializer->m_root.FindByPtrInHandles(ptrs[i]);
468 }
469 }

472void CSerializedValue::ReplaceHandles()
473{
474 if( m_handlePtr )
475 {
476 // Find the object that the handle is referring to
477 CSerializedValue *handle_to = m_serializer->m_root.FindByPtr(m_handlePtr);
478
479 // If the object hasn't been stored yet...
480 if( handle_to == 0 )
481 {
482 // Store the object now
483 asITypeInfo *type = GetType();
484 CSerializedValue *need_create = new CSerializedValue(this, m_name, m_nameSpace, m_handlePtr, type->GetTypeId());
485
486 // Make sure all other handles that point to the same object
487 // are updated, so we don't end up creating duplicates
488 CancelDuplicates(need_create);
489
490 m_children.push_back(need_create);
491 }
492 }
493
494 // Replace the handles in the children too
495 for( size_t i = 0; i < m_children.size(); ++i )
496 m_children[i]->ReplaceHandles();
497}

499void CSerializedValue::RestoreHandles()
500{
501 if( m_typeId & asTYPEID_OBJHANDLE )
502 {
503 if( m_handlePtr )
504 {
505 // Find the object the handle is supposed to point to
506 CSerializedValue *handleTo = m_serializer->m_root.FindByPtr(m_handlePtr);
507
508 if( m_restorePtr && handleTo && handleTo->m_restorePtr )
509 {
510 asITypeInfo *type = m_serializer->m_engine->GetTypeInfoById(m_typeId);
511
512 // If the handle is already pointing to something it must be released first
513 if( *(void**)m_restorePtr )
514 m_serializer->m_engine->ReleaseScriptObject(*(void**)m_restorePtr, type);
515
516 // Update the internal pointer
517 *(void**)m_restorePtr = handleTo->m_restorePtr;
518
519 // Increase the reference
520 m_serializer->m_engine->AddRefScriptObject(handleTo->m_restorePtr, type);
521 }
522 }
523 else
524 {
525 // If the handle is pointing to something, we must release it to restore the null pointer
526 if( m_restorePtr && *(void**)m_restorePtr )
527 {
528 m_serializer->m_engine->ReleaseScriptObject(*(void**)m_restorePtr, m_serializer->m_engine->GetTypeInfoById(m_typeId));
529 *(void**)m_restorePtr = 0;
530 }
531 }
532 }
533
534 // Do the same for the children
535 for( size_t i = 0; i < m_children.size(); ++i )
536 m_children[i]->RestoreHandles();
537}
