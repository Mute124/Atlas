1// dear imgui: Platform Backend for Windows (standard windows API for 32-bits AND 64-bits applications)
2// This needs to be used along with a Renderer (e.g. DirectX11, OpenGL3, Vulkan..)
3
4// Implemented features:
5// [X] Platform: Clipboard support (for Win32 this is actually part of core dear imgui)
6// [X] Platform: Mouse support. Can discriminate Mouse/TouchScreen/Pen.
7// [X] Platform: Keyboard support. Since 1.87 we are using the io.AddKeyEvent() function. Pass ImGuiKey values to all key functions e.g. ImGui::IsKeyPressed(ImGuiKey_Space). [Legacy VK_* values are obsolete since 1.87 and not supported since 1.91.5]

152static bool ImGui_ImplWin32_InitEx(void* hwnd, bool platform_has_own_dc)
153{
154 ImGuiIO& io = ImGui::GetIO();
155 IMGUI_CHECKVERSION();
156 IM_ASSERT(io.BackendPlatformUserData == nullptr && "Already initialized a platform backend!");
157
158 INT64 perf_frequency, perf_counter;
159 if (!::QueryPerformanceFrequency((LARGE_INTEGER*)&perf_frequency))
160 return false;
161 if (!::QueryPerformanceCounter((LARGE_INTEGER*)&perf_counter))
162 return false;
163
164 // Setup backend capabilities flags
165 ImGui_ImplWin32_Data* bd = IM_NEW(ImGui_ImplWin32_Data)();
166 io.BackendPlatformUserData = (void*)bd;
167 io.BackendPlatformName = "imgui_impl_win32";
168 io.BackendFlags |= ImGuiBackendFlags_HasMouseCursors; // We can honor GetMouseCursor() values (optional)
169 io.BackendFlags |= ImGuiBackendFlags_HasSetMousePos; // We can honor io.WantSetMousePos requests (optional, rarely used)
170
171 bd->hWnd = (HWND)hwnd;
172 bd->TicksPerSecond = perf_frequency;
173 bd->Time = perf_counter;
174 bd->LastMouseCursor = ImGuiMouseCursor_COUNT;
175 ImGui_ImplWin32_UpdateKeyboardCodePage(io);
176
177 // Set platform dependent data in viewport
178 ImGuiViewport* main_viewport = ImGui::GetMainViewport();
179 main_viewport->PlatformHandle = main_viewport->PlatformHandleRaw = (void*)bd->hWnd;
180 IM_UNUSED(platform_has_own_dc); // Used in 'docking' branch
181
182 // Dynamically load XInput library
183#ifndef IMGUI_IMPL_WIN32_DISABLE_GAMEPAD
184 bd->WantUpdateHasGamepad = true;
185 const char* xinput_dll_names[] =
186 {
187 "xinput1_4.dll", // Windows 8+
188 "xinput1_3.dll", // DirectX SDK
189 "xinput9_1_0.dll", // Windows Vista, Windows 7
190 "xinput1_2.dll", // DirectX SDK
191 "xinput1_1.dll" // DirectX SDK
192 };
193 for (int n = 0; n < IM_ARRAYSIZE(xinput_dll_names); n++)
194 if (HMODULE dll = ::LoadLibraryA(xinput_dll_names[n]))
195 {
196 bd->XInputDLL = dll;
197 bd->XInputGetCapabilities = (PFN_XInputGetCapabilities)::GetProcAddress(dll, "XInputGetCapabilities");
198 bd->XInputGetState = (PFN_XInputGetState)::GetProcAddress(dll, "XInputGetState");
199 break;
200 }
201#endif // IMGUI_IMPL_WIN32_DISABLE_GAMEPAD
202
203 return true;
204}

235static bool ImGui_ImplWin32_UpdateMouseCursor(ImGuiIO& io, ImGuiMouseCursor imgui_cursor)
236{
237 if (io.ConfigFlags & ImGuiConfigFlags_NoMouseCursorChange)
238 return false;
239
240 if (imgui_cursor == ImGuiMouseCursor_None || io.MouseDrawCursor)
241 {
242 // Hide OS mouse cursor if imgui is drawing it or if it wants no cursor
243 ::SetCursor(nullptr);
244 }
245 else
246 {
247 // Show OS mouse cursor
248 LPTSTR win32_cursor = IDC_ARROW;
249 switch (imgui_cursor)
250 {
251 case ImGuiMouseCursor_Arrow: win32_cursor = IDC_ARROW; break;
252 case ImGuiMouseCursor_TextInput: win32_cursor = IDC_IBEAM; break;
253 case ImGuiMouseCursor_ResizeAll: win32_cursor = IDC_SIZEALL; break;
254 case ImGuiMouseCursor_ResizeEW: win32_cursor = IDC_SIZEWE; break;
255 case ImGuiMouseCursor_ResizeNS: win32_cursor = IDC_SIZENS; break;
256 case ImGuiMouseCursor_ResizeNESW: win32_cursor = IDC_SIZENESW; break;
257 case ImGuiMouseCursor_ResizeNWSE: win32_cursor = IDC_SIZENWSE; break;
258 case ImGuiMouseCursor_Hand: win32_cursor = IDC_HAND; break;
259 case ImGuiMouseCursor_NotAllowed: win32_cursor = IDC_NO; break;
260 }
261 ::SetCursor(::LoadCursor(nullptr, win32_cursor));
262 }
263 return true;
264}

278static void ImGui_ImplWin32_ProcessKeyEventsWorkarounds(ImGuiIO& io)
279{
280 // Left & right Shift keys: when both are pressed together, Windows tend to not generate the WM_KEYUP event for the first released one.
281 if (ImGui::IsKeyDown(ImGuiKey_LeftShift) && !IsVkDown(VK_LSHIFT))
282 ImGui_ImplWin32_AddKeyEvent(io, ImGuiKey_LeftShift, false, VK_LSHIFT);
283 if (ImGui::IsKeyDown(ImGuiKey_RightShift) && !IsVkDown(VK_RSHIFT))
284 ImGui_ImplWin32_AddKeyEvent(io, ImGuiKey_RightShift, false, VK_RSHIFT);
285
286 // Sometimes WM_KEYUP for Win key is not passed down to the app (e.g. for Win+V on some setups, according to GLFW).
287 if (ImGui::IsKeyDown(ImGuiKey_LeftSuper) && !IsVkDown(VK_LWIN))
288 ImGui_ImplWin32_AddKeyEvent(io, ImGuiKey_LeftSuper, false, VK_LWIN);
289 if (ImGui::IsKeyDown(ImGuiKey_RightSuper) && !IsVkDown(VK_RWIN))
290 ImGui_ImplWin32_AddKeyEvent(io, ImGuiKey_RightSuper, false, VK_RWIN);

301static void ImGui_ImplWin32_UpdateMouseData(ImGuiIO& io)
302{
303 ImGui_ImplWin32_Data* bd = ImGui_ImplWin32_GetBackendData(io);
304 IM_ASSERT(bd->hWnd != 0);
305
306 HWND focused_window = ::GetForegroundWindow();
307 const bool is_app_focused = (focused_window == bd->hWnd);
308 if (is_app_focused)
309 {
310 // (Optional) Set OS mouse position from Dear ImGui if requested (rarely used, only when io.ConfigNavMoveSetMousePos is enabled by user)
311 if (io.WantSetMousePos)
312 {
313 POINT pos = { (int)io.MousePos.x, (int)io.MousePos.y };
314 if (::ClientToScreen(bd->hWnd, &pos))
315 ::SetCursorPos(pos.x, pos.y);
316 }
317
318 // (Optional) Fallback to provide mouse position when focused (WM_MOUSEMOVE already provides this when hovered or captured)
319 // This also fills a short gap when clicking non-client area: WM_NCMOUSELEAVE -> modal OS move -> gap -> WM_NCMOUSEMOVE
320 if (!io.WantSetMousePos && bd->MouseTrackedArea == 0)
321 {
322 POINT pos;
323 if (::GetCursorPos(&pos) && ::ScreenToClient(bd->hWnd, &pos))
324 io.AddMousePosEvent((float)pos.x, (float)pos.y);
325 }
326 }

562// Helper to obtain the source of mouse messages.
563// See https://learn.microsoft.com/en-us/windows/win32/tablet/system-events-and-mouse-messages
564// Prefer to call this at the top of the message handler to avoid the possibility of other Win32 calls interfering with this.
565static ImGuiMouseSource ImGui_ImplWin32_GetMouseSourceFromMessageExtraInfo()
566{
567 LPARAM extra_info = ::GetMessageExtraInfo();
568 if ((extra_info & 0xFFFFFF80) == 0xFF515700)
569 return ImGuiMouseSource_Pen;
570 if ((extra_info & 0xFFFFFF80) == 0xFF515780)
571 return ImGuiMouseSource_TouchScreen;
572 return ImGuiMouseSource_Mouse;

575// Win32 message handler (process Win32 mouse/keyboard inputs, etc.)
576// Call from your application's message handler. Keep calling your message handler unless this function returns TRUE.
577// When implementing your own backend, you can read the io.WantCaptureMouse, io.WantCaptureKeyboard flags to tell if Dear ImGui wants to use your inputs.
578// - When io.WantCaptureMouse is true, do not dispatch mouse input data to your main application, or clear/overwrite your copy of the mouse data.
579// - When io.WantCaptureKeyboard is true, do not dispatch keyboard input data to your main application, or clear/overwrite your copy of the keyboard data.
580// Generally you may always pass all inputs to Dear ImGui, and hide them from your application based on those two flags.
581// PS: We treat DBLCLK messages as regular mouse down messages, so this code will work on windows classes that have the CS_DBLCLKS flag set. Our own example app code doesn't set this flag.
582
583// Copy either line into your .cpp file to forward declare the function:
584extern IMGUI_IMPL_API LRESULT ImGui_ImplWin32_WndProcHandler(HWND hWnd, UINT msg, WPARAM wParam, LPARAM lParam); // Use ImGui::GetCurrentContext()
585extern IMGUI_IMPL_API LRESULT ImGui_ImplWin32_WndProcHandlerEx(HWND hWnd, UINT msg, WPARAM wParam, LPARAM lParam, ImGuiIO& io); // Doesn't use ImGui::GetCurrentContext()

587IMGUI_IMPL_API LRESULT ImGui_ImplWin32_WndProcHandler(HWND hwnd, UINT msg, WPARAM wParam, LPARAM lParam)
588{
589 // Most backends don't have silent checks like this one, but we need it because WndProc are called early in CreateWindow().
590 // We silently allow both context or just only backend data to be nullptr.
591 if (ImGui::GetCurrentContext() == NULL)
592 return 0;
593 return ImGui_ImplWin32_WndProcHandlerEx(hwnd, msg, wParam, lParam, ImGui::GetIO());

596// This version is in theory thread-safe in the sense that no path should access ImGui::GetCurrentContext().
597IMGUI_IMPL_API LRESULT ImGui_ImplWin32_WndProcHandlerEx(HWND hwnd, UINT msg, WPARAM wParam, LPARAM lParam, ImGuiIO& io)
598{
599 ImGui_ImplWin32_Data* bd = ImGui_ImplWin32_GetBackendData(io);
600 if (bd == NULL)
601 return 0;
602 switch (msg)
603 {
604 case WM_MOUSEMOVE:
605 case WM_NCMOUSEMOVE:
606 {
607 // We need to call TrackMouseEvent in order to receive WM_MOUSELEAVE events
608 ImGuiMouseSource mouse_source = ImGui_ImplWin32_GetMouseSourceFromMessageExtraInfo();
609 const int area = (msg == WM_MOUSEMOVE) ? 1 : 2;
610 bd->MouseHwnd = hwnd;
611 if (bd->MouseTrackedArea != area)
612 {
613 TRACKMOUSEEVENT tme_cancel = { sizeof(tme_cancel), TME_CANCEL, hwnd, 0 };
614 TRACKMOUSEEVENT tme_track = { sizeof(tme_track), (DWORD)((area == 2) ? (TME_LEAVE | TME_NONCLIENT) : TME_LEAVE), hwnd, 0 };
615 if (bd->MouseTrackedArea != 0)
616 ::TrackMouseEvent(&tme_cancel);
617 ::TrackMouseEvent(&tme_track);
618 bd->MouseTrackedArea = area;
619 }
620 POINT mouse_pos = { (LONG)GET_X_LPARAM(lParam), (LONG)GET_Y_LPARAM(lParam) };
621 if (msg == WM_NCMOUSEMOVE && ::ScreenToClient(hwnd, &mouse_pos) == FALSE) // WM_NCMOUSEMOVE are provided in absolute coordinates.
622 return 0;
623 io.AddMouseSourceEvent(mouse_source);
624 io.AddMousePosEvent((float)mouse_pos.x, (float)mouse_pos.y);
625 return 0;
626 }
627 case WM_MOUSELEAVE:
628 case WM_NCMOUSELEAVE:
629 {
630 const int area = (msg == WM_MOUSELEAVE) ? 1 : 2;
631 if (bd->MouseTrackedArea == area)
632 {
633 if (bd->MouseHwnd == hwnd)
634 bd->MouseHwnd = nullptr;
635 bd->MouseTrackedArea = 0;
636 io.AddMousePosEvent(-FLT_MAX, -FLT_MAX);
637 }
638 return 0;
639 }
640 case WM_DESTROY:
641 if (bd->MouseHwnd == hwnd && bd->MouseTrackedArea != 0)
642 {
643 TRACKMOUSEEVENT tme_cancel = { sizeof(tme_cancel), TME_CANCEL, hwnd, 0 };
644 ::TrackMouseEvent(&tme_cancel);
645 bd->MouseHwnd = nullptr;
646 bd->MouseTrackedArea = 0;
647 io.AddMousePosEvent(-FLT_MAX, -FLT_MAX);
648 }
649 return 0;
650 case WM_LBUTTONDOWN: case WM_LBUTTONDBLCLK:
651 case WM_RBUTTONDOWN: case WM_RBUTTONDBLCLK:
652 case WM_MBUTTONDOWN: case WM_MBUTTONDBLCLK:
653 case WM_XBUTTONDOWN: case WM_XBUTTONDBLCLK:
654 {
655 ImGuiMouseSource mouse_source = ImGui_ImplWin32_GetMouseSourceFromMessageExtraInfo();
656 int button = 0;
657 if (msg == WM_LBUTTONDOWN || msg == WM_LBUTTONDBLCLK) { button = 0; }
658 if (msg == WM_RBUTTONDOWN || msg == WM_RBUTTONDBLCLK) { button = 1; }
659 if (msg == WM_MBUTTONDOWN || msg == WM_MBUTTONDBLCLK) { button = 2; }
660 if (msg == WM_XBUTTONDOWN || msg == WM_XBUTTONDBLCLK) { button = (GET_XBUTTON_WPARAM(wParam) == XBUTTON1) ? 3 : 4; }
661 if (bd->MouseButtonsDown == 0 && ::GetCapture() == nullptr)
662 ::SetCapture(hwnd); // Allow us to read mouse coordinates when dragging mouse outside of our window bounds.
663 bd->MouseButtonsDown |= 1 << button;
664 io.AddMouseSourceEvent(mouse_source);
665 io.AddMouseButtonEvent(button, true);
666 return 0;
667 }
668 case WM_LBUTTONUP:
669 case WM_RBUTTONUP:
670 case WM_MBUTTONUP:
671 case WM_XBUTTONUP:
672 {
673 ImGuiMouseSource mouse_source = ImGui_ImplWin32_GetMouseSourceFromMessageExtraInfo();
674 int button = 0;
675 if (msg == WM_LBUTTONUP) { button = 0; }
676 if (msg == WM_RBUTTONUP) { button = 1; }
677 if (msg == WM_MBUTTONUP) { button = 2; }
678 if (msg == WM_XBUTTONUP) { button = (GET_XBUTTON_WPARAM(wParam) == XBUTTON1) ? 3 : 4; }
679 bd->MouseButtonsDown &= ~(1 << button);
680 if (bd->MouseButtonsDown == 0 && ::GetCapture() == hwnd)
681 ::ReleaseCapture();
682 io.AddMouseSourceEvent(mouse_source);
683 io.AddMouseButtonEvent(button, false);
684 return 0;
685 }
686 case WM_MOUSEWHEEL:
687 io.AddMouseWheelEvent(0.0f, (float)GET_WHEEL_DELTA_WPARAM(wParam) / (float)WHEEL_DELTA);
688 return 0;
689 case WM_MOUSEHWHEEL:
690 io.AddMouseWheelEvent(-(float)GET_WHEEL_DELTA_WPARAM(wParam) / (float)WHEEL_DELTA, 0.0f);
691 return 0;
692 case WM_KEYDOWN:
693 case WM_KEYUP:
694 case WM_SYSKEYDOWN:
695 case WM_SYSKEYUP:
696 {
697 const bool is_key_down = (msg == WM_KEYDOWN || msg == WM_SYSKEYDOWN);
698 if (wParam < 256)
699 {
700 // Submit modifiers
701 ImGui_ImplWin32_UpdateKeyModifiers(io);
702
703 // Obtain virtual key code and convert to ImGuiKey
704 const ImGuiKey key = ImGui_ImplWin32_KeyEventToImGuiKey(wParam, lParam);
705 const int vk = (int)wParam;
706 const int scancode = (int)LOBYTE(HIWORD(lParam));
707
708 // Special behavior for VK_SNAPSHOT / ImGuiKey_PrintScreen as Windows doesn't emit the key down event.
709 if (key == ImGuiKey_PrintScreen && !is_key_down)
710 ImGui_ImplWin32_AddKeyEvent(io, key, true, vk, scancode);
711
712 // Submit key event
713 if (key != ImGuiKey_None)
714 ImGui_ImplWin32_AddKeyEvent(io, key, is_key_down, vk, scancode);
715
716 // Submit individual left/right modifier events
717 if (vk == VK_SHIFT)
718 {
719 // Important: Shift keys tend to get stuck when pressed together, missing key-up events are corrected in ImGui_ImplWin32_ProcessKeyEventsWorkarounds()
720 if (IsVkDown(VK_LSHIFT) == is_key_down) { ImGui_ImplWin32_AddKeyEvent(io, ImGuiKey_LeftShift, is_key_down, VK_LSHIFT, scancode); }
721 if (IsVkDown(VK_RSHIFT) == is_key_down) { ImGui_ImplWin32_AddKeyEvent(io, ImGuiKey_RightShift, is_key_down, VK_RSHIFT, scancode); }
722 }
723 else if (vk == VK_CONTROL)
724 {
725 if (IsVkDown(VK_LCONTROL) == is_key_down) { ImGui_ImplWin32_AddKeyEvent(io, ImGuiKey_LeftCtrl, is_key_down, VK_LCONTROL, scancode); }
726 if (IsVkDown(VK_RCONTROL) == is_key_down) { ImGui_ImplWin32_AddKeyEvent(io, ImGuiKey_RightCtrl, is_key_down, VK_RCONTROL, scancode); }
727 }
728 else if (vk == VK_MENU)
729 {
730 if (IsVkDown(VK_LMENU) == is_key_down) { ImGui_ImplWin32_AddKeyEvent(io, ImGuiKey_LeftAlt, is_key_down, VK_LMENU, scancode); }
731 if (IsVkDown(VK_RMENU) == is_key_down) { ImGui_ImplWin32_AddKeyEvent(io, ImGuiKey_RightAlt, is_key_down, VK_RMENU, scancode); }
732 }
733 }
734 return 0;
735 }
736 case WM_SETFOCUS:
737 case WM_KILLFOCUS:
738 io.AddFocusEvent(msg == WM_SETFOCUS);
739 return 0;
740 case WM_INPUTLANGCHANGE:
741 ImGui_ImplWin32_UpdateKeyboardCodePage(io);
742 return 0;
743 case WM_CHAR:
744 if (::IsWindowUnicode(hwnd))
745 {
746 // You can also use ToAscii()+GetKeyboardState() to retrieve characters.
747 if (wParam > 0 && wParam < 0x10000)
748 io.AddInputCharacterUTF16((unsigned short)wParam);
749 }
750 else
751 {
752 wchar_t wch = 0;
753 ::MultiByteToWideChar(bd->KeyboardCodePage, MB_PRECOMPOSED, (char*)&wParam, 1, &wch, 1);
754 io.AddInputCharacter(wch);
755 }
756 return 0;
757 case WM_SETCURSOR:
758 // This is required to restore cursor when transitioning from e.g resize borders to client area.
759 if (LOWORD(lParam) == HTCLIENT && ImGui_ImplWin32_UpdateMouseCursor(io, bd->LastMouseCursor))
760 return 1;
761 return 0;
762 case WM_DEVICECHANGE:
763#ifndef IMGUI_IMPL_WIN32_DISABLE_GAMEPAD
764 if ((UINT)wParam == DBT_DEVNODES_CHANGED)
765 bd->WantUpdateHasGamepad = true;
766#endif
767 return 0;
768 }
769 return 0;
770}

775//--------------------------------------------------------------------------------------------------------
776// - Use to enable DPI awareness without having to create an application manifest.
777// - Your own app may already do this via a manifest or explicit calls. This is mostly useful for our examples/ apps.
778// - In theory we could call simple functions from Windows SDK such as SetProcessDPIAware(), SetProcessDpiAwareness(), etc.
779// but most of the functions provided by Microsoft require Windows 8.1/10+ SDK at compile time and Windows 8/10+ at runtime,
780// neither we want to require the user to have. So we dynamically select and load those functions to avoid dependencies.
781//---------------------------------------------------------------------------------------------------------
782// This is the scheme successfully used by GLFW (from which we borrowed some of the code) and other apps aiming to be highly portable.
783// ImGui_ImplWin32_EnableDpiAwareness() is just a helper called by main.cpp, we don't call it automatically.

787// Perform our own check with RtlVerifyVersionInfo() instead of using functions from <VersionHelpers.h> as they
788// require a manifest to be functional for checks above 8.1. See https://github.com/ocornut/imgui/issues/4200
789static BOOL _IsWindowsVersionOrGreater(WORD major, WORD minor, WORD)
790{
791 typedef LONG(WINAPI* PFN_RtlVerifyVersionInfo)(OSVERSIONINFOEXW*, ULONG, ULONGLONG);
792 static PFN_RtlVerifyVersionInfo RtlVerifyVersionInfoFn = nullptr;
793 if (RtlVerifyVersionInfoFn == nullptr)
794 if (HMODULE ntdllModule = ::GetModuleHandleA("ntdll.dll"))
795 RtlVerifyVersionInfoFn = (PFN_RtlVerifyVersionInfo)GetProcAddress(ntdllModule, "RtlVerifyVersionInfo");
796 if (RtlVerifyVersionInfoFn == nullptr)
797 return FALSE;
798
799 RTL_OSVERSIONINFOEXW versionInfo = { };
800 ULONGLONG conditionMask = 0;
801 versionInfo.dwOSVersionInfoSize = sizeof(RTL_OSVERSIONINFOEXW);
802 versionInfo.dwMajorVersion = major;
803 versionInfo.dwMinorVersion = minor;
804 VER_SET_CONDITION(conditionMask, VER_MAJORVERSION, VER_GREATER_EQUAL);
805 VER_SET_CONDITION(conditionMask, VER_MINORVERSION, VER_GREATER_EQUAL);
806 return (RtlVerifyVersionInfoFn(&versionInfo, VER_MAJORVERSION | VER_MINORVERSION, conditionMask) == 0) ? TRUE : FALSE;
807}
