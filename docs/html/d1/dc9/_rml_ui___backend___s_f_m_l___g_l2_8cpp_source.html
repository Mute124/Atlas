60 Rml::TextureHandle LoadTexture(Rml::Vector2i& texture_dimensions, const Rml::String& source) override
61 {
62 Rml::FileInterface* file_interface = Rml::GetFileInterface();
63 Rml::FileHandle file_handle = file_interface->Open(source);
64 if (!file_handle)
65 return false;
66
67 file_interface->Seek(file_handle, 0, SEEK_END);
68 size_t buffer_size = file_interface->Tell(file_handle);
69 file_interface->Seek(file_handle, 0, SEEK_SET);
70
71 using Rml::byte;
72 Rml::UniquePtr<byte[]> buffer(new byte[buffer_size]);
73 file_interface->Read(buffer.get(), buffer_size, file_handle);
74 file_interface->Close(file_handle);
75
76 sf::Image image;
77 if (!image.loadFromMemory(buffer.get(), buffer_size))
78 return false;
79
80 // Convert colors to premultiplied alpha, which is necessary for correct alpha compositing.
81 for (unsigned int x = 0; x < image.getSize().x; x++)
82 {
83 for (unsigned int y = 0; y < image.getSize().y; y++)
84 {
85 sf::Color color = image.getPixel(x, y);
86 color.r = (sf::Uint8)((color.r * color.a) / 255);
87 color.g = (sf::Uint8)((color.g * color.a) / 255);
88 color.b = (sf::Uint8)((color.b * color.a) / 255);
89 image.setPixel(x, y, color);
90 }
91 }
92
93 sf::Texture* texture = new sf::Texture();
94 texture->setSmooth(true);
95
96 if (!texture->loadFromImage(image))
97 {
98 delete texture;
99 return false;
100 }
101
102 texture_dimensions = Rml::Vector2i(texture->getSize().x, texture->getSize().y);
103 return (Rml::TextureHandle)texture;
104 }

106 Rml::TextureHandle GenerateTexture(Rml::Span<const Rml::byte> source, Rml::Vector2i source_dimensions) override
107 {
108 sf::Texture* texture = new sf::Texture();
109 texture->setSmooth(true);
110
111 if (!texture->create(source_dimensions.x, source_dimensions.y))
112 {
113 delete texture;
114 return false;
115 }
116
117 texture->update(source.data(), source_dimensions.x, source_dimensions.y, 0, 0);

125// Updates the viewport and context dimensions, should be called whenever the window size changes.
126static void UpdateWindowDimensions(sf::RenderWindow& window, RenderInterface_GL2_SFML& render_interface, Rml::Context* context)
127{
128 const int width = (int)window.getSize().x;
129 const int height = (int)window.getSize().y;
130
131 if (context)
132 context->SetDimensions(Rml::Vector2i(width, height));
133
134 sf::View view(sf::FloatRect(0.f, 0.f, (float)width, (float)height));
135 window.setView(view);
136
137 render_interface.SetViewport(width, height);
138}

153bool Backend::Initialize(const char* window_name, int width, int height, bool allow_resize)
154{
155 RMLUI_ASSERT(!data);
156
157 data = Rml::MakeUnique<BackendData>();
158
159 // Create the window.
160 sf::RenderWindow out_window;
161 sf::ContextSettings context_settings;
162 context_settings.stencilBits = 8;
163 context_settings.antialiasingLevel = 2;
164
165 const sf::Uint32 style = (allow_resize ? sf::Style::Default : (sf::Style::Titlebar | sf::Style::Close));
166
167 data->window.create(sf::VideoMode(width, height), window_name, style, context_settings);
168 data->window.setVerticalSyncEnabled(true);
169
170 if (!data->window.isOpen())
171 {
172 data.reset();
173 return false;
174 }
175
176 // Optionally apply the SFML window to the system interface so that it can change its mouse cursor.
177 data->system_interface.SetWindow(&data->window);
178
179 UpdateWindowDimensions(data->window, data->render_interface, nullptr);
180
181 return true;
182}

201bool Backend::ProcessEvents(Rml::Context* context, KeyDownCallback key_down_callback, bool power_save)
202{
203 RMLUI_ASSERT(data && context);
204
205 // SFML does not seem to provide a way to wait for events with a timeout.
206 (void)power_save;
207
208 // The contents of this function is intended to be copied directly into your main loop.
209 bool result = data->running;
210 data->running = true;
211
212 sf::Event ev;
213 while (data->window.pollEvent(ev))
214 {
215 switch (ev.type)
216 {
217 case sf::Event::Resized: UpdateWindowDimensions(data->window, data->render_interface, context); break;
218 case sf::Event::KeyPressed:
219 {
220 const Rml::Input::KeyIdentifier key = RmlSFML::ConvertKey(ev.key.code);
221 const int key_modifier = RmlSFML::GetKeyModifierState();
222 const float native_dp_ratio = 1.f;
223
224 // See if we have any global shortcuts that take priority over the context.
225 if (key_down_callback && !key_down_callback(context, key, key_modifier, native_dp_ratio, true))
226 break;
227 // Otherwise, hand the event over to the context by calling the input handler as normal.
228 if (!RmlSFML::InputHandler(context, ev))
229 break;
230 // The key was not consumed by the context either, try keyboard shortcuts of lower priority.
231 if (key_down_callback && !key_down_callback(context, key, key_modifier, native_dp_ratio, false))
232 break;
233 }
234 break;
235 case sf::Event::Closed: result = false; break;
236 default: RmlSFML::InputHandler(context, ev); break;
237 }
238 }
239
240 return result;
241}
