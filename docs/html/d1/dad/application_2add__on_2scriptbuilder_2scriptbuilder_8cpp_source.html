176int CScriptBuilder::LoadScriptSection(const char* filename)
177{
178 // Open the script file
179 string scriptFile = filename;
180#if _MSC_VER >= 1500 && !defined(__S3E__)
181 #ifdef _WIN32
182 // Convert the filename from UTF8 to UTF16
183 wchar_t bufUTF16_name[10000] = {0};
184 wchar_t bufUTF16_mode[10] = {0};
185 MultiByteToWideChar(CP_UTF8, 0, filename, -1, bufUTF16_name, 10000);
186 MultiByteToWideChar(CP_UTF8, 0, "rb", -1, bufUTF16_mode, 10);
187
188 FILE *f = 0;
189 _wfopen_s(&f, bufUTF16_name, bufUTF16_mode);
190 #else
191 FILE* f = 0;
192 fopen_s(&f, scriptFile.c_str(), "rb");
193 #endif
194#else
195 FILE *f = fopen(scriptFile.c_str(), "rb");
196#endif
197 if( f == 0 )
198 {
199 // Write a message to the engine's message callback
200 string msg = "Failed to open script file '" + GetAbsolutePath(scriptFile) + "'";
201 engine->WriteMessage(filename, 0, 0, asMSGTYPE_ERROR, msg.c_str());
202
203 // TODO: Write the file where this one was included from
204
205 return -1;
206 }
207
208 // Determine size of the file
209 fseek(f, 0, SEEK_END);
210 int len = ftell(f);
211 fseek(f, 0, SEEK_SET);
212
213 // On Win32 it is possible to do the following instead
214 // int len = _filelength(_fileno(f));
215
216 // Read the entire file
217 string code;
218 size_t c = 0;
219 if( len > 0 )
220 {
221 code.resize(len);
222 c = fread(&code[0], len, 1, f);
223 }
224
225 fclose(f);
226
227 if( c == 0 && len > 0 )
228 {
229 // Write a message to the engine's message callback
230 string msg = "Failed to load script file '" + GetAbsolutePath(scriptFile) + "'";
231 engine->WriteMessage(filename, 0, 0, asMSGTYPE_ERROR, msg.c_str());
232 return -1;
233 }
234
235 // Process the script section even if it is zero length so that the name is registered
236 return ProcessScriptSection(code.c_str(), (unsigned int)(code.length()), filename, 0);
237}

239int CScriptBuilder::ProcessScriptSection(const char *script, unsigned int length, const char *sectionname, int lineOffset)
240{
241 vector<string> includes;
242
243 // Perform a superficial parsing of the script first to store the metadata
244 if( length )
245 modifiedScript.assign(script, length);
246 else
247 modifiedScript = script;
248
249 // First perform the checks for #if directives to exclude code that shouldn't be compiled
250 unsigned int pos = 0;
251 int nested = 0;
252 while( pos < modifiedScript.size() )
253 {
254 asUINT len = 0;
255 asETokenClass t = engine->ParseToken(&modifiedScript[pos], modifiedScript.size() - pos, &len);
256 if( t == asTC_UNKNOWN && modifiedScript[pos] == '#' && (pos + 1 < modifiedScript.size()) )
257 {
258 int start = pos++;
259
260 // Is this an #if directive?
261 t = engine->ParseToken(&modifiedScript[pos], modifiedScript.size() - pos, &len);
262
263 string token;
264 token.assign(&modifiedScript[pos], len);
265
266 pos += len;
267
268 if( token == "if" )
269 {
270 t = engine->ParseToken(&modifiedScript[pos], modifiedScript.size() - pos, &len);
271 if( t == asTC_WHITESPACE )
272 {
273 pos += len;
274 t = engine->ParseToken(&modifiedScript[pos], modifiedScript.size() - pos, &len);
275 }
276
277 if( t == asTC_IDENTIFIER )
278 {
279 string word;
280 word.assign(&modifiedScript[pos], len);
281
282 // Overwrite the #if directive with space characters to avoid compiler error
283 pos += len;
284 OverwriteCode(start, pos-start);
285
286 // Has this identifier been defined by the application or not?
287 if( definedWords.find(word) == definedWords.end() )
288 {
289 // Exclude all the code until and including the #endif
290 pos = ExcludeCode(pos);
291 }
292 else
293 {
294 nested++;
295 }
296 }
297 }
298 else if( token == "endif" )
299 {
300 // Only remove the #endif if there was a matching #if
301 if( nested > 0 )
302 {
303 OverwriteCode(start, pos-start);
304 nested--;
305 }
306 }
307 }
308 else
309 pos += len;
310 }
311
312#if AS_PROCESS_METADATA == 1
313 // Preallocate memory
314 string name, declaration;
315 vector<string> metadata;
316 declaration.reserve(100);
317#endif
318
319 // Then check for meta data and pre-processor directives
320 pos = 0;
321 while( pos < modifiedScript.size() )
322 {
323 asUINT len = 0;
324 asETokenClass t = engine->ParseToken(&modifiedScript[pos], modifiedScript.size() - pos, &len);
325 if( t == asTC_COMMENT || t == asTC_WHITESPACE )
326 {
327 pos += len;
328 continue;
329 }
330 string token;
331 token.assign(&modifiedScript[pos], len);
332
333#if AS_PROCESS_METADATA == 1
334 // Skip possible decorators before class and interface declarations
335 if (token == "shared" || token == "abstract" || token == "mixin" || token == "external")
336 {
337 pos += len;
338 continue;
339 }
340
341 // Check if class or interface so the metadata for members can be gathered
342 if( currentClass == "" && (token == "class" || token == "interface") )
343 {
344 // Get the identifier after "class"
345 do
346 {
347 pos += len;
348 if( pos >= modifiedScript.size() )
349 {
350 t = asTC_UNKNOWN;
351 break;
352 }
353 t = engine->ParseToken(&modifiedScript[pos], modifiedScript.size() - pos, &len);
354 } while(t == asTC_COMMENT || t == asTC_WHITESPACE);
355
356 if( t == asTC_IDENTIFIER )
357 {
358 currentClass = modifiedScript.substr(pos,len);
359
360 // Search until first { or ; is encountered
361 while( pos < modifiedScript.length() )
362 {
363 engine->ParseToken(&modifiedScript[pos], modifiedScript.size() - pos, &len);
364
365 // If start of class section encountered stop
366 if( modifiedScript[pos] == '{' )
367 {
368 pos += len;
369 break;
370 }
371 else if (modifiedScript[pos] == ';')
372 {
373 // The class declaration has ended and there are no children
374 currentClass = "";
375 pos += len;
376 break;
377 }
378
379 // Check next symbol
380 pos += len;
381 }
382 }
383
384 continue;
385 }
386
387 // Check if end of class
388 if( currentClass != "" && token == "}" )
389 {
390 currentClass = "";
391 pos += len;
392 continue;
393 }
394
395 // Check if namespace so the metadata for members can be gathered
396 if( token == "namespace" )
397 {
398 // Get the identifier after "namespace"
399 do
400 {
401 pos += len;
402 t = engine->ParseToken(&modifiedScript[pos], modifiedScript.size() - pos, &len);
403 } while(t == asTC_COMMENT || t == asTC_WHITESPACE);
404
405 if( currentNamespace != "" )
406 currentNamespace += "::";
407 currentNamespace += modifiedScript.substr(pos,len);
408
409 // Search until first { is encountered
410 while( pos < modifiedScript.length() )
411 {
412 engine->ParseToken(&modifiedScript[pos], modifiedScript.size() - pos, &len);
413
414 // If start of namespace section encountered stop
415 if( modifiedScript[pos] == '{' )
416 {
417 pos += len;
418 break;
419 }
420
421 // Check next symbol
422 pos += len;
423 }
424
425 continue;
426 }
427
428 // Check if end of namespace
429 if( currentNamespace != "" && token == "}" )
430 {
431 size_t found = currentNamespace.rfind( "::" );
432 if( found != string::npos )
433 {
434 currentNamespace.erase( found );
435 }
436 else
437 {
438 currentNamespace = "";
439 }
440 pos += len;
441 continue;
442 }
443
444 // Is this the start of metadata?
445 if( token == "[" )
446 {
447 // Get the metadata string
448 pos = ExtractMetadata(pos, metadata);
449
450 // Determine what this metadata is for
451 int type;
452 ExtractDeclaration(pos, name, declaration, type);
453
454 // Store away the declaration in a map for lookup after the build has completed
455 if( type > 0 )
456 {
457 SMetadataDecl decl(metadata, name, declaration, type, currentClass, currentNamespace);
458 foundDeclarations.push_back(decl);
459 }
460 }
461 else
462#endif
463 // Is this a preprocessor directive?
464 if( token == "#" && (pos + 1 < modifiedScript.size()) )
465 {
466 int start = pos++;
467
468 t = engine->ParseToken(&modifiedScript[pos], modifiedScript.size() - pos, &len);
469 if (t == asTC_IDENTIFIER)
470 {
471 token.assign(&modifiedScript[pos], len);
472 if (token == "include")
473 {
474 pos += len;
475 t = engine->ParseToken(&modifiedScript[pos], modifiedScript.size() - pos, &len);
476 if (t == asTC_WHITESPACE)
477 {
478 pos += len;
479 t = engine->ParseToken(&modifiedScript[pos], modifiedScript.size() - pos, &len);
480 }
481
482 if (t == asTC_VALUE && len > 2 && (modifiedScript[pos] == '"' || modifiedScript[pos] == '\''))
483 {
484 // Get the include file
485 string includefile;
486 includefile.assign(&modifiedScript[pos + 1], len - 2);
487 pos += len;
488
489 // Make sure the includeFile doesn't contain any line breaks
490 size_t p = includefile.find('\n');
491 if (p != string::npos)
492 {
493 // TODO: Show the correct line number for the error
494 string str = "Invalid file name for #include; it contains a line-break: '" + includefile.substr(0, p) + "'";
495 engine->WriteMessage(sectionname, 0, 0, asMSGTYPE_ERROR, str.c_str());
496 }
497 else
498 {
499 // Store it for later processing
500 includes.push_back(includefile);
501
502 // Overwrite the include directive with space characters to avoid compiler error
503 OverwriteCode(start, pos - start);
504 }
505 }
506 }
507 else if (token == "pragma")
508 {
509 // Read until the end of the line
510 pos += len;
511 for (; pos < modifiedScript.size() && modifiedScript[pos] != '\n'; pos++);
512
513 // Call the pragma callback
514 string pragmaText(&modifiedScript[start + 7], pos - start - 7);
515 int r = pragmaCallback ? pragmaCallback(pragmaText, *this, pragmaParam) : -1;
516 if (r < 0)
517 {
518 // TODO: Report the correct line number
519 engine->WriteMessage(sectionname, 0, 0, asMSGTYPE_ERROR, "Invalid #pragma directive");
520 return r;
521 }
522
523 // Overwrite the pragma directive with space characters to avoid compiler error
524 OverwriteCode(start, pos - start);
525 }
526 }
527 else
528 {
529 // Check for lines starting with #!, e.g. shebang interpreter directive. These will be treated as comments and removed by the preprocessor
530 if (modifiedScript[pos] == '!')
531 {
532 // Read until the end of the line
533 pos += len;
534 for (; pos < modifiedScript.size() && modifiedScript[pos] != '\n'; pos++);
535
536 // Overwrite the directive with space characters to avoid compiler error
537 OverwriteCode(start, pos - start);
538 }
539 }
540 }
541 // Don't search for metadata/includes within statement blocks or between tokens in statements
542 else
543 {
544 pos = SkipStatement(pos);
545 }
546 }
547
548 // Build the actual script
549 engine->SetEngineProperty(asEP_COPY_SCRIPT_SECTIONS, true);
550 module->AddScriptSection(sectionname, modifiedScript.c_str(), modifiedScript.size(), lineOffset);
551
552 if( includes.size() > 0 )
553 {
554 // If the callback has been set, then call it for each included file
555 if( includeCallback )
556 {
557 for( int n = 0; n < (int)includes.size(); n++ )
558 {
559 int r = includeCallback(includes[n].c_str(), sectionname, this, includeParam);
560 if( r < 0 )
561 return r;
562 }
563 }
564 else
565 {
566 // By default we try to load the included file from the relative directory of the current file
567
568 // Determine the path of the current script so that we can resolve relative paths for includes
569 string path = sectionname;
570 size_t posOfSlash = path.find_last_of("/\\");
571 if( posOfSlash != string::npos )
572 path.resize(posOfSlash+1);
573 else
574 path = "";
575
576 // Load the included scripts
577 for( int n = 0; n < (int)includes.size(); n++ )
578 {
579 // If the include is a relative path, then prepend the path of the originating script
580 if( includes[n].find_first_of("/\\") != 0 &&
581 includes[n].find_first_of(":") == string::npos )
582 {
583 includes[n] = path + includes[n];
584 }
585
586 // Include the script section
587 int r = AddSectionFromFile(includes[n].c_str());
588 if( r < 0 )
589 return r;
590 }
591 }
592 }
593
594 return 0;
595}

597int CScriptBuilder::Build()
598{
599 int r = module->Build();
600 if( r < 0 )
601 return r;
602
603#if AS_PROCESS_METADATA == 1
604 // After the script has been built, the metadata strings should be
605 // stored for later lookup by function id, type id, and variable index
606 for( int n = 0; n < (int)foundDeclarations.size(); n++ )
607 {
608 SMetadataDecl *decl = &foundDeclarations[n];
609 module->SetDefaultNamespace(decl->nameSpace.c_str());
610 if( decl->type == MDT_TYPE )
611 {
612 // Find the type id
613 int typeId = module->GetTypeIdByDecl(decl->declaration.c_str());
614 assert( typeId >= 0 );
615 if( typeId >= 0 )
616 typeMetadataMap.insert(map<int, vector<string> >::value_type(typeId, decl->metadata));
617 }
618 else if( decl->type == MDT_FUNC )
619 {
620 if( decl->parentClass == "" )
621 {
622 // Find the function id
623 asIScriptFunction *func = module->GetFunctionByDecl(decl->declaration.c_str());
624 assert( func );
625 if( func )
626 funcMetadataMap.insert(map<int, vector<string> >::value_type(func->GetId(), decl->metadata));
627 }
628 else
629 {
630 // Find the method id
631 int typeId = module->GetTypeIdByDecl(decl->parentClass.c_str());
632 assert( typeId > 0 );
633 map<int, SClassMetadata>::iterator it = classMetadataMap.find(typeId);
634 if( it == classMetadataMap.end() )
635 {
636 classMetadataMap.insert(map<int, SClassMetadata>::value_type(typeId, SClassMetadata(decl->parentClass)));
637 it = classMetadataMap.find(typeId);
638 }
639
640 asITypeInfo *type = engine->GetTypeInfoById(typeId);
641 asIScriptFunction *func = type->GetMethodByDecl(decl->declaration.c_str());
642 assert( func );
643 if( func )
644 it->second.funcMetadataMap.insert(map<int, vector<string> >::value_type(func->GetId(), decl->metadata));
645 }
646 }
647 else if( decl->type == MDT_VIRTPROP )
648 {
649 if( decl->parentClass == "" )
650 {
651 // Find the global virtual property accessors
652 asIScriptFunction *func = module->GetFunctionByName(("get_" + decl->declaration).c_str());
653 if( func )
654 funcMetadataMap.insert(map<int, vector<string> >::value_type(func->GetId(), decl->metadata));
655 func = module->GetFunctionByName(("set_" + decl->declaration).c_str());
656 if( func )
657 funcMetadataMap.insert(map<int, vector<string> >::value_type(func->GetId(), decl->metadata));
658 }
659 else
660 {
661 // Find the method virtual property accessors
662 int typeId = module->GetTypeIdByDecl(decl->parentClass.c_str());
663 assert( typeId > 0 );
664 map<int, SClassMetadata>::iterator it = classMetadataMap.find(typeId);
665 if( it == classMetadataMap.end() )
666 {
667 classMetadataMap.insert(map<int, SClassMetadata>::value_type(typeId, SClassMetadata(decl->parentClass)));
668 it = classMetadataMap.find(typeId);
669 }
670
671 asITypeInfo *type = engine->GetTypeInfoById(typeId);
672 asIScriptFunction *func = type->GetMethodByName(("get_" + decl->declaration).c_str());
673 if( func )
674 it->second.funcMetadataMap.insert(map<int, vector<string> >::value_type(func->GetId(), decl->metadata));
675 func = type->GetMethodByName(("set_" + decl->declaration).c_str());
676 if( func )
677 it->second.funcMetadataMap.insert(map<int, vector<string> >::value_type(func->GetId(), decl->metadata));
678 }
679 }
680 else if( decl->type == MDT_VAR )
681 {
682 if( decl->parentClass == "" )
683 {
684 // Find the global variable index
685 int varIdx = module->GetGlobalVarIndexByName(decl->declaration.c_str());
686 assert( varIdx >= 0 );
687 if( varIdx >= 0 )
688 varMetadataMap.insert(map<int, vector<string> >::value_type(varIdx, decl->metadata));
689 }
690 else
691 {
692 int typeId = module->GetTypeIdByDecl(decl->parentClass.c_str());
693 assert( typeId > 0 );
694
695 // Add the classes if needed
696 map<int, SClassMetadata>::iterator it = classMetadataMap.find(typeId);
697 if( it == classMetadataMap.end() )
698 {
699 classMetadataMap.insert(map<int, SClassMetadata>::value_type(typeId, SClassMetadata(decl->parentClass)));
700 it = classMetadataMap.find(typeId);
701 }
702
703 // Add the variable to class
704 asITypeInfo *objectType = engine->GetTypeInfoById(typeId);
705 int idx = -1;
706
707 // Search through all properties to get proper declaration
708 for( asUINT i = 0; i < (asUINT)objectType->GetPropertyCount(); ++i )
709 {
710 const char *name;
711 objectType->GetProperty(i, &name);
712 if( decl->declaration == name )
713 {
714 idx = i;
715 break;
716 }
717 }
718
719 // If found, add it
720 assert( idx >= 0 );
721 if( idx >= 0 ) it->second.varMetadataMap.insert(map<int, vector<string> >::value_type(idx, decl->metadata));
722 }
723 }
724 else if (decl->type == MDT_FUNC_OR_VAR)
725 {
726 if (decl->parentClass == "")
727 {
728 // Find the global variable index
729 int varIdx = module->GetGlobalVarIndexByName(decl->name.c_str());
730 if (varIdx >= 0)
731 varMetadataMap.insert(map<int, vector<string> >::value_type(varIdx, decl->metadata));
732 else
733 {
734 asIScriptFunction *func = module->GetFunctionByDecl(decl->declaration.c_str());
735 assert(func);
736 if (func)
737 funcMetadataMap.insert(map<int, vector<string> >::value_type(func->GetId(), decl->metadata));
738 }
739 }
740 else
741 {
742 int typeId = module->GetTypeIdByDecl(decl->parentClass.c_str());
743 assert(typeId > 0);
744
745 // Add the classes if needed
746 map<int, SClassMetadata>::iterator it = classMetadataMap.find(typeId);
747 if (it == classMetadataMap.end())
748 {
749 classMetadataMap.insert(map<int, SClassMetadata>::value_type(typeId, SClassMetadata(decl->parentClass)));
750 it = classMetadataMap.find(typeId);
751 }
752
753 // Add the variable to class
754 asITypeInfo *objectType = engine->GetTypeInfoById(typeId);
755 int idx = -1;
756
757 // Search through all properties to get proper declaration
758 for (asUINT i = 0; i < (asUINT)objectType->GetPropertyCount(); ++i)
759 {
760 const char *name;
761 objectType->GetProperty(i, &name);
762 if (decl->name == name)
763 {
764 idx = i;
765 break;
766 }
767 }
768
769 // If found, add it
770 if (idx >= 0)
771 it->second.varMetadataMap.insert(map<int, vector<string> >::value_type(idx, decl->metadata));
772 else
773 {
774 // Look for the matching method instead
775 asITypeInfo *type = engine->GetTypeInfoById(typeId);
776 asIScriptFunction *func = type->GetMethodByDecl(decl->declaration.c_str());
777 if (func)
778 it->second.funcMetadataMap.insert(map<int, vector<string> >::value_type(func->GetId(), decl->metadata));
779 }
780 }
781 }
782 }
783 module->SetDefaultNamespace("");
784#endif
785
786 return 0;

789int CScriptBuilder::SkipStatement(int pos)
790{
791 asUINT len = 0;
792
793 // Skip until ; or { whichever comes first
794 while( pos < (int)modifiedScript.length() && modifiedScript[pos] != ';' && modifiedScript[pos] != '{' )
795 {
796 engine->ParseToken(&modifiedScript[pos], modifiedScript.size() - pos, &len);
797 pos += len;
798 }
799
800 // Skip entire statement block
801 if( pos < (int)modifiedScript.length() && modifiedScript[pos] == '{' )
802 {
803 pos += 1;
804
805 // Find the end of the statement block
806 int level = 1;
807 while( level > 0 && pos < (int)modifiedScript.size() )
808 {

827// Overwrite all code with blanks until the matching #endif
828int CScriptBuilder::ExcludeCode(int pos)
829{
830 asUINT len = 0;
831 int nested = 0;
832 while( pos < (int)modifiedScript.size() )
833 {
834 engine->ParseToken(&modifiedScript[pos], modifiedScript.size() - pos, &len);
835 if( modifiedScript[pos] == '#' )
836 {
837 modifiedScript[pos] = ' ';
838 pos++;
839
840 // Is it an #if or #endif directive?
841 engine->ParseToken(&modifiedScript[pos], modifiedScript.size() - pos, &len);
842 string token;
843 token.assign(&modifiedScript[pos], len);
844 OverwriteCode(pos, len);
845
846 if( token == "if" )
847 {
848 nested++;
849 }
850 else if( token == "endif" )
851 {
852 if( nested-- == 0 )
853 {
854 pos += len;
855 break;
856 }
857 }
858 }
859 else if( modifiedScript[pos] != '\n' )
860 {
861 OverwriteCode(pos, len);
862 }
863 pos += len;
864 }
865
866 return pos;
867}

938int CScriptBuilder::ExtractDeclaration(int pos, string &name, string &declaration, int &type)
939{
940 declaration = "";
941 type = 0;
942
943 int start = pos;
944
945 std::string token;
946 asUINT len = 0;
947 asETokenClass t = asTC_WHITESPACE;
948
949 // Skip white spaces, comments, and leading decorators
950 do
951 {
952 pos += len;
953 t = engine->ParseToken(&modifiedScript[pos], modifiedScript.size() - pos, &len);
954 token.assign(&modifiedScript[pos], len);
955 } while ( t == asTC_WHITESPACE || t == asTC_COMMENT ||
956 token == "private" || token == "protected" ||
957 token == "shared" || token == "external" ||
958 token == "final" || token == "abstract" );
959
960 // We're expecting, either a class, interface, function, or variable declaration
961 if( t == asTC_KEYWORD || t == asTC_IDENTIFIER )
962 {
963 token.assign(&modifiedScript[pos], len);
964 if( token == "interface" || token == "class" || token == "enum" )
965 {
966 // Skip white spaces and comments
967 do
968 {
969 pos += len;
970 t = engine->ParseToken(&modifiedScript[pos], modifiedScript.size() - pos, &len);
971 } while ( t == asTC_WHITESPACE || t == asTC_COMMENT );
972
973 if( t == asTC_IDENTIFIER )
974 {
975 type = MDT_TYPE;
976 declaration.assign(&modifiedScript[pos], len);
977 pos += len;
978 return pos;
979 }
980 }
981 else
982 {
983 // For function declarations, store everything up to the start of the
984 // statement block, except for succeeding decorators (final, override, etc)
985
986 // For variable declaration store just the name as there can only be one
987
988 // We'll only know if the declaration is a variable or function declaration
989 // when we see the statement block, or absense of a statement block.
990 bool hasParenthesis = false;
991 int nestedParenthesis = 0;
992 declaration.append(&modifiedScript[pos], len);
993 pos += len;
994 for(; pos < (int)modifiedScript.size();)
995 {
996 t = engine->ParseToken(&modifiedScript[pos], modifiedScript.size() - pos, &len);
997 token.assign(&modifiedScript[pos], len);
998 if (t == asTC_KEYWORD)
999 {
1000 if (token == "{" && nestedParenthesis == 0)
1001 {
1002 if (hasParenthesis)
1003 {
1004 // We've found the end of a function signature
1005 type = MDT_FUNC;
1006 }
1007 else
1008 {
1009 // We've found a virtual property. Just keep the name
1010 declaration = name;
1011 type = MDT_VIRTPROP;
1012 }
1013 return pos;
1014 }
1015 if ((token == "=" && !hasParenthesis) || token == ";")
1016 {
1017 if (hasParenthesis)
1018 {
1019 // The declaration is ambigous. It can be a variable with initialization, or a function prototype
1020 type = MDT_FUNC_OR_VAR;
1021 }
1022 else
1023 {
1024 // Substitute the declaration with just the name
1025 declaration = name;
1026 type = MDT_VAR;
1027 }
1028 return pos;
1029 }
1030 else if (token == "(")
1031 {
1032 nestedParenthesis++;
1033
1034 // This is the first parenthesis we encounter. If the parenthesis isn't followed
1035 // by a statement block, then this is a variable declaration, in which case we
1036 // should only store the type and name of the variable, not the initialization parameters.
1037 hasParenthesis = true;
1038 }
1039 else if (token == ")")
1040 {
1041 nestedParenthesis--;
1042 }
1043 }
1044 else if( t == asTC_IDENTIFIER )
1045 {
1046 name = token;
1047 }
1048
1049 // Skip trailing decorators
1050 if( !hasParenthesis || nestedParenthesis > 0 || t != asTC_IDENTIFIER || (token != "final" && token != "override" && token != "delete" && token != "property"))
1051 declaration += token;
1052
1053 pos += len;
1054 }
1055 }
1056 }
1057
1058 return start;
1059}

1119string GetAbsolutePath(const string &file)
1120{
1121 string str = file;
1122
1123 // If this is a relative path, complement it with the current path
1124 if( !((str.length() > 0 && (str[0] == '/' || str[0] == '\\')) ||
1125 str.find(":") != string::npos) )
1126 {
1127 str = GetCurrentDir() + "/" + str;
1128 }
1129
1130 // Replace backslashes for forward slashes
1131 size_t pos = 0;
1132 while( (pos = str.find("\\", pos)) != string::npos )
1133 str[pos] = '/';
1134
1135 // Replace /./ with /
1136 pos = 0;
1137 while( (pos = str.find("/./", pos)) != string::npos )
1138 str.erase(pos+1, 2);
1139
1140 // For each /../ remove the parent dir and the /../
1141 pos = 0;
1142 while( (pos = str.find("/../")) != string::npos )
1143 {
1144 size_t pos2 = str.rfind("/", pos-1);
1145 if( pos2 != string::npos )
1146 str.erase(pos2, pos+3-pos2);
1147 else
1148 {
1149 // The path is invalid
1150 break;
1151 }
1152 }
1153
1154 return str;
1155}
