55Rml::String PlatformExtensions::FindSamplesRoot()
56{
57#ifdef RMLUI_PLATFORM_WIN32
58 // Test various relative paths to the "Samples" directory, based on common build and install locations.
59 const char* candidate_paths[] = {
60 "",
61 "..\\Samples\\",
62 "..\\share\\Samples\\",
63 "..\\..\\Samples\\",
64 "..\\..\\..\\Samples\\",
65 "..\\..\\..\\..\\Samples\\",
66 };
67
68 // Fetch the path of the executable, test the candidate paths appended to that.
69 char executable_file_name[MAX_PATH];
70 if (GetModuleFileNameA(NULL, executable_file_name, MAX_PATH) >= MAX_PATH && GetLastError() == ERROR_INSUFFICIENT_BUFFER)
71 {
72 executable_file_name[0] = 0;
73 }
74
75 Rml::String executable_path(executable_file_name);
76 executable_path = executable_path.substr(0, executable_path.rfind('\\') + 1);
77
78 // We assume we have found the correct path if we can find the lookup file from it
79 const char* lookup_file = "assets\\rml.rcss";
80
81 for (const char* relative_path : candidate_paths)
82 {
83 Rml::String absolute_path = executable_path + relative_path;
84
85 if (PathFileExistsA(Rml::String(absolute_path + lookup_file).c_str()))
86 {
87 char canonical_path[MAX_PATH];
88 if (!PathCanonicalizeA(canonical_path, absolute_path.c_str()))
89 canonical_path[0] = 0;
90
91 return Rml::String(canonical_path);
92 }
93 }
94
95 Rml::Log::Message(Rml::Log::LT_ERROR, "Failed to find the path to the samples root");
96
97 return Rml::String();
98
99#elif defined RMLUI_PLATFORM_MACOSX
100
101 Rml::String path = "../Samples/";
102
103 // Find the location of the executable.
104 CFBundleRef bundle = CFBundleGetMainBundle();
105 CFURLRef executable_url = CFBundleCopyExecutableURL(bundle);
106 CFStringRef executable_posix_file_name = CFURLCopyFileSystemPath(executable_url, kCFURLPOSIXPathStyle);
107 CFIndex max_length = CFStringGetMaximumSizeOfFileSystemRepresentation(executable_posix_file_name);
108 char* executable_file_name = new char[max_length];
109 if (!CFStringGetFileSystemRepresentation(executable_posix_file_name, executable_file_name, max_length))
110 executable_file_name[0] = 0;
111
112 Rml::String executable_path = Rml::String(executable_file_name);
113 executable_path = executable_path.substr(0, executable_path.rfind("/") + 1);
114
115 delete[] executable_file_name;
116 CFRelease(executable_posix_file_name);
117 CFRelease(executable_url);
118
119 return executable_path + "../../../" + path;
120
121#elif defined RMLUI_PLATFORM_EMSCRIPTEN
122
123 return Rml::String("Samples/");
124
125#elif defined RMLUI_PLATFORM_UNIX
126
127 char executable_file_name[PATH_MAX];
128 ssize_t len = readlink("/proc/self/exe", executable_file_name, PATH_MAX);
129 if (len == -1)
130 {
131 Rml::Log::Message(Rml::Log::LT_ERROR, "Failed to determine the executable path");
132 executable_file_name[0] = 0;
133 }
134 else
135 {
136 // readlink() does not append a null byte to buf.
137 executable_file_name[len] = 0;
138 }
139 Rml::String executable_path = Rml::String(executable_file_name);
140 executable_path = executable_path.substr(0, executable_path.rfind("/") + 1);
141
142 // We assume we have found the correct path if we can find the lookup file from it.
143 const char* lookup_file = "assets/rml.rcss";
144
145 // Test various relative paths to the "Samples" directory, based on common build and install locations.
146 const char* candidate_paths[] = {
147 "",
148 "../",
149 "../Samples/",
150 "../share/Samples/",
151 "../../Samples/",
152 "../../../Samples/",
153 "../../../../Samples/",
154 };
155
156 auto isRegularFile = [](const Rml::String& path) -> bool {
157 struct stat sb;
158 return stat(path.c_str(), &sb) == 0 && S_ISREG(sb.st_mode);
159 };
160
161 for (const char* relative_path : candidate_paths)
162 {
163 Rml::String absolute_path = executable_path + relative_path;
164 Rml::String absolute_lookup_file = absolute_path + lookup_file;
165
166 if (isRegularFile(absolute_lookup_file))
167 {
168 return absolute_path;
169 }
170 }
171
172 Rml::Log::Message(Rml::Log::LT_ERROR, "Failed to find the path to the samples root");
173
174 return Rml::String();
175
176#else
177
178 return Rml::String();
179
180#endif
181}

185static Rml::StringList ListFilesOrDirectories(ListType type, const Rml::String& directory, const Rml::String& extension)
186{
187 if (directory.empty())
188 return Rml::StringList();
189
190 Rml::StringList result;
191
192#ifdef RMLUI_PLATFORM_WIN32
193
194 const Rml::String find_path = directory + "/*." + (extension.empty() ? Rml::String("*") : extension);
195
196 _finddata_t find_data;
197 intptr_t find_handle = _findfirst(find_path.c_str(), &find_data);
198 if (find_handle != -1)
199 {
200 do
201 {
202 if (strcmp(find_data.name, ".") == 0 || strcmp(find_data.name, "..") == 0)
203 continue;
204
205 bool is_directory = ((find_data.attrib & _A_SUBDIR) == _A_SUBDIR);
206 bool is_file = (!is_directory && ((find_data.attrib & _A_NORMAL) == _A_NORMAL));
207
208 if (((type == ListType::Files) && is_file) || ((type == ListType::Directories) && is_directory))
209 {
210 result.push_back(find_data.name);
211 }
212
213 } while (_findnext(find_handle, &find_data) == 0);
214
215 _findclose(find_handle);
216 }
217
218#else
219
220 struct dirent** file_list = nullptr;
221 const int file_count = scandir(directory.c_str(), &file_list, 0, alphasort);
222 if (file_count == -1)
223 return Rml::StringList();
224
225 for (int i = 0; i < file_count; i++)
226 {
227 if (strcmp(file_list[i]->d_name, ".") == 0 || strcmp(file_list[i]->d_name, "..") == 0)
228 continue;
229
230 bool is_directory = ((file_list[i]->d_type & DT_DIR) == DT_DIR);
231 bool is_file = ((file_list[i]->d_type & DT_REG) == DT_REG);
232
233 if (!extension.empty())
234 {
235 const char* last_dot = strrchr(file_list[i]->d_name, '.');
236 if (!last_dot || strcmp(last_dot + 1, extension.c_str()) != 0)
237 continue;
238 }
239
240 if ((type == ListType::Files && is_file) || (type == ListType::Directories && is_directory))
241 {
242 result.push_back(file_list[i]->d_name);
243 }
244 }
245 for (int i = 0; i < file_count; i++)
246 free(file_list[i]);
247
248 free(file_list);
249
250#endif
251
252 return result;
253}
