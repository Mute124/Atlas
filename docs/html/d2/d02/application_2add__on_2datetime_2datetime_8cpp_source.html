30static bool tm_to_time_point(const tm &_tm, std::chrono::time_point<std::chrono::system_clock> &tp)
31{
32 tm localTm = _tm;
33
34 // Do not rely on timezone, as it is not portable
35 // ref: https://stackoverflow.com/questions/38298261/why-there-is-no-inverse-function-for-gmtime-in-libc
36 // ref: https://stackoverflow.com/questions/8558919/mktime-and-tm-isdst
37 // TODO: mktime is not threadsafe
38 time_t t = mktime(&localTm);
39 if (t == -1)
40 return false;
41
42 // Adjust the time_t since epoch with the difference of the local timezone to the universal timezone
43 // TODO: localtime, gmtime, and mktime are not threadsafe
44 t += (mktime(localtime(&t)) - mktime(gmtime(&t)));
45
46 tp = system_clock::from_time_t(t);
47 return true;
48}

100bool CDateTime::setDate(asUINT year, asUINT month, asUINT day)
101{
102 tm local = time_point_to_tm(tp);
103 local.tm_year = int(year) - 1900;
104 local.tm_mon = month - 1;
105 local.tm_mday = day;
106
107 std::chrono::time_point<std::chrono::system_clock> newTp;
108 if (!tm_to_time_point(local, newTp))
109 return false;
110
111 // Check if the date was actually valid
112 tm local2 = time_point_to_tm(newTp);
113
114 if (local.tm_year != local2.tm_year ||
115 local.tm_mon != local2.tm_mon ||
116 local.tm_mday != local2.tm_mday)
117 return false;
118
119 tp = newTp;
120 return true;
121}
