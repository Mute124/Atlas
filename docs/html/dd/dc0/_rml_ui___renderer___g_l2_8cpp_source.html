60void RenderInterface_GL2::BeginFrame()
61{
62 RMLUI_ASSERT(viewport_width >= 0 && viewport_height >= 0);
63 glViewport(0, 0, viewport_width, viewport_height);
64
65 glEnableClientState(GL_VERTEX_ARRAY);
66 glEnableClientState(GL_COLOR_ARRAY);
67 glDisableClientState(GL_TEXTURE_COORD_ARRAY);
68
69 glEnable(GL_BLEND);
70 glBlendFunc(GL_ONE, GL_ONE_MINUS_SRC_ALPHA);
71
72 glEnable(GL_STENCIL_TEST);
73 glStencilFunc(GL_ALWAYS, 1, GLuint(-1));
74 glStencilMask(GLuint(-1));
75 glStencilOp(GL_KEEP, GL_KEEP, GL_KEEP);
76
77 Rml::Matrix4f projection = Rml::Matrix4f::ProjectOrtho(0, (float)viewport_width, (float)viewport_height, 0, -10000, 10000);
78 glMatrixMode(GL_PROJECTION);
79 glLoadMatrixf(projection.data());
80 glMatrixMode(GL_TEXTURE);
81 glLoadIdentity();
82 glMatrixMode(GL_MODELVIEW);
83 glLoadIdentity();
84
85 transform_enabled = false;
86}

108void RenderInterface_GL2::RenderGeometry(Rml::CompiledGeometryHandle handle, Rml::Vector2f translation, Rml::TextureHandle texture)
109{
110 const GeometryView* geometry = reinterpret_cast<GeometryView*>(handle);
111 const Rml::Vertex* vertices = geometry->vertices.data();
112 const int* indices = geometry->indices.data();
113 const int num_indices = (int)geometry->indices.size();
114
115 glPushMatrix();
116 glTranslatef(translation.x, translation.y, 0);
117
118 glVertexPointer(2, GL_FLOAT, sizeof(Rml::Vertex), &vertices[0].position);
119 glColorPointer(4, GL_UNSIGNED_BYTE, sizeof(Rml::Vertex), &vertices[0].colour);
120
121 if (!texture)
122 {
123 glDisable(GL_TEXTURE_2D);
124 glDisableClientState(GL_TEXTURE_COORD_ARRAY);
125 }
126 else
127 {
128 glEnable(GL_TEXTURE_2D);
129
130 if (texture != TextureEnableWithoutBinding)
131 glBindTexture(GL_TEXTURE_2D, (GLuint)texture);
132
133 glEnableClientState(GL_TEXTURE_COORD_ARRAY);
134 glTexCoordPointer(2, GL_FLOAT, sizeof(Rml::Vertex), &vertices[0].tex_coord);
135 }
136
137 glDrawElements(GL_TRIANGLES, num_indices, GL_UNSIGNED_INT, indices);
138
139 glPopMatrix();
140}

163void RenderInterface_GL2::RenderToClipMask(Rml::ClipMaskOperation operation, Rml::CompiledGeometryHandle geometry, Rml::Vector2f translation)
164{
165 RMLUI_ASSERT(glIsEnabled(GL_STENCIL_TEST));
166 using Rml::ClipMaskOperation;
167
168 const bool clear_stencil = (operation == ClipMaskOperation::Set || operation == ClipMaskOperation::SetInverse);
169 if (clear_stencil)
170 {
171 // @performance Increment the reference value instead of clearing each time.
172 glClear(GL_STENCIL_BUFFER_BIT);
173 }
174
175 GLint stencil_test_value = 0;
176 glGetIntegerv(GL_STENCIL_REF, &stencil_test_value);
177
178 glColorMask(GL_FALSE, GL_FALSE, GL_FALSE, GL_FALSE);
179 glStencilFunc(GL_ALWAYS, GLint(1), GLuint(-1));
180
181 switch (operation)
182 {
183 case ClipMaskOperation::Set:
184 {
185 glStencilOp(GL_KEEP, GL_KEEP, GL_REPLACE);
186 stencil_test_value = 1;
187 }
188 break;
189 case ClipMaskOperation::SetInverse:
190 {
191 glStencilOp(GL_KEEP, GL_KEEP, GL_REPLACE);
192 stencil_test_value = 0;
193 }
194 break;
195 case ClipMaskOperation::Intersect:
196 {
197 glStencilOp(GL_KEEP, GL_KEEP, GL_INCR);
198 stencil_test_value += 1;
199 }
200 break;
201 }
202
203 RenderGeometry(geometry, translation, {});
204
205 // Restore state
206 // @performance Cache state so we don't toggle it unnecessarily.
207 glColorMask(GL_TRUE, GL_TRUE, GL_TRUE, GL_TRUE);
208 glStencilOp(GL_KEEP, GL_KEEP, GL_KEEP);
209 glStencilFunc(GL_EQUAL, stencil_test_value, GLuint(-1));
210}

231Rml::TextureHandle RenderInterface_GL2::LoadTexture(Rml::Vector2i& texture_dimensions, const Rml::String& source)
232{
233 Rml::FileInterface* file_interface = Rml::GetFileInterface();
234 Rml::FileHandle file_handle = file_interface->Open(source);
235 if (!file_handle)
236 {
237 return false;
238 }
239
240 file_interface->Seek(file_handle, 0, SEEK_END);
241 size_t buffer_size = file_interface->Tell(file_handle);
242 file_interface->Seek(file_handle, 0, SEEK_SET);
243
244 if (buffer_size <= sizeof(TGAHeader))
245 {
246 Rml::Log::Message(Rml::Log::LT_ERROR, "Texture file size is smaller than TGAHeader, file is not a valid TGA image.");
247 file_interface->Close(file_handle);
248 return false;
249 }
250
251 using Rml::byte;
252 Rml::UniquePtr<byte[]> buffer(new byte[buffer_size]);
253 file_interface->Read(buffer.get(), buffer_size, file_handle);
254 file_interface->Close(file_handle);
255
256 TGAHeader header;
257 memcpy(&header, buffer.get(), sizeof(TGAHeader));
258
259 int color_mode = header.bitsPerPixel / 8;
260 const size_t image_size = header.width * header.height * 4; // We always make 32bit textures
261
262 if (header.dataType != 2)
263 {
264 Rml::Log::Message(Rml::Log::LT_ERROR, "Only 24/32bit uncompressed TGAs are supported.");
265 return false;
266 }
267
268 // Ensure we have at least 3 colors
269 if (color_mode < 3)
270 {
271 Rml::Log::Message(Rml::Log::LT_ERROR, "Only 24 and 32bit textures are supported.");
272 return false;
273 }
274
275 const byte* image_src = buffer.get() + sizeof(TGAHeader);
276 Rml::UniquePtr<byte[]> image_dest_buffer(new byte[image_size]);
277 byte* image_dest = image_dest_buffer.get();
278
279 // Targa is BGR, swap to RGB, flip Y axis, and convert to premultiplied alpha.
280 for (long y = 0; y < header.height; y++)
281 {
282 long read_index = y * header.width * color_mode;
283 long write_index = ((header.imageDescriptor & 32) != 0) ? read_index : (header.height - y - 1) * header.width * 4;
284 for (long x = 0; x < header.width; x++)
285 {
286 image_dest[write_index] = image_src[read_index + 2];
287 image_dest[write_index + 1] = image_src[read_index + 1];
288 image_dest[write_index + 2] = image_src[read_index];
289 if (color_mode == 4)
290 {
291 const byte alpha = image_src[read_index + 3];
292 for (size_t j = 0; j < 3; j++)
293 image_dest[write_index + j] = byte((image_dest[write_index + j] * alpha) / 255);
294 image_dest[write_index + 3] = alpha;
295 }
296 else
297 image_dest[write_index + 3] = 255;
298
299 write_index += 4;
300 read_index += color_mode;
301 }
302 }
303
304 texture_dimensions.x = header.width;
305 texture_dimensions.y = header.height;
306
307 return GenerateTexture({image_dest, image_size}, texture_dimensions);
308}

310Rml::TextureHandle RenderInterface_GL2::GenerateTexture(Rml::Span<const Rml::byte> source, Rml::Vector2i source_dimensions)
311{
312 GLuint texture_id = 0;
313 glGenTextures(1, &texture_id);
314 if (texture_id == 0)
315 {
316 Rml::Log::Message(Rml::Log::LT_ERROR, "Failed to generate texture.");
317 return {};
318 }
319
320 glBindTexture(GL_TEXTURE_2D, texture_id);
321
322 glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA8, source_dimensions.x, source_dimensions.y, 0, GL_RGBA, GL_UNSIGNED_BYTE, source.data());
323 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
324 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
325
326 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
327 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
328
329 return (Rml::TextureHandle)texture_id;
330}
