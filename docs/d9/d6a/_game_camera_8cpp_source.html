94void Atlas::GameCamera::update()
95{
96 // This is being set here so that the function can follow the principles of DRY -- Don't Repeat Yourself
97 size_t convertedForwards = static_cast<size_t>(GameCamera::EMovementDirection::FORWARD_DIRECTION);
98 size_t convertedBackwards = static_cast<size_t>(GameCamera::EMovementDirection::BACKWARD_DIRECTION);
99 size_t convertedLeft = static_cast<size_t>(GameCamera::EMovementDirection::LEFT_DIRECTION);
100 size_t convertedRight = static_cast<size_t>(GameCamera::EMovementDirection::RIGHT_DIRECTION);
101 size_t convertedUp = static_cast<size_t>(GameCamera::EMovementDirection::UP_DIRECTION);
102 size_t convertedDown = static_cast<size_t>(GameCamera::EMovementDirection::DOWN_DIRECTION);
103
104 //UpdateCamera(&mCameraData, CAMERA_FIRST_PERSON);
105 UpdateCameraPro(&mCameraData,
106 Vector3 {
107 // Move forward-backward
108 this->mDirectionalInput[convertedForwards] -
109 this->mDirectionalInput[convertedBackwards],
110
111 // Move right-left
112 this->mDirectionalInput[convertedRight] -
113 this->mDirectionalInput[convertedLeft],
114
115 // Move up-down
116 this->mDirectionalInput[convertedUp] -
117 this->mDirectionalInput[convertedDown]
118 },
119 Vector3 {
120 GetMouseDelta().x * 0.05f, // Rotation: yaw
121 GetMouseDelta().y * 0.05f, // Rotation: pitch
122 0.0f // Rotation: roll
123 },
124 GetMouseWheelMove() * 2.0f // Move to target (zoom)
125 );
126 /****************************************************
127 * Clear directional input
128 *****************************************************/
129
130 this->mDirectionalInput[convertedForwards] = 0.0f;
131 this->mDirectionalInput[convertedBackwards] = 0.0f;
132
133 this->mDirectionalInput[convertedLeft] = 0.0f;
134 this->mDirectionalInput[convertedRight] = 0.0f;
135
136 this->mDirectionalInput[convertedUp] = 0.0f;
137 this->mDirectionalInput[convertedDown] = 0.0f;
138
139}
